78ThreadsafeImuBuffer::QueryResult ThreadsafeImuBuffer::isDataAvailableUpToImpl(
79 const Timestamp& timestamp_ns_from,
80 const Timestamp& timestamp_ns_to) const {
81 CHECK_LT(timestamp_ns_from, timestamp_ns_to);
82
83 if (shutdown_) {
84 return QueryResult::kQueueShutdown;
85 }
86
87 if (buffer_.empty()) {
88 return QueryResult::kDataNotYetAvailable;
89 }
90
91 ImuMeasurement value;
92 if (buffer_.getNewestValue(&value) && value.timestamp_ < timestamp_ns_to) {
93 // This is triggered if the timestamp_ns_to requested exceeds the newest
94 // IMU measurement, meaning that there is data still to arrive to reach the
95 // requested point in time.
96 return QueryResult::kDataNotYetAvailable;
97 }
98
99 if (buffer_.getOldestValue(&value) && timestamp_ns_from < value.timestamp_) {
100 // This is triggered if the user requests data previous to the oldest IMU
101 // measurement present in the buffer, meaning that there is missing data
102 // from the timestamp_ns_from requested to the oldest stored timestamp.
103 return QueryResult::kDataNeverAvailable;
104 }
105 return QueryResult::kDataAvailable;
106}

108void ThreadsafeImuBuffer::linearInterpolate(const Timestamp& t0,
109 const ImuAccGyr& y0,
110 const Timestamp& t1,
111 const ImuAccGyr& y1,
112 const Timestamp& t, ImuAccGyr* y) {
113 CHECK_NOTNULL(y);
114 CHECK_LE(t0, t);
115 CHECK_LE(t, t1);
116 *y = t0 == t1 ? y0 : // y0 if t0 == t1, interpolate otherwise:
117 y0 + (y1 - y0) * static_cast<double>(t - t0) /
118 static_cast<double>(t1 - t0);
119}

121ThreadsafeImuBuffer::QueryResult ThreadsafeImuBuffer::getImuDataBtwTimestamps(
122 const Timestamp& timestamp_ns_from, const Timestamp& timestamp_ns_to,
123 Timestamps* imu_timestamps, ImuAccGyrs* imu_measurements,
124 bool get_lower_bound) {
125 CHECK_NOTNULL(imu_timestamps);
126 CHECK_NOTNULL(imu_measurements);
127 DCHECK_LT(timestamp_ns_from, timestamp_ns_to);
128 QueryResult query_result =
129 isDataAvailableUpToImpl(timestamp_ns_from, timestamp_ns_to);
130 if (query_result != QueryResult::kDataAvailable) {
131 imu_timestamps->resize(Eigen::NoChange, 0);
132 imu_measurements->resize(Eigen::NoChange, 0);
133 return query_result;
134 }
135
136 // TODO can we avoid copying and have the temporal buffer store pairs of
137 // timestamps/accgyr data?
138 // Copy the data with timestamp_up_to <= timestamps_buffer from the buffer.
139 Aligned<std::vector, ImuMeasurement> between_values;
140 CHECK(buffer_.getValuesBetweenTimes(timestamp_ns_from, timestamp_ns_to,
141 &between_values, get_lower_bound));
142
143 if (between_values.empty()) {
144 LOG(WARNING) << "No IMU measurements available strictly between time "
145 << timestamp_ns_from << "[ns] and " << timestamp_ns_to
146 << "[ns].";
147 imu_timestamps->resize(Eigen::NoChange, 0);
148 imu_measurements->resize(Eigen::NoChange, 0);
149 return QueryResult::kTooFewMeasurementsAvailable;
150 }
151
152 const size_t num_measurements = between_values.size();
153 imu_timestamps->resize(Eigen::NoChange, num_measurements);
154 imu_measurements->resize(Eigen::NoChange, num_measurements);
155
156 for (size_t idx = 0u; idx < num_measurements; ++idx) {
157 (*imu_timestamps)(idx) = between_values[idx].timestamp_;
158 (*imu_measurements).col(idx) = between_values[idx].acc_gyr_;
159 }
160
161 return query_result;
162}

164ThreadsafeImuBuffer::QueryResult
165ThreadsafeImuBuffer::getImuDataInterpolatedUpperBorder(
166 const Timestamp& timestamp_ns_from, const Timestamp& timestamp_ns_to,
167 Timestamps* imu_timestamps, ImuAccGyrs* imu_measurements) {
168 CHECK_NOTNULL(imu_timestamps);
169 CHECK_NOTNULL(imu_measurements);
170 DCHECK_LT(timestamp_ns_from, timestamp_ns_to);
171 // Get data.
172 QueryResult query_result = getImuDataBtwTimestamps(
173 timestamp_ns_from, timestamp_ns_to, imu_timestamps, imu_measurements,
174 true); // Get lower bound.
175 // Early exit if there is no data.
176 if (query_result != QueryResult::kDataAvailable) {
177 imu_timestamps->resize(Eigen::NoChange, 0);
178 imu_measurements->resize(Eigen::NoChange, 0);
179 return query_result;
180 }
181
182 // Interpolate upper border.
183 ImuAccGyr interpolated_upper_border;
184 interpolateValueAtTimestamp(timestamp_ns_to, &interpolated_upper_border);
185
186 DCHECK_EQ(imu_timestamps->rows(), 1);
187 DCHECK_EQ(imu_measurements->rows(), 6);
188 // The last measurement will correspond to the interpolated data.
189 const size_t num_measurements = imu_timestamps->cols() + 1u;
190 imu_timestamps->conservativeResize(Eigen::NoChange, num_measurements);
191 imu_measurements->conservativeResize(Eigen::NoChange, num_measurements);
192 // Append upper border.
193 imu_timestamps->rightCols<1>()(0) = timestamp_ns_to;
194 imu_measurements->rightCols<1>() = interpolated_upper_border;
195
196 return query_result;
197}

199ThreadsafeImuBuffer::QueryResult
200ThreadsafeImuBuffer::getImuDataInterpolatedBorders(
201 const Timestamp& timestamp_ns_from, const Timestamp& timestamp_ns_to,
202 Timestamps* imu_timestamps, ImuAccGyrs* imu_measurements) {
203 CHECK_NOTNULL(imu_timestamps);
204 CHECK_NOTNULL(imu_measurements);
205 // Get data.
206 Timestamps imu_timestamps_tmp;
207 ImuAccGyrs imu_measurements_tmp;
208 QueryResult query_result =
209 getImuDataBtwTimestamps(timestamp_ns_from, timestamp_ns_to,
210 &imu_timestamps_tmp, &imu_measurements_tmp);
211
212 // Early exit if there is no data.
213 if (query_result != QueryResult::kDataAvailable) {
214 imu_timestamps->resize(Eigen::NoChange, 0);
215 imu_measurements->resize(Eigen::NoChange, 0);
216 return query_result;
217 }
218
219 // Interpolate lower border.
220 ImuAccGyr interpolated_lower_border;
221 interpolateValueAtTimestamp(timestamp_ns_from, &interpolated_lower_border);
222 // Interpolate upper border.
223 ImuAccGyr interpolated_upper_border;
224 interpolateValueAtTimestamp(timestamp_ns_to, &interpolated_upper_border);
225
226 DCHECK_EQ(imu_timestamps->rows(), 1);
227 DCHECK_EQ(imu_measurements->rows(), 6);
228 // The first and last measurements will correspond to the interpolated data.
229 const size_t num_measurements = imu_timestamps_tmp.cols() + 2u;
230 imu_timestamps->resize(Eigen::NoChange, num_measurements);
231 imu_measurements->resize(Eigen::NoChange, num_measurements);
232 // Prepend lower border.
233 imu_timestamps->leftCols<1>()(0) = timestamp_ns_from;
234 imu_measurements->leftCols<1>() = interpolated_lower_border;
235 // Add measurements.
236 imu_timestamps->middleCols(1, imu_timestamps_tmp.cols()) = imu_timestamps_tmp;
237 imu_measurements->middleCols(1, imu_measurements_tmp.cols()) =
238 imu_measurements_tmp;
239 // Append upper border.
240 imu_timestamps->rightCols<1>()(0) = timestamp_ns_to;
241 imu_measurements->rightCols<1>() = interpolated_upper_border;
242
243 return query_result;
244}

246void ThreadsafeImuBuffer::interpolateValueAtTimestamp(
247 const Timestamp& timestamp_ns, ImuAccGyr* interpolated_imu_measurement) {
248 CHECK_NOTNULL(interpolated_imu_measurement);
249 Timestamp pre_border_timestamp, post_border_timestamp;
250 ImuMeasurement pre_border_value, post_border_value;
251 CHECK(buffer_.getValueAtOrBeforeTime(timestamp_ns, &pre_border_timestamp,
252 &pre_border_value))
253 << "The IMU buffer seems not to contain measurements at or before time: "
254 << timestamp_ns;
255 CHECK_EQ(pre_border_timestamp, pre_border_value.timestamp_);
256 CHECK(buffer_.getValueAtOrAfterTime(timestamp_ns, &post_border_timestamp,
257 &post_border_value))
258 << "The IMU buffer seems not to contain measurements at or after time: "
259 << timestamp_ns;
260 CHECK_EQ(post_border_timestamp, post_border_value.timestamp_);
261 linearInterpolate(pre_border_value.timestamp_, pre_border_value.acc_gyr_,
262 post_border_value.timestamp_, post_border_value.acc_gyr_,
263 timestamp_ns, interpolated_imu_measurement);
264}

266ThreadsafeImuBuffer::QueryResult
267ThreadsafeImuBuffer::getImuDataInterpolatedBordersBlocking(
268 const Timestamp& timestamp_ns_from, const Timestamp& timestamp_ns_to,
269 long int wait_timeout_nanoseconds, Timestamps* imu_timestamps,
270 ImuAccGyrs* imu_measurements) {
271 CHECK_NOTNULL(imu_timestamps);
272 CHECK_NOTNULL(imu_measurements);
273
274 // Wait for the IMU buffer to contain the required measurements within a
275 // timeout.
276 auto tic = utils::Timer::tic();
277 QueryResult query_result;
278 {
279 std::unique_lock<std::mutex> lock(m_buffer_);
280 while ((query_result =
281 isDataAvailableUpToImpl(timestamp_ns_from, timestamp_ns_to)) !=
282 QueryResult::kDataAvailable) {
283 cv_new_measurement_.wait_for(
284 lock, std::chrono::nanoseconds(wait_timeout_nanoseconds));
285
286 if (shutdown_) {
287 imu_timestamps->resize(Eigen::NoChange, 0);
288 imu_measurements->resize(Eigen::NoChange, 0);
289 return QueryResult::kQueueShutdown;
290 }
291
292 // Check if we hit the max. time allowed to wait for the required data.
293 auto toc = utils::Timer::toc<std::chrono::nanoseconds>(tic);
294 if (toc.count() >= wait_timeout_nanoseconds) {
295 imu_timestamps->resize(Eigen::NoChange, 0);
296 imu_measurements->resize(Eigen::NoChange, 0);
297 LOG(WARNING) << "Timeout reached while trying to get the requested "
298 << "IMU data. Requested range: " << timestamp_ns_from
299 << " to " << timestamp_ns_to << ".";
300 if (query_result == QueryResult::kDataNotYetAvailable) {
301 LOG(WARNING) << "The relevant IMU data is not yet available.";
302 } else if (query_result == QueryResult::kDataNeverAvailable) {
303 LOG(WARNING) << "The relevant IMU data will never be available. "
304 << "Either the buffer is too small or a sync issue "
305 << "occurred.";
306 } else {
307 LOG(FATAL) << "Unknown query result error.";
308 }
309 return query_result;
310 }
311 }
312 }
313 return getImuDataInterpolatedBorders(timestamp_ns_from, timestamp_ns_to,
314 imu_timestamps, imu_measurements);
315}
