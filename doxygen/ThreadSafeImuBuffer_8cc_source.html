73ThreadsafeImuBuffer::QueryResult ThreadsafeImuBuffer::isDataAvailableUpToImpl(
74 const Timestamp& timestamp_ns_from,
75 const Timestamp& timestamp_ns_to) const {
76 CHECK_LT(timestamp_ns_from, timestamp_ns_to);
77
78 if (shutdown_) {
79 return QueryResult::kQueueShutdown;
80 }
81
82 if (buffer_.empty()) {
83 return QueryResult::kDataNotYetAvailable;
84 }
85
86 ImuMeasurement value;
87 if (buffer_.getNewestValue(&value) && value.timestamp_ < timestamp_ns_to) {
88 // This is triggered if the timestamp_ns_to requested exceeds the newest
89 // IMU measurement, meaning that there is data still to arrive to reach the
90 // requested point in time.
91 return QueryResult::kDataNotYetAvailable;
92 }
93
94 if (buffer_.getOldestValue(&value) && timestamp_ns_from < value.timestamp_) {
95 // This is triggered if the user requests data previous to the oldest IMU
96 // measurement present in the buffer, meaning that there is missing data
97 // from the timestamp_ns_from requested to the oldest stored timestamp.
98 return QueryResult::kDataNeverAvailable;
99 }
100 return QueryResult::kDataAvailable;
101}

103void ThreadsafeImuBuffer::linearInterpolate(const Timestamp& t0,
104 const ImuAccGyr& y0,
105 const Timestamp& t1,
106 const ImuAccGyr& y1,
107 const Timestamp& t,
108 ImuAccGyr* y) {
109 CHECK_NOTNULL(y);
110 CHECK_LE(t0, t);
111 CHECK_LE(t, t1);
112 *y = t0 == t1 ? y0 : // y0 if t0 == t1, interpolate otherwise:
113 y0 + (y1 - y0) * static_cast<double>(t - t0) /
114 static_cast<double>(t1 - t0);
115}

117ThreadsafeImuBuffer::QueryResult ThreadsafeImuBuffer::getImuDataBtwTimestamps(
118 const Timestamp& timestamp_ns_from,
119 const Timestamp& timestamp_ns_to,
120 Timestamps* imu_timestamps,
121 ImuAccGyrs* imu_measurements,
122 bool get_lower_bound) {
123 CHECK_NOTNULL(imu_timestamps);
124 CHECK_NOTNULL(imu_measurements);
125 DCHECK_LT(timestamp_ns_from, timestamp_ns_to);
126 QueryResult query_result =
127 isDataAvailableUpToImpl(timestamp_ns_from, timestamp_ns_to);
128 if (query_result != QueryResult::kDataAvailable) {
129 imu_timestamps->resize(Eigen::NoChange, 0);
130 imu_measurements->resize(Eigen::NoChange, 0);
131 return query_result;
132 }
133
134 // TODO can we avoid copying and have the temporal buffer store pairs of
135 // timestamps/accgyr data?
136 // Copy the data with timestamp_up_to <= timestamps_buffer from the buffer.
137 Aligned<std::vector, ImuMeasurement> between_values;
138 CHECK(buffer_.getValuesBetweenTimes(timestamp_ns_from, timestamp_ns_to,
139 &between_values, get_lower_bound));
140
141 if (between_values.empty()) {
142 LOG(WARNING) << "No IMU measurements available strictly between time "
143 << timestamp_ns_from << "[ns] and " << timestamp_ns_to
144 << "[ns].";
145 imu_timestamps->resize(Eigen::NoChange, 0);
146 imu_measurements->resize(Eigen::NoChange, 0);
147 return QueryResult::kTooFewMeasurementsAvailable;
148 }
149
150 const size_t num_measurements = between_values.size();
151 imu_timestamps->resize(Eigen::NoChange, num_measurements);
152 imu_measurements->resize(Eigen::NoChange, num_measurements);
153
154 for (size_t idx = 0u; idx < num_measurements; ++idx) {
155 (*imu_timestamps)(idx) = between_values[idx].timestamp_;
156 (*imu_measurements).col(idx) = between_values[idx].acc_gyr_;
157 }
158
159 return query_result;
160}

162ThreadsafeImuBuffer::QueryResult
163ThreadsafeImuBuffer::getImuDataInterpolatedUpperBorder(
164 const Timestamp& timestamp_ns_from,
165 const Timestamp& timestamp_ns_to,
166 Timestamps* imu_timestamps,
167 ImuAccGyrs* imu_measurements) {
168 CHECK_NOTNULL(imu_timestamps);
169 CHECK_NOTNULL(imu_measurements);
170 DCHECK_LT(timestamp_ns_from, timestamp_ns_to);
171 // Get data.
172 QueryResult query_result = getImuDataBtwTimestamps(
173 timestamp_ns_from, timestamp_ns_to, imu_timestamps, imu_measurements,
174 true); // Get lower bound.
175 // Early exit if there is no data.
176 if (query_result != QueryResult::kDataAvailable) {
177 imu_timestamps->resize(Eigen::NoChange, 0);
178 imu_measurements->resize(Eigen::NoChange, 0);
179 return query_result;
180 }
181
182 // Interpolate upper border.
183 ImuAccGyr interpolated_upper_border;
184 interpolateValueAtTimestamp(timestamp_ns_to, &interpolated_upper_border);
185
186 DCHECK_EQ(imu_timestamps->rows(), 1);
187 DCHECK_EQ(imu_measurements->rows(), 6);
188 // The last measurement will correspond to the interpolated data.
189 const size_t num_measurements = imu_timestamps->cols() + 1u;
190 imu_timestamps->conservativeResize(Eigen::NoChange, num_measurements);
191 imu_measurements->conservativeResize(Eigen::NoChange, num_measurements);
192 // Append upper border.
193 imu_timestamps->rightCols<1>()(0) = timestamp_ns_to;
194 imu_measurements->rightCols<1>() = interpolated_upper_border;
195
196 return query_result;
197}

199ThreadsafeImuBuffer::QueryResult
200ThreadsafeImuBuffer::getImuDataInterpolatedBorders(
201 const Timestamp& timestamp_ns_from,
202 const Timestamp& timestamp_ns_to,
203 Timestamps* imu_timestamps,
204 ImuAccGyrs* imu_measurements) {
205 CHECK_NOTNULL(imu_timestamps);
206 CHECK_NOTNULL(imu_measurements);
207 // Get data.
208 Timestamps imu_timestamps_tmp;
209 ImuAccGyrs imu_measurements_tmp;
210 QueryResult query_result =
211 getImuDataBtwTimestamps(timestamp_ns_from, timestamp_ns_to,
212 &imu_timestamps_tmp, &imu_measurements_tmp);
213
214 // Early exit if there is no data.
215 if (query_result != QueryResult::kDataAvailable) {
216 imu_timestamps->resize(Eigen::NoChange, 0);
217 imu_measurements->resize(Eigen::NoChange, 0);
218 return query_result;
219 }
220
221 // Interpolate lower border.
222 ImuAccGyr interpolated_lower_border;
223 interpolateValueAtTimestamp(timestamp_ns_from, &interpolated_lower_border);
224 // Interpolate upper border.
225 ImuAccGyr interpolated_upper_border;
226 interpolateValueAtTimestamp(timestamp_ns_to, &interpolated_upper_border);
227
228 DCHECK_EQ(imu_timestamps->rows(), 1);
229 DCHECK_EQ(imu_measurements->rows(), 6);
230 // The first and last measurements will correspond to the interpolated data.
231 const size_t num_measurements = imu_timestamps_tmp.cols() + 2u;
232 imu_timestamps->resize(Eigen::NoChange, num_measurements);
233 imu_measurements->resize(Eigen::NoChange, num_measurements);
234 // Prepend lower border.
235 imu_timestamps->leftCols<1>()(0) = timestamp_ns_from;
236 imu_measurements->leftCols<1>() = interpolated_lower_border;
237 // Add measurements.
238 imu_timestamps->middleCols(1, imu_timestamps_tmp.cols()) = imu_timestamps_tmp;
239 imu_measurements->middleCols(1, imu_measurements_tmp.cols()) =
240 imu_measurements_tmp;
241 // Append upper border.
242 imu_timestamps->rightCols<1>()(0) = timestamp_ns_to;
243 imu_measurements->rightCols<1>() = interpolated_upper_border;
244
245 return query_result;
246}

248void ThreadsafeImuBuffer::interpolateValueAtTimestamp(
249 const Timestamp& timestamp_ns,
250 ImuAccGyr* interpolated_imu_measurement) {
251 CHECK_NOTNULL(interpolated_imu_measurement);
252 Timestamp pre_border_timestamp, post_border_timestamp;
253 ImuMeasurement pre_border_value, post_border_value;
254 CHECK(buffer_.getValueAtOrBeforeTime(timestamp_ns, &pre_border_timestamp,
255 &pre_border_value))
256 << "The IMU buffer seems not to contain measurements at or before time: "
257 << timestamp_ns;
258 CHECK_EQ(pre_border_timestamp, pre_border_value.timestamp_);
259 CHECK(buffer_.getValueAtOrAfterTime(timestamp_ns, &post_border_timestamp,
260 &post_border_value))
261 << "The IMU buffer seems not to contain measurements at or after time: "
262 << timestamp_ns;
263 CHECK_EQ(post_border_timestamp, post_border_value.timestamp_);
264 linearInterpolate(pre_border_value.timestamp_,
265 pre_border_value.acc_gyr_,
266 post_border_value.timestamp_,
267 post_border_value.acc_gyr_,
268 timestamp_ns,
269 interpolated_imu_measurement);
270}

272ThreadsafeImuBuffer::QueryResult
273ThreadsafeImuBuffer::getImuDataInterpolatedBordersBlocking(
274 const Timestamp& timestamp_ns_from,
275 const Timestamp& timestamp_ns_to,
276 long int wait_timeout_nanoseconds,
277 Timestamps* imu_timestamps,
278 ImuAccGyrs* imu_measurements) {
279 CHECK_NOTNULL(imu_timestamps);
280 CHECK_NOTNULL(imu_measurements);
281
282 // Wait for the IMU buffer to contain the required measurements within a
283 // timeout.
284 auto tic = utils::Timer::tic();
285 QueryResult query_result;
286 {
287 std::unique_lock<std::mutex> lock(m_buffer_);
288 while ((query_result =
289 isDataAvailableUpToImpl(timestamp_ns_from, timestamp_ns_to)) !=
290 QueryResult::kDataAvailable) {
291 cv_new_measurement_.wait_for(
292 lock, std::chrono::nanoseconds(wait_timeout_nanoseconds));
293
294 if (shutdown_) {
295 imu_timestamps->resize(Eigen::NoChange, 0);
296 imu_measurements->resize(Eigen::NoChange, 0);
297 return QueryResult::kQueueShutdown;
298 }
299
300 // Check if we hit the max. time allowed to wait for the required data.
301 auto toc = utils::Timer::toc<std::chrono::nanoseconds>(tic);
302 if (toc.count() >= wait_timeout_nanoseconds) {
303 imu_timestamps->resize(Eigen::NoChange, 0);
304 imu_measurements->resize(Eigen::NoChange, 0);
305 LOG(WARNING) << "Timeout reached while trying to get the requested "
306 << "IMU data. Requested range: " << timestamp_ns_from
307 << " to " << timestamp_ns_to << ".";
308 if (query_result == QueryResult::kDataNotYetAvailable) {
309 LOG(WARNING) << "The relevant IMU data is not yet available.";
310 } else if (query_result == QueryResult::kDataNeverAvailable) {
311 LOG(WARNING) << "The relevant IMU data will never be available. "
312 << "Either the buffer is too small or a sync issue "
313 << "occurred.";
314 } else {
315 LOG(FATAL) << "Unknown query result error.";
316 }
317 return query_result;
318 }
319 }
320 }
321 return getImuDataInterpolatedBorders(timestamp_ns_from, timestamp_ns_to,
322 imu_timestamps, imu_measurements);
323}
