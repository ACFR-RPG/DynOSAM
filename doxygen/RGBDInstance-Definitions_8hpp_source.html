62 RGBDInstanceOutputPacket(
63 const StatusKeypointVector& static_keypoint_measurements,
64 const StatusKeypointVector& dynamic_keypoint_measurements,
65 const StatusLandmarkVector& static_landmarks,
66 const StatusLandmarkVector& dynamic_landmarks,
67 const gtsam::Pose3 T_world_camera, const Timestamp timestamp,
68 const FrameId frame_id, const MotionEstimateMap& estimated_motions,
69 const ObjectPoseMap propogated_object_poses = {},
70 const gtsam::Pose3Vector camera_poses = {},
71 const Camera::Ptr camera = nullptr,
72 const GroundTruthInputPacket::Optional& gt_packet = std::nullopt,
73 const DebugImagery::Optional& debug_imagery = std::nullopt)
74 : FrontendOutputPacketBase(
75 FrontendType::kRGBD, static_keypoint_measurements,
76 dynamic_keypoint_measurements, T_world_camera, timestamp, frame_id,
77 camera, gt_packet, debug_imagery),
78 static_landmarks_(static_landmarks),
79 dynamic_landmarks_(dynamic_landmarks),
80 estimated_motions_(estimated_motions),
81 propogated_object_poses_(propogated_object_poses),
82 camera_poses_(camera_poses) {
83 // they need to be the same size as we expect a 1-to-1 relation between the
84 // keypoint and the landmark (which acts as an initalisation point)
85 CHECK_EQ(static_landmarks_.size(), static_keypoint_measurements_.size());
86 CHECK_EQ(dynamic_landmarks_.size(), dynamic_keypoint_measurements_.size());
87
88 object_clouds_ = groupObjectCloud(dynamic_landmarks_, T_world_camera_);
89 for (const auto& [object_id, this_object_cloud] : object_clouds_) {
90 pcl::PointCloud<pcl::PointXYZRGB>::Ptr obj_cloud_ptr =
91 pcl::make_shared<pcl::PointCloud<pcl::PointXYZRGB>>(
92 this_object_cloud);
93 // ObjectBBX this_object_bbx = findOBBFromCloud(obj_cloud_ptr);
94
95 // TODO: currently axis-aligned bounding box (not orientated bbx)
96 ObjectBBX this_object_bbx =
97 findAABBFromCloud<pcl::PointXYZRGB>(obj_cloud_ptr);
98 object_bbxes_.insert2(object_id, this_object_bbx);
99 }
100 }
