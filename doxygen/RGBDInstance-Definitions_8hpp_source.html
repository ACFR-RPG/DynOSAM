91 const PointCloudLabelRGB::Ptr dense_labelled_cloud = nullptr,
92 const FrameIdTimestampMap& involved_timestamps = {})
93 : FrontendOutputPacketBase(
94 FrontendType::kRGBD, static_keypoint_measurements,
95 dynamic_keypoint_measurements, T_world_camera, timestamp, frame_id,
96 camera, gt_packet, debug_imagery),
97 static_landmarks_(static_landmarks),
98 dynamic_landmarks_(dynamic_landmarks),
99 object_motions_(object_motions),
100 propogated_object_poses_(propogated_object_poses),
101 camera_poses_(camera_poses),
102 dense_labelled_cloud_(dense_labelled_cloud),
103 involved_timestamps_(involved_timestamps) {
104 // they need to be the same size as we expect a 1-to-1 relation between the
105 // keypoint and the landmark (which acts as an initalisation point)
106 CHECK_EQ(static_landmarks_.size(), static_keypoint_measurements_.size());
107 CHECK_EQ(dynamic_landmarks_.size(), dynamic_keypoint_measurements_.size());
108
109 object_clouds_ = groupObjectCloud(dynamic_landmarks_, T_world_camera_);
110 for (const auto& [object_id, this_object_cloud] : object_clouds_) {
111 pcl::PointCloud<pcl::PointXYZRGB>::Ptr obj_cloud_ptr =
112 pcl::make_shared<pcl::PointCloud<pcl::PointXYZRGB>>(
113 this_object_cloud);
114
115 // TODO: currently axis-aligned bounding box (not orientated bbx)
116 ObjectBBX this_object_bbx =
117 findAABBFromCloud<pcl::PointXYZRGB>(obj_cloud_ptr);
118 object_bbxes_.insert2(object_id, this_object_bbx);
119 object_ids_.push_back(object_id);
120 }
121 }
