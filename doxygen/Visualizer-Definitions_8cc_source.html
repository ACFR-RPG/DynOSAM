44void OpenCVImageDisplayQueue::process() {
45 bool queue_state = false;
46 if (parallel_run_) {
47 ImageToDisplay image_to_display;
48 queue_state = display_queue_->popBlockingWithTimeout(image_to_display, 2);
49
50 if (queue_state) {
51 // cv::namedWindow(image_to_display.name_);
52 cv::imshow(image_to_display.name_, image_to_display.image_);
53 cv::waitKey(1); // Not needed because we are using startWindowThread()
54 }
55 } else {
56 std::vector<ImageToDisplay> images_to_display;
57 queue_state = display_queue_->popAll(images_to_display, 2);
58
59 if (queue_state) {
60 for (const auto& image_to_display : images_to_display) {
61 cv::imshow(image_to_display.name_, image_to_display.image_);
62 }
63 cv::waitKey(1);
64 }
65 }
66}
