51ObjectOdometryMap DSDTransport::constructObjectOdometries(
52 const MotionEstimateMap& motions_k, const ObjectPoseMap& poses,
53 FrameId frame_id_k, Timestamp timestamp_k,
54 const std::string& frame_id_link) {
55 // need to get poses for k-1
56 // TODO: no way to ensure that the motions are for frame k
57 // this is a weird data-structure to use and motions are per frame and
58 // ObjectPoseMap is over all k to K
59 // const FrameId frame_id_k_1 = frame_id_k - 1u;
60 ObjectOdometryMap object_odom_map;
61 for (const auto& [object_id, object_motion] : motions_k) {
62 const gtsam::Pose3& motion_k = object_motion;
63
64 if (!poses.exists(object_id, frame_id_k)) {
65 LOG(WARNING) << "Cannot construct ObjectOdometry for object " << object_id
66 << ", at frame " << frame_id_k
67 << " Missing entry in ObjectPoseMap";
68 continue;
69 }
70
71 const gtsam::Pose3& pose_k = poses.at(object_id, frame_id_k);
72
73 const std::string child_frame_id_link = constructObjectFrameLink(object_id);
74
75 object_odom_map.insert2(
76 child_frame_id_link,
77 constructObjectOdometry(motion_k, pose_k, object_id, timestamp_k,
78 frame_id_link, child_frame_id_link));
79 }
80
81 return object_odom_map;
82}

84MultiObjectOdometryPath DSDTransport::constructMultiObjectOdometryPaths(
85 const MotionEstimateMap& motions_k, const ObjectPoseMap& poses,
86 FrameId frame_id_k, Timestamp timestamp_k, rclcpp::Time ros_time_now,
87 const std::string& frame_id_link) {
88 MultiObjectOdometryPath multi_path;
89 multi_path.header.stamp = ros_time_now;
90 multi_path.header.frame_id = frame_id_link;
91
92 // TODO: right now dont have the motion for every timestep so... just leave
93 // blank?
94 for (const auto& [object_id, frame_pose_map] : poses) {
95 const std::string child_frame_id_link = constructObjectFrameLink(object_id);
96 FrameId previous_frame_id = -1;
97 bool first = true;
98 int path_segment = 0;
99
100 std_msgs::msg::ColorRGBA colour_msg;
101 convert(Color::uniqueId(object_id), colour_msg);
102
103 // paths for this object, broken into segments
104 gtsam::FastMap<int, ObjectOdometryPath> segmented_paths;
105
106 for (const auto& [frame_id, object_pose] : frame_pose_map) {
107 if (!first && frame_id != previous_frame_id + 1) {
108 path_segment++;
109 }
110 first = false;
111 previous_frame_id = frame_id;
112
113 // RIGHT NOW MOTION IDENTITY
114 // timestamp is wring
115 gtsam::Pose3 motion;
116 const ObjectOdometry object_odometry =
117 constructObjectOdometry(motion, object_pose, object_id, timestamp_k,
118 frame_id_link, child_frame_id_link);
119
120 if (!segmented_paths.exists(path_segment)) {
121 ObjectOdometryPath path;
122 path.colour = colour_msg;
123 path.object_id = object_id;
124 path.path_segment = path_segment;
125
126 path.header = multi_path.header;
127
128 segmented_paths.insert2(path_segment, path);
129 }
130
131 ObjectOdometryPath& path = segmented_paths.at(path_segment);
132 path.object_odometries.push_back(object_odometry);
133 }
134
135 for (const auto& [_, path] : segmented_paths) {
136 multi_path.paths.push_back(path);
137 }
138 }
139
140 return multi_path;
141}
