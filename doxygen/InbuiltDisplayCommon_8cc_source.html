68void InbuiltDisplayCommon::publishObjectPositions(
69 MarkerArrayPub::SharedPtr pub, const ObjectPoseMap& object_positions,
70 FrameId frame_id, Timestamp latest_timestamp,
71 const std::string& prefix_marker_namespace, bool draw_labels,
72 double scale) {
73 visualization_msgs::msg::MarkerArray object_pose_marker_array;
74 static visualization_msgs::msg::Marker delete_marker;
75 delete_marker.action = visualization_msgs::msg::Marker::DELETEALL;
76
77 auto ros_time = utils::toRosTime(latest_timestamp);
78
79 object_pose_marker_array.markers.push_back(delete_marker);
80
81 for (const auto& [object_id, poses_map] : object_positions) {
82 // do not draw if in current frame
83 if (!poses_map.exists(frame_id)) {
84 continue;
85 }
86
87 const gtsam::Pose3& pose = poses_map.at(frame_id);
88
89 // assume
90 // object centroid per frame
91 visualization_msgs::msg::Marker marker;
92 marker.header.frame_id = params_.world_frame_id;
93 marker.ns = prefix_marker_namespace + "_object_positions";
94 marker.id = object_id;
95 marker.type = visualization_msgs::msg::Marker::SPHERE;
96 marker.action = visualization_msgs::msg::Marker::ADD;
97 marker.header.stamp = ros_time;
98 marker.pose.position.x = pose.x();
99 marker.pose.position.y = pose.y();
100 marker.pose.position.z = pose.z();
101 marker.scale.x = scale;
102 marker.scale.y = scale;
103 marker.scale.z = scale;
104 marker.color.a = 1.0; // Don't forget to set the alpha!
105
106 const cv::Scalar colour = Color::uniqueId(object_id);
107 marker.color.r = colour(0) / 255.0;
108 marker.color.g = colour(1) / 255.0;
109 marker.color.b = colour(2) / 255.0;
110
111 object_pose_marker_array.markers.push_back(marker);
112
113 if (draw_labels) {
114 visualization_msgs::msg::Marker text_marker = marker;
115 text_marker.ns = prefix_marker_namespace + "_object_labels";
116 text_marker.type = visualization_msgs::msg::Marker::TEXT_VIEW_FACING;
117 text_marker.text = std::to_string(object_id);
118 text_marker.pose.position.z += 1.0; // make it higher than the pose
119 // marker
120 text_marker.scale.z = 0.7;
121 object_pose_marker_array.markers.push_back(text_marker);
122 }
123 }
124
125 pub->publish(object_pose_marker_array);
126}

128void InbuiltDisplayCommon::publishObjectPaths(
129 MarkerArrayPub::SharedPtr pub, const ObjectPoseMap& object_positions,
130 FrameId frame_id, Timestamp latest_timestamp,
131 const std::string& prefix_marker_namespace, const int min_poses) {
132 visualization_msgs::msg::MarkerArray object_path_marker_array;
133 static visualization_msgs::msg::Marker delete_marker;
134 delete_marker.action = visualization_msgs::msg::Marker::DELETEALL;
135
136 auto ros_time = utils::toRosTime(latest_timestamp);
137 object_path_marker_array.markers.push_back(delete_marker);
138
139 for (const auto& [object_id, poses_map] : object_positions) {
140 if (poses_map.size() < 2u) {
141 continue;
142 }
143
144 // draw a line list for viz
145 visualization_msgs::msg::Marker line_list_marker;
146 line_list_marker.type = visualization_msgs::msg::Marker::LINE_STRIP;
147 line_list_marker.header.frame_id = "world";
148 line_list_marker.ns = prefix_marker_namespace + "_object_path";
149 line_list_marker.id = object_id;
150 line_list_marker.header.stamp = ros_time;
151 line_list_marker.scale.x = 0.05;
152
153 line_list_marker.pose.orientation.x = 0;
154 line_list_marker.pose.orientation.y = 0;
155 line_list_marker.pose.orientation.z = 0;
156 line_list_marker.pose.orientation.w = 1;
157
158 const cv::Scalar colour = Color::uniqueId(object_id);
159 line_list_marker.color.r = colour(0) / 255.0;
160 line_list_marker.color.g = colour(1) / 255.0;
161 line_list_marker.color.b = colour(2) / 255.0;
162 line_list_marker.color.a = 1;
163
164 size_t traj_size;
165 // draw all the poses
166 if (min_poses == -1) {
167 traj_size = poses_map.size();
168 } else {
169 traj_size = std::min(min_poses, static_cast<int>(poses_map.size()));
170 }
171
172 // totally assume in order
173 auto map_iter = poses_map.end();
174 // equivalant to map_iter-traj_size + 1
175 // we want to go backwards to the starting point specified by trajectory
176 // size
177 std::advance(map_iter, (-traj_size + 1));
178 for (; map_iter != poses_map.end(); map_iter++) {
179 auto prev_iter = std::prev(map_iter, 1);
180 const gtsam::Pose3& prev_pose = prev_iter->second;
181 ;
182 const gtsam::Pose3& curr_pose = map_iter->second;
183
184 // check frames are consequative
185 // CHECK_EQ(prev_iter->first + 1, map_iter->first) << " For object " <<
186 // object_id;
187 LOG_IF(INFO, prev_iter->first + 1 != map_iter->first)
188 << " Frames not consequative for object " << object_id;
189
190 {
191 geometry_msgs::msg::Point p;
192 p.x = prev_pose.x();
193 p.y = prev_pose.y();
194 p.z = prev_pose.z();
195
196 line_list_marker.points.push_back(p);
197 }
198
199 {
200 geometry_msgs::msg::Point p;
201 p.x = curr_pose.x();
202 p.y = curr_pose.y();
203 p.z = curr_pose.z();
204
205 line_list_marker.points.push_back(p);
206 }
207 }
208
209 object_path_marker_array.markers.push_back(line_list_marker);
210 }
211
212 pub->publish(object_path_marker_array);
213}

235void InbuiltDisplayCommon::createAxisMarkers(
236 const gtsam::Pose3& pose, MarkerArray& axis_markers, Timestamp timestamp,
237 const cv::Scalar& colour, const std::string& frame, const std::string& ns,
238 double length, double radius) {
239 const auto pose_matrix = pose.matrix();
240 Eigen::Isometry3d x = Eigen::Translation3d(length / 2.0, 0, 0) *
241 Eigen::AngleAxisd(M_PI / 2.0, Eigen::Vector3d::UnitY());
242 gtsam::Pose3 x_pose(pose_matrix * x.matrix());
243
244 // Publish y axis
245 Eigen::Isometry3d y = Eigen::Translation3d(0, length / 2.0, 0) *
246 Eigen::AngleAxisd(M_PI / 2.0, Eigen::Vector3d::UnitX());
247 gtsam::Pose3 y_pose(pose_matrix * y.matrix());
248
249 Eigen::Isometry3d z = Eigen::Translation3d(0, 0, length / 2.0) *
250 Eigen::AngleAxisd(0, Eigen::Vector3d::UnitZ());
251 gtsam::Pose3 z_pose(pose_matrix * z.matrix());
252
253 const auto ros_time = utils::toRosTime(timestamp);
254
255 auto make_cylinder =
256 [&length, &radius, &ros_time, &ns, &frame,
257 &colour](const gtsam::Pose3& pose) -> visualization_msgs::msg::Marker {
258 visualization_msgs::msg::Marker marker;
259 marker.header.stamp = ros_time;
260 marker.ns = ns;
261 marker.header.frame_id = frame;
262
263 geometry_msgs::msg::Pose pose_msg;
264 dyno::convert(pose, pose_msg);
265
266 marker.pose = pose_msg;
267 marker.scale.x = radius;
268 marker.scale.y = radius;
269 marker.scale.z = length;
270
271 marker.color.r = colour(0) / 255.0;
272 marker.color.g = colour(1) / 255.0;
273 marker.color.b = colour(2) / 255.0;
274
275 return marker;
276 };
277
278 axis_markers.markers.push_back(make_cylinder(x_pose));
279 axis_markers.markers.push_back(make_cylinder(y_pose));
280 axis_markers.markers.push_back(make_cylinder(z_pose));
281}

283void InbuiltDisplayCommon::constructBoundingBoxeMarkers(
284 const CloudPerObject& cloud_per_object, MarkerArray& aabb_markers,
285 MarkerArray& obb_markers, Timestamp timestamp,
286 const std::string& prefix_marker_namespace) {
287 auto ros_time = utils::toRosTime(timestamp);
288
289 for (const auto& [object_id, obj_cloud] : cloud_per_object) {
290 pcl::PointXYZ centroid;
291 pcl::computeCentroid(obj_cloud, centroid);
292
293 const cv::Scalar colour = Color::uniqueId(object_id);
294
295 visualization_msgs::msg::Marker txt_marker;
296 txt_marker.header.frame_id = "world";
297 txt_marker.ns = prefix_marker_namespace + "_object_id_txt";
298 txt_marker.id = object_id;
299 txt_marker.type = visualization_msgs::msg::Marker::TEXT_VIEW_FACING;
300 txt_marker.action = visualization_msgs::msg::Marker::ADD;
301 txt_marker.header.stamp = ros_time;
302
303 txt_marker.scale.z = 0.5;
304 // txt_marker.scale.z = 2.0;
305 txt_marker.color.r = colour(0) / 255.0;
306 txt_marker.color.g = colour(1) / 255.0;
307 txt_marker.color.b = colour(2) / 255.0;
308 txt_marker.color.a = 1;
309 txt_marker.text = "Obj " + std::to_string(object_id);
310 txt_marker.pose.position.x = centroid.x;
311 // txt_marker.pose.position.y = centroid.y - 2.0;
312 // txt_marker.pose.position.z = centroid.z - 1.0;
313 txt_marker.pose.position.y = centroid.y - 0.6;
314 txt_marker.pose.position.z = centroid.z - 0.5;
315 aabb_markers.markers.push_back(txt_marker);
316
317 pcl::PointCloud<pcl::PointXYZRGB>::Ptr obj_cloud_ptr =
318 pcl::make_shared<pcl::PointCloud<pcl::PointXYZRGB> >(obj_cloud);
319 ObjectBBX aabb = findAABBFromCloud<pcl::PointXYZRGB>(obj_cloud_ptr);
320 ObjectBBX obb = findOBBFromCloud<pcl::PointXYZRGB>(obj_cloud_ptr);
321
322 visualization_msgs::msg::Marker aabb_marker;
323 aabb_marker.header.frame_id = "world";
324 aabb_marker.ns = prefix_marker_namespace + "object_aabb";
325 aabb_marker.id = object_id;
326 aabb_marker.type = visualization_msgs::msg::Marker::LINE_LIST;
327 aabb_marker.action = visualization_msgs::msg::Marker::ADD;
328 aabb_marker.header.stamp = ros_time;
329 aabb_marker.scale.x = 0.03;
330
331 aabb_marker.pose.orientation.x = 0;
332 aabb_marker.pose.orientation.y = 0;
333 aabb_marker.pose.orientation.z = 0;
334 aabb_marker.pose.orientation.w = 1;
335
336 aabb_marker.color.r = colour(0) / 255.0;
337 aabb_marker.color.g = colour(1) / 255.0;
338 aabb_marker.color.b = colour(2) / 255.0;
339 aabb_marker.color.a = 1;
340
341 for (const pcl::PointXYZ& this_line_list_point :
342 findLineListPointsFromAABBMinMax(aabb.min_bbx_point_,
343 aabb.max_bbx_point_)) {
344 geometry_msgs::msg::Point p;
345 p.x = this_line_list_point.x;
346 p.y = this_line_list_point.y;
347 p.z = this_line_list_point.z;
348 aabb_marker.points.push_back(p);
349 }
350 aabb_markers.markers.push_back(aabb_marker);
351
352 visualization_msgs::msg::Marker obbx_marker;
353 obbx_marker.header.frame_id = "world";
354 obbx_marker.ns = prefix_marker_namespace + "object_obb";
355 obbx_marker.id = object_id;
356 obbx_marker.type = visualization_msgs::msg::Marker::CUBE;
357 obbx_marker.action = visualization_msgs::msg::Marker::ADD;
358 obbx_marker.header.stamp = ros_time;
359 obbx_marker.scale.x = obb.max_bbx_point_.x() - obb.min_bbx_point_.x();
360 obbx_marker.scale.y = obb.max_bbx_point_.y() - obb.min_bbx_point_.y();
361 obbx_marker.scale.z = obb.max_bbx_point_.z() - obb.min_bbx_point_.z();
362
363 obbx_marker.pose.position.x = obb.bbx_position_.x();
364 obbx_marker.pose.position.y = obb.bbx_position_.y();
365 obbx_marker.pose.position.z = obb.bbx_position_.z();
366
367 const gtsam::Quaternion& q = obb.orientation_.toQuaternion();
368 obbx_marker.pose.orientation.x = q.x();
369 obbx_marker.pose.orientation.y = q.y();
370 obbx_marker.pose.orientation.z = q.z();
371 obbx_marker.pose.orientation.w = q.w();
372
373 obbx_marker.color.r = colour(0) / 255.0;
374 obbx_marker.color.g = colour(1) / 255.0;
375 obbx_marker.color.b = colour(2) / 255.0;
376 obbx_marker.color.a = 0.2;
377
378 obb_markers.markers.push_back(obbx_marker);
379 }

382MarkerArray InbuiltDisplayCommon::createCameraMarker(
383 const gtsam::Pose3& T_world_x, Timestamp timestamp, const std::string& ns,
384 const cv::Scalar& colour, double marker_scale) {
385 auto transform_odom_to =
386 [&T_world_x](
387 const geometry_msgs::msg::Pose& msg) -> geometry_msgs::msg::Pose {
388 // convert msg to pose
389 gtsam::Pose3 msg_as_pose;
390 dyno::convert(msg, msg_as_pose);
391 // transform pose into world frame
392 gtsam::Pose3 transformed_pose = T_world_x * msg_as_pose;
393 // transform pose back to ROS msg and return
394 geometry_msgs::msg::Pose transformed_msg;
395 dyno::convert(transformed_pose, transformed_msg);
396 return transformed_msg;
397 };
398
399 auto ros_time = utils::toRosTime(timestamp);
400
401 // make rectangles as frame
402 const double r_w = 1.0;
403 const double z_plane = (r_w / 2.0) * marker_scale;
404
405 static constexpr double sqrt2_2 = sqrt(2) / 2;
406
407 MarkerArray marker_array;
408 visualization_msgs::msg::Marker marker;
409
410 // the marker will be displayed in frame_id
411 marker.header.frame_id = params_.world_frame_id;
412 marker.header.stamp = ros_time;
413 marker.ns = ns;
414 marker.action = 0;
415 marker.id = 8; // 8 of 8 markers
416
417 marker.type = visualization_msgs::msg::Marker::CUBE;
418 marker.scale.x = r_w * marker_scale;
419 marker.scale.y = 0.04 * marker_scale;
420 marker.scale.z = 0.04 * marker_scale;
421 marker.color.r = colour(0) / 255.0;
422 marker.color.g = colour(1) / 255.0;
423 marker.color.b = colour(2) / 255.0;
424 marker.color.a = 1.0;
425
426 marker.pose.orientation.x = 0;
427 marker.pose.orientation.y = 0;
428 marker.pose.orientation.z = 0;
429 marker.pose.orientation.w = 1;
430 marker.id--;
431 marker_array.markers.push_back(marker);
432 marker.pose.position.y = -(r_w / 4.0) * marker_scale;
433 marker.id--;
434 marker_array.markers.push_back(marker);
435
436 marker.scale.x = (r_w / 2.0) * marker_scale;
437 marker.pose.position.x = (r_w / 2.0) * marker_scale;
438 marker.pose.position.y = 0;
439 marker.pose.orientation.w = sqrt2_2;
440 marker.pose.orientation.z = sqrt2_2;
441 marker.id--;
442 marker_array.markers.push_back(marker);
443 marker.pose.position.x = -(r_w / 2.0) * marker_scale;
444 marker.id--;
445 marker_array.markers.push_back(marker);
446
447 // make pyramid edges
448 marker.scale.x = (3.0 * r_w / 4.0) * marker_scale;
449 marker.pose.position.z = 0.5 * z_plane;
450
451 marker.pose.position.x = (r_w / 4.0) * marker_scale;
452 marker.pose.position.y = (r_w / 8.0) * marker_scale;
453 // 0.08198092, -0.34727674, 0.21462883, 0.9091823
454 marker.pose.orientation.x = 0.08198092;
455 marker.pose.orientation.y = -0.34727674;
456 marker.pose.orientation.z = 0.21462883;
457 marker.pose.orientation.w = 0.9091823;
458 marker.id--;
459 marker_array.markers.push_back(marker);
460
461 marker.pose.position.x = -(r_w / 4.0) * marker_scale;
462 marker.pose.position.y = (r_w / 8.0) * marker_scale;
463 // -0.27395078, -0.22863284, 0.9091823 , 0.21462883
464 marker.pose.orientation.x = 0.08198092;
465 marker.pose.orientation.y = 0.34727674;
466 marker.pose.orientation.z = -0.21462883;
467 marker.pose.orientation.w = 0.9091823;
468 marker.id--;
469 marker_array.markers.push_back(marker);
470
471 marker.pose.position.x = -(r_w / 4.0) * marker_scale;
472 marker.pose.position.y = -(r_w / 8.0) * marker_scale;
473 // -0.08198092, 0.34727674, 0.21462883, 0.9091823
474 marker.pose.orientation.x = -0.08198092;
475 marker.pose.orientation.y = 0.34727674;
476 marker.pose.orientation.z = 0.21462883;
477 marker.pose.orientation.w = 0.9091823;
478 marker.id--;
479 marker_array.markers.push_back(marker);
480
481 marker.pose.position.x = (r_w / 4.0) * marker_scale;
482 marker.pose.position.y = -(r_w / 8.0) * marker_scale;
483 // -0.08198092, -0.34727674, -0.21462883, 0.9091823
484 marker.pose.orientation.x = -0.08198092;
485 marker.pose.orientation.y = -0.34727674;
486 marker.pose.orientation.z = -0.21462883;
487 marker.pose.orientation.w = 0.9091823;
488 marker.id--;
489 marker_array.markers.push_back(marker);
490
491 for (auto& marker : marker_array.markers) {
492 // put in world frame
493 marker.pose = transform_odom_to(marker.pose);
494 }
495 return marker_array;
496}
