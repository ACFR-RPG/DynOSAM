102template <typename POINTER, typename GTSAM_VALUE>
103bool equateGtsamPointerLikeValues(const POINTER& a, const POINTER& b,
104 double tol = 1e-8) {
105 if (a && b) {
106 return gtsam::traits<GTSAM_VALUE>::Equals(*a, *b, tol);
107 }
108
109 if (!a && !b) {
110 return true;
111 }
112
113 return false;
114}

201template <typename T>
202T perturbWithNoise(const T& t, const gtsam::Vector& sigmas, int32_t seed = 42) {
203 CHECK_EQ(gtsam::traits<T>::dimension, sigmas.size());
206 // TODO: (jesse)yeah, but we pass different sigmas and seeds to it, so wont
207 // this mean it just doesnt get updated?
208 static gtsam::Sampler sample(sigmas, seed);
209
210 gtsam::Vector delta(sample.sample()); // delta should be the tangent vector
211 return gtsam::traits<T>::Retract(t, delta);
212}

214template <typename T>
215T perturbWithNoise(const T& t, double sigma, int32_t seed = 42) {
216 gtsam::Vector sigmas =
217 gtsam::Vector::Constant(gtsam::traits<T>::dimension, sigma);
218 return perturbWithNoise<T>(t, sigmas, seed);
219}
