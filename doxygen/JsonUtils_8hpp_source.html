151template <typename Scalar, int Rows, int Cols>
152struct adl_serializer<Eigen::Matrix<Scalar, Rows, Cols>> {
153 // static_assert(Rows != Eigen::Dynamic, "Not implemented for dynamic rows");
154 // static_assert(Cols != Eigen::Dynamic, "Not implemented for dynamic cols");
155
156 static void to_json(json& j,
157 const Eigen::Matrix<Scalar, Rows, Cols>& matrix) {
158 for (int row = 0; row < matrix.rows(); ++row) {
159 json column = json::array();
160 for (int col = 0; col < matrix.cols(); ++col) {
161 column.push_back(matrix(row, col));
162 }
163 j.push_back(column);
164 }
165 }
166
167 static void from_json(const json& j,
168 Eigen::Matrix<Scalar, Rows, Cols>& matrix) {
169 for (std::size_t row = 0; row < j.size(); ++row) {
170 const auto& jrow = j.at(row);
171 for (std::size_t col = 0; col < jrow.size(); ++col) {
172 const auto& value = jrow.at(col);
173 value.get_to(matrix(row, col));
174 }
175 }
176 }
177};

182 static void to_json(json& j, const gtsam::Pose3& pose) {
183 j["tx"] = pose.x();
184 j["ty"] = pose.y();
185 j["tz"] = pose.z();
186
187 const auto& rotation = pose.rotation();
188 const auto& quaternion = rotation.toQuaternion();
189
190 j["qx"] = quaternion.x();
191 j["qy"] = quaternion.y();
192 j["qz"] = quaternion.z();
193 j["qw"] = quaternion.w();
194 }
195
196 static void from_json(const json& j, gtsam::Pose3& pose) {
197 gtsam::Point3 translation(j["tx"].template get<double>(),
198 j["ty"].template get<double>(),
199 j["tz"].template get<double>());
200
201 // Note: w, x, y, z order
202 gtsam::Rot3 rotation(
203 j["qw"].template get<double>(), j["qx"].template get<double>(),
204 j["qy"].template get<double>(), j["qz"].template get<double>());
205
206 pose = gtsam::Pose3(rotation, translation);
207 }
208};

236template <typename KEY, typename VALUE>
237struct StringMapConverter : public StringMapConverterBase<KEY, VALUE> {
238 using Base = StringMapConverterBase<KEY, VALUE>;
239 using Stdmap_with_allocator_KEY = typename Base::Stdmap_with_allocator_KEY;
240 using Stdmap_with_allocator_STRING =
241 typename Base::Stdmap_with_allocator_STRING;
242
243 static Stdmap_with_allocator_STRING to_string_map(
244 const Stdmap_with_allocator_KEY& typed_map) {
245 Stdmap_with_allocator_STRING string_map;
246 for (const auto& [key, value] : typed_map) {
247 string_map[std::to_string(key)] = value;
248 }
249 return string_map;
250 }
251
252 static Stdmap_with_allocator_KEY from_string_map(
253 const Stdmap_with_allocator_STRING& string_map) {
254 Stdmap_with_allocator_KEY typed_map;
255 for (const auto& [string_key, value] : string_map) {
256 KEY key;
257
258 if constexpr (std::is_integral_v<KEY>) {
259 key = static_cast<KEY>(std::stoll(string_key));
260 } else if constexpr (std::is_floating_point_v<KEY>) {
261 key = static_cast<KEY>(std::stod(string_key));
262 } else {
263 throw std::invalid_argument(
264 "Unsupported key type for deserialization. Type is " +
265 dyno::type_name<KEY>());
266 }
267
268 typed_map[key] = value;
269 }
270 return typed_map;
271 }
272};

431template <typename T>
432struct adl_serializer<dyno::MeasurementWithCovariance<T>> {
433 static void to_json(json& j,
434 const dyno::MeasurementWithCovariance<T>& value) {
435 j["measurement"] = value.measurement();
436 if (value.hasModel()) {
437 j["covariance"] = value.covariance();
438 }
439 }
440 static void from_json(const json& j,
441 dyno::MeasurementWithCovariance<T>& value) {
442 T measurement = j["measurement"].template get<T>();
443
444 // check if the json has a covariance value, if it does not it means the
445 // original object did not have a sensor model. We should then use the
446 // measurement only constructor to ennsure the new model does not have a
447 // model eitehr and hasModel() returns false!
448 if (j.contains("covariance")) {
449 typename dyno::MeasurementWithCovariance<T>::Covariance cov =
450 j["covariance"]
451 .template get<
452 typename dyno::MeasurementWithCovariance<T>::Covariance>();
453 value = dyno::MeasurementWithCovariance<T>(measurement, cov);
454 return;
455 } else {
456 // measurement only constructor
457 value = dyno::MeasurementWithCovariance<T>(measurement);
458 }
459 }
460};
