93template <typename T>
94struct adl_serializer<std::optional<T>> {
95 static void to_json(json& j, const std::optional<T>& opt) {
96 if (opt == std::nullopt) {
97 j = nullptr;
98 } else {
99 j = *opt; // this will call adl_serializer<T>::to_json which will
100 // find the free function to_json in T's namespace!
101 }
102 }
103
104 static void from_json(const json& j, std::optional<T>& opt) {
105 if (j.is_null()) {
106 opt = std::nullopt;
107 } else {
108 opt = j.template get<T>(); // same as above, but with
109 // adl_serializer<T>::from_json
110 }
111 }
112};

150template <typename Scalar, int Rows, int Cols>
151struct adl_serializer<Eigen::Matrix<Scalar, Rows, Cols>> {
152 // static_assert(Rows != Eigen::Dynamic, "Not implemented for dynamic rows");
153 // static_assert(Cols != Eigen::Dynamic, "Not implemented for dynamic cols");
154
155 static void to_json(json& j,
156 const Eigen::Matrix<Scalar, Rows, Cols>& matrix) {
157 for (int row = 0; row < matrix.rows(); ++row) {
158 json column = json::array();
159 for (int col = 0; col < matrix.cols(); ++col) {
160 column.push_back(matrix(row, col));
161 }
162 j.push_back(column);
163 }
164 }
165
166 static void from_json(const json& j,
167 Eigen::Matrix<Scalar, Rows, Cols>& matrix) {
168 for (std::size_t row = 0; row < j.size(); ++row) {
169 const auto& jrow = j.at(row);
170 for (std::size_t col = 0; col < jrow.size(); ++col) {
171 const auto& value = jrow.at(col);
172 value.get_to(matrix(row, col));
173 }
174 }
175 }
176};

179template <>
180struct adl_serializer<gtsam::Pose3> {
181 static void to_json(json& j, const gtsam::Pose3& pose) {
182 j["tx"] = pose.x();
183 j["ty"] = pose.y();
184 j["tz"] = pose.z();
185
186 const auto& rotation = pose.rotation();
187 const auto& quaternion = rotation.toQuaternion();
188
189 j["qx"] = quaternion.x();
190 j["qy"] = quaternion.y();
191 j["qz"] = quaternion.z();
192 j["qw"] = quaternion.w();
193 }
194
195 static void from_json(const json& j, gtsam::Pose3& pose) {
196 gtsam::Point3 translation(j["tx"].template get<double>(),
197 j["ty"].template get<double>(),
198 j["tz"].template get<double>());
199
200 // Note: w, x, y, z order
201 gtsam::Rot3 rotation(
202 j["qw"].template get<double>(), j["qx"].template get<double>(),
203 j["qy"].template get<double>(), j["qz"].template get<double>());
204
205 pose = gtsam::Pose3(rotation, translation);
206 }
207};

289template <typename T>
290struct adl_serializer<dyno::MeasurementWithCovariance<T>> {
291 static void to_json(json& j,
292 const dyno::MeasurementWithCovariance<T>& value) {
293 j["measurement"] = value.measurement();
294 if (value.hasModel()) {
295 j["covariance"] = value.covariance();
296 }
297 }
298 static void from_json(const json& j,
299 dyno::MeasurementWithCovariance<T>& value) {
300 T measurement = j["measurement"].template get<T>();
301
302 // check if the json has a covariance value, if it does not it means the
303 // original object did not have a sensor model. We should then use the
304 // measurement only constructor to ennsure the new model does not have a
305 // model eitehr and hasModel() returns false!
306 if (j.contains("covariance")) {
307 typename dyno::MeasurementWithCovariance<T>::Covariance cov =
308 j["covariance"]
309 .template get<
310 typename dyno::MeasurementWithCovariance<T>::Covariance>();
311 value = dyno::MeasurementWithCovariance<T>(measurement, cov);
312 return;
313 } else {
314 // measurement only constructor
315 value = dyno::MeasurementWithCovariance<T>(measurement);
316 }
317 }
318};

321struct adl_serializer<dyno::TrackedValueStatus<T>> {
322 static void to_json(json& j, const dyno::TrackedValueStatus<T>& status) {
323 using nlohmann::to_json;
324 // expect value to be seralizable
325 j["value"] = (json)status.value();
326 j["frame_id"] = status.frameId();
327 j["tracklet_id"] = status.trackletId();
328 j["object_id"] = status.objectId();
329 j["reference_frame"] = status.referenceFrame();
330 }

332 typename dyno::TrackedValueStatus<T>::Value value =
333 j["value"].template get<typename dyno::TrackedValueStatus<T>::Value>();
334
335 using namespace dyno;
336 FrameId frame_id = j["frame_id"].template get<FrameId>();
337 TrackletId tracklet_id = j["tracklet_id"].template get<TrackletId>();
338 ObjectId object_id = j["object_id"].template get<ObjectId>();
339 ReferenceFrame rf = j["reference_frame"].template get<ReferenceFrame>();
340 status = dyno::TrackedValueStatus<T>(value, frame_id, tracklet_id,
341 object_id, rf);
