45template <typename T, typename Enable = void>
46struct color_traits;

48template <typename T>
49struct _floating_point_colour_traits {

67template <typename T>
68struct RGBA {
69 // Data.
70 T r = 0;
71 T g = 0;
72 T b = 0;
73 T a = max();
74
75 RGBA() = default;
76 RGBA(T r, T g, T b, T a = max()) : r(r), g(g), b(b), a(a) {}
77
78 template <typename U>
79 RGBA(const RGBA<U>& other) {
80 // check if we need to go up (1 -> 255) or down (255 -> 1)
81 constexpr auto other_max = RGBA<U>::max();
82
83 // this is lower than other so we need to normalize using other max
84 if constexpr (this->max() < other_max) {
85 this->r = is_zero(other.r) ? 0 : static_cast<T>(other.r / other_max);
86 this->g = is_zero(other.g) ? 0 : static_cast<T>(other.g / other_max);
87 this->b = is_zero(other.b) ? 0 : static_cast<T>(other.b / other_max);
88 this->a = is_zero(other.a) ? 0 : static_cast<T>(other.a / other_max);
89 CHECK_LE(this->a, max());
90 }
91 // this is higher than other so we need to normalize using this max
92 else {
93 constexpr U cast_max = static_cast<U>(max());
94 this->r = is_zero(other.r) ? 0 : static_cast<T>(other.r * cast_max);
95 this->g = is_zero(other.g) ? 0 : static_cast<T>(other.g * cast_max);
96 this->b = is_zero(other.b) ? 0 : static_cast<T>(other.b * cast_max);
97 this->a = is_zero(other.a) ? 0 : static_cast<T>(other.a * cast_max);
98 CHECK_LE(this->a, max());
99 }
100 }
101
104 static constexpr inline T max() { return color_traits<T>::max(); }
105};
