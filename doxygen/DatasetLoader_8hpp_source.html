342 using TypedDataFolderTuple = std::tuple<typename DataFolder<DataTypes>::Ptr...>;
343
344 DYNO_POINTER_TYPEDEFS(This)

346
347 template <size_t I>
348 using DataType = std::tuple_element_t<I, DataTypeTuple>;
349
350 template <size_t I>
351 using DataFolderType = std::tuple_element_t<I, TypedDataFolderTuple>;
352
353 DataFolderStructure(const fs::path& dataset_path, typename DataFolder<DataTypes>::Ptr... data_folders) : dataset_path_(dataset_path), data_folders_(data_folders...) {}
354
355 //will be of type DataFolder<T>::Ptr where T is the type returned by the data folder as index at DataTypeTuple<I>
356 template<size_t I>
357 auto getDataFolder() const {
358 return std::get<I>(data_folders_);
359 }
360
361 template<size_t I>
362 fs::path getAbsoluteFolderPath() const {
363 const auto data_folder = this->template getDataFolder<I>();
364 fs::path folder_path = dataset_path_;
365 //technically this can also be a file path
366 folder_path /= fs::path(data_folder->getFolderName());
367 return fs::absolute(folder_path);
368 }
369
377 const fs::path& getDatasetPath() const { return dataset_path_; }
378
379
380protected:
381 const fs::path dataset_path_;
382 TypedDataFolderTuple data_folders_;

394template<typename... DataTypes>
395class GenericDataset : public DataFolderStructure<DataTypes...> {
396
397public:
398 using This = GenericDataset<DataTypes...>;
399 using Base = DataFolderStructure<DataTypes...>;
400
401
405 using DataStorageTuple = std::tuple<std::vector<DataTypes>...>;
406
407 template<size_t I>
408 using DataStorageVector = std::vector<typename Base::DataType<I>>;
409
410 DYNO_POINTER_TYPEDEFS(This)
411
412
413 GenericDataset(const fs::path& dataset_path, typename DataFolder<DataTypes>::Ptr... data_folders)
414 : Base(dataset_path, data_folders...)
415 {
416 //this will also set the absolute folder/file path on each DataFolder
417 validateFoldersExist();
418 LOG(INFO) << "Done folder validate";
419 }
420
421 template<size_t I>
422 DataStorageVector<I>& getDataVector() {
423 return std::get<I>(data_);
424 }
425
426 virtual ~GenericDataset() {}
427
436 bool load(size_t idx) {
437 if(idx < 1u) {
438 throw std::runtime_error("Failure when loading data in GenericDataset - loading index's start at 1!");
439 }
440 for (size_t i = 0; i < Base::N; i++) {
441 internal::select_apply<Base::N>(i, [&](auto stream_index) {
442 using Type = typename Base::template DataType<stream_index>;
443 auto data_folder = this->template getDataFolder<stream_index>();
444 //loade at idx-1 since we expect all the indexing and loading to be appropiately zero idnexed
445 Type loaded_data = data_folder->getItem(idx-1);
446
447 auto& data_vector = this->template getDataVector<stream_index>();
448
449 // add loaded data to data vector. resize if necessary but we assume in most cases data will be loaded in order
450 if(idx > data_vector.size()) {
451 data_vector.resize(idx);
452 }
453
454 //index the data storage vector at I-1 to account for the zero indexing
455 data_vector.at(idx-1) = loaded_data;
456 });
457 }
458
459 return true;//?
460 }
461
462
463
464private:
472 void validateFoldersExist() {
473 for (size_t i = 0; i < Base::N; i++) {
474 internal::select_apply<Base::N>(i, [&](auto stream_index) {
475 const fs::path absolute_file_path = this->template getAbsoluteFolderPath<stream_index>();
476
477 if(!fs::exists(absolute_file_path)) {
478 throw std::runtime_error("Error loading datafolder - absolute file path " + std::string(absolute_file_path) + " does not exist");
479 }
480
481 LOG(INFO) << "Validated dataset folder: " << absolute_file_path;
482 auto data_folder = this->template getDataFolder<stream_index>();
483 data_folder->initAbsolutePath(absolute_file_path);
484
485 });
486 }
487 }
