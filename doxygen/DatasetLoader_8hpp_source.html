342template <typename... DataTypes>
343class DataFolderStructure {
344 public:
345 using This = DataFolderStructure<DataTypes...>;
346 using DataTypeTuple = std::tuple<DataTypes...>;
347 using TypedDataFolderTuple =

350 DYNO_POINTER_TYPEDEFS(This)
351 static constexpr size_t N = sizeof...(DataTypes);
352
353 template <size_t I>
354 using DataType = std::tuple_element_t<I, DataTypeTuple>;
355
356 template <size_t I>
357 using DataFolderType = std::tuple_element_t<I, TypedDataFolderTuple>;
358
359 DataFolderStructure(const fs::path& dataset_path,
360 typename DataFolder<DataTypes>::Ptr... data_folders)
361 : dataset_path_(dataset_path), data_folders_(data_folders...) {}
362
363 // will be of type DataFolder<T>::Ptr where T is the type returned by the data
364 // folder as index at DataTypeTuple<I>
365 template <size_t I>
366 auto getDataFolder() const {
367 return std::get<I>(data_folders_);
368 }
369
370 template <size_t I>
371 fs::path getAbsoluteFolderPath() const {
372 const auto data_folder = this->template getDataFolder<I>();
373 fs::path folder_path = dataset_path_;
374 // technically this can also be a file path
375 folder_path /= fs::path(data_folder->getFolderName());
376 return fs::absolute(folder_path);
377 }
378
386 const fs::path& getDatasetPath() const { return dataset_path_; }
387
388 protected:
389 const fs::path
390 dataset_path_;
391 TypedDataFolderTuple data_folders_;

402template <typename... DataTypes>
403class GenericDataset : public DataFolderStructure<DataTypes...> {
404 public:
405 using This = GenericDataset<DataTypes...>;
406 using Base = DataFolderStructure<DataTypes...>;
407
411 using DataStorageTuple = std::tuple<std::vector<DataTypes>...>;
412
413 template <size_t I>
414 using DataStorageVector = std::vector<typename Base::DataType<I>>;
415
416 DYNO_POINTER_TYPEDEFS(This)
417
418 GenericDataset(const fs::path& dataset_path,
419 typename DataFolder<DataTypes>::Ptr... data_folders)
420 : Base(dataset_path, data_folders...) {
421 // this will also set the absolute folder/file path on each DataFolder
422 validateFoldersExist();
423 LOG(INFO) << "Done folder validate";
424 }
425
426 template <size_t I>
427 DataStorageVector<I>& getDataVector() {
428 return std::get<I>(data_);
429 }
430
431 virtual ~GenericDataset() {}
432
442 bool load(size_t idx) {
443 if (idx < 1u) {
444 throw std::runtime_error(
445 "Failure when loading data in GenericDataset - loading index's start "
446 "at 1!");
447 }
448 for (size_t i = 0; i < Base::N; i++) {
449 internal::select_apply<Base::N>(i, [&](auto stream_index) {
450 using Type = typename Base::template DataType<stream_index>;
451 auto data_folder = this->template getDataFolder<stream_index>();
452 // loade at idx-1 since we expect all the indexing and loading to be
453 // appropiately zero idnexed
454 Type loaded_data = data_folder->getItem(idx - 1);
455
456 auto& data_vector = this->template getDataVector<stream_index>();
457
458 // add loaded data to data vector. resize if necessary but we assume in
459 // most cases data will be loaded in order
460 if (idx > data_vector.size()) {
461 data_vector.resize(idx);
462 }
463
464 // index the data storage vector at I-1 to account for the zero indexing
465 data_vector.at(idx - 1) = loaded_data;
466 });
467 }
468
469 return true; //?
470 }
471
472 private:
481 void validateFoldersExist() {
482 for (size_t i = 0; i < Base::N; i++) {
483 internal::select_apply<Base::N>(i, [&](auto stream_index) {
484 const fs::path absolute_file_path =
485 this->template getAbsoluteFolderPath<stream_index>();
486
487 if (!fs::exists(absolute_file_path)) {
488 throw std::runtime_error(
489 "Error loading datafolder - absolute file path " +
490 std::string(absolute_file_path) + " does not exist");
491 }
492
493 LOG(INFO) << "Validated dataset folder: " << absolute_file_path;
494 auto data_folder = this->template getDataFolder<stream_index>();
495 data_folder->initAbsolutePath(absolute_file_path);
496 });
497 }
498 }
499
500 private:
501 DataStorageTuple data_;
502};
