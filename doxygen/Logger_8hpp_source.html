165class JsonConverter {
166 public:
167 using json = nlohmann::json;
168
169 enum Format { NORMAL, BSON };
170
171 // T must be json seralizable
172 template <typename T>
173 static void WriteOutJson(const T& value, const std::string& filepath,
174 const Format& fmt = Format::BSON) {
175 if (fmt == Format::BSON) {
176 WriteBson<T>(value, filepath);
177 } else {
178 CHECK(false) << "normal json not implemented";
179 }
180 }
181
182 template <typename T>
183 static bool ReadInJson(T& value, const std::string& filepath,
184 const Format& fmt = Format::BSON) {
185 if (fmt == Format::BSON) {
186 return ReadBson(value, filepath);
187 } else {
188 CHECK(false) << "normal json not implemented";
189 }
190 }
191
192 private:
193 template <typename T>
194 static void WriteBson(const T& value, const std::string& filepath) {
195 VLOG(5) << "Writing bson data to filepath: " << filepath;
196
197 json j;
198 j["data"] = value;
199 auto v_bson = json::to_bson(j);
200
201 std::ofstream bsonfile(filepath.c_str(),
202 std::ios_base::out | std::ios::binary);
203 bsonfile.write((char*)&v_bson[0], v_bson.size() * sizeof(v_bson[0]));
204 bsonfile.close();
205 }
206
207 template <typename T>
208 static bool ReadBson(T& value, const std::string& filepath) {
209 VLOG(5) << "Reading bson data from filepath: " << filepath;
210 std::ifstream file(filepath, std::ios::binary);
211
212 if (!file.is_open()) {
213 return false;
214 }
215
216 // Get the size of the file
217 file.seekg(0, std::ios::end);
218 const std::streamsize file_size = file.tellg();
219 file.seekg(0, std::ios::beg);
220
221 // Create a vector to store the file content
222 std::vector<std::uint8_t> content;
223 content.reserve(file_size); // Reserve space to avoid reallocation
224
225 // Read the file content into the vector
226 content.insert(content.begin(), std::istreambuf_iterator<char>(file),
227 std::istreambuf_iterator<char>());
228 const json json_instance_read = json::from_bson(content);
229
230 value = json_instance_read["data"].template get<T>();
231 return true;
232 }
