165class JsonConverter {
166 public:
167 using json = nlohmann::json;
168
169 enum Format { NORMAL, BSON };
170
171 // T must be json seralizable
172 template <typename T>
173 static void WriteOutJson(const T& value, const std::string& filepath,
174 const Format& fmt = Format::BSON,
175 bool append = false) {
176 if (fmt == Format::BSON) {
177 WriteBson<T>(value, filepath, append);
178 } else {
179 CHECK(false) << "normal json not implemented";
180 }
181 }
182
183 template <typename T>
184 static bool ReadInJson(T& value, const std::string& filepath,
185 const Format& fmt = Format::BSON) {
186 if (fmt == Format::BSON) {
187 return ReadBson(value, filepath);
188 } else {
189 CHECK(false) << "normal json not implemented";
190 }
191 }
192
193 private:
194 template <typename T>
195 static void WriteBson(const T& value, const std::string& filepath,
196 bool append) {
197 VLOG(5) << "Writing bson data to filepath: " << filepath;
198
199 json j;
200 j["data"] = value;
201 auto v_bson = json::to_bson(j);
202
203 std::ofstream bsonfile;
204 if (!append) {
205 bsonfile.open(filepath.c_str(), std::ios_base::out | std::ios::binary);
206 } else {
207 bsonfile.open(filepath.c_str(),
208 std::ios_base::app | std::ios_base::out | std::ios::binary);
209 }
210
211 bsonfile.write((char*)&v_bson[0], v_bson.size() * sizeof(v_bson[0]));
212 bsonfile.close();
213 }
214
215 template <typename T>
216 static bool ReadBson(T& value, const std::string& filepath) {
217 VLOG(5) << "Reading bson data from filepath: " << filepath;
218 std::ifstream file(filepath, std::ios::binary);
219
220 if (!file.is_open()) {
221 return false;
222 }
223
224 // Get the size of the file
225 file.seekg(0, std::ios::end);
226 const std::streamsize file_size = file.tellg();
227 file.seekg(0, std::ios::beg);
228
229 // Create a vector to store the file content
230 std::vector<std::uint8_t> content;
231 content.reserve(file_size); // Reserve space to avoid reallocation
232
233 // Read the file content into the vector
234 content.insert(content.begin(), std::istreambuf_iterator<char>(file),
235 std::istreambuf_iterator<char>());
236 const json json_instance_read = json::from_bson(content);
237
238 value = json_instance_read["data"].template get<T>();
239 return true;
240 }
