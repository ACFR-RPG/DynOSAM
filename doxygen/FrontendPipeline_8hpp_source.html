60 explicit FrontendOfflinePipeline(const std::string& module_name, const std::string& bsjon_file_path, int ending_frame = -1)
61 : Base(module_name) {
62
63 if(!JsonConverter::ReadInJson(frontend_output_packets_, bsjon_file_path, JsonConverter::BSON)) {
64 LOG(FATAL) << "Failed to load frontend bson from: " << bsjon_file_path;
65 }
66 current_packet_ = frontend_output_packets_.cbegin();
67 ending_frame_ = getFrontendOutputPackets().rbegin()->first;
68 if(ending_frame != -1) {
69 ending_frame_ = (FrameId)std::min(ending_frame, (int)ending_frame_);
70 }
71 LOG(INFO) << "Setting frontend offline ending frame = " << ending_frame_;
72 }
73
74 InputConstSharedPtr getInputPacket() override {
75 static auto output = std::make_shared<NullPipelinePayload>();
76 return output;
77 }
78
79 inline OutputConstSharedPtr process(const InputConstSharedPtr&) override {
80 if(!hasWork()) {
81 this->shutdown();
82 return nullptr;
83 }
84 auto output = current_packet_->second;
85 current_packet_++;
86 return output;
87 }
88
89 inline const FrontendPacketMap& getFrontendOutputPackets() const { return frontend_output_packets_; }
90 inline FrameId endingFrame() const { return (FrameId)ending_frame_; }
91
92protected:
93 void shutdownQueues() override { current_packet_ = frontend_output_packets_.end(); }
94 bool hasWork() const override { return current_packet_ != frontend_output_packets_.cend() && current_packet_->first <= (FrameId)ending_frame_; }
