420template <typename INPUT, typename OUTPUT,
421 typename = std::enable_if_t<
422 rosidl_generator_traits::is_message<OUTPUT>::value>>
423bool convertWithHeader(
424 const INPUT &input, OUTPUT &output, Timestamp timestamp,
425 const std::string &frame_id,
426 std::optional<std::string> child_frame_id = std::nullopt) {
427 if (!dyno::convert<INPUT, OUTPUT>(input, output)) return false;
428
429 // does actually have a header
430 if constexpr (internal::HasMsgHeader<OUTPUT>::value) {
431 output.header.stamp = toRosTime(timestamp);
432 output.header.frame_id = frame_id;
433 }
434
435 // if has child frame id
436 if constexpr (internal::HasChildFrame<OUTPUT>::value) {
437 if (child_frame_id) {
438 output.child_frame_id = child_frame_id.value();
439 }
440 }
441
442 return true;
443}
