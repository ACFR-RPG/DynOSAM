37OnlineDataProviderRos::OnlineDataProviderRos(

51void OnlineDataProviderRos::shutdown() {
52 shutdown_ = true;
53 // shutdown synchronizer
54 RCLCPP_INFO_STREAM(node_->get_logger(),
55 "Shutting down OnlineDataProviderRos");
56 if (sync_) sync_.reset();
57
58 rgb_image_sub_.unsubscribe();
59 depth_image_sub_.unsubscribe();
60 flow_image_sub_.unsubscribe();
61 mask_image_sub_.unsubscribe();
62}

64void OnlineDataProviderRos::connect() {
65 rclcpp::Node *node_ptr = node_.get();
66 CHECK_NOTNULL(node_ptr);
67 rgb_image_sub_.subscribe(node_ptr, "image/rgb");
68 depth_image_sub_.subscribe(node_ptr, "image/depth");
69 flow_image_sub_.subscribe(node_ptr, "image/flow");
70 mask_image_sub_.subscribe(node_ptr, "image/mask");
71
72 if (sync_) sync_.reset();
73
74 static constexpr size_t kQueueSize = 20u;
75 sync_ = std::make_shared<message_filters::Synchronizer<SyncPolicy>>(
76 SyncPolicy(kQueueSize), rgb_image_sub_, depth_image_sub_, flow_image_sub_,
77 mask_image_sub_);
78
79 sync_->registerCallback(std::bind(
80 &OnlineDataProviderRos::imageSyncCallback, this, std::placeholders::_1,
81 std::placeholders::_2, std::placeholders::_3, std::placeholders::_4));
82
83 RCLCPP_INFO_STREAM(
84 node_->get_logger(),
85 "OnlineDataProviderRos has been connected. Subscribed to image topics: "
86 << rgb_image_sub_.getSubscriber()->get_topic_name() << " "
87 << depth_image_sub_.getSubscriber()->get_topic_name() << " "
88 << flow_image_sub_.getSubscriber()->get_topic_name() << " "
89 << mask_image_sub_.getSubscriber()->get_topic_name() << ".");
90
91 shutdown_ = false;
92}

94void OnlineDataProviderRos::imageSyncCallback(
95 const sensor_msgs::msg::Image::ConstSharedPtr &rgb_msg,
96 const sensor_msgs::msg::Image::ConstSharedPtr &depth_msg,
97 const sensor_msgs::msg::Image::ConstSharedPtr &flow_msg,
98 const sensor_msgs::msg::Image::ConstSharedPtr &mask_msg) {
99 if (!image_container_callback_) {
100 RCLCPP_ERROR_THROTTLE(node_->get_logger(), *node_->get_clock(), 1000,
101 "Image Sync callback triggered but "
102 "image_container_callback_ is not registered!");
103 return;
104 }
105
106 const cv::Mat rgb = readRgbRosImage(rgb_msg);
107 const cv::Mat depth = readDepthRosImage(depth_msg);
108 const cv::Mat flow = readFlowRosImage(flow_msg);
109 const cv::Mat mask = readMaskRosImage(mask_msg);
110
111 const Timestamp timestamp = utils::fromRosTime(rgb_msg->header.stamp);
112 const FrameId frame_id = frame_id_;
113 frame_id_++;
114
115 ImageContainer::Ptr image_container = ImageContainer::Create(
116 timestamp, frame_id, ImageWrapper<ImageType::RGBMono>(rgb),
117 ImageWrapper<ImageType::Depth>(depth),
118 ImageWrapper<ImageType::OpticalFlow>(flow),
119 ImageWrapper<ImageType::MotionMask>(mask));
120 CHECK(image_container);
121
122 cv::Mat of_viz, motion_viz, depth_viz;
123 of_viz = ImageType::OpticalFlow::toRGB(flow);
124 motion_viz = ImageType::MotionMask::toRGB(mask);
125 depth_viz = ImageType::Depth::toRGB(depth);
126
127 // cv::imshow("Optical Flow", of_viz);
128 // cv::imshow("Motion mask", motion_viz);
129 // cv::imshow("Depth", depth_viz);
130 // cv::waitKey(1);
131 // trigger callback to send data to the DataInterface!
132 image_container_callback_(image_container);
133}
