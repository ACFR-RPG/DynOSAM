65void OnlineDataProviderRos::connectImages() {
66 rclcpp::Node &node_ref = *node_;
67
68 static const std::array<std::string, 4> &topics = {
69 "image/rgb", "image/depth", "image/flow", "image/mask"};
70
71 std::shared_ptr<MultiImageSync4> multi_image_sync =
72 std::make_shared<MultiImageSync4>(node_ref, topics, 20);
73 multi_image_sync->registerCallback(
74 [this](const sensor_msgs::msg::Image::ConstSharedPtr &rgb_msg,
75 const sensor_msgs::msg::Image::ConstSharedPtr &depth_msg,
76 const sensor_msgs::msg::Image::ConstSharedPtr &flow_msg,
77 const sensor_msgs::msg::Image::ConstSharedPtr &mask_msg) {
78 if (!image_container_callback_) {
79 RCLCPP_ERROR_THROTTLE(node_->get_logger(), *node_->get_clock(), 1000,
80 "Image Sync callback triggered but "
81 "image_container_callback_ is not registered!");
82 return;
83 }
84
85 const cv::Mat rgb = readRgbRosImage(rgb_msg);
86 const cv::Mat depth = readDepthRosImage(depth_msg);
87 const cv::Mat flow = readFlowRosImage(flow_msg);
88 const cv::Mat mask = readMaskRosImage(mask_msg);
89
90 const Timestamp timestamp = utils::fromRosTime(rgb_msg->header.stamp);
91 const FrameId frame_id = frame_id_;
92 frame_id_++;
93
94 ImageContainer image_container(frame_id, timestamp);
95 image_container.rgb(rgb)
96 .depth(depth)
97 .opticalFlow(flow)
98 .objectMotionMask(mask);
99
100 image_container_callback_(
101 std::make_shared<ImageContainer>(image_container));
102 });
103 CHECK(multi_image_sync->connect());
104 image_subscriber_ = multi_image_sync;
105}

107void OnlineDataProviderRos::connectImu() {
108 if (imu_sub_) imu_sub_.reset();
109
110 imu_callback_group_ = node_->create_callback_group(
111 rclcpp::CallbackGroupType::MutuallyExclusive);
112
113 rclcpp::SubscriptionOptions imu_sub_options;
114 imu_sub_options.callback_group = imu_callback_group_;
115
116 imu_sub_ = node_->create_subscription<ImuAdaptedType>(
117 "imu", rclcpp::SensorDataQoS(),
118 [&](const dyno::ImuMeasurement &imu) -> void {
119 if (!imu_single_input_callback_) {
120 RCLCPP_ERROR_THROTTLE(
121 node_->get_logger(), *node_->get_clock(), 1000,
122 "Imu callback triggered but "
123 "imu_single_input_callback_ is not registered!");
124 return;
125 }
126 imu_single_input_callback_(imu);
127 },
128 imu_sub_options);
129}
