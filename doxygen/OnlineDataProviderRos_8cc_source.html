91void OnlineDataProviderRos::subscribeImu() {
92 if (imu_sub_) imu_sub_.reset();
93
94 imu_callback_group_ =
95 node_->create_callback_group(rclcpp::CallbackGroupType::Reentrant);
96
97 rclcpp::SubscriptionOptions imu_sub_options;
98 imu_sub_options.callback_group = imu_callback_group_;
99

101 "imu", rclcpp::SensorDataQoS(),
102 [&](const dyno::ImuMeasurement& imu) -> void {
103 if (!imu_single_input_callback_) {
104 RCLCPP_ERROR_THROTTLE(

106 "Imu callback triggered but "
107 "imu_single_input_callback_ is not registered!");
108 return;
109 }
110 imu_single_input_callback_(imu);
111 },
112 imu_sub_options);
113}

115RGBDTypeCalibrationHelper::RGBDTypeCalibrationHelper(
116 rclcpp::Node::SharedPtr node, const OnlineDataProviderRosParams& params)
117 : node_(node) {
118 if (params.wait_for_camera_params) {

136void RGBDTypeCalibrationHelper::processRGB(const cv::Mat& src, cv::Mat& dst) {
137 cv::remap(src, dst, mapx_, mapy_, cv::INTER_LINEAR);
138 // output will have the same type as mapx/y so covnert back to required type
139 dst.convertTo(dst, src.type());
140}
141void RGBDTypeCalibrationHelper::processDepth(const cv::Mat& src, cv::Mat& dst) {
142 cv::remap(src, dst, mapx_, mapy_, cv::INTER_LINEAR);
143 CHECK(src.type() == ImageType::Depth::OpenCVType);
144 // output will have the same type as mapx/y so covnert back to required type
145 dst.convertTo(dst, src.type());
146
147 // convert the depth map to metirc scale
148 // data-type shoule match
149 src *= depth_scale_;
150}

161void RGBDTypeCalibrationHelper::setupNewCameraParams(
162 const CameraParams& original_camera_params, CameraParams& new_camera_params,
163 const int& rescale_width, const int& rescale_height) {
164 const auto original_size = original_camera_params.imageSize();
165 const cv::Size rescale_size = cv::Size(rescale_width, rescale_height);
166 cv::Mat original_K = original_camera_params.getCameraMatrix();
167 const cv::Mat distortion = original_camera_params.getDistortionCoeffs();
168
169 cv::Mat new_K = cv::getOptimalNewCameraMatrix(
170 original_K, distortion, original_size, 1.0, rescale_size);
171
172 cv::initUndistortRectifyMap(original_K, distortion, cv::Mat(), new_K,
173 rescale_size, CV_32FC1, mapx_, mapy_);
174
175 dyno::CameraParams::IntrinsicsCoeffs intrinsics;
176 cv::Mat K_double;
177 new_K.convertTo(K_double, CV_64F);
178 dyno::CameraParams::convertKMatrixToIntrinsicsCoeffs(K_double, intrinsics);
179 dyno::CameraParams::DistortionCoeffs zero_distortion(4, 0);
180
181 new_camera_params = CameraParams(intrinsics, zero_distortion, rescale_size,
182 original_camera_params.getDistortionModel(),
183 original_camera_params.getExtrinsics());
184}

186void RGBDTypeCalibrationHelper::getParamsFromRos(
187 const CameraParams& original_camera_params, int& rescale_width,
188 int& rescale_height, double& depth_scale) {
189 rescale_width = ParameterConstructor(node_.get(), "rescale_width",
190 original_camera_params.ImageWidth())
191 .description(
192 "Image width to rescale to. If not provided or -1 "
193 "image will be inchanged")
194 .finish()
195 .get<int>();
196 if (rescale_width == -1) {
197 rescale_width = original_camera_params.ImageWidth();
198 }
199
200 rescale_height = ParameterConstructor(node_.get(), "rescale_height",
201 original_camera_params.ImageHeight())
202 .description(
203 "Image height to rescale to. If not provided or -1 "
204 "image will be inchanged")
205 .finish()
206 .get<int>();
207 if (rescale_height == -1) {
208 rescale_height = original_camera_params.ImageHeight();
209 }
210
211 depth_scale = ParameterConstructor(node_.get(), "depth_scale", 0.001)
212 .description(
213 "Value to scale the depth image from a disparity map "
214 "to metric depth")
215 .finish()
216 .get<double>();
217}

236AllImagesOnlineProviderRos::AllImagesOnlineProviderRos(
237 rclcpp::Node::SharedPtr node, const OnlineDataProviderRosParams& params)
238 : OnlineDataProviderRos(node, params) {
239 LOG(INFO) << "Creating AllImagesOnlineProviderRos";
240 calibration_helper_ =
241 std::make_unique<RGBDTypeCalibrationHelper>(node, params);
242
243 // All Images only works with undisroted images as the pre-processing must be
244 // done on undistorted
245 // images particularly for optical flow
246 auto original_camera_params = calibration_helper_->getOriginalCameraParams();
247 if (original_camera_params) {
248 const cv::Mat distortion = original_camera_params->getDistortionCoeffs();
249 if (cv::countNonZero(distortion.reshape(1)) != 0) {
250 // not all zeros
251 DYNO_THROW_MSG(DynosamException)
252 << "In AllImagesOnlineProviderRos the original camera params has "
253 "distortion coeffs which means the images "
254 " propvided have not been undisroted!";
255 }
256 } else {
257 DYNO_THROW_MSG(DynosamException)
258 << "No original camera params found for AllImagesOnlineProviderRos";
259 }
260}

262void AllImagesOnlineProviderRos::subscribeImages() {
263 rclcpp::Node& node_ref = *node_;
264 static const std::array<std::string, 4>& topics = {
265 "image/rgb", "image/depth", "image/flow", "image/mask"};
266
267 MultiSyncConfig config;
268 config.queue_size = 20;
269 // config.subscriber_options.callback_group =
270 // node_ref.create_callback_group(rclcpp::CallbackGroupType::Reentrant);
271
272 std::shared_ptr<MultiImageSync4> multi_image_sync =
273 std::make_shared<MultiImageSync4>(node_ref, topics, config);
274 multi_image_sync->registerCallback(
275 [this](const sensor_msgs::msg::Image::ConstSharedPtr& rgb_msg,
276 const sensor_msgs::msg::Image::ConstSharedPtr& depth_msg,
277 const sensor_msgs::msg::Image::ConstSharedPtr& flow_msg,
278 const sensor_msgs::msg::Image::ConstSharedPtr& mask_msg) {
279 if (!image_container_callback_) {
280 RCLCPP_ERROR_THROTTLE(node_->get_logger(), *node_->get_clock(), 1000,
281 "Image Sync callback triggered but "
282 "image_container_callback_ is not registered!");
283 return;
284 }
285
286 const cv::Mat rgb = readRgbRosImage(rgb_msg);
287 const cv::Mat depth = readDepthRosImage(depth_msg);
288 const cv::Mat flow = readFlowRosImage(flow_msg);
289 const cv::Mat mask = readMaskRosImage(mask_msg);
290
291 const Timestamp timestamp = utils::fromRosTime(rgb_msg->header.stamp);
292 const FrameId frame_id = frame_id_;
293 frame_id_++;
294
295 ImageContainer image_container(frame_id, timestamp);
296 image_container.rgb(rgb)
297 .depth(depth)
298 .opticalFlow(flow)
299 .objectMotionMask(mask);
300
301 image_container_callback_(
302 std::make_shared<ImageContainer>(image_container));
303 });
304 CHECK(multi_image_sync->connect());
305 image_subscriber_ = multi_image_sync;
306}

324void RGBDOnlineProviderRos::subscribeImages() {
325 rclcpp::Node& node_ref = *node_;
326 static const std::array<std::string, 2>& topics = {"image/rgb",
327 "image/depth"};
328
329 MultiSyncConfig config;
330 config.queue_size = 20;
331 // config.subscriber_options.callback_group =
332 // node_ref.create_callback_group(rclcpp::CallbackGroupType::Reentrant);
333
334 std::shared_ptr<MultiImageSync2> multi_image_sync =
335 std::make_shared<MultiImageSync2>(node_ref, topics, config);
336 multi_image_sync->registerCallback(
337 [this](const sensor_msgs::msg::Image::ConstSharedPtr& rgb_msg,
338 const sensor_msgs::msg::Image::ConstSharedPtr& depth_msg) {
339 if (!image_container_callback_) {
340 RCLCPP_ERROR_THROTTLE(node_->get_logger(), *node_->get_clock(), 1000,
341 "Image Sync callback triggered but "
342 "image_container_callback_ is not registered!");
343 return;
344 }
345
346 cv::Mat rgb = readRgbRosImage(rgb_msg);
347 cv::Mat depth = readDepthRosImage(depth_msg);
348
349 calibration_helper_->processRGB(rgb, rgb);
350 calibration_helper_->processDepth(depth, depth);
351
352 const Timestamp timestamp = utils::fromRosTime(rgb_msg->header.stamp);
353 const FrameId frame_id = frame_id_;
354 frame_id_++;
355
356 ImageContainer image_container(frame_id, timestamp);
357 image_container.rgb(rgb).depth(depth);
358
359 image_container_callback_(
360 std::make_shared<ImageContainer>(image_container));
361 });
362 CHECK(multi_image_sync->connect());
363 image_subscriber_ = multi_image_sync;
364}
