117template <typename TIMING_GENERATOR>
118class BaseTimingStatsCollector {
119 public:
120 using TimingGenerator = TIMING_GENERATOR;
121
122 DYNO_POINTER_TYPEDEFS(BaseTimingStatsCollector)
123
124 protected:
125 BaseTimingStatsCollector(const TimingGenerator& timing_generator,
126 const std::string& tag, int glog_level,
127 bool construct_stopped)
128 : timing_generator_(timing_generator),
129 tag_(tag + " [ms]"),
130 glog_level_(glog_level),
131 is_timing_(false) {
132 if (!construct_stopped) {
133 start();
134 }
135 }
136
137 public:
138 ~BaseTimingStatsCollector() { stop(); }
139
140 void start() {
141 timing_generator_.onStart();
142 is_timing_ = true;
143 }
144
145 void stop() {
146 if (isTiming() && shouldGlog()) {
147 log();
148 }
149
150 timing_generator_.onStop();
151 is_timing_ = false;
152 }
153
154 bool isTiming() const { return is_timing_; }
155 void discardTiming() { is_timing_ = false; }
156
157 // Time delta (dt) in nano seconds
158 inline double delta() const { return timing_generator_.calcDelta(); }
159
160 // not that it will log to glog, but that the glog verbosity level is set
161 // such that it will log to the collector
162 bool shouldGlog() const {
163 if (glog_level_ == 0) {
164 return true;
165 }
166 return VLOG_IS_ON(glog_level_);
167 }
168
169 private:
178 void log() {
179 const double nanoseconds = delta();
180 const double milliseconds = nanoseconds / 1e6;
181
182 if (!collector_) {
183 collector_ = std::make_unique<StatsCollector>(tag_);
184 }
185 collector_->AddSample(milliseconds);
186 }
187
188 private:
189 TimingGenerator timing_generator_;
190 const std::string tag_;
191 const int glog_level_;
193 std::atomic_bool is_timing_;
197 std::unique_ptr<StatsCollector> collector_;
198};
