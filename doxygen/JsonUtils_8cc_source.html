81adl_serializer<dyno::RGBDInstanceOutputPacket>::from_json(const json& j) {
82 using namespace dyno;
83
84 StatusKeypointVector static_keypoints =
85 j["static_keypoints"].template get<StatusKeypointVector>();
86 StatusKeypointVector dynamic_keypoints =
87 j["dynamic_keypoints"].template get<StatusKeypointVector>();
88
89 gtsam::Pose3 T_world_camera =
90 j["T_world_camera"].template get<gtsam::Pose3>();
91
92 Timestamp timestamp = j["timestamp"].template get<Timestamp>();
93 FrameId frame_id = j["frame_id"].template get<FrameId>();
94
95 StatusLandmarkVector static_landmarks =
96 j["static_landmarks"].template get<StatusLandmarkVector>();
97 StatusLandmarkVector dynamic_landmarks =
98 j["dynamic_landmarks"].template get<StatusLandmarkVector>();
99 // Base is a std::map with the right custom allocation
100 // we use the std::map version so that all the automagic with nlohmann::json
101 // can work
102 ObjectMotionMap estimated_motions(
103 j["estimated_motions"].template get<ObjectMotionMap>());
104 ObjectPoseMap propogated_object_poses =
105 j["propogated_object_poses"].template get<ObjectPoseMap>();
106 gtsam::Pose3Vector camera_poses =
107 j["camera_poses"].template get<gtsam::Pose3Vector>();
108
109 GroundTruthInputPacket::Optional gt_packet =
110 j["ground_truth"].template get<GroundTruthInputPacket::Optional>();
111
112 return dyno::RGBDInstanceOutputPacket(
113 static_keypoints, dynamic_keypoints, static_landmarks, dynamic_landmarks,
114 T_world_camera, timestamp, frame_id, estimated_motions,
115 propogated_object_poses, camera_poses, nullptr, /* no camera*/
116 gt_packet, std::nullopt /* no debug imagery */
117 );
118}
