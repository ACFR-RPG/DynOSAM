52CloudPerObject DisplayRos::publishPointCloud(
53 PointCloud2Pub::SharedPtr pub, const StatusLandmarkVector& landmarks,
54 const gtsam::Pose3& T_world_camera) {
55 pcl::PointCloud<pcl::PointXYZRGB> cloud;
56
57 CloudPerObject clouds_per_obj;
58
59 for (const auto& status_estimate : landmarks) {
60 Landmark lmk_world = status_estimate.value();
61 const ObjectId object_id = status_estimate.objectId();
62 if (status_estimate.referenceFrame() == ReferenceFrame::LOCAL) {
63 lmk_world = T_world_camera * status_estimate.value();
64 } else if (status_estimate.referenceFrame() == ReferenceFrame::OBJECT) {
65 throw DynosamException(
66 "Cannot display object point in the object reference frame");
67 }
68
69 pcl::PointXYZRGB pt;
70 if (status_estimate.isStatic()) {
71 // publish static lmk's as white
72 pt = pcl::PointXYZRGB(lmk_world(0), lmk_world(1), lmk_world(2), 0, 0, 0);
73 } else {
74 const cv::Scalar colour = Color::uniqueId(object_id);
75 pt = pcl::PointXYZRGB(lmk_world(0), lmk_world(1), lmk_world(2), colour(0),
76 colour(1), colour(2));
77 }
78 cloud.points.push_back(pt);
79 clouds_per_obj[object_id].push_back(pt);
80 }
81
82 sensor_msgs::msg::PointCloud2 pc2_msg;
83 pcl::toROSMsg(cloud, pc2_msg);
84 pc2_msg.header.frame_id = params_.world_frame_id_;
85 pub->publish(pc2_msg);
86
87 return clouds_per_obj;
88}

115void DisplayRos::publishObjectPositions(
116 MarkerArrayPub::SharedPtr pub, const ObjectPoseMap& object_positions,
117 FrameId frame_id, Timestamp latest_timestamp,
118 const std::string& prefix_marker_namespace, bool draw_labels,
119 double scale) {
120 visualization_msgs::msg::MarkerArray object_pose_marker_array;
121 static visualization_msgs::msg::Marker delete_marker;
122 delete_marker.action = visualization_msgs::msg::Marker::DELETEALL;
123
124 auto ros_time = utils::toRosTime(latest_timestamp);
125
126 object_pose_marker_array.markers.push_back(delete_marker);
127
128 for (const auto& [object_id, poses_map] : object_positions) {
129 // do not draw if in current frame
130 if (!poses_map.exists(frame_id)) {
131 continue;
132 }
133
134 const gtsam::Pose3& pose = poses_map.at(frame_id);
135
136 // assume
137 // object centroid per frame
138 visualization_msgs::msg::Marker marker;
139 marker.header.frame_id = params_.world_frame_id_;
140 marker.ns = prefix_marker_namespace + "_object_positions";
141 marker.id = object_id;
142 marker.type = visualization_msgs::msg::Marker::SPHERE;
143 marker.action = visualization_msgs::msg::Marker::ADD;
144 marker.header.stamp = ros_time;
145 marker.pose.position.x = pose.x();
146 marker.pose.position.y = pose.y();
147 marker.pose.position.z = pose.z();
148 marker.scale.x = scale;
149 marker.scale.y = scale;
150 marker.scale.z = scale;
151 marker.color.a = 1.0; // Don't forget to set the alpha!
152
153 const cv::Scalar colour = Color::uniqueId(object_id);
154 marker.color.r = colour(0) / 255.0;
155 marker.color.g = colour(1) / 255.0;
156 marker.color.b = colour(2) / 255.0;
157
158 object_pose_marker_array.markers.push_back(marker);
159
160 if (draw_labels) {
161 visualization_msgs::msg::Marker text_marker = marker;
162 text_marker.ns = prefix_marker_namespace + "_object_labels";
163 text_marker.type = visualization_msgs::msg::Marker::TEXT_VIEW_FACING;
164 text_marker.text = std::to_string(object_id);
165 text_marker.pose.position.z += 1.0; // make it higher than the pose
166 // marker
167 text_marker.scale.z = 0.7;
168 object_pose_marker_array.markers.push_back(text_marker);
169 }
170 }
171
172 pub->publish(object_pose_marker_array);
173}

175void DisplayRos::publishObjectPaths(MarkerArrayPub::SharedPtr pub,
176 const ObjectPoseMap& object_positions,
177 FrameId frame_id,
178 Timestamp latest_timestamp,
179 const std::string& prefix_marker_namespace,
180 const int min_poses) {
181 visualization_msgs::msg::MarkerArray object_path_marker_array;
182 static visualization_msgs::msg::Marker delete_marker;
183 delete_marker.action = visualization_msgs::msg::Marker::DELETEALL;
184
185 auto ros_time = utils::toRosTime(latest_timestamp);
186 object_path_marker_array.markers.push_back(delete_marker);
187
188 for (const auto& [object_id, poses_map] : object_positions) {
189 if (poses_map.size() < 2u) {
190 continue;
191 }
192
193 // draw a line list for viz
194 visualization_msgs::msg::Marker line_list_marker;
195 line_list_marker.type = visualization_msgs::msg::Marker::LINE_STRIP;
196 line_list_marker.header.frame_id = "world";
197 line_list_marker.ns = prefix_marker_namespace + "_object_path";
198 line_list_marker.id = object_id;
199 line_list_marker.header.stamp = ros_time;
200 line_list_marker.scale.x = 0.05;
201
202 line_list_marker.pose.orientation.x = 0;
203 line_list_marker.pose.orientation.y = 0;
204 line_list_marker.pose.orientation.z = 0;
205 line_list_marker.pose.orientation.w = 1;
206
207 const cv::Scalar colour = Color::uniqueId(object_id);
208 line_list_marker.color.r = colour(0) / 255.0;
209 line_list_marker.color.g = colour(1) / 255.0;
210 line_list_marker.color.b = colour(2) / 255.0;
211 line_list_marker.color.a = 1;
212
213 size_t traj_size;
214 // draw all the poses
215 if (min_poses == -1) {
216 traj_size = poses_map.size();
217 } else {
218 traj_size = std::min(min_poses, static_cast<int>(poses_map.size()));
219 }
220
221 // totally assume in order
222 auto map_iter = poses_map.end();
223 // equivalant to map_iter-traj_size + 1
224 // we want to go backwards to the starting point specified by trajectory
225 // size
226 std::advance(map_iter, (-traj_size + 1));
227 for (; map_iter != poses_map.end(); map_iter++) {
228 auto prev_iter = std::prev(map_iter, 1);
229 const gtsam::Pose3& prev_pose = prev_iter->second;
230 ;
231 const gtsam::Pose3& curr_pose = map_iter->second;
232
233 // check frames are consequative
234 // CHECK_EQ(prev_iter->first + 1, map_iter->first) << " For object " <<
235 // object_id;
236 LOG_IF(INFO, prev_iter->first + 1 != map_iter->first)
237 << " Frames not consequative for object " << object_id;
238
239 {
240 geometry_msgs::msg::Point p;
241 p.x = prev_pose.x();
242 p.y = prev_pose.y();
243 p.z = prev_pose.z();
244
245 line_list_marker.points.push_back(p);
246 }
247
248 {
249 geometry_msgs::msg::Point p;
250 p.x = curr_pose.x();
251 p.y = curr_pose.y();
252 p.z = curr_pose.z();
253
254 line_list_marker.points.push_back(p);
255 }
256 }
257
258 object_path_marker_array.markers.push_back(line_list_marker);
259 }
260
261 pub->publish(object_path_marker_array);
262}

284void DisplayRos::createAxisMarkers(
285 const gtsam::Pose3& pose, MarkerArray& axis_markers, Timestamp timestamp,
286 const cv::Scalar& colour, const std::string& frame, const std::string& ns,
287 double length, double radius) {
288 const auto pose_matrix = pose.matrix();
289 Eigen::Isometry3d x = Eigen::Translation3d(length / 2.0, 0, 0) *
290 Eigen::AngleAxisd(M_PI / 2.0, Eigen::Vector3d::UnitY());
291 gtsam::Pose3 x_pose(pose_matrix * x.matrix());
292
293 // Publish y axis
294 Eigen::Isometry3d y = Eigen::Translation3d(0, length / 2.0, 0) *
295 Eigen::AngleAxisd(M_PI / 2.0, Eigen::Vector3d::UnitX());
296 gtsam::Pose3 y_pose(pose_matrix * y.matrix());
297
298 Eigen::Isometry3d z = Eigen::Translation3d(0, 0, length / 2.0) *
299 Eigen::AngleAxisd(0, Eigen::Vector3d::UnitZ());
300 gtsam::Pose3 z_pose(pose_matrix * z.matrix());
301
302 const auto ros_time = utils::toRosTime(timestamp);
303
304 auto make_cylinder =
305 [&length, &radius, &ros_time, &ns, &frame,
306 &colour](const gtsam::Pose3& pose) -> visualization_msgs::msg::Marker {
307 visualization_msgs::msg::Marker marker;
308 marker.header.stamp = ros_time;
309 marker.ns = ns;
310 marker.header.frame_id = frame;
311
312 geometry_msgs::msg::Pose pose_msg;
313 dyno::convert(pose, pose_msg);
314
315 marker.pose = pose_msg;
316 marker.scale.x = radius;
317 marker.scale.y = radius;
318 marker.scale.z = length;
319
320 marker.color.r = colour(0) / 255.0;
321 marker.color.g = colour(1) / 255.0;
322 marker.color.b = colour(2) / 255.0;
323
324 return marker;
325 };
326
327 axis_markers.markers.push_back(make_cylinder(x_pose));
328 axis_markers.markers.push_back(make_cylinder(y_pose));
329 axis_markers.markers.push_back(make_cylinder(z_pose));
330}

332void DisplayRos::constructBoundingBoxeMarkers(
333 const CloudPerObject& cloud_per_object, MarkerArray& aabb_markers,
334 MarkerArray& obb_markers, Timestamp timestamp,
335 const std::string& prefix_marker_namespace) {
336 auto ros_time = utils::toRosTime(timestamp);
337
338 for (const auto& [object_id, obj_cloud] : cloud_per_object) {
339 pcl::PointXYZ centroid;
340 pcl::computeCentroid(obj_cloud, centroid);
341
342 const cv::Scalar colour = Color::uniqueId(object_id);
343
344 visualization_msgs::msg::Marker txt_marker;
345 txt_marker.header.frame_id = "world";
346 txt_marker.ns = prefix_marker_namespace + "_object_id_txt";
347 txt_marker.id = object_id;
348 txt_marker.type = visualization_msgs::msg::Marker::TEXT_VIEW_FACING;
349 txt_marker.action = visualization_msgs::msg::Marker::ADD;
350 txt_marker.header.stamp = ros_time;
351
352 txt_marker.scale.z = 0.5;
353 // txt_marker.scale.z = 2.0;
354 txt_marker.color.r = colour(0) / 255.0;
355 txt_marker.color.g = colour(1) / 255.0;
356 txt_marker.color.b = colour(2) / 255.0;
357 txt_marker.color.a = 1;
358 txt_marker.text = "Obj " + std::to_string(object_id);
359 txt_marker.pose.position.x = centroid.x;
360 // txt_marker.pose.position.y = centroid.y - 2.0;
361 // txt_marker.pose.position.z = centroid.z - 1.0;
362 txt_marker.pose.position.y = centroid.y - 0.6;
363 txt_marker.pose.position.z = centroid.z - 0.5;
364 aabb_markers.markers.push_back(txt_marker);
365
366 pcl::PointCloud<pcl::PointXYZRGB>::Ptr obj_cloud_ptr =
367 pcl::make_shared<pcl::PointCloud<pcl::PointXYZRGB> >(obj_cloud);
368 ObjectBBX aabb = findAABBFromCloud<pcl::PointXYZRGB>(obj_cloud_ptr);
369 ObjectBBX obb = findOBBFromCloud<pcl::PointXYZRGB>(obj_cloud_ptr);
370
371 visualization_msgs::msg::Marker aabb_marker;
372 aabb_marker.header.frame_id = "world";
373 aabb_marker.ns = prefix_marker_namespace + "object_aabb";
374 aabb_marker.id = object_id;
375 aabb_marker.type = visualization_msgs::msg::Marker::LINE_LIST;
376 aabb_marker.action = visualization_msgs::msg::Marker::ADD;
377 aabb_marker.header.stamp = ros_time;
378 aabb_marker.scale.x = 0.03;
379
380 aabb_marker.pose.orientation.x = 0;
381 aabb_marker.pose.orientation.y = 0;
382 aabb_marker.pose.orientation.z = 0;
383 aabb_marker.pose.orientation.w = 1;
384
385 aabb_marker.color.r = colour(0) / 255.0;
386 aabb_marker.color.g = colour(1) / 255.0;
387 aabb_marker.color.b = colour(2) / 255.0;
388 aabb_marker.color.a = 1;
389
390 for (const pcl::PointXYZ& this_line_list_point :
391 findLineListPointsFromAABBMinMax(aabb.min_bbx_point_,
392 aabb.max_bbx_point_)) {
393 geometry_msgs::msg::Point p;
394 p.x = this_line_list_point.x;
395 p.y = this_line_list_point.y;
396 p.z = this_line_list_point.z;
397 aabb_marker.points.push_back(p);
398 }
399 aabb_markers.markers.push_back(aabb_marker);
400
401 visualization_msgs::msg::Marker obbx_marker;
402 obbx_marker.header.frame_id = "world";
403 obbx_marker.ns = prefix_marker_namespace + "object_obb";
404 obbx_marker.id = object_id;
405 obbx_marker.type = visualization_msgs::msg::Marker::CUBE;
406 obbx_marker.action = visualization_msgs::msg::Marker::ADD;
407 obbx_marker.header.stamp = ros_time;
408 obbx_marker.scale.x = obb.max_bbx_point_.x() - obb.min_bbx_point_.x();
409 obbx_marker.scale.y = obb.max_bbx_point_.y() - obb.min_bbx_point_.y();
410 obbx_marker.scale.z = obb.max_bbx_point_.z() - obb.min_bbx_point_.z();
411
412 obbx_marker.pose.position.x = obb.bbx_position_.x();
413 obbx_marker.pose.position.y = obb.bbx_position_.y();
414 obbx_marker.pose.position.z = obb.bbx_position_.z();
415
416 const gtsam::Quaternion& q = obb.orientation_.toQuaternion();
417 obbx_marker.pose.orientation.x = q.x();
418 obbx_marker.pose.orientation.y = q.y();
419 obbx_marker.pose.orientation.z = q.z();
420 obbx_marker.pose.orientation.w = q.w();
421
422 obbx_marker.color.r = colour(0) / 255.0;
423 obbx_marker.color.g = colour(1) / 255.0;
424 obbx_marker.color.b = colour(2) / 255.0;
425 obbx_marker.color.a = 0.2;
426
427 obb_markers.markers.push_back(obbx_marker);
428 }

431DisplayRos::MarkerArray DisplayRos::createCameraMarker(
432 const gtsam::Pose3& T_world_x, Timestamp timestamp, const std::string& ns,
433 const cv::Scalar& colour, double marker_scale) {
434 auto transform_odom_to =
435 [&T_world_x](
436 const geometry_msgs::msg::Pose& msg) -> geometry_msgs::msg::Pose {
437 // convert msg to pose
438 gtsam::Pose3 msg_as_pose;
439 dyno::convert(msg, msg_as_pose);
440 // transform pose into world frame
441 gtsam::Pose3 transformed_pose = T_world_x * msg_as_pose;
442 // transform pose back to ROS msg and return
443 geometry_msgs::msg::Pose transformed_msg;
444 dyno::convert(transformed_pose, transformed_msg);
445 return transformed_msg;
446 };
447
448 auto ros_time = utils::toRosTime(timestamp);
449
450 // make rectangles as frame
451 const double r_w = 1.0;
452 const double z_plane = (r_w / 2.0) * marker_scale;
453
454 static constexpr double sqrt2_2 = sqrt(2) / 2;
455
456 MarkerArray marker_array;
457 visualization_msgs::msg::Marker marker;
458
459 // the marker will be displayed in frame_id
460 marker.header.frame_id = params_.world_frame_id_;
461 marker.header.stamp = ros_time;
462 marker.ns = ns;
463 marker.action = 0;
464 marker.id = 8; // 8 of 8 markers
465
466 marker.type = visualization_msgs::msg::Marker::CUBE;
467 marker.scale.x = r_w * marker_scale;
468 marker.scale.y = 0.04 * marker_scale;
469 marker.scale.z = 0.04 * marker_scale;
470 marker.color.r = colour(0) / 255.0;
471 marker.color.g = colour(1) / 255.0;
472 marker.color.b = colour(2) / 255.0;
473 marker.color.a = 1.0;
474
475 marker.pose.orientation.x = 0;
476 marker.pose.orientation.y = 0;
477 marker.pose.orientation.z = 0;
478 marker.pose.orientation.w = 1;
479 marker.id--;
480 marker_array.markers.push_back(marker);
481 marker.pose.position.y = -(r_w / 4.0) * marker_scale;
482 marker.id--;
483 marker_array.markers.push_back(marker);
484
485 marker.scale.x = (r_w / 2.0) * marker_scale;
486 marker.pose.position.x = (r_w / 2.0) * marker_scale;
487 marker.pose.position.y = 0;
488 marker.pose.orientation.w = sqrt2_2;
489 marker.pose.orientation.z = sqrt2_2;
490 marker.id--;
491 marker_array.markers.push_back(marker);
492 marker.pose.position.x = -(r_w / 2.0) * marker_scale;
493 marker.id--;
494 marker_array.markers.push_back(marker);
495
496 // make pyramid edges
497 marker.scale.x = (3.0 * r_w / 4.0) * marker_scale;
498 marker.pose.position.z = 0.5 * z_plane;
499
500 marker.pose.position.x = (r_w / 4.0) * marker_scale;
501 marker.pose.position.y = (r_w / 8.0) * marker_scale;
502 // 0.08198092, -0.34727674, 0.21462883, 0.9091823
503 marker.pose.orientation.x = 0.08198092;
504 marker.pose.orientation.y = -0.34727674;
505 marker.pose.orientation.z = 0.21462883;
506 marker.pose.orientation.w = 0.9091823;
507 marker.id--;
508 marker_array.markers.push_back(marker);
509
510 marker.pose.position.x = -(r_w / 4.0) * marker_scale;
511 marker.pose.position.y = (r_w / 8.0) * marker_scale;
512 // -0.27395078, -0.22863284, 0.9091823 , 0.21462883
513 marker.pose.orientation.x = 0.08198092;
514 marker.pose.orientation.y = 0.34727674;
515 marker.pose.orientation.z = -0.21462883;
516 marker.pose.orientation.w = 0.9091823;
517 marker.id--;
518 marker_array.markers.push_back(marker);
519
520 marker.pose.position.x = -(r_w / 4.0) * marker_scale;
521 marker.pose.position.y = -(r_w / 8.0) * marker_scale;
522 // -0.08198092, 0.34727674, 0.21462883, 0.9091823
523 marker.pose.orientation.x = -0.08198092;
524 marker.pose.orientation.y = 0.34727674;
525 marker.pose.orientation.z = 0.21462883;
526 marker.pose.orientation.w = 0.9091823;
527 marker.id--;
528 marker_array.markers.push_back(marker);
529
530 marker.pose.position.x = (r_w / 4.0) * marker_scale;
531 marker.pose.position.y = -(r_w / 8.0) * marker_scale;
532 // -0.08198092, -0.34727674, -0.21462883, 0.9091823
533 marker.pose.orientation.x = -0.08198092;
534 marker.pose.orientation.y = -0.34727674;
535 marker.pose.orientation.z = -0.21462883;
536 marker.pose.orientation.w = 0.9091823;
537 marker.id--;
538 marker_array.markers.push_back(marker);
539
540 for (auto& marker : marker_array.markers) {
541 // put in world frame
542 marker.pose = transform_odom_to(marker.pose);
543 }
544 return marker_array;
545}
