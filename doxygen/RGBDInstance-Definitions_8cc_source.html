41GenericTrackedStatusVector<LandmarkKeypointStatus>
42collectLandmarkKeypointMeasurementsHelper(
43 const StatusLandmarkVector& landmarks,
44 const StatusKeypointVector& keypoints) {
45 CHECK_EQ(landmarks.size(), keypoints.size());
46
47 GenericTrackedStatusVector<LandmarkKeypointStatus> collection;
48
49 for (size_t i = 0; i < landmarks.size(); i++) {
50 const auto& lmk_status = landmarks.at(i);
51 const auto& kp_status = keypoints.at(i);
52
53 CHECK_EQ(lmk_status.trackletId(), kp_status.trackletId());
54 CHECK_EQ(lmk_status.objectId(), kp_status.objectId());
55 CHECK_EQ(lmk_status.frameId(), kp_status.frameId());
56 // expect visual measurements being sent to the back-end to be local
57 CHECK_EQ(lmk_status.referenceFrame(), ReferenceFrame::LOCAL);
58
59 collection.push_back(LandmarkKeypointStatus(
60 LandmarkKeypoint(lmk_status.value(), kp_status.value()),
61 lmk_status.frameId(), lmk_status.trackletId(), lmk_status.objectId(),
62 lmk_status.referenceFrame()));
63 }
64
65 return collection;
66}

84void RGBDFrontendLogger::logTrackingLengthHistogram(const Frame::Ptr frame) {
85 gtsam::FastMap<ObjectId, Histogram> histograms =
86 vision_tools::makeTrackletLengthHistorgram(frame);
87 // collect histograms per object and then nest them per frame
88 // must cast keys (object id, frame id) to string to get the json library to
89 // properly construct nested maps
90 json per_object_hist;
91 for (const auto& [object_id, hist] : histograms) {
92 per_object_hist[std::to_string(object_id)] = hist;
93 }
94 tracklet_length_json_[std::to_string(frame->getFrameId())] = per_object_hist;
95}
