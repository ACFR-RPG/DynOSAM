46template <class T1, class T2>
47size_t hashPair(const std::pair<T1, T2>& p) {
48 // note: used to use boost::hash_combine
49 const auto h1 = std::hash<T1>{}(p.first);
50 const auto h2 = std::hash<T2>{}(p.second);

147template <typename T>
148T mod(T x, T y) {
149 static_assert(!std::numeric_limits<T>::is_exact,
150 "Mod: floating-point type expected");
151
152 if (0. == y) return x;
153
154 double m = x - y * floor(x / y);
155
156 // handle boundary cases resulted from floating-point cut off:
157 if (y > 0) // modulo range: [0..y)
158 {
159 if (m >= y) // Mod(-1e-16 , 360. ): m= 360.
160 return 0;
161
162 if (m < 0) {
163 if (y + m == y)
164 return 0; // just in case...
165 else
166 return y + m; // Mod(106.81415022205296 , _TWO_PI ): m= -1.421e-14
167 }
168 } else // modulo range: (y..0]
169 {
170 if (m <= y) // Mod(1e-16 , -360. ): m= -360.
171 return 0;
172
173 if (m > 0) {
174 if (y + m == y)
175 return 0; // just in case...
176 else
177 return y + m; // Mod(-106.81415022205296, -_TWO_PI): m= 1.421e-14
178 }
179 }
180
181 return m;
182}

195inline bool equals_with_abs_tol(const std::vector<T>& vec1,
196 const std::vector<T>& vec2, double tol = 1e-9) {
197 if (vec1.size() != vec2.size()) return false;
198
199 const size_t m = vec1.size();
200 for (size_t i = 0; i < m; i++) {
201 if (!fpEqual(vec1[i], vec2[i], tol)) {
202 return false;
203 }
204 }
205
206 return true;
207}

227template <typename T>
228inline T computeCentroid(const std::vector<T>& vec) {
229 T sum{};
230 for (const T& t : vec) {

232 }

237inline T computeCentroid(

246template <typename T>
247inline double calculateStandardDeviation(
248 const std::vector<T, Eigen::aligned_allocator<T>>& vec) {
249 const T mean = computeCentroid<T>(vec);
250
251 double sum = 0;
252 for (const T& t : vec) {
253 T sub = t.colwise() - mean;
254 sum += sub.norm();
255 }
256
257 return std::sqrt(sum / static_cast<double>(vec.size()));
258}

273inline bool saveMatrixAsUpperTriangular(std::ostream& os,
274 const gtsam::Matrix& matrix) {
275 const size_t rows = matrix.rows();
276 const size_t cols = matrix.cols();
277
278 if (rows != cols) {
279 throw std::runtime_error(
280 "Attempting to save matrix as upper triangular but input size was not "
281 "square");
282 }
283
284 for (size_t i = 0; i < rows; i++) {
285 for (size_t j = i; j < cols; j++) {
286 os << " " << matrix(i, j);
287 }
288 }
289 return os.good();
290}
