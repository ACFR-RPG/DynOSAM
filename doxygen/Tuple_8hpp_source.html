69template <class Function, std::size_t... Is>
70void for_each_apply_impl(Function f, std::index_sequence<Is...>)
71{

165template<typename T, typename Tuple>
166struct tuple_element_index_helper;
167
168template<typename T>
169struct tuple_element_index_helper<T, std::tuple<>>
170{
171 static constexpr std::size_t value = 0;
172};
173
174
175template<typename T, typename... Rest>
176struct tuple_element_index_helper<T, std::tuple<T, Rest...>>
177{
178 static constexpr std::size_t value = 0;
179 using RestTuple = std::tuple<Rest...>;
180 static_assert(
181 tuple_element_index_helper<T, RestTuple>::value ==
182 std::tuple_size_v<RestTuple>,
183 "type appears more than once in tuple");
184};
185
186template<typename T, typename First, typename... Rest>
187struct tuple_element_index_helper<T, std::tuple<First, Rest...>>
188{
189 using RestTuple = std::tuple<Rest...>;
190 static constexpr std::size_t value = 1 +
191 tuple_element_index_helper<T, RestTuple>::value;
192};
193
194template<typename T, typename Tuple>
195struct tuple_element_index
196{
197 static constexpr std::size_t value =
198 tuple_element_index_helper<T, Tuple>::value;
199 static_assert(value < std::tuple_size_v<Tuple>,
200 "type does not appear in tuple");
201};
