117 template <typename C, typename F, typename T, typename... A>
118 static inline auto apply(C&& c, F&& f, T&& t, A&&... a)
119 -> decltype(ApplyMember<N - 1>::apply(forward<C>(c), forward<F>(f),

159template <typename T, typename Tuple>

162template <typename T>
163struct tuple_element_index_helper<T, std::tuple<>> {
164 static constexpr std::size_t value = 0;
165};
166
167template <typename T, typename... Rest>
168struct tuple_element_index_helper<T, std::tuple<T, Rest...>> {
169 static constexpr std::size_t value = 0;
170 using RestTuple = std::tuple<Rest...>;
171 static_assert(tuple_element_index_helper<T, RestTuple>::value ==
172 std::tuple_size_v<RestTuple>,
173 "type appears more than once in tuple");
174};
175
176template <typename T, typename First, typename... Rest>
177struct tuple_element_index_helper<T, std::tuple<First, Rest...>> {
178 using RestTuple = std::tuple<Rest...>;
179 static constexpr std::size_t value =
180 1 + tuple_element_index_helper<T, RestTuple>::value;
181};
