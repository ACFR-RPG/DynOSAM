65void Track::update(const DetectionBase::Ptr& matched_detection, size_t frame_id) {
66 detection_->set_rect(matched_detection->rect());
67 detection_->set_score(matched_detection->score());
68 predicted_rect_ = kalman_filter_.update(matched_detection->rect());
69
70 // If the track was actively tracked, just increment the tracklet length
71 // Otherwise, mark the track as tracked again and reset the tracklet length
72 if (state_ == TrackState::Tracked) {
73 tracklet_len_++;
74 } else {
75 state_ = TrackState::Tracked;
76 tracklet_len_ = 0;
77 }
78 is_confirmed_ = true;
79 frame_id_ = frame_id;
80}
