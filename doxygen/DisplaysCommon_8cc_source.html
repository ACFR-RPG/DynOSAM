13CloudPerObject DisplayCommon::publishPointCloud(
14 PointCloud2Pub::SharedPtr pub, const StatusLandmarkVector& landmarks,
15 const gtsam::Pose3& T_world_camera, const std::string& frame_id) {
16 pcl::PointCloud<pcl::PointXYZRGB> cloud;
17 CloudPerObject clouds_per_obj =
18 groupObjectCloud(landmarks, T_world_camera,
19 [&cloud](const pcl::PointXYZRGB& point, ObjectId) {
20 cloud.points.push_back(point);
21 });
22
23 pcl::PointCloud<pcl::PointXYZRGB> filtered_and_merged_cloud;
24 for (auto& [_, obj_cloud] : clouds_per_obj) {
25 auto cloud = pcl::make_shared<pcl::PointCloud<pcl::PointXYZRGB>>(obj_cloud);
26 pcl::PointCloud<pcl::PointXYZRGB>::Ptr cloud_filtered(
27 new pcl::PointCloud<pcl::PointXYZRGB>);
28 pcl::StatisticalOutlierRemoval<pcl::PointXYZRGB> sor;
29 sor.setInputCloud(cloud);
30 sor.setMeanK(50); // Number of neighbors to analyze
31 sor.setStddevMulThresh(1.0); // Threshold based on std dev
32 sor.filter(*cloud_filtered);
33
34 // update object cloud with the one that is actually published
35 obj_cloud = *cloud_filtered;
36 filtered_and_merged_cloud += (*cloud_filtered);
37 }
38
39 sensor_msgs::msg::PointCloud2 pc2_msg;
40 pcl::toROSMsg(filtered_and_merged_cloud, pc2_msg);
41 pc2_msg.header.frame_id = frame_id;
42 pub->publish(pc2_msg);
43
44 return clouds_per_obj;
45}

74std::vector<Marker> DisplayCommon::objectBBXToRvizMarker(
75 const ObjectBBX& bounding_box, const ObjectId object_id,
76 const Timestamp latest_timestamp, const std::string& frame_id) {
77 std::vector<Marker> markers;
78
79 Marker marker;
80 marker.header.frame_id = frame_id;
81 marker.header.stamp = utils::toRosTime(latest_timestamp);
82 marker.ns = "object_bbx";
83 marker.id = object_id * 2; // even ids for boxes
84 marker.type = visualization_msgs::msg::Marker::CUBE;
85 marker.action = visualization_msgs::msg::Marker::ADD;
86 marker.lifetime = rclcpp::Duration::from_seconds(3.0);
87
88 std_msgs::msg::ColorRGBA colour_msg;
89 convert(Color::uniqueId(object_id), colour_msg);
90 marker.color = colour_msg;
91 marker.color.a = 0.25f; // translucent so points/objects remain visible
92
93 // Compute scale from min/max points
94 gtsam::Point3 size =
95 bounding_box.max_bbx_point_ - bounding_box.min_bbx_point_;
96 marker.scale.x = std::abs(size.x());
97 marker.scale.y = std::abs(size.y());
98 marker.scale.z = std::abs(size.z());
99
100 geometry_msgs::msg::Pose pose_msg;
101 // convert(bounding_box.pose(), pose);
102 gtsam::Pose3 bbx_pose(gtsam::Rot3::Identity(), bounding_box.bbx_position_);
103 convert(bbx_pose, pose_msg);
104
105 marker.pose = pose_msg;
106 markers.push_back(marker);
107
108 // SOLID EDGES (wireframe outline)
109 Marker edges;
110 edges.header = marker.header;
111 edges.ns = "object_bbx_edges";
112 edges.id = object_id * 2 + 1; // odd ids for edges
113 edges.type = visualization_msgs::msg::Marker::LINE_LIST;
114 edges.action = visualization_msgs::msg::Marker::ADD;
115 edges.lifetime = rclcpp::Duration::from_seconds(3.0);
116
117 edges.scale.x = 0.015; // line thickness
118 edges.color = marker.color;
119 edges.color.a = 1.0f; // fully opaque edges
120
121 // Compute 8 corners
122 std::vector<Eigen::Vector3d> corners;
123 corners.reserve(8);
124 Eigen::Vector3d min(bounding_box.min_bbx_point_.x(),
125 bounding_box.min_bbx_point_.y(),
126 bounding_box.min_bbx_point_.z());
127 Eigen::Vector3d max(bounding_box.max_bbx_point_.x(),
128 bounding_box.max_bbx_point_.y(),
129 bounding_box.max_bbx_point_.z());
130
131 for (int i = 0; i < 8; ++i)
132 corners.emplace_back((i & 1) ? max.x() : min.x(),
133 (i & 2) ? max.y() : min.y(),
134 (i & 4) ? max.z() : min.z());
135
136 // Transform for OBB
137 const Eigen::Matrix3d R = bbx_pose.rotation().matrix();
138 const Eigen::Vector3d t = bbx_pose.translation();
139 for (auto& c : corners) c = R * (c - Eigen::Vector3d::Zero()) + t;
140
141 auto add_edge = [&](int i, int j) {
142 geometry_msgs::msg::Point p1, p2;
143 p1.x = corners[i].x();
144 p1.y = corners[i].y();
145 p1.z = corners[i].z();
146 p2.x = corners[j].x();
147 p2.y = corners[j].y();
148 p2.z = corners[j].z();
149 edges.points.push_back(p1);
150 edges.points.push_back(p2);
151 };
152
153 // 12 edges of a cube
154 add_edge(0, 1);
155 add_edge(1, 3);
156 add_edge(3, 2);
157 add_edge(2, 0);
158 add_edge(4, 5);
159 add_edge(5, 7);
160 add_edge(7, 6);
161 add_edge(6, 4);
162 add_edge(0, 4);
163 add_edge(1, 5);
164 add_edge(2, 6);
165 add_edge(3, 7);
166
167 markers.push_back(edges);
168
169 return markers;
170}
