54template <class MAP>
55MotionEstimateMap AccessorT<MAP>::getObjectMotions(FrameId frame_id) const {
56 MotionEstimateMap motion_estimates;
57
58 const auto frame_node = map()->getFrame(frame_id);
59 if (!frame_node) {
60 return motion_estimates;
61 }
62
63 const auto object_seen =
64 frame_node->objects_seen.template collectIds<ObjectId>();
65 for (ObjectId object_id : object_seen) {
66 StateQuery<Motion3ReferenceFrame> motion_query =
67 this->getObjectMotionReferenceFrame(frame_id, object_id);
68 if (motion_query) {
69 motion_estimates.insert2(object_id, motion_query.get());
70 }
71 }
72 return motion_estimates;
73}

75template <class MAP>
76StateQuery<Motion3ReferenceFrame> AccessorT<MAP>::getObjectMotionReferenceFrame(
77 FrameId frame_id, ObjectId object_id) const {
78 StateQuery<Motion3> motion_query = this->getObjectMotion(frame_id, object_id);
79 if (motion_query) {
80 return StateQuery<Motion3ReferenceFrame>(
81 motion_query.key_,
82 Motion3ReferenceFrame(motion_query.get(),
83 Motion3ReferenceFrame::Style::F2F,
84 ReferenceFrame::GLOBAL, frame_id - 1u, frame_id));
85 } else {
86 return StateQuery<Motion3ReferenceFrame>(motion_query.key_,
87 motion_query.status_);
88 }
89}

115template <class MAP>
116ObjectPoseMap AccessorT<MAP>::getObjectPoses() const {
117 ObjectPoseMap object_poses;
118 for (FrameId frame_id : map()->getFrameIds()) {
119 EstimateMap<ObjectId, gtsam::Pose3> per_object_pose =
120 this->getObjectPoses(frame_id);
121
122 for (const auto& [object_id, pose] : per_object_pose) {
123 if (!object_poses.exists(object_id)) {
124 object_poses.insert2(object_id,
125 gtsam::FastMap<FrameId, gtsam::Pose3>{});
126 }
127
128 auto& per_frame_pose = object_poses.at(object_id);
129 per_frame_pose.insert2(frame_id, pose);
130 }
131 }
132 return object_poses;
133}

136ObjectMotionMap AccessorT<MAP>::getObjectMotions() const {
137 ObjectMotionMap object_motions;
138 for (FrameId frame_id : map()->getFrameIds()) {
139 MotionEstimateMap per_object_motions = this->getObjectMotions(frame_id);
140

142 if (!object_motions.exists(object_id)) {

145
146 auto& per_frame_motion = object_motions.at(object_id);
147 per_frame_motion.insert2(frame_id, motion);
148 }
149 }
150 return object_motions;

168template <class MAP>
169StatusLandmarkVector AccessorT<MAP>::getDynamicLandmarkEstimates(
170 FrameId frame_id, ObjectId object_id) const {
171 const auto frame_node = map()->getFrame(frame_id);
172 CHECK_NOTNULL(frame_node);
173
174 if (!frame_node->objectObserved(object_id)) {
175 return StatusLandmarkVector{};
176 }
177
178 StatusLandmarkVector estimates;
179 const auto& dynamic_landmarks = frame_node->dynamic_landmarks;
180 for (auto lmk_node : dynamic_landmarks) {
181 const auto tracklet_id = lmk_node->tracklet_id;
182
183 if (object_id != lmk_node->object_id) {
184 continue;
185 }
186
187 // user defined function should put point in the world frame
188 StateQuery<gtsam::Point3> lmk_query =
189 this->getDynamicLandmark(frame_id, tracklet_id);
190 if (lmk_query) {
191 estimates.push_back(LandmarkStatus::DynamicInGLobal(
192 Point3Measurement(lmk_query.get()), // estimate
193 frame_id, tracklet_id,
194 object_id) // status
195 );
196 }
197 }
198 return estimates;
199}

201template <class MAP>
202StatusLandmarkVector AccessorT<MAP>::getStaticLandmarkEstimates(
203 FrameId frame_id) const {
204 // dont go over the frames as this contains references to the landmarks
205 // multiple times
206 // e.g. the ones seen in that frame
207 StatusLandmarkVector estimates;
208
209 const auto frame_node = map()->getFrame(frame_id);
210 CHECK_NOTNULL(frame_node);
211
212 for (const auto& landmark_node : frame_node->static_landmarks) {
213 if (landmark_node->isStatic()) {
214 StateQuery<gtsam::Point3> lmk_query =
215 getStaticLandmark(landmark_node->tracklet_id);
216 if (lmk_query) {
217 estimates.push_back(LandmarkStatus::StaticInGlobal(
218 Point3Measurement(lmk_query.get()), // estimate
219 LandmarkStatus::MeaninglessFrame,
220 landmark_node->getId() // tracklet id
221 ) // status
222 );
223 }
224 }
225 }
226 return estimates;
227}

285template <class MAP>
286gtsam::FastMap<ObjectId, gtsam::Point3> AccessorT<MAP>::computeObjectCentroids(
287 FrameId frame_id) const {
288 gtsam::FastMap<ObjectId, gtsam::Point3> centroids;
289
290 const auto frame_node = map()->getFrame(frame_id);
291 if (!frame_node) {
292 return centroids;
293 }
294
295 const auto object_seen =
296 frame_node->objects_seen.template collectIds<ObjectId>();
297 for (ObjectId object_id : object_seen) {
298 const auto [centroid, result] = computeObjectCentroid(frame_id, object_id);
299
300 if (result) {
301 centroids.insert2(object_id, centroid);
302 }
303 }
304 return centroids;
305}

314template <class MAP>
315template <typename ValueType>
316StateQuery<ValueType> AccessorT<MAP>::query(gtsam::Key key) const {
317 const gtsam::Values* theta = shared_data_.values;
318 CHECK_NOTNULL(theta);
319 if (theta->exists(key)) {
320 return StateQuery<ValueType>(key, theta->at<ValueType>(key));
321 } else {
322 return StateQuery<ValueType>::NotInMap(key);
323 }
