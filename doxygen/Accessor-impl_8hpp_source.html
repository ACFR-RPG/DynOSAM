54template <class MAP>
55MotionEstimateMap Accessor<MAP>::getObjectMotions(FrameId frame_id) const {
56 MotionEstimateMap motion_estimates;
57
58 const auto frame_node = map()->getFrame(frame_id);
59 if (!frame_node) {
60 return motion_estimates;
61 }
62
63 const auto object_seen =
64 frame_node->objects_seen.template collectIds<ObjectId>();
65 for (ObjectId object_id : object_seen) {
66 StateQuery<Motion3ReferenceFrame> motion_query =
67 this->getObjectMotionReferenceFrame(frame_id, object_id);
68 if (motion_query) {
69 motion_estimates.insert2(object_id, motion_query.get());
70 }
71 }
72 return motion_estimates;
73}

75template <class MAP>
76StateQuery<Motion3ReferenceFrame> Accessor<MAP>::getObjectMotionReferenceFrame(
77 FrameId frame_id, ObjectId object_id) const {
78 StateQuery<Motion3> motion_query = this->getObjectMotion(frame_id, object_id);
79 if (motion_query) {
80 return StateQuery<Motion3ReferenceFrame>(
81 motion_query.key_,
82 Motion3ReferenceFrame(motion_query.get(),
83 Motion3ReferenceFrame::Style::F2F,
84 ReferenceFrame::GLOBAL, frame_id - 1u, frame_id));
85 } else {
86 return StateQuery<Motion3ReferenceFrame>(motion_query.key_,
87 motion_query.status_);

115template <class MAP>
116ObjectPoseMap Accessor<MAP>::getObjectPoses() const {
117 ObjectPoseMap object_poses;
118 for (FrameId frame_id : map()->getFrameIds()) {
119 EstimateMap<ObjectId, gtsam::Pose3> per_object_pose =
120 this->getObjectPoses(frame_id);
121
122 for (const auto& [object_id, pose] : per_object_pose) {
123 if (!object_poses.exists(object_id)) {
124 object_poses.insert2(object_id,
125 gtsam::FastMap<FrameId, gtsam::Pose3>{});
126 }
127
128 auto& per_frame_pose = object_poses.at(object_id);
129 per_frame_pose.insert2(frame_id, pose);
130 }
131 }
132 return object_poses;
133}

168template <class MAP>
169StatusLandmarkVector Accessor<MAP>::getDynamicLandmarkEstimates(
170 FrameId frame_id, ObjectId object_id) const {
171 const auto frame_node = map()->getFrame(frame_id);
172 CHECK_NOTNULL(frame_node);
173
174 if (!frame_node->objectObserved(object_id)) {
175 return StatusLandmarkVector{};
176 }
177
178 StatusLandmarkVector estimates;
179 const auto& dynamic_landmarks = frame_node->dynamic_landmarks;
180 for (auto lmk_node : dynamic_landmarks) {
181 const auto tracklet_id = lmk_node->tracklet_id;
182
183 if (object_id != lmk_node->object_id) {
184 continue;
185 }
186
187 // user defined function should put point in the world frame
188 StateQuery<gtsam::Point3> lmk_query =
189 this->getDynamicLandmark(frame_id, tracklet_id);
190 if (lmk_query) {
191 estimates.push_back(LandmarkStatus::DynamicInGLobal(
192 Point3Measurement(lmk_query.get()), // estimate
193 frame_id, tracklet_id,
194 object_id) // status
195 );
196 }
197 }
198 return estimates;
199}

201template <class MAP>
202StatusLandmarkVector Accessor<MAP>::getStaticLandmarkEstimates(
203 FrameId frame_id) const {
204 // dont go over the frames as this contains references to the landmarks
205 // multiple times
206 // e.g. the ones seen in that frame
207 StatusLandmarkVector estimates;
208
209 const auto frame_node = map()->getFrame(frame_id);
210 CHECK_NOTNULL(frame_node);
211
212 for (const auto& landmark_node : frame_node->static_landmarks) {
213 if (landmark_node->isStatic()) {
214 StateQuery<gtsam::Point3> lmk_query =
215 getStaticLandmark(landmark_node->tracklet_id);
216 if (lmk_query) {
217 estimates.push_back(LandmarkStatus::StaticInGlobal(
218 Point3Measurement(lmk_query.get()), // estimate
219 LandmarkStatus::MeaninglessFrame,
220 landmark_node->getId() // tracklet id
221 ) // status
222 );
223 }
224 }
225 }
226 return estimates;
227}

307template <class MAP>
308gtsam::FastMap<ObjectId, gtsam::Point3> Accessor<MAP>::computeObjectCentroids(
309 FrameId frame_id) const {
310 gtsam::FastMap<ObjectId, gtsam::Point3> centroids;
311
312 const auto frame_node = map()->getFrame(frame_id);
313 if (!frame_node) {
314 return centroids;
315 }
316
317 const auto object_seen =
318 frame_node->objects_seen.template collectIds<ObjectId>();
319 for (ObjectId object_id : object_seen) {
320 const auto [centroid, result] = computeObjectCentroid(frame_id, object_id);
321
322 if (result) {
323 centroids.insert2(object_id, centroid);
324 }
325 }
326 return centroids;
327}

329template <class MAP>
330std::tuple<gtsam::Point3, bool> Accessor<MAP>::computeObjectCentroid(
331 FrameId frame_id, ObjectId object_id) const {
332 const StatusLandmarkVector& dynamic_lmks =
333 this->getDynamicLandmarkEstimates(frame_id, object_id);
334
335 // convert to point cloud - should be a map with only one map in it
336 CloudPerObject object_clouds =
337 groupObjectCloud(dynamic_lmks, this->getSensorPose(frame_id).get());
338 if (object_clouds.size() == 0) {
339 VLOG(20) << "Cannot collect object clouds from dynamic landmarks of "
340 << object_id << " and frame " << frame_id << "!! "
341 << " # Dynamic lmks in the map for this object at this frame was "
342 << dynamic_lmks.size(); //<< " but reocrded lmks was " <<
343 // dynamic_landmarks.size();
344 return {gtsam::Point3{}, false};
345 }
346 CHECK_EQ(object_clouds.size(), 1);
347 CHECK(object_clouds.exists(object_id));
348
349 const auto dynamic_point_cloud = object_clouds.at(object_id);
350 pcl::PointXYZ centroid;
351 pcl::computeCentroid(dynamic_point_cloud, centroid);
352 // TODO: outlier reject?
353 gtsam::Point3 translation = pclPointToGtsam(centroid);
354 return {translation, true};
355}

357template <class MAP>
358EstimateMap<ObjectId, std::pair<Motion3, gtsam::Pose3>>
359Accessor<MAP>::getRequestedObjectPosePair(FrameId motion_frame_id,
360 FrameId pose_frame_id) const {
361 const MotionEstimateMap motions_k = this->getObjectMotions(motion_frame_id);
362 const EstimateMap<ObjectId, gtsam::Pose3> poses_k =
363 this->getObjectPoses(pose_frame_id);
364
365 EstimateMap<ObjectId, std::pair<Motion3, gtsam::Pose3>> result;
366 // collect common objects
367 for (const auto& [object_id, _] : motions_k) {
368 if (poses_k.exists(object_id)) {
369 auto motion_reference_values = motions_k.at(object_id);
370 auto pose_reference_values = poses_k.at(object_id);
371
372 CHECK_EQ(motion_reference_values.frame_, pose_reference_values.frame_);
373
374 ReferenceFrameValue<std::pair<Motion3, gtsam::Pose3>> estimate(
375 std::make_pair(motion_reference_values.estimate_,
376 pose_reference_values.estimate_),
377 motion_reference_values.frame_);
378 result.insert2(object_id, estimate);
379 }
380 }
381 return result;
382}
