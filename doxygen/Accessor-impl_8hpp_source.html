50template <class MAP>
51MotionEstimateMap Accessor<MAP>::getObjectMotions(FrameId frame_id) const {
52 MotionEstimateMap motion_estimates;
53
54 const auto frame_node = map()->getFrame(frame_id);
55 if (!frame_node) {
56 return motion_estimates;
57 }
58
59 const auto object_seen =
60 frame_node->objects_seen.template collectIds<ObjectId>();
61 for (ObjectId object_id : object_seen) {
62 StateQuery<Motion3> motion_query =
63 this->getObjectMotion(frame_id, object_id);
64 if (motion_query) {
65 motion_estimates.insert2(
66 object_id, ReferenceFrameValue<Motion3>(motion_query.get(),
67 ReferenceFrame::GLOBAL));
68 }
69 }
70 return motion_estimates;
71}

97template <class MAP>
98ObjectPoseMap Accessor<MAP>::getObjectPoses() const {
99 ObjectPoseMap object_poses;
100 for (FrameId frame_id : map()->getFrameIds()) {
101 EstimateMap<ObjectId, gtsam::Pose3> per_object_pose =
102 this->getObjectPoses(frame_id);
103
104 for (const auto& [object_id, pose] : per_object_pose) {
105 if (!object_poses.exists(object_id)) {
106 object_poses.insert2(object_id,
107 gtsam::FastMap<FrameId, gtsam::Pose3>{});
108 }
109
110 auto& per_frame_pose = object_poses.at(object_id);
111 per_frame_pose.insert2(frame_id, pose);
112 }
113 }
114 return object_poses;
115}

132template <class MAP>
133StatusLandmarkVector Accessor<MAP>::getDynamicLandmarkEstimates(
134 FrameId frame_id, ObjectId object_id) const {
135 const auto frame_node = map()->getFrame(frame_id);
136 CHECK_NOTNULL(frame_node);
137
138 if (!frame_node->objectObserved(object_id)) {
139 return StatusLandmarkVector{};
140 }
141
142 StatusLandmarkVector estimates;
143 const auto& dynamic_landmarks = frame_node->dynamic_landmarks;
144 for (auto lmk_node : dynamic_landmarks) {
145 const auto tracklet_id = lmk_node->tracklet_id;
146
147 if (object_id != lmk_node->object_id) {
148 continue;
149 }
150
151 // user defined function should put point in the world frame
152 StateQuery<gtsam::Point3> lmk_query =
153 this->getDynamicLandmark(frame_id, tracklet_id);
154 if (lmk_query) {
155 estimates.push_back(
156 LandmarkStatus::DynamicInGLobal(lmk_query.get(), // estimate
157 frame_id, tracklet_id,
158 object_id) // status
159 );
160 }
161 }
162 return estimates;
163}

165template <class MAP>
166StatusLandmarkVector Accessor<MAP>::getStaticLandmarkEstimates(
167 FrameId frame_id) const {
168 // dont go over the frames as this contains references to the landmarks
169 // multiple times
170 // e.g. the ones seen in that frame
171 StatusLandmarkVector estimates;
172
173 const auto frame_node = map()->getFrame(frame_id);
174 CHECK_NOTNULL(frame_node);
175
176 for (const auto& landmark_node : frame_node->static_landmarks) {
177 if (landmark_node->isStatic()) {
178 StateQuery<gtsam::Point3> lmk_query =
179 getStaticLandmark(landmark_node->tracklet_id);
180 if (lmk_query) {
181 estimates.push_back(LandmarkStatus::StaticInGlobal(
182 lmk_query.get(), // estimate
183 LandmarkStatus::MeaninglessFrame,
184 landmark_node->getId() // tracklet id
185 ) // status
186 );
187 }
188 }
189 }
190 return estimates;
191}

195 // dont go over the frames as this contains references to the landmarks
196 // multiple times e.g. the ones seen in that frame

209 ) // status
210 );
211 }
212 }
213 }

271template <class MAP>
272gtsam::FastMap<ObjectId, gtsam::Point3> Accessor<MAP>::computeObjectCentroids(
273 FrameId frame_id) const {
274 gtsam::FastMap<ObjectId, gtsam::Point3> centroids;
275
276 const auto frame_node = map()->getFrame(frame_id);
277 if (!frame_node) {
278 return centroids;
279 }
280
281 const auto object_seen =
282 frame_node->objects_seen.template collectIds<ObjectId>();
283 for (ObjectId object_id : object_seen) {
284 const auto [centroid, result] = computeObjectCentroid(frame_id, object_id);
285
286 if (result) {
287 centroids.insert2(object_id, centroid);
288 }
289 }
290 return centroids;
291}

293template <class MAP>
294std::tuple<gtsam::Point3, bool> Accessor<MAP>::computeObjectCentroid(
295 FrameId frame_id, ObjectId object_id) const {
296 const StatusLandmarkVector& dynamic_lmks =
297 this->getDynamicLandmarkEstimates(frame_id, object_id);
298
299 // convert to point cloud - should be a map with only one map in it
300 CloudPerObject object_clouds =
301 groupObjectCloud(dynamic_lmks, this->getSensorPose(frame_id).get());
302 if (object_clouds.size() == 0) {
303 // TODO: why does this happen so much!!!
304 VLOG(20) << "Cannot collect object clouds from dynamic landmarks of "
305 << object_id << " and frame " << frame_id << "!! "
306 << " # Dynamic lmks in the map for this object at this frame was "
307 << dynamic_lmks.size(); //<< " but reocrded lmks was " <<
308 // dynamic_landmarks.size();
309 return {gtsam::Point3{}, false};
310 }
311 CHECK_EQ(object_clouds.size(), 1);
312 CHECK(object_clouds.exists(object_id));
313
314 const auto dynamic_point_cloud = object_clouds.at(object_id);
315 pcl::PointXYZ centroid;
316 pcl::computeCentroid(dynamic_point_cloud, centroid);
317 // TODO: outlier reject?
318 gtsam::Point3 translation = pclPointToGtsam(centroid);
319 return {translation, true};
320}

322template <class MAP>
323EstimateMap<ObjectId, std::pair<Motion3, gtsam::Pose3>>
324Accessor<MAP>::getRequestedObjectPosePair(FrameId motion_frame_id,
325 FrameId pose_frame_id) const {
326 const MotionEstimateMap motions_k = this->getObjectMotions(motion_frame_id);
327 const EstimateMap<ObjectId, gtsam::Pose3> poses_k =
328 this->getObjectPoses(pose_frame_id);
329
330 EstimateMap<ObjectId, std::pair<Motion3, gtsam::Pose3>> result;
331 // collect common objects
332 for (const auto& [object_id, _] : motions_k) {
333 if (poses_k.exists(object_id)) {
334 auto motion_reference_values = motions_k.at(object_id);
335 auto pose_reference_values = poses_k.at(object_id);
336
337 CHECK_EQ(motion_reference_values.frame_, pose_reference_values.frame_);
338
339 ReferenceFrameValue<std::pair<Motion3, gtsam::Pose3>> estimate(
340 std::make_pair(motion_reference_values.estimate_,
341 pose_reference_values.estimate_),
342 motion_reference_values.frame_);
343 result.insert2(object_id, estimate);
344 }
345 }
346 return result;
347}
