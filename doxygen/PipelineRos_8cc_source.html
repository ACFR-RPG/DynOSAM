60dyno::DataProvider::Ptr DynoNode::createDataProvider() {
61 const bool online =
62 ParameterConstructor(this, "online", false)
63 .description("If the online DataProvider should be used")
64 .finish()
65 .get<bool>();
66
67 if (online) {
68 RCLCPP_INFO_STREAM(
69 this->get_logger(),
70 "Online DataProvider selected. Waiting for ROS topics...");
71 OnlineDataProviderRosParams online_params;
72 online_params.wait_for_camera_params =
73 ParameterConstructor(this, "wait_for_camera_params",
74 online_params.wait_for_camera_params)
75 .description(
76 "If the online DataProvider should wait for the camera params "
77 "on a ROS topic!")
78 .finish()
79 .get<bool>();
80 online_params.camera_params_timeout =
81 ParameterConstructor(this, "camera_params_timeout",
82 online_params.camera_params_timeout)
83 .description(
84 "When waiting for camera params, how long the online "
85 "DataProvider should wait before time out (ms)")
86 .finish()
87 .get<int>();
88 return std::make_shared<OnlineDataProviderRos>(
89 this->create_sub_node("dataprovider"), online_params);
90 }
91
92 auto params_path = getParamsPath();
93 auto dataset_path = getDatasetPath();
94 auto dyno_params = getDynoParams();
95
96 RCLCPP_INFO_STREAM(this->get_logger(),
97 "Loading dataset from: " << dataset_path);
98
99 dyno::DataProvider::Ptr data_loader = dyno::DataProviderFactory::Create(
100 dataset_path, params_path,
101 static_cast<dyno::DatasetType>(dyno_params.dataProviderType()));
102 RCLCPP_INFO_STREAM(this->get_logger(), "Constructed data loader");
103 return data_loader;
104}

106std::string DynoNode::searchForPathWithParams(const std::string& param_name,
107 const std::string& default_path,
108 const std::string& description) {
109 // check if we've alrady declared this param
110 // use non-default version so that ParameterConstructor throws exception if no
111 // parameter is provided on the param server
112 const std::string path = ParameterConstructor(this, param_name)
113 .description(description)
114 .finish()
115 .get<std::string>();
116 throwExceptionIfPathInvalid(path);
117 return path;
118}

124void DynoPipelineManagerRos::initalisePipeline() {
125 RCLCPP_INFO_STREAM(this->get_logger(), "Starting DynoPipelineManagerRos");
126
127 auto params = getDynoParams();
128
129 // setup display params
130 DisplayParams display_params;
131 display_params.camera_frame_id =
132 ParameterConstructor(this, "camera_frame_id",
133 display_params.camera_frame_id)
134 .description(
135 "ROS frame id for the camera (ie. the measured odometry)")
136 .finish()
137 .get<std::string>();
138 display_params.world_frame_id =
139 ParameterConstructor(this, "world_frame_id",
140 display_params.world_frame_id)
141 .description("ROS frame id for the static workd frame (ie. odometry)")
142 .finish()
143 .get<std::string>();
144
145 auto frontend_display = std::make_shared<dyno::FrontendDisplayRos>(
146 display_params, this->create_sub_node("frontend"));
147 auto backend_display = std::make_shared<dyno::BackendDisplayRos>(
148 display_params, this->create_sub_node("backend"));
149
150 auto data_loader = createDataProvider();
151 pipeline_ = std::make_unique<DynoPipelineManager>(
152 params, data_loader, frontend_display, backend_display);
153}
