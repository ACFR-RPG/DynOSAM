66std::string DynoNode::searchForPathWithParams(const std::string& param_name, const std::string& default_path, const std::string& description) {
67 // check if we've alrady declared this param
68 if (!this->has_parameter(param_name)) {
69 auto param_desc = rcl_interfaces::msg::ParameterDescriptor{};
70 param_desc.description = description;
71
72 this->declare_parameter(param_name, default_path, param_desc);
73 }
74
75 std::string path;
76 if(this->get_parameter(param_name, path)) {
77 return path;
78 }
79 else {
80 throw std::runtime_error("ROS param `" + param_name + "` expected but not found");
81 }
82
83 throwExceptionIfPathInvalid(path);
84 return path;
85}
