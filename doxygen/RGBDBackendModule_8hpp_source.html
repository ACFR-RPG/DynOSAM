114 // previous_trigger_frame starts at overlap
115 SlidingWindow(const int window, const int overlap)
116 : sliding_window(window),
117 overlap_size(overlap),
118 previous_trigger_frame(overlap) {}
119
120 Result check(FrameId frame_k) {
121 if (first_frame == -1) {
122 first_frame = static_cast<int>(frame_k);
123 CHECK_GE(first_frame, 0);
124 }
125
126 auto frame = static_cast<int>(frame_k) - first_frame;
127 const bool condition =
128 (previous_trigger_frame - (frame - sliding_window)) == overlap_size;
129 if (condition) {
130 previous_trigger_frame = frame;
131 }
132
133 Result result;
134 result.condition = condition;
135 result.ending_frame = frame_k;
136 int starting_frame = frame_k - sliding_window;
137
138 // some logic checks
139 if (condition) {
140 CHECK_GE(starting_frame, first_frame);
141 }
142 result.starting_frame = static_cast<FrameId>(starting_frame);
143 return result;
144 }
145 };
