39std::vector<int> findMovingObject(const GroundTruthInputPacket& prev_packet_vector, const GroundTruthInputPacket& curr_packet_vector, double tol_m = 0.4) {
40 //find object in previous
41 std::vector<int> moving_labels;
42
43 for(const ObjectPoseGT& object_pose_gt : curr_packet_vector.object_poses_) {
44 const ObjectId object_id = object_pose_gt.object_id_;
45 auto it = std::find_if(prev_packet_vector.object_poses_.begin(), prev_packet_vector.object_poses_.end(),
46 [=](const ObjectPoseGT& gt_object) { return gt_object.object_id_ == object_id; });
47 if(it == prev_packet_vector.object_poses_.end()) {
48 continue;
49 }
50
51 const ObjectPoseGT& object_pose_gt_in_previous = *it;
52 const gtsam::Pose3 prev_L_world = prev_packet_vector.X_world_ * object_pose_gt_in_previous.L_camera_;
53
54 const gtsam::Pose3 curr_L_world = curr_packet_vector.X_world_ * object_pose_gt.L_camera_;
55
56 gtsam::Vector3 t_diff = curr_L_world.translation() - prev_L_world.translation();
57
58 // const gtsam::Pose3 pose_diff = prev_L_world.inverse() * curr_L_world;
59 // const gtsam::Point3& translation_error = pose_diff.translation();
60 // L2 norm - ie. magnitude
61 double t_error = t_diff.norm();
62
63 if(t_error > tol_m) {
64 LOG(INFO) << "Object " << object_id << " is moving!";
65 moving_labels.push_back(object_id);
66 }
67 else {
68 LOG(INFO) << "Object " << object_id << " is NOT moving!";
69 }
70 }
71
72 return moving_labels;
73}

76cv::Mat constructMotionMask(const cv::Mat& instance_mask, const std::vector<int>& moving_labels) {
77 cv::Mat motion_mask;
78 instance_mask.copyTo(motion_mask);
79
80 for (int i = 0; i < motion_mask.rows; i++)
81 {
82 for (int j = 0; j < motion_mask.cols; j++)
83 {
84
85 int label = motion_mask.at<int>(i, j);
86 if(label == 0) {
87 continue;
88 }
89
90 //check if label is in moving labels. if not, make zero!
91 auto it = std::find(moving_labels.begin(), moving_labels.end(), label);
92 if(it == moving_labels.end()) {
93 motion_mask.at<int>(i, j) = 0;
94 }
95 }
96 }
97 return motion_mask;
98}
