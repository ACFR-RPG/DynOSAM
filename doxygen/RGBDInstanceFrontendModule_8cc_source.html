114FrontendModule::SpinReturn RGBDInstanceFrontendModule::nominalSpin(
115 FrontendInputPacketBase::ConstPtr input) {
116 ImageContainer::Ptr image_container = input->image_container_;
117
118 const bool has_imu = input->imu_measurements.has_value();
119 const bool has_stereo = image_container->hasRightRgb();
120
121 std::optional<gtsam::Rot3> R_curr_ref;
122 ImuFrontend::PimPtr pim;
123 if (has_imu) {
124 pim = imu_frontend_.preintegrateImuMeasurements(
125 input->imu_measurements.value());
126
127 nav_state_ =
128 pim->predict(previous_nav_state_, gtsam::imuBias::ConstantBias{});
129
130 last_imu_nav_state_update_ = input->getFrameId();
131
132 R_curr_ref =
133 previous_nav_state_.attitude().inverse() * nav_state_.attitude();
134 }
135
136 Frame::Ptr frame = tracker_->track(input->getFrameId(), input->getTimestamp(),
137 *image_container, R_curr_ref);
138
139 Frame::Ptr previous_frame = tracker_->getPreviousFrame();
140 CHECK(previous_frame);
141
142 VLOG(1) << to_string(tracker_->getTrackerInfo());
143
144 {
145 // this will mark some points as invalid if they are out of depth range
146 utils::TimingStatsCollector update_depths_timer("depth_updater");
147 frame->updateDepths();
148 }
149
150 bool stereo_result = false;
151 std::shared_ptr<RGBDCamera> rgbd_camera = camera_->safeGetRGBDCamera();
152
153 if (has_stereo) {
154 CHECK(rgbd_camera) << "Stereo imagery provided but rgbd camera is null!";
155 }
156
157 if (has_stereo) {
158 const cv::Mat& left_rgb = image_container->rgb();
159 const cv::Mat& right_rgb = image_container->rightRgb();
160
161 auto usable_iterator = frame->static_features_.beginUsable();
162 LOG(INFO) << "Counted intliers "
163 << std::distance(usable_iterator.begin(), usable_iterator.end());
164
165 FeaturePtrs stereo_features_1;
166 stereo_result =
167 tracker_->stereoTrack(stereo_features_1, frame->static_features_,
168 left_rgb, right_rgb, rgbd_camera->baseline());
169 }
170
171 // this includes the refine correspondances with joint optical flow
172 // TODO: lots of internal logic around how the actual pose gets predicted.
173 // should streamline this and tell backend how pose was selected!!
174 if (!solveCameraMotion(frame, previous_frame, R_curr_ref)) {
175 LOG(ERROR) << "Could not solve for camera";
176 }
177
178 if (has_stereo && stereo_result) {
179 // need to match aagain after optical flow used to update the keypoints
180 // wow this seems to make a pretty big difference!!
181 const cv::Mat& left_rgb = image_container->rgb();
182 const cv::Mat& right_rgb = image_container->rightRgb();
183 FeaturePtrs stereo_features_2;
184 stereo_result &=
185 tracker_->stereoTrack(stereo_features_2, frame->static_features_,
186 left_rgb, right_rgb, rgbd_camera->baseline());
187 }
188
189 // VERY important calculation
190 const gtsam::Pose3 T_k_1_k =
191 previous_nav_state_.pose().inverse() * frame->T_world_camera_;
192 vo_velocity_ = T_k_1_k;
193
194 // we currently use the frame pose as the nav state - this value can come from
195 // either the VO OR the IMU, depending on the result from the
196 // solveCameraMotion this is only relevant since we dont solve incremental so
197 // the backend is not immediately updating the frontend at which point we can
198 // just use the best estimate in the case of the VO, the nav_state velocity
199 // will be wrong (currently!!)
200 previous_nav_state_ =
201 gtsam::NavState(frame->T_world_camera_, nav_state_.velocity());
202 // previous_nav_state_ = nav_state_;
203
204 if (R_curr_ref) {
205 imu_frontend_.resetIntegration();
206 }
207
208 // if (FLAGS_use_dynamic_track) {
209 // utils::TimingStatsCollector track_dynamic_timer("tracking_dynamic");
210 // vision_tools::trackDynamic(getFrontendParams(), *previous_frame, frame);
211 // }
212
213 const auto [object_motions, object_poses] =
214 object_motion_solver_->solve(frame, previous_frame);
215
216 VisionImuPacket::Ptr vision_imu_packet = std::make_shared<VisionImuPacket>();
217 vision_imu_packet->frameId(frame->getFrameId());
218 vision_imu_packet->timestamp(frame->getTimestamp());
219 vision_imu_packet->pim(pim);
220 vision_imu_packet->groundTruthPacket(input->optional_gt_);
221 fillOutputPacketWithTracks(vision_imu_packet, *frame, T_k_1_k, object_motions,
222 object_poses);
223
224 DebugImagery debug_imagery;
225 debug_imagery.tracking_image =
226 createTrackingImage(frame, previous_frame, object_poses);
227 const ImageContainer& processed_image_container = frame->image_container_;
228 debug_imagery.rgb_viz =
229 ImageType::RGBMono::toRGB(processed_image_container.rgb());
230 debug_imagery.flow_viz =
231 ImageType::OpticalFlow::toRGB(processed_image_container.opticalFlow());
232 debug_imagery.mask_viz = ImageType::MotionMask::toRGB(
233 processed_image_container.objectMotionMask());
234 debug_imagery.depth_viz =
235 ImageType::Depth::toRGB(processed_image_container.depth());
236
237 if (display_queue_) {
238 display_queue_->push(
239 ImageToDisplay("Tracks", debug_imagery.tracking_image));
240
241 // cv::Mat stereo_matches;
242 // if(tracker_->drawStereoMatches(stereo_matches, *frame)){
243 // display_queue_->push(
244 // ImageToDisplay("Stereo Matches", stereo_matches));
245 // }
246 }
247
248 vision_imu_packet->debugImagery(debug_imagery);
249
250 // // const cv::Mat& board_detection_mask =
251 // tracker_->getBoarderDetectionMask(); PointCloudLabelRGB::Ptr
252 // dense_labelled_cloud =
253 // frame->projectToDenseCloud(&board_detection_mask);
254 PointCloudLabelRGB::Ptr dense_labelled_cloud = nullptr;
255
256 // if (FLAGS_save_frontend_json)
257 // output_packet_record_.insert({output->getFrameId(), output});
258
259 sendToFrontendLogger(frame, vision_imu_packet);
260
261 // if (FLAGS_log_projected_masks)
262 // vision_tools::writeOutProjectMaskAndDepthMap(
263 // frame->image_container_.depth(),
264 // frame->image_container_.objectMotionMask(), *frame->getCamera(),
265 // frame->getFrameId());
266
267 return {State::Nominal, vision_imu_packet};
268}

270bool RGBDInstanceFrontendModule::solveCameraMotion(
271 Frame::Ptr frame_k, const Frame::Ptr& frame_k_1,
272 std::optional<gtsam::Rot3> R_curr_ref) {
273 utils::TimingStatsCollector timer("frontend.solve_camera_motion");
274 Pose3SolverResult result;
275
276 const auto& frontend_params = getFrontendParams();
277 if (frontend_params.use_ego_motion_pnp) {
278 result = motion_solver_.geometricOutlierRejection3d2d(frame_k_1, frame_k,
279 R_curr_ref);
280 } else {
281 // TODO: untested
282 LOG(FATAL) << "Not tested";
283 // result = motion_solver_.geometricOutlierRejection3d3d(frame_k_1,frame_k);
284 }
285
286 VLOG(15) << (frontend_params.use_ego_motion_pnp ? "3D2D" : "3D3D")
287 << "camera pose estimate at frame " << frame_k->frame_id_
288 << (result.status == TrackingStatus::VALID ? " success "
289 : " failure ")
290 << ":\n"
291 << "- Tracking Status: " << to_string(result.status) << '\n'
292 << "- Total Correspondences: "
293 << result.inliers.size() + result.outliers.size() << '\n'
294 << "\t- # inliers: " << result.inliers.size() << '\n'
295 << "\t- # outliers: " << result.outliers.size() << '\n';
296
297 // collect all usable tracklets
298 TrackletIds tracklets = frame_k->static_features_.collectTracklets();
299 CHECK_GE(tracklets.size(),
300 result.inliers.size() +
301 result.outliers.size()); // tracklets shoudl be more (or same
302 // as) correspondances as there will
303 // be new points untracked
304 frame_k->static_features_.markOutliers(result.outliers);
305
306 // was 60!
307 if (result.status != TrackingStatus::VALID || result.inliers.size() < 30) {
308 // TODO: fix code structure - nav state should be passed in?
309 // use nav state which we assume is updated by IMU
310 std::stringstream ss;
311 ss << "Number usable static feature < 30 or status is invalid. ";
312
313 // check if we have a nav state update from the IMU (this is also a cheap
314 // way of checking that we HAVE an imu). If we do we can use the nav state
315 // directly to update the current pose as the nav state is the forward
316 // prediction from the IMU
317 if (last_imu_nav_state_update_ == frame_k->getFrameId()) {
318 frame_k->T_world_camera_ = nav_state_.pose();
319 ss << "Nav state was previous updated with IMU. Using predicted pose to "
320 "set camera transform; k"
321 << frame_k->getFrameId();
322 } else {
323 // no IMU for forward prediction, use constant velocity model to propogate
324 // pose expect previous_nav_state_ to always be updated with the best
325 // pose!
326 frame_k->T_world_camera_ = previous_nav_state_.pose() * vo_velocity_;
327 ss << "Nav state has no information from imu. Using constant velocity "
328 "model to propofate pose; k"
329 << frame_k->getFrameId();
330 }
331
332 VLOG(10) << ss.str();
333
334 // if fails should we mark current inliers as outliers?
335
336 // TODO: should almost definitely do this in future, but right now we use
337 // measurements to construct a framenode in the backend so if there are no
338 // measurements we get a frame_node null.... for now... make hack and set
339 // all ages of inliers to 1!!! since we need n measurements in the backend
340 // this will ensure that they dont get added to the
341 // optimisation problem but will get added to the map...
342 for (const auto& inlier : result.inliers) {
343 frame_k->static_features_.getByTrackletId(inlier)->age(1u);
344 }
345 // frame_k->static_features_.markOutliers(result.inliers);
346
347 // for some reason using tracklets to mark all features gives error as a
348 // tracklet id
349 // seems to be not actually in static features. Dont know why
350 // maybe remeber that tracklets (fromc collectTracklets) is actually just
351 // the usable tracklets...?
352 // frame_k->static_features_.markOutliers(tracklets);
353 return false;
354 } else {
355 frame_k->T_world_camera_ = result.best_result;
356
357 if (frontend_params.refine_camera_pose_with_joint_of) {
358 VLOG(10) << "Refining camera pose with joint of";
359 utils::TimingStatsCollector timer(
360 "frontend.solve_camera_motion.of_refine");
361 OpticalFlowAndPoseOptimizer flow_optimizer(
362 frontend_params.object_motion_solver_params.joint_of_params);
363
364 auto flow_opt_result = flow_optimizer.optimizeAndUpdate<CalibrationType>(
365 frame_k_1, frame_k, result.inliers, result.best_result);
366 frame_k->T_world_camera_ = flow_opt_result.best_result.refined_pose;
367 VLOG(15) << "Refined camera pose with optical flow - error before: "
368 << flow_opt_result.error_before.value_or(NaN)
369 << " error_after: " << flow_opt_result.error_after.value_or(NaN);
370 }
371 return true;
372 }

375void RGBDInstanceFrontendModule::fillOutputPacketWithTracks(
376 VisionImuPacket::Ptr vision_imu_packet, const Frame& frame,
377 const gtsam::Pose3& T_k_1_k, const ObjectMotionMap& object_motions,
378 const ObjectPoseMap& object_poses) const {
379 CHECK(vision_imu_packet);
380 const auto frame_id = frame.getFrameId();
381 // construct image tracks
382 const double& static_pixel_sigma =
383 params_.backend_params_.static_pixel_noise_sigma;
384 const double& static_point_sigma =
385 params_.backend_params_.static_point_noise_sigma;
386
387 const double& dynamic_pixel_sigma =
388 params_.backend_params_.dynamic_pixel_noise_sigma;
389 const double& dynamic_point_sigma =
390 params_.backend_params_.dynamic_point_noise_sigma;
391
392 gtsam::Vector2 static_pixel_sigmas;
393 static_pixel_sigmas << static_pixel_sigma, static_pixel_sigma;
394
395 gtsam::Vector2 dynamic_pixel_sigmas;
396 dynamic_pixel_sigmas << dynamic_pixel_sigma, dynamic_pixel_sigma;
397
398 auto& camera = *this->camera_;
399 auto fill_camera_measurements =
400 [&camera](FeatureFilterIterator it,
401 CameraMeasurementStatusVector* measurements, FrameId frame_id,
402 const gtsam::Vector2& pixel_sigmas, double depth_sigma) {
403 std::shared_ptr<RGBDCamera> rgbd_camera = camera.safeGetRGBDCamera();
404
405 for (const Feature::Ptr& f : it) {
406 const TrackletId tracklet_id = f->trackletId();
407 const Keypoint& kp = f->keypoint();
408 const ObjectId object_id = f->objectId();
409 CHECK_EQ(f->objectId(), object_id);
410 CHECK(Feature::IsUsable(f));
411
412 MeasurementWithCovariance<Keypoint> kp_measurement =
413 MeasurementWithCovariance<Keypoint>::FromSigmas(kp, pixel_sigmas);
414 CameraMeasurement camera_measurement(kp_measurement);
415
416 // This can come from either stereo or rgbd
417 if (f->hasDepth()) {
418 // MeasurementWithCovariance<Landmark> landmark_measurement(
419 // // assume sigma_u and sigma_v are identical
420 // vision_tools::backProjectAndCovariance(
421 // *f, camera, pixel_sigmas(0), depth_sigma));
422 // camera_measurement.landmark(landmark_measurement);
423 Landmark landmark;
424 camera.backProject(kp, f->depth(), &landmark);
425
426 gtsam::Vector3 sigmas;
427 sigmas << depth_sigma, depth_sigma, depth_sigma;
428
429 MeasurementWithCovariance<Landmark> landmark_measurement =
430 MeasurementWithCovariance<Landmark>::FromSigmas(landmark,
431 sigmas);
432 camera_measurement.landmark(landmark_measurement);
433 }
434
435 if (f->hasRightKeypoint()) {
436 CHECK(f->hasDepth())
437 << "Right keypoint set for feature but no depth!";
438 MeasurementWithCovariance<Keypoint> right_kp_measurement =
439 MeasurementWithCovariance<Keypoint>::FromSigmas(
440 f->rightKeypoint(), pixel_sigmas);
441 camera_measurement.rightKeypoint(right_kp_measurement);
442 }
443 // no right keypoint and has rgbd camera and has depth, project
444 // keypoint into right camera
445 else if (rgbd_camera && f->hasDepth()) {
446 bool right_projection_result = rgbd_camera->projectRight(f);
447 if (!right_projection_result) {
448 // TODO: for now mark as outlier and ignore point
449 f->markOutlier();
450 continue;
451 }
452
453 CHECK(f->hasRightKeypoint());
454 MeasurementWithCovariance<Keypoint> right_kp_measurement =
455 MeasurementWithCovariance<Keypoint>::FromSigmas(
456 f->rightKeypoint(), pixel_sigmas);
457 camera_measurement.rightKeypoint(right_kp_measurement);
458 }
459
460 if (f->keypointType() == KeyPointType::STATIC) {
461 CHECK_EQ(object_id, background_label);
462 } else {
463 CHECK_NE(object_id, background_label);
464 }
465
466 measurements->push_back(
467 CameraMeasurementStatus(camera_measurement, frame_id, tracklet_id,
468 object_id, ReferenceFrame::LOCAL));
469 }
470 };
471
472 // TODO: fill ttracking status?
473 VisionImuPacket::CameraTracks camera_tracks;
474 auto* static_measurements = &camera_tracks.measurements;
475 fill_camera_measurements(frame.usableStaticFeaturesBegin(),
476 static_measurements, frame_id, static_pixel_sigmas,
477 static_point_sigma);
478 camera_tracks.X_W_k = frame.getPose();
479 camera_tracks.T_k_1_k = T_k_1_k;
480 vision_imu_packet->cameraTracks(camera_tracks);
481
482 // First collect all dynamic measurements then split them by object
483 // This is a bit silly
484 CameraMeasurementStatusVector dynamic_measurements;
485 fill_camera_measurements(frame.usableDynamicFeaturesBegin(),
486 &dynamic_measurements, frame_id,
487 dynamic_pixel_sigmas, dynamic_point_sigma);
488
489 VisionImuPacket::ObjectTrackMap object_tracks;
490 // motions in this frame (ie. new motions!!)
491 MotionEstimateMap motion_estimates = object_motions.toEstimateMap(frame_id);
492 auto pose_estimates = object_poses.toEstimateMap(frame_id);
493
494 // fill object tracks based on valid motions
495 for (const auto& [object_id, motion_reference_estimate] : motion_estimates) {
496 CHECK(pose_estimates.exists(object_id))
497 << "Object pose missing " << info_string(frame_id, object_id)
498 << " but frontend motion available";
499 const auto& L_W_k = pose_estimates.at(object_id);
500
501 VisionImuPacket::ObjectTracks object_track;
502 object_track.H_W_k_1_k = motion_reference_estimate;
503 object_track.L_W_k = L_W_k;
504 object_tracks.insert2(object_id, object_track);
505 }
506
507 for (const auto& dm : dynamic_measurements) {
508 const auto& object_id = dm.objectId();
509 // throw out features detected on objects where the tracking failed
510 if (object_tracks.exists(object_id)) {
511 VisionImuPacket::ObjectTracks& object_track = object_tracks.at(object_id);
512 object_track.measurements.push_back(dm);
513 }
514 }
515 vision_imu_packet->objectTracks(object_tracks);
516}

535cv::Mat RGBDInstanceFrontendModule::createTrackingImage(
536 const Frame::Ptr& frame_k, const Frame::Ptr& frame_k_1,
537 const ObjectPoseMap& object_poses) const {
538 cv::Mat tracking_image = tracker_->computeImageTracks(
539 *frame_k_1, *frame_k, getFrontendParams().image_tracks_vis_params);
540
541 const auto& camera_params = camera_->getParams();
542 const auto& K = camera_params.getCameraMatrix();
543 const auto& D = camera_params.getDistortionCoeffs();
544
545 const gtsam::Pose3& X_k = frame_k->getPose();
546
547 // poses are expected to be in the world frame
548 gtsam::FastMap<ObjectId, gtsam::Pose3> poses_k_map =
549 object_poses.collectByFrame(frame_k->getFrameId());
550 std::vector<gtsam::Pose3> poses_k_vec;
551 std::transform(poses_k_map.begin(), poses_k_map.end(),
552 std::back_inserter(poses_k_vec),
553 [&X_k](const std::pair<ObjectId, gtsam::Pose3>& pair) {
554 // put object pose into the camera frame so it can be
555 // projected into the image
556 return X_k.inverse() * pair.second;
557 });
558
559 // TODO: bring back when visualisation is unified with incremental solver!!
560 // utils::drawObjectPoseAxes(tracking_image, K, D, poses_k_vec);
561 return tracking_image;
562}
