104FrontendModule::SpinReturn RGBDInstanceFrontendModule::nominalSpin(
105 FrontendInputPacketBase::ConstPtr input) {
106 ImageContainer::Ptr image_container = input->image_container_;
107
108 Frame::Ptr frame = nullptr;
109 {
110 utils::TimingStatsCollector tracking_timer("tracking_timer");
111 frame = tracker_->track(input->getFrameId(), input->getTimestamp(),
112 *image_container);
113 }
114 CHECK(frame);
115
116 Frame::Ptr previous_frame = tracker_->getPreviousFrame();
117 CHECK(previous_frame);
118
119 LOG(INFO) << to_string(tracker_->getTrackerInfo());
120
121 {
122 utils::TimingStatsCollector update_depths_timer("depth_updater");
123 frame->updateDepths();
124 }
125 // updates frame->T_world_camera_
126 if (!solveCameraMotion(frame, previous_frame)) {
127 LOG(ERROR) << "Could not solve for camera";
128 }
129
130 if (FLAGS_use_dynamic_track) {
131 // TODO: bring back byte tracker??
132 utils::TimingStatsCollector track_dynamic_timer("tracking_dynamic");
133 vision_tools::trackDynamic(base_params_, *previous_frame, frame);
134 }
135
136 MotionEstimateMap motion_estimates;
137 solveObjectMotions(frame, previous_frame, motion_estimates);
138
139 // update the object_poses trajectory map which will be send to the viz
140 propogateObjectPoses(motion_estimates, frame->getFrameId());
141
142 if (logger_) {
143 auto ground_truths = this->getGroundTruthPackets();
144 logger_->logCameraPose(frame->getFrameId(), frame->getPose(),
145 ground_truths);
146 logger_->logObjectMotion(frame->getFrameId(), motion_estimates,
147 ground_truths);
148 logger_->logTrackingLengthHistogram(frame);
149 logger_->logFrameIdToTimestamp(frame->getFrameId(), frame->getTimestamp());
150 }
151
152 DebugImagery debug_imagery;
153 debug_imagery.tracking_image =
154 tracker_->computeImageTracks(*previous_frame, *frame);
155 if (display_queue_)
156 display_queue_->push(
157 ImageToDisplay("tracks", debug_imagery.tracking_image));
158
159 debug_imagery.detected_bounding_boxes = frame->drawDetectedObjectBoxes();
160
161 const ImageContainer& processed_image_container = frame->image_container_;
162 debug_imagery.rgb_viz = ImageType::RGBMono::toRGB(
163 processed_image_container.get<ImageType::RGBMono>());
164 debug_imagery.flow_viz = ImageType::OpticalFlow::toRGB(
165 processed_image_container.get<ImageType::OpticalFlow>());
166 debug_imagery.mask_viz = ImageType::MotionMask::toRGB(
167 processed_image_container.get<ImageType::MotionMask>());
168 debug_imagery.depth_viz = ImageType::Depth::toRGB(
169 processed_image_container.get<ImageType::Depth>());
170
171 RGBDInstanceOutputPacket::Ptr output =
172 constructOutput(*frame, motion_estimates, frame->T_world_camera_,
173 input->optional_gt_, debug_imagery);
174
175 if (FLAGS_save_frontend_json)
176 output_packet_record_.insert({output->getFrameId(), output});
177
178 if (FLAGS_log_projected_masks)
179 vision_tools::writeOutProjectMaskAndDepthMap(
180 frame->image_container_.get<ImageType::Depth>(),
181 frame->image_container_.get<ImageType::MotionMask>(),
182 *frame->getCamera(), frame->getFrameId());
183
184 if (logger_) {
185 auto ground_truths = this->getGroundTruthPackets();
186 logger_->logPoints(output->getFrameId(), output->T_world_camera_,
187 output->dynamic_landmarks_);
188 // object_poses_ are in frontend module
189 logger_->logObjectPose(output->getFrameId(), object_poses_, ground_truths);
190 logger_->logObjectBbxes(output->getFrameId(), output->getObjectBbxes());
191 }
192 return {State::Nominal, output};
193}

195bool RGBDInstanceFrontendModule::solveCameraMotion(
196 Frame::Ptr frame_k, const Frame::Ptr& frame_k_1) {
197 Pose3SolverResult result;
198 if (base_params_.use_ego_motion_pnp) {
199 result = motion_solver_.geometricOutlierRejection3d2d(frame_k_1, frame_k);
200 } else {
201 // TODO: untested
202 LOG(FATAL) << "Not tested";
203 result = motion_solver_.geometricOutlierRejection3d3d(frame_k_1, frame_k);
204 }
205
206 VLOG(15) << (base_params_.use_ego_motion_pnp ? "3D2D" : "3D3D")
207 << "camera pose estimate at frame " << frame_k->frame_id_
208 << (result.status == TrackingStatus::VALID ? " success "
209 : " failure ")
210 << ":\n"
211 << "- Tracking Status: " << to_string(result.status) << '\n'
212 << "- Total Correspondences: "
213 << result.inliers.size() + result.outliers.size() << '\n'
214 << "\t- # inliers: " << result.inliers.size() << '\n'
215 << "\t- # outliers: " << result.outliers.size() << '\n';
216
217 if (result.status == TrackingStatus::VALID) {
218 frame_k->T_world_camera_ = result.best_result;
219 TrackletIds tracklets = frame_k->static_features_.collectTracklets();
220 CHECK_GE(tracklets.size(),
221 result.inliers.size() +
222 result.outliers.size()); // tracklets shoudl be more (or same
223 // as) correspondances as there will
224 // be new points untracked
225 frame_k->static_features_.markOutliers(result.outliers);
226
227 if (base_params_.refine_camera_pose_with_joint_of) {
228 OpticalFlowAndPoseOptimizer flow_optimizer(
229 base_params_.object_motion_solver_params.joint_of_params);
230
231 auto flow_opt_result = flow_optimizer.optimizeAndUpdate<CalibrationType>(
232 frame_k_1, frame_k, result.inliers, result.best_result);
233 frame_k->T_world_camera_ = flow_opt_result.best_result.refined_pose;
234 VLOG(15) << "Refined camera pose with optical flow - error before: "
235 << flow_opt_result.error_before.value_or(NaN)
236 << " error_after: " << flow_opt_result.error_after.value_or(NaN);
237 }
238 return true;
239 } else {
240 frame_k->T_world_camera_ = gtsam::Pose3::Identity();
241 return false;
242 }

245bool RGBDInstanceFrontendModule::solveObjectMotion(
246 Frame::Ptr frame_k, Frame::Ptr frame_k_1, ObjectId object_id,
247 MotionEstimateMap& motion_estimates) {
248 Pose3SolverResult result;
249 if (base_params_.use_object_motion_pnp) {
250 result = object_motion_solver_.geometricOutlierRejection3d2d(
251 frame_k_1, frame_k, frame_k->T_world_camera_, object_id);
252 } else {
253 result = object_motion_solver_.geometricOutlierRejection3d3d(
254 frame_k_1, frame_k, frame_k->T_world_camera_, object_id);
255 }
256
257 VLOG(15) << (base_params_.use_object_motion_pnp ? "3D2D" : "3D3D")
258 << " object motion estimate " << object_id << " at frame "
259 << frame_k->frame_id_
260 << (result.status == TrackingStatus::VALID ? " success "
261 : " failure ")
262 << ":\n"
263 << "- Tracking Status: " << to_string(result.status) << '\n'
264 << "- Total Correspondences: "
265 << result.inliers.size() + result.outliers.size() << '\n'
266 << "\t- # inliers: " << result.inliers.size() << '\n'
267 << "\t- # outliers: " << result.outliers.size() << '\n';
268 // sanity check
269 // if valid, remove outliers and add to motion estimation
270 if (result.status == TrackingStatus::VALID) {
271 frame_k->dynamic_features_.markOutliers(result.outliers);
272 ReferenceFrameValue<Motion3> estimate(result.best_result,
273 ReferenceFrame::GLOBAL);
274 motion_estimates.insert({object_id, estimate});
275 return true;
276 } else {
277 return false;
278 }

281void RGBDInstanceFrontendModule::solveObjectMotions(
282 Frame::Ptr frame_k, Frame::Ptr frame_k_1,
283 MotionEstimateMap& motion_estimates) {
284 ObjectIds failed_object_tracks;
285
286 // if only 1 object, no point parallelising
287 if (motion_estimates.size() <= 1) {
288 for (const auto& [object_id, observations] :
289 frame_k->object_observations_) {
290 if (!solveObjectMotion(frame_k, frame_k_1, object_id, motion_estimates)) {
291 VLOG(5) << "Could not solve motion for object " << object_id
292 << " from frame " << frame_k_1->getFrameId() << " -> "
293 << frame_k->getFrameId();
294 failed_object_tracks.push_back(object_id);
295 }
296 }
297 } else {
298 std::mutex mutex;
299 // paralleilise the process of each function call.
300 tbb::parallel_for_each(
301 frame_k->object_observations_.begin(),
302 frame_k->object_observations_.end(),
303 [&](const std::pair<ObjectId, DynamicObjectObservation>& pair) {
304 const auto object_id = pair.first;
305 if (!solveObjectMotion(frame_k, frame_k_1, object_id,
306 motion_estimates)) {
307 VLOG(5) << "Could not solve motion for object " << object_id
308 << " from frame " << frame_k_1->getFrameId() << " -> "
309 << frame_k->getFrameId();
310
311 std::lock_guard<std::mutex> lk(mutex);
312 failed_object_tracks.push_back(object_id);
313 }
314 });
315 }
316
318 // does not remove the features etc but stops the object being propogated to
319 // the backend as we loop over the object observations in the constructOutput
320 // function
321 for (auto object_id : failed_object_tracks) {
322 frame_k->object_observations_.erase(object_id);
323 }
324}

326RGBDInstanceOutputPacket::Ptr RGBDInstanceFrontendModule::constructOutput(
327 const Frame& frame, const MotionEstimateMap& estimated_motions,
328 const gtsam::Pose3& T_world_camera,
329 const GroundTruthInputPacket::Optional& gt_packet,
330 const DebugImagery::Optional& debug_imagery) {
331 StatusKeypointVector static_keypoint_measurements;
332 StatusLandmarkVector static_landmarks;
333 for (const Feature::Ptr& f : frame.usableStaticFeaturesBegin()) {
334 const TrackletId tracklet_id = f->trackletId();
335 const Keypoint kp = f->keypoint();
336 CHECK(f->isStatic());
337 CHECK(Feature::IsUsable(f));
338
339 // dont include features that have only been seen once as we havent had a
340 // chance to validate it yet
341 if (f->age() < 1) {
342 continue;
343 }
344
345 MeasurementWithCovariance<Keypoint> kp_measurement(kp);
346 MeasurementWithCovariance<Landmark> landmark_measurement(
347 vision_tools::backProjectAndCovariance(*f, *camera_, 0.2, 0.1));
348
349 static_keypoint_measurements.push_back(KeypointStatus::StaticInLocal(
350 kp_measurement, frame.getFrameId(), tracklet_id));
351
352 static_landmarks.push_back(LandmarkStatus::StaticInLocal(
353 landmark_measurement, frame.getFrameId(), tracklet_id));
354 }
355
356 StatusKeypointVector dynamic_keypoint_measurements;
357 StatusLandmarkVector dynamic_landmarks;
358 for (const auto& [object_id, obs] : frame.object_observations_) {
359 CHECK_EQ(object_id, obs.instance_label_);
360 // TODO: add back in?
361 // CHECK(obs.marked_as_moving_);
362
363 for (const TrackletId tracklet : obs.object_features_) {
364 if (frame.isFeatureUsable(tracklet)) {
365 const Feature::Ptr f = frame.at(tracklet);
366 CHECK(!f->isStatic());
367 CHECK_EQ(f->objectId(), object_id);
368
369 // dont include features that have only been seen once as we havent had
370 // a chance to validate it yet
371 if (f->age() < 1) {
372 continue;
373 }
374
375 const TrackletId tracklet_id = f->trackletId();
376 const Keypoint kp = f->keypoint();
377
378 MeasurementWithCovariance<Keypoint> kp_measurement(kp);
379 MeasurementWithCovariance<Landmark> landmark_measurement(
380 vision_tools::backProjectAndCovariance(*f, *camera_, 0.2, 0.1));
381
382 dynamic_keypoint_measurements.push_back(KeypointStatus::DynamicInLocal(
383 kp_measurement, frame.frame_id_, tracklet_id, object_id));
384
385 dynamic_landmarks.push_back(LandmarkStatus::DynamicInLocal(
386 landmark_measurement, frame.frame_id_, tracklet_id, object_id));
387 }
388 }
389 }
390
391 // update trajectory of camera poses to be visualised by the frontend viz
392 // module
393 camera_poses_.push_back(T_world_camera);
394
395 return std::make_shared<RGBDInstanceOutputPacket>(
396 static_keypoint_measurements, dynamic_keypoint_measurements,
397 static_landmarks, dynamic_landmarks, T_world_camera, frame.timestamp_,
398 frame.frame_id_, estimated_motions, object_poses_, camera_poses_, camera_,
399 gt_packet, debug_imagery);
400}

402void RGBDInstanceFrontendModule::propogateObjectPoses(
403 const MotionEstimateMap& motion_estimates, FrameId frame_id) {
404 gtsam::Point3Vector object_centroids_k_1, object_centroids_k;
405
406 for (const auto& [object_id, motion_estimate] : motion_estimates) {
407 const auto frame_k_1 = tracker_->getPreviousFrame();
408 const auto frame_k = tracker_->getCurrentFrame();
409
410 auto object_points = FeatureFilterIterator(
411 const_cast<FeatureContainer&>(frame_k_1->dynamic_features_),
412 [object_id, &frame_k](const Feature::Ptr& f) -> bool {
413 return Feature::IsUsable(f) && f->objectId() == object_id &&
414 frame_k->exists(f->trackletId()) &&
415 frame_k->isFeatureUsable(f->trackletId());
416 });
417
418 gtsam::Point3 centroid_k_1(0, 0, 0);
419 gtsam::Point3 centroid_k(0, 0, 0);
420 size_t count = 0;
421 for (const auto& feature : object_points) {
422 gtsam::Point3 lmk_k_1 =
423 frame_k_1->backProjectToCamera(feature->trackletId());
424 centroid_k_1 += lmk_k_1;
425
426 gtsam::Point3 lmk_k = frame_k->backProjectToCamera(feature->trackletId());
427 centroid_k += lmk_k;
428
429 count++;
430 }
431
432 centroid_k_1 /= count;
433 centroid_k /= count;
434
435 centroid_k_1 = frame_k_1->getPose() * centroid_k_1;
436 centroid_k = frame_k->getPose() * centroid_k;
437
438 object_centroids_k_1.push_back(centroid_k_1);
439 object_centroids_k.push_back(centroid_k);
440 }
441
442 if (FLAGS_init_object_pose_from_gt) {
443 dyno::propogateObjectPoses(object_poses_, motion_estimates,
444 object_centroids_k_1, object_centroids_k,
445 frame_id, getGroundTruthPackets());
446 } else {
447 dyno::propogateObjectPoses(object_poses_, motion_estimates,
448 object_centroids_k_1, object_centroids_k,
449 frame_id);
450 }
451}

453TrackingInputImages RGBDInstanceFrontendModule::constructTrackingImages(
454 const ImageContainer::Ptr image_container) {
455 // if we only have instance semgentation (not motion) then we need to make a
456 // motion mask out of the semantic mask we cannot do this for the first frame
457 // so we will just treat the semantic mask and the motion mask and then
458 // subsequently elimate non-moving objects later on
459 TrackingInputImages tracking_images;
460 if (image_container->hasSemanticMask()) {
461 CHECK(!image_container->hasMotionMask());
462 // TODO: some bug when going from semantic mask to motion mask as motion
463 // mask is empty in the tracker after this process!!! its becuase we dont
464 // actually use the tracking_images!!
465 auto intermediate_tracking_images =
466 image_container->makeSubset<ImageType::RGBMono, ImageType::OpticalFlow,
467 ImageType::SemanticMask>();
468 tracking_images = TrackingInputImages(
469 intermediate_tracking_images.getImageWrapper<ImageType::RGBMono>(),
470 intermediate_tracking_images.getImageWrapper<ImageType::OpticalFlow>(),
471 ImageWrapper<ImageType::MotionMask>(
472 intermediate_tracking_images.get<ImageType::SemanticMask>()));
473 } else {
474 tracking_images =
475 image_container->makeSubset<ImageType::RGBMono, ImageType::OpticalFlow,
476 ImageType::MotionMask>();
477 }
478 return tracking_images;
479}
