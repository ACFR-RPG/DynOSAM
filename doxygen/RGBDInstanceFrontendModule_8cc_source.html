115FrontendModule::SpinReturn RGBDInstanceFrontendModule::nominalSpin(
116 FrontendInputPacketBase::ConstPtr input) {
117 ImageContainer::Ptr image_container = input->image_container_;
118
119 const bool has_imu = input->imu_measurements.has_value();
120 const bool has_stereo = image_container->hasRightRgb();
121
122 std::optional<gtsam::Rot3> R_curr_ref;
123 ImuFrontend::PimPtr pim;
124 if (has_imu) {
125 pim = imu_frontend_.preintegrateImuMeasurements(
126 input->imu_measurements.value());
127
128 nav_state_ =
129 pim->predict(previous_nav_state_, gtsam::imuBias::ConstantBias{});
130
131 last_imu_nav_state_update_ = input->getFrameId();
132
133 R_curr_ref =
134 previous_nav_state_.attitude().inverse() * nav_state_.attitude();
135 }
136
137 std::set<ObjectId> keyframed_objects;
138 Frame::Ptr frame =
139 tracker_->track(input->getFrameId(), input->getTimestamp(),
140 *image_container, keyframed_objects, R_curr_ref);
141
142 Frame::Ptr previous_frame = tracker_->getPreviousFrame();
143 CHECK(previous_frame);
144
145 VLOG(1) << to_string(tracker_->getTrackerInfo());
146
147 {
148 utils::TimingStatsCollector update_depths_timer("depth_updater");
149 frame->updateDepths();
150 }
151
152 bool stereo_result = false;
153 static const double base_line = 0.05; // VIODE
154 // static const double base_line = 0.12; // ZED
155
156 if (has_stereo) {
157 const cv::Mat& left_rgb = image_container->rgb();
158 const cv::Mat& right_rgb = image_container->rightRgb();
159
160 FeaturePtrs stereo_features_1;
161 stereo_result =
162 tracker_->stereoTrack(stereo_features_1, frame->static_features_,
163 left_rgb, right_rgb, base_line);
164 }
165
166 // this includes the refine correspondances with joint optical flow
167 // TODO: lots of internal logic around how the actual pose gets predicted.
168 // should streamline this and tell backend how pose was selected!!
169 if (!solveCameraMotion(frame, previous_frame, R_curr_ref)) {
170 LOG(ERROR) << "Could not solve for camera";
171 }
172
173 if (has_stereo && stereo_result) {
174 // need to match aagain after optical flow used to update the keypoints
175 // wow this seems to make a pretty big difference!!
176 const cv::Mat& left_rgb = image_container->rgb();
177 const cv::Mat& right_rgb = image_container->rightRgb();
178 FeaturePtrs stereo_features_2;
179 stereo_result &=
180 tracker_->stereoTrack(stereo_features_2, frame->static_features_,
181 left_rgb, right_rgb, base_line);
182 }
183
184 // VERY important calculation
185 const gtsam::Pose3 T_k_1_k =
186 previous_nav_state_.pose().inverse() * frame->T_world_camera_;
187
188 // we currently use the frame pose as the nav state - this value can come from
189 // either the VO OR the IMU, depending on the result from the
190 // solveCameraMotion this is only relevant since we dont solve incremental so
191 // the backend is not immediately updating the frontend at which point we can
192 // just use the best estimate in the case of the VO, the nav_state velocity
193 // will be wrong (currently!!)
194 previous_nav_state_ =
195 gtsam::NavState(frame->T_world_camera_, nav_state_.velocity());
196
197 if (R_curr_ref) {
198 imu_frontend_.resetIntegration();
199 }
200
201 if (FLAGS_use_dynamic_track) {
202 utils::TimingStatsCollector track_dynamic_timer("tracking_dynamic");
203 vision_tools::trackDynamic(base_params_, *previous_frame, frame);
204 }
205
206 ObjectMotionMap object_motions;
207 ObjectPoseMap object_poses;
208 // MotionEstimateMap motion_estimates;
209 std::tie(object_motions, object_poses) =
210 object_motion_solver_->solve(frame, previous_frame);
211
212 // motions in this frame (ie. new motions!!)
213 MotionEstimateMap motion_estimates =
214 object_motions.toEstimateMap(frame->getFrameId());
215
216 // const cv::Mat& board_detection_mask = tracker_->getBoarderDetectionMask();
217 // PointCloudLabelRGB::Ptr dense_labelled_cloud =
218 // frame->projectToDenseCloud(&board_detection_mask);
219 PointCloudLabelRGB::Ptr dense_labelled_cloud = nullptr;
220
221 if (logger_) {
222 auto ground_truths = this->shared_module_info.getGroundTruthPackets();
223 logger_->logCameraPose(frame->getFrameId(), frame->getPose(),
224 ground_truths);
225 // TODO: this is now wrong as we dont log all the corrected motions!!!!
226 logger_->logObjectMotion(frame->getFrameId(), motion_estimates,
227 ground_truths);
228 logger_->logTrackingLengthHistogram(frame);
229 logger_->logFrameIdToTimestamp(frame->getFrameId(), frame->getTimestamp());
230 }
231
232 DebugImagery debug_imagery;
233 debug_imagery.tracking_image =
234 createTrackingImage(frame, previous_frame, object_poses);
235
236 if (display_queue_)
237 display_queue_->push(
238 ImageToDisplay("Tracks", debug_imagery.tracking_image));
239
240 debug_imagery.detected_bounding_boxes = frame->drawDetectedObjectBoxes();
241
242 const ImageContainer& processed_image_container = frame->image_container_;
243 debug_imagery.rgb_viz =
244 ImageType::RGBMono::toRGB(processed_image_container.rgb());
245 debug_imagery.flow_viz =
246 ImageType::OpticalFlow::toRGB(processed_image_container.opticalFlow());
247 debug_imagery.mask_viz = ImageType::MotionMask::toRGB(
248 processed_image_container.objectMotionMask());
249 debug_imagery.depth_viz =
250 ImageType::Depth::toRGB(processed_image_container.depth());
251
252 LOG(INFO) << "Done debug imagery";
253
254 RGBDInstanceOutputPacket::Ptr output = constructOutput(
255 *frame, object_motions, object_poses, frame->T_world_camera_,
256 input->optional_gt_, debug_imagery, dense_labelled_cloud);
257
258 output->pim_ = pim;
259 output->T_k_1_k_ = T_k_1_k;
260 vo_velocity_ = T_k_1_k;
261 // output->is_keyframe_ = keyframed_objects;
262
263 if (FLAGS_save_frontend_json)
264 output_packet_record_.insert({output->getFrameId(), output});
265
266 if (FLAGS_log_projected_masks)
267 vision_tools::writeOutProjectMaskAndDepthMap(
268 frame->image_container_.depth(),
269 frame->image_container_.objectMotionMask(), *frame->getCamera(),
270 frame->getFrameId());
271
272 if (logger_) {
273 auto ground_truths = this->shared_module_info.getGroundTruthPackets();
274 logger_->logPoints(output->getFrameId(), output->T_world_camera_,
275 output->dynamic_landmarks_);
276 // object_poses_ are in frontend module
277 logger_->logObjectPose(output->getFrameId(), object_poses, ground_truths);
278 logger_->logObjectBbxes(output->getFrameId(), output->getObjectBbxes());
279 }
280 return {State::Nominal, output};
281}

283bool RGBDInstanceFrontendModule::solveCameraMotion(
284 Frame::Ptr frame_k, const Frame::Ptr& frame_k_1,
285 std::optional<gtsam::Rot3> R_curr_ref) {
286 utils::TimingStatsCollector timer("frontend.solve_camera_motion");
287 Pose3SolverResult result;
288 if (base_params_.use_ego_motion_pnp) {
289 result = motion_solver_.geometricOutlierRejection3d2d(frame_k_1, frame_k,
290 R_curr_ref);
291 } else {
292 // TODO: untested
293 LOG(FATAL) << "Not tested";
294 // result = motion_solver_.geometricOutlierRejection3d3d(frame_k_1,
295 // frame_k);
296 }
297
298 VLOG(15) << (base_params_.use_ego_motion_pnp ? "3D2D" : "3D3D")
299 << "camera pose estimate at frame " << frame_k->frame_id_
300 << (result.status == TrackingStatus::VALID ? " success "
301 : " failure ")
302 << ":\n"
303 << "- Tracking Status: " << to_string(result.status) << '\n'
304 << "- Total Correspondences: "
305 << result.inliers.size() + result.outliers.size() << '\n'
306 << "\t- # inliers: " << result.inliers.size() << '\n'
307 << "\t- # outliers: " << result.outliers.size() << '\n';
308
309 // collect all usable tracklets
310 TrackletIds tracklets = frame_k->static_features_.collectTracklets();
311 CHECK_GE(tracklets.size(),
312 result.inliers.size() +
313 result.outliers.size()); // tracklets shoudl be more (or same
314 // as) correspondances as there will
315 // be new points untracked
316 frame_k->static_features_.markOutliers(result.outliers);
317
318 // was 60!
319 if (result.status != TrackingStatus::VALID || result.inliers.size() < 30) {
320 // TODO: fix code structure - nav state should be passed in?
321 // use nav state which we assume is updated by IMU
322 std::stringstream ss;
323 ss << "Number usable static feature < 30 or status is invalid. ";
324
325 // check if we have a nav state update from the IMU (this is also a cheap
326 // way of checking that we HAVE an imu). If we do we can use the nav state
327 // directly to update the current pose as the nav state is the forward
328 // prediction from the IMU
329 if (last_imu_nav_state_update_ == frame_k_1->getFrameId()) {
330 frame_k->T_world_camera_ = nav_state_.pose();
331 ss << "Nav state was previous updated with IMU. Using predicted pose to "
332 "set camera transform; k"
333 << frame_k->getFrameId();
334 } else {
335 // no IMU for forward prediction, use constant velocity model to propogate
336 // pose expect previous_nav_state_ to always be updated with the best
337 // pose!
338 frame_k->T_world_camera_ = previous_nav_state_.pose() * vo_velocity_;
339 ss << "Nav state has no information from imu. Using constant velocity "
340 "model to propofate pose; k"
341 << frame_k->getFrameId();
342 }
343
344 VLOG(10) << ss.str();
345
346 // if fails should we mark current inliers as outliers?
347
348 // TODO: should almost definitely do this in future, but right now we use
349 // measurements to construct a framenode in the backend so if there are no
350 // measurements we get a frame_node null.... for now... make hack and set
351 // all ages of inliers to 1!!! since we need n measurements in the backend
352 // this will ensure that they dont get added to the
353 // optimisation problem but will get added to the map...
354 for (const auto& inlier : result.inliers) {
355 frame_k->static_features_.getByTrackletId(inlier)->age(1u);
356 }
357 // frame_k->static_features_.markOutliers(result.inliers);
358
359 // for some reason using tracklets to mark all features gives error as a
360 // tracklet id
361 // seems to be not actually in static features. Dont know why
362 // maybe remeber that tracklets (fromc collectTracklets) is actually just
363 // the usable tracklets...?
364 // frame_k->static_features_.markOutliers(tracklets);
365 return false;
366 } else {
367 frame_k->T_world_camera_ = result.best_result;
368
369 if (base_params_.refine_camera_pose_with_joint_of) {
370 VLOG(10) << "Refining camera pose with joint of";
371 OpticalFlowAndPoseOptimizer flow_optimizer(
372 base_params_.object_motion_solver_params.joint_of_params);
373
374 auto flow_opt_result = flow_optimizer.optimizeAndUpdate<CalibrationType>(
375 frame_k_1, frame_k, result.inliers, result.best_result);
376 frame_k->T_world_camera_ = flow_opt_result.best_result.refined_pose;
377 VLOG(15) << "Refined camera pose with optical flow - error before: "
378 << flow_opt_result.error_before.value_or(NaN)
379 << " error_after: " << flow_opt_result.error_after.value_or(NaN);
380 }
381 return true;
382 }

385RGBDInstanceOutputPacket::Ptr RGBDInstanceFrontendModule::constructOutput(
386 const Frame& frame, const ObjectMotionMap& object_motions,
387 const ObjectPoseMap& object_poses, const gtsam::Pose3& T_world_camera,
388 const GroundTruthInputPacket::Optional& gt_packet,
389 const DebugImagery::Optional& debug_imagery,
390 const PointCloudLabelRGB::Ptr dense_labelled_cloud) {
391 StatusKeypointVector static_keypoint_measurements;
392 StatusLandmarkVector static_landmarks;
393 for (const Feature::Ptr& f : frame.usableStaticFeaturesBegin()) {
394 const TrackletId tracklet_id = f->trackletId();
395 const Keypoint kp = f->keypoint();
396 CHECK(f->isStatic());
397 CHECK(Feature::IsUsable(f));
398
399 // dont include features that have only been seen once as we havent had a
400 // // chance to validate it yet
401 if (f->age() < 1) {
402 continue;
403 }
404
405 MeasurementWithCovariance<Keypoint> kp_measurement(kp);
406 MeasurementWithCovariance<Landmark> landmark_measurement(
407 vision_tools::backProjectAndCovariance(*f, *camera_, 0.2, 0.1));
408
409 static_keypoint_measurements.push_back(KeypointStatus::StaticInLocal(
410 kp_measurement, frame.getFrameId(), tracklet_id));
411
412 static_landmarks.push_back(LandmarkStatus::StaticInLocal(
413 landmark_measurement, frame.getFrameId(), tracklet_id));
414 }
415
416 StatusKeypointVector dynamic_keypoint_measurements;
417 StatusLandmarkVector dynamic_landmarks;
418 for (const auto& [object_id, obs] : frame.object_observations_) {
419 CHECK_EQ(object_id, obs.instance_label_);
420 // TODO: add back in?
421 // CHECK(obs.marked_as_moving_);
422
423 for (const TrackletId tracklet : obs.object_features_) {
424 if (frame.isFeatureUsable(tracklet)) {
425 const Feature::Ptr f = frame.at(tracklet);
426 CHECK(!f->isStatic());
427 CHECK_EQ(f->objectId(), object_id);
428
429 // dont include features that have only been seen once as we havent had
430 // a chance to validate it yet
431 if (f->age() < 1) {
432 continue;
433 }
434
435 const TrackletId tracklet_id = f->trackletId();
436 const Keypoint kp = f->keypoint();
437
438 MeasurementWithCovariance<Keypoint> kp_measurement(kp);
439 MeasurementWithCovariance<Landmark> landmark_measurement(
440 vision_tools::backProjectAndCovariance(*f, *camera_, 0.2, 0.1));
441
442 dynamic_keypoint_measurements.push_back(KeypointStatus::DynamicInLocal(
443 kp_measurement, frame.frame_id_, tracklet_id, object_id));
444
445 dynamic_landmarks.push_back(LandmarkStatus::DynamicInLocal(
446 landmark_measurement, frame.frame_id_, tracklet_id, object_id));
447 }
448 }
449 }
450
451 // update trajectory of camera poses to be visualised by the frontend viz
452 // module
453 camera_poses_.push_back(T_world_camera);
454
455 // create timestamps for the provided frames
456 // TODO: maybe a more efficient way later!!!
457 const FrameIdTimestampMap& all_timestamp_map =
458 shared_module_info.getTimestampMap();
459 FrameIdTimestampMap involved_timestamps;
460 for (const FrameId& frame_id : object_motions.getInvovledFrameIds()) {
461 CHECK(all_timestamp_map.exists(frame_id));
462 involved_timestamps.insert2(frame_id, all_timestamp_map.at(frame_id));
463 }
464
465 return std::make_shared<RGBDInstanceOutputPacket>(
466 static_keypoint_measurements, dynamic_keypoint_measurements,
467 static_landmarks, dynamic_landmarks, T_world_camera, frame.timestamp_,
468 frame.frame_id_, object_motions, object_poses, camera_poses_, camera_,
469 gt_packet, debug_imagery, dense_labelled_cloud, involved_timestamps);
470}

472cv::Mat RGBDInstanceFrontendModule::createTrackingImage(
473 const Frame::Ptr& frame_k, const Frame::Ptr& frame_k_1,
474 const ObjectPoseMap& object_poses) const {
475 cv::Mat tracking_image = tracker_->computeImageTracks(
476 *frame_k_1, *frame_k, base_params_.image_tracks_vis_params);
477
478 const auto& camera_params = camera_->getParams();
479 const auto& K = camera_params.getCameraMatrix();
480 const auto& D = camera_params.getDistortionCoeffs();
481
482 const gtsam::Pose3& X_k = frame_k->getPose();
483
484 // poses are expected to be in the world frame
485 gtsam::FastMap<ObjectId, gtsam::Pose3> poses_k_map =
486 object_poses.collectByFrame(frame_k->getFrameId());
487 std::vector<gtsam::Pose3> poses_k_vec;
488 std::transform(poses_k_map.begin(), poses_k_map.end(),
489 std::back_inserter(poses_k_vec),
490 [&X_k](const std::pair<ObjectId, gtsam::Pose3>& pair) {
491 // put object pose into the camera frame so it can be
492 // projected into the image
493 return X_k.inverse() * pair.second;
494 });
495
496 // TODO: bring back when visualisation is unified with incremental solver!!
497 // utils::drawObjectPoseAxes(tracking_image, K, D, poses_k_vec);
498 return tracking_image;
499}
