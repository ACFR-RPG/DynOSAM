37cv::Mat ByteObjectTracker::track(const cv::Mat& masks, FrameId frame_id) {
38 const ObjectIds instance_labels = vision_tools::getObjectLabels(masks);
39 std::vector<byte_track::DetectionBase::Ptr> input_detections;
40 for (auto instance_label : instance_labels) {
41 cv::Rect bounding_box;
42 if (vision_tools::findObjectBoundingBox(masks, instance_label,
43 bounding_box)) {
44 input_detections.push_back(
45 std::make_shared<byte_track::Detection>(bounding_box, 1.0));
46 } else {
47 VLOG(30) << "Could not find bb for instance label " << instance_label
48 << " at frame " << frame_id << " and therefore could not track!";
49 }
50 }
51
52 auto object_tracks = impl_tracker_.update(input_detections, frame_id);
53 CHECK_EQ(object_tracks.size(), input_detections.size());
54
55 ObjectIds old_labels = instance_labels;
56 ObjectIds new_labels; // the new tracked labels from the EKF
57 for (const auto& track : object_tracks) {
58 new_labels.push_back(track->get_track_id());
59 }
60
61 // update tracking labels
62 cv::Mat updated_masks;
63 vision_tools::relabelMasks(masks, updated_masks, old_labels, new_labels);
64 return updated_masks;
65}
