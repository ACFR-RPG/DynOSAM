42std::vector<SingleDetectionResult> ByteObjectTracker::track(
43 const std::vector<ObjectDetection>& detections) {
44 std::vector<byte_track::STrackPtr> object_tracks =
45 impl_tracker_.update(detections);
46
47 std::vector<SingleDetectionResult> output;
48 for (size_t i = 0; i < object_tracks.size(); i++) {
49 const auto& object_track = object_tracks.at(i);
50
51 SingleDetectionResult single_result;
52 single_result.bounding_box = fromByeTrackRect(object_track->getRect());
53 single_result.mask = object_track->getMask();
54 // single_result.class_name = detection.class_name;
55 single_result.confidence = object_track->getScore();
56 single_result.object_id = static_cast<ObjectId>(object_track->getTrackId());
57 // //is this the only indicater we should be using!?
58 single_result.well_tracked = object_track->isActivated();
59
60 output.push_back(single_result);
61 }
62
63 return output;
64}
