63const cv_bridge::CvImageConstPtr DataProviderRos::readRosImage(
64 const sensor_msgs::msg::Image::ConstSharedPtr& img_msg) const {
65 CHECK(img_msg);
66 cv_bridge::CvImageConstPtr cv_ptr;
67 try {
68 // important to copy to ensure that memory does not go out of scope (which
69 // it seems to !!!)
70 cv_ptr = cv_bridge::toCvCopy(img_msg);
71 } catch (cv_bridge::Exception& exception) {
72 RCLCPP_FATAL(node_->get_logger(), "cv_bridge exception: %s",
73 exception.what());
74 rclcpp::shutdown();
75 }
76 return cv_ptr;
77}
