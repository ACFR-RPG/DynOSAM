64const cv_bridge::CvImageConstPtr DataProviderRos::readRosImage(
65 const sensor_msgs::msg::Image::ConstSharedPtr& img_msg) const {
66 CHECK(img_msg);
67 cv_bridge::CvImageConstPtr cv_ptr;
68 try {
69 // important to copy to ensure that memory does not go out of scope (which
70 // it seems to !!!)
71 cv_ptr = cv_bridge::toCvCopy(img_msg);
72 } catch (cv_bridge::Exception& exception) {
73 RCLCPP_FATAL(node_->get_logger(), "cv_bridge exception: %s",
74 exception.what());
75 rclcpp::shutdown();
76 }
77 return cv_ptr;
78}
