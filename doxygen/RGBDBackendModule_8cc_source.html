78RGBDBackendModule::RGBDBackendModule(const BackendParams& backend_params,
79 Map3d2d::Ptr map, Camera::Ptr camera,
80 const UpdaterType& updater_type,
81 ImageDisplayQueue* display_queue)
82 : Base(backend_params, map, display_queue),
83 camera_(CHECK_NOTNULL(camera)),
84 updater_type_(updater_type) {
85 CHECK_NOTNULL(map);
86 CHECK_NOTNULL(map_);
87
88 // TODO: functioanlise and streamline with BackendModule
89 noise_models_.static_point_noise = gtsam::noiseModel::Isotropic::Sigma(
90 3u, backend_params.static_point_noise_sigma_);
91 noise_models_.dynamic_point_noise = gtsam::noiseModel::Isotropic::Sigma(
92 3u, backend_params.dynamic_point_noise_sigma_);
93 // set in base!
94 noise_models_.landmark_motion_noise = gtsam::noiseModel::Isotropic::Sigma(
95 3u, backend_params.motion_ternary_factor_noise_sigma_);
96
97 if (backend_params.use_robust_kernals_) {
98 noise_models_.static_point_noise = gtsam::noiseModel::Robust::Create(
99 gtsam::noiseModel::mEstimator::Huber::Create(
100 backend_params.k_huber_3d_points_),
101 noise_models_.static_point_noise);
102
103 noise_models_.dynamic_point_noise = gtsam::noiseModel::Robust::Create(
104 gtsam::noiseModel::mEstimator::Huber::Create(
105 backend_params.k_huber_3d_points_),
106 noise_models_.dynamic_point_noise);
107
108 // TODO: not k_huber_3d_points_ not just used for 3d points
109 noise_models_.landmark_motion_noise = gtsam::noiseModel::Robust::Create(
110 gtsam::noiseModel::mEstimator::Huber::Create(
111 backend_params.k_huber_3d_points_),
112 noise_models_.landmark_motion_noise);
113 }
114
115 CHECK_NOTNULL(noise_models_.static_point_noise);
116 CHECK_NOTNULL(noise_models_.dynamic_point_noise);
117 CHECK_NOTNULL(noise_models_.landmark_motion_noise);
118
119 noise_models_.dynamic_point_noise->print("Dynamic Point Noise");
120 noise_models_.landmark_motion_noise->print("Landmark motion noise");
121 // CHECK(false);
122
123 new_updater_ = std::move(makeUpdater());
124 sliding_window_condition_ = std::make_unique<SlidingWindow>(
125 FLAGS_opt_window_size, FLAGS_opt_window_overlap);
126}

143 RGBDInstanceOutputPacket::ConstPtr input) {
144 const FrameId frame_k = input->getFrameId();
145 first_frame_id_ = frame_k;
146 CHECK_EQ(spin_state_.frame_id, frame_k);
147 LOG(INFO) << "Running backend " << frame_k;
148 CHECK(!(bool)sliding_window_condition_->check(
149 frame_k)); // trigger the check to update the first frame call. Bit
150 // gross!
151
152 // Pose estimate from the front-end

154 updateMap(T_world_cam_k_frontend, input);
155
156 gtsam::Values new_values;
157 gtsam::NonlinearFactorGraph new_factors;
158
159 new_updater_->setInitialPose(T_world_cam_k_frontend, frame_k, new_values);
160 new_updater_->setInitialPosePrior(T_world_cam_k_frontend, frame_k,
161 new_factors);
162 return {State::Nominal, nullptr};

165RGBDBackendModule::SpinReturn RGBDBackendModule::nominalSpinImpl(
166 RGBDInstanceOutputPacket::ConstPtr input) {
167 const FrameId frame_k = input->getFrameId();
168 LOG(INFO) << "Running backend " << frame_k;
169 CHECK_EQ(spin_state_.frame_id, frame_k);
170
171 // Pose estimate from the front-end
172 gtsam::Pose3 T_world_cam_k_frontend;
173 updateMap(T_world_cam_k_frontend, input);
174
175 gtsam::Values new_values;
176 gtsam::NonlinearFactorGraph new_factors;
177
178 new_updater_->addOdometry(frame_k, T_world_cam_k_frontend, new_values,
179 new_factors);
180
181 UpdateObservationParams update_params;
182 update_params.enable_debug_info = true;
183 update_params.do_backtrack =
184 false; // apparently this is v important for making the results == ICRA
185
186 {
187 LOG(INFO) << "Starting updateStaticObservations";
188 utils::TimingStatsCollector timer("backend.update_static_obs");
189 new_updater_->updateStaticObservations(frame_k, new_values, new_factors,
190 update_params);
191 }
192
193 {
194 LOG(INFO) << "Starting updateDynamicObservations";
195 utils::TimingStatsCollector timer("backend.update_dynamic_obs");
196 new_updater_->updateDynamicObservations(frame_k, new_values, new_factors,
197 update_params);
198 }
199
200 if (FLAGS_use_full_batch_opt) {
201 LOG(INFO) << " full batch frame " << base_params_.full_batch_frame;
202 if (base_params_.full_batch_frame - 1 == (int)frame_k) {
203 LOG(INFO) << " Doing full batch at frame " << frame_k;
204
205 // graph.error(values);
206 gtsam::LevenbergMarquardtParams opt_params;
207 opt_params.verbosity = gtsam::NonlinearOptimizerParams::Verbosity::ERROR;
208
209 const auto theta = new_updater_->getTheta();
210 const auto graph = new_updater_->getGraph();
211 utils::StatsCollector(new_updater_->getFullyQualifiedName() +
212 ".full_batch_opt_num_vars_all")
213 .AddSample(theta.size());
214
215 double error_before = graph.error(theta);
216 utils::TimingStatsCollector timer(new_updater_->getFullyQualifiedName() +
217 ".full_batch_opt");
218
219 gtsam::LevenbergMarquardtOptimizer problem(graph, theta, opt_params);
220 gtsam::Values optimised_values = problem.optimize();
221 double error_after = graph.error(optimised_values);
222
223 utils::StatsCollector(new_updater_->getFullyQualifiedName() +
224 ".inner_iterations")
225 .AddSample(problem.getInnerIterations());
226 utils::StatsCollector(new_updater_->getFullyQualifiedName() +
227 ".iterations")
228 .AddSample(problem.iterations());
229
230 new_updater_->updateTheta(optimised_values);
231 LOG(INFO) << " Error before sliding window: " << error_before
232 << " error after: " << error_after;
233 }
234 } else {
235 double error_before, error_after;
236 gtsam::Values optimised_values;
237 if (buildSlidingWindowOptimisation(frame_k, optimised_values, error_before,
238 error_after)) {
239 LOG(INFO) << "Updating values with opt!";
240 new_updater_->updateTheta(optimised_values);
241 LOG(INFO) << " Error before sliding window: " << error_before
242 << " error after: " << error_after;
243 }
244 }
245
246 auto accessor = new_updater_->accessorFromTheta();
247
248 utils::TimingStatsCollector timer(new_updater_->getFullyQualifiedName() +
249 ".post_update");
250 BackendMetaData backend_info;
251 backend_info.params = base_params_;
252 backend_info.ground_truth_packets = this->getGroundTruthPackets();
253 new_updater_->accessorFromTheta()->postUpdateCallback(
254 backend_info); // force update every time (slow! and just for testing)
255
256 auto backend_output = std::make_shared<BackendOutputPacket>();
257 backend_output->timestamp_ = input->getTimestamp();
258 backend_output->frame_id_ = input->getFrameId();
259 backend_output->T_world_camera_ = accessor->getSensorPose(frame_k).get();
260 backend_output->static_landmarks_ = accessor->getFullStaticMap();
261 backend_output->dynamic_landmarks_ =
262 accessor->getDynamicLandmarkEstimates(frame_k);
263
264 for (FrameId frame_id : map_->getFrameIds()) {
265 backend_output->optimized_poses_.push_back(
266 accessor->getSensorPose(frame_id).get());
267 }
268
269 backend_output->composed_object_poses = accessor->getObjectPoses();
270
271 debug_info_ = DebugInfo();
272
273 return {State::Nominal, backend_output};
274}

276void RGBDBackendModule::updateMap(gtsam::Pose3& T_world_cam,
277 RGBDInstanceOutputPacket::ConstPtr input) {
278 utils::TimingStatsCollector timer("map.update_observations");
279
280 const gtsam::Pose3 T_world_cam_k_frontend = input->T_world_camera_;
281 const FrameId frame_k = input->getFrameId();
282
283 map_->updateObservations(input->collectStaticLandmarkKeypointMeasurements());
284 map_->updateObservations(input->collectDynamicLandmarkKeypointMeasurements());

287 map_->updateObjectMotionMeasurements(frame_k, input->estimated_motions_);
288
289 T_world_cam = T_world_cam_k_frontend;

292std::tuple<gtsam::Values, gtsam::NonlinearFactorGraph>
293RGBDBackendModule::constructGraph(FrameId from_frame, FrameId to_frame,
294 bool set_initial_camera_pose_prior,
295 std::optional<gtsam::Values> initial_theta) {
296 CHECK_LT(from_frame, to_frame);
297 gtsam::Values new_values;
298 gtsam::NonlinearFactorGraph new_factors;
299
300 auto updater = std::move(makeUpdater());
301
302 if (initial_theta) {
303 // update initial linearisation points (could be from a previous
304 // optimisation)
305 // TODO: currently cannot set theta becuase this will be ALL the previous
306 // values and not just the ones in
307 // TODO: for now dont do this as we have to handle covariance/santiy checks
308 // differently as some modules expect values to be new and will check that
309 // a value does not exist yet (becuase it shouldn't in that iteration, but
310 // overall it may!)
311 // updater->setTheta(*initial_theta);
312 }
313
314 UpdateObservationParams update_params;
315 update_params.do_backtrack = false;
316 update_params.enable_debug_info = true;
317
318 UpdateObservationResult results;
319
320 CHECK_GE(from_frame, map_->firstFrameId());
321 CHECK_LE(to_frame, map_->lastFrameId());
322
323 // TODO: pick how new values are going to be used as right now they get
324 // appened internallty to the thera and I think this is very slow so we should
325 // not do this...
326
327 for (auto frame_id = from_frame; frame_id <= to_frame; frame_id++) {
328 LOG(INFO) << "Constructing dynamic graph at frame " << frame_id
329 << " in loop (" << from_frame << " -> " << to_frame << ")";
330
331 // pose estimate from frontend
332 gtsam::Pose3 T_world_camera_k;
333 CHECK(map_->hasInitialSensorPose(frame_id, &T_world_camera_k));
334
335 // if first frame
336 if (frame_id == from_frame) {
337 // add first pose
338 updater->setInitialPose(T_world_camera_k, frame_id, new_values);
339
340 if (set_initial_camera_pose_prior)
341 updater->setInitialPosePrior(T_world_camera_k, frame_id, new_factors);
342 } else {
343 updater->addOdometry(frame_id, T_world_camera_k, new_values, new_factors);
344 // no backtrack
345 results += updater->updateDynamicObservations(frame_id, new_values,
346 new_factors, update_params);
347 }
348 results += updater->updateStaticObservations(frame_id, new_values,
349 new_factors, update_params);
350 }
351
352 return {new_values, new_factors};
353}

355bool RGBDBackendModule::buildSlidingWindowOptimisation(
356 FrameId frame_k, gtsam::Values& optimised_values, double& error_before,
357 double& error_after) {
358 auto condition_result = sliding_window_condition_->check(frame_k);
359 if (condition_result) {
360 const auto start_frame = condition_result.starting_frame;
361 const auto end_frame = condition_result.ending_frame;
362 LOG(INFO) << "Running dynamic slam window on between frames " << start_frame
363 << " - " << end_frame;
364
365 gtsam::Values values;
366 gtsam::NonlinearFactorGraph graph;
367 {
368 utils::TimingStatsCollector timer(new_updater_->getFullyQualifiedName() +
369 ".sliding_window_construction");
370 std::tie(values, graph) = constructGraph(start_frame, end_frame, true,
371 new_updater_->getTheta());
372 LOG(INFO) << " Finished graph construction";
373 }
374
375 error_before = graph.error(values);
376 gtsam::LevenbergMarquardtParams opt_params;
377 if (VLOG_IS_ON(20))
378 opt_params.verbosity = gtsam::NonlinearOptimizerParams::Verbosity::ERROR;
379
380 utils::TimingStatsCollector timer(new_updater_->getFullyQualifiedName() +
381 ".sliding_window_optimise");
382 utils::StatsCollector(new_updater_->getFullyQualifiedName() +
383 ".sliding_window_optimise_num_vars_all")
384 .AddSample(values.size());
385 try {
386 optimised_values =
387 gtsam::LevenbergMarquardtOptimizer(graph, values, opt_params)
388 .optimize();
389 LOG(INFO) << "Finished op!";
390 } catch (const gtsam::ValuesKeyDoesNotExist& e) {
391 LOG(FATAL)
392 << "gtsam::ValuesKeyDoesNotExist: key does not exist in the values"
393 << DynoLikeKeyFormatter(e.key());
394 }
395 error_after = graph.error(optimised_values);
396
397 return true;
398 }
399 return false;
400}

402Formulation<RGBDBackendModule::RGBDMap>::UniquePtr
403RGBDBackendModule::makeUpdater() {
404 FormulationParams formulation_params;
405 formulation_params.min_static_observations = base_params_.min_static_obs_;
406 formulation_params.min_dynamic_observations = base_params_.min_dynamic_obs_;
407 formulation_params.suffix = FLAGS_updater_suffix;
408
409 if (updater_type_ == UpdaterType::MotionInWorld) {
410 LOG(INFO) << "Using MotionInWorld";
411 return std::make_unique<WorldMotionFormulation>(formulation_params,
412 getMap(), noise_models_);
413
414 } else if (updater_type_ == UpdaterType::LLWorld) {
415 LOG(INFO) << "Using LLWorld";
416 return std::make_unique<WorldPoseFormulation>(formulation_params, getMap(),
417 noise_models_);
418 } else {
419 CHECK(false) << "Not implemented";
420 }
421}
