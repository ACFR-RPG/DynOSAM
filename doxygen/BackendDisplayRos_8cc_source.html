74void BackendDisplayRos::spinOnce(
75 const BackendOutputPacket::ConstPtr& backend_output) {
76 publishPointCloud(static_tracked_points_pub_,
77 backend_output->static_landmarks_,
78 backend_output->T_world_camera_);
79 CloudPerObject clouds_per_obj = publishPointCloud(
80 dynamic_tracked_points_pub_, backend_output->dynamic_landmarks_,
81 backend_output->T_world_camera_);
82
83 publishObjectBoundingBox(object_aabb_pub_, nullptr, /* no Obb publisher */
84 clouds_per_obj, utils::fromRosTime(node_->now()),
85 "backend");
86 publishObjectPositions(
87 object_pose_pub_, backend_output->composed_object_poses,
88 backend_output->frame_id_, backend_output->timestamp_, "backend");
89
90 publishObjectPaths(
91 object_pose_path_pub_, backend_output->composed_object_poses,
92 backend_output->frame_id_, backend_output->timestamp_, "backend", 60);
93
94 {
95 nav_msgs::msg::Odometry odom_msg;
96 utils::convertWithHeader(backend_output->T_world_camera_, odom_msg,
97 backend_output->timestamp_, "world", "camera");
98 odometry_pub_->publish(odom_msg);
99 }
100
101 {
102 nav_msgs::msg::Path odom_path_msg;
103
104 for (const gtsam::Pose3& T_world_camera :
105 backend_output->optimized_poses_) {
106 // optimized camera traj
107 geometry_msgs::msg::PoseStamped pose_stamped;
108 utils::convertWithHeader(T_world_camera, pose_stamped,
109 backend_output->timestamp_, "world");
110
111 static std_msgs::msg::Header header;
112 header.stamp = utils::toRosTime(backend_output->timestamp_);
113 header.frame_id = "world";
114 odom_path_msg.header = header;
115 odom_path_msg.poses.push_back(pose_stamped);
116 }
117
118 odometry_path_pub_->publish(odom_path_msg);
119 }
120}
