65void declare_config(CameraParams& config) {
66 using namespace config;
67 name("Camera Params");
68
69 cv::Size image_size;
70 field(image_size, "resolution");
71
72 // camera to robot pose
73 std::vector<double> vector_pose;
74 field(vector_pose, "T_BS");
75 checkCondition(
76 vector_pose.size() == 16u,
77 "param 'T_BS' must be a 16 length vector in homogenous matrix form");
78 gtsam::Pose3 T_robot_camera = utils::poseVectorToGtsamPose3(vector_pose);
79
80 std::vector<double> intrinsics_v;
81 field(intrinsics_v, "intrinsics");
82 checkCondition(intrinsics_v.size() == 4,
83 "param 'intrinsics' must be a 4 length vector.");
84 CameraParams::IntrinsicsCoeffs intrinsics;
85 intrinsics.resize(4u);
86 // Move elements from one to the other.
87 std::copy_n(std::make_move_iterator(intrinsics_v.begin()), intrinsics.size(),
88 intrinsics.begin());
89
90 CameraParams::DistortionCoeffs distortion;
91 field(distortion, "distortion_coefficients");
92
93 std::string distortion_model, camera_model;
94 field(distortion_model, "distortion_model");
95 field(camera_model, "camera_model");
96
97 DistortionModel model =
98 CameraParams::stringToDistortion(distortion_model, camera_model);
99 if (model == DistortionModel::NONE) {
100 throw InvalidCameraCalibration(
101 "Invalid distortion model/camera model combination when constructing "
102 "camera params from yaml file - "
103 " distortion model: " +
104 distortion_model + ", camera model: " + camera_model);
105 }
106
107 config =
108 CameraParams(intrinsics, distortion, image_size, model, T_robot_camera);
109}

158void CameraParams::convertDistortionVectorToMatrix(
159 const DistortionCoeffs& distortion_coeffs, cv::Mat* distortion_coeffs_mat) {
160 *distortion_coeffs_mat = cv::Mat::zeros(1, distortion_coeffs.size(), CV_64F);
161 for (int k = 0; k < distortion_coeffs_mat->cols; k++) {
162 distortion_coeffs_mat->at<double>(0, k) = distortion_coeffs[k];
163 }
164}

166void CameraParams::convertIntrinsicsVectorToMatrix(
167 const IntrinsicsCoeffs& intrinsics, cv::Mat* camera_matrix) {
168 CHECK_GT(intrinsics.at(0), 0.0) << "fx cannot be zero";
169 CHECK_GT(intrinsics.at(1), 0.0) << "fy cannot be zero";
170 CHECK_GT(intrinsics.at(2), 0.0) << "cx cannot be zero";
171 CHECK_GT(intrinsics.at(3), 0.0) << "cy cannot be zero";
172
173 *camera_matrix = cv::Mat::eye(3, 3, CV_64F);
174 camera_matrix->at<double>(0, 0) = intrinsics[0];
175 camera_matrix->at<double>(1, 1) = intrinsics[1];
176 camera_matrix->at<double>(0, 2) = intrinsics[2];
177 camera_matrix->at<double>(1, 2) = intrinsics[3];
178}

180void CameraParams::convertKMatrixToIntrinsicsCoeffs(
181 const cv::Mat& K, IntrinsicsCoeffs& intrinsics) {
182 CHECK_EQ(K.type(), CV_64F) << "Cannot convert: Camera Matrix (K) is expected "
183 "to be of type CV_64F but is "
184 << utils::cvTypeToString(K.type());
185 CHECK(K.rows == 3 && K.cols) << "Cannot convert: Camera Matrix (K) is "
186 "expected to be of size (3 x 3) but is ["
187 << K.rows << " x " << K.cols << "]";
188 ;
189
190 intrinsics.resize(4);
191 intrinsics[0] = K.at<double>(0, 0);
192 intrinsics[1] = K.at<double>(1, 1);
193 intrinsics[2] = K.at<double>(0, 2);
194 intrinsics[3] = K.at<double>(1, 2);
195}

197DistortionModel CameraParams::stringToDistortion(
198 const std::string& distortion_model, const std::string& camera_model) {
199 std::string lower_case_distortion_model = distortion_model;
200 std::string lower_case_camera_model = camera_model;
201
202 std::transform(lower_case_distortion_model.begin(),
203 lower_case_distortion_model.end(),
204 lower_case_distortion_model.begin(), ::tolower);
205 std::transform(lower_case_camera_model.begin(), lower_case_camera_model.end(),
206 lower_case_camera_model.begin(), ::tolower);
207
208 if (lower_case_camera_model == "pinhole") {
209 if (lower_case_distortion_model == "none") {
210 return DistortionModel::NONE;
211 } else if ((lower_case_distortion_model == "plumb_bob") ||
212 (lower_case_distortion_model == "radial_tangential") ||
213 (lower_case_distortion_model == "radtan")) {
214 return DistortionModel::RADTAN;
215 } else if (lower_case_distortion_model == "equidistant") {
216 return DistortionModel::EQUIDISTANT;
217 } else if (lower_case_distortion_model == "kannala_brandt") {
218 return DistortionModel::FISH_EYE;
219 } else {
220 LOG(ERROR) << "Unrecognized distortion model for pinhole camera. Valid "
221 "pinhole distortion model options are 'none', 'radtan', "
222 "'equidistant', 'fish eye'.";
223 }
224 } else {
225 LOG(ERROR)
226 << "Unrecognized camera model. Valid camera models are 'pinhole'";
227 }
228 // Return no distortion model if invalid model is provided
229 return DistortionModel::NONE;
230}

232bool CameraParams::distortionModelToString(const DistortionModel& model,
233 std::string& distortion_model,
234 std::string& camera_model) {
235 switch (model) {
236 case DistortionModel::NONE:
237 return false;
238 case DistortionModel::RADTAN:
239 // in ROS only plumb_bob is defined so we use this
240 distortion_model = "plumb_bob";
241 camera_model = "pinhole";
242 return true;
243 case DistortionModel::EQUIDISTANT:
244 distortion_model = "equidistant";
245 camera_model = "pinhole";
246 return true;
247 case DistortionModel::FISH_EYE:
248 distortion_model = "kannala_brandt";
249 camera_model = "pinhole";
250 return true;
251 default:
252 LOG(WARNING) << "Cannot convert distortion model to string: "
253 << to_string(model);
254 return false;
255 }

258bool CameraParams::equals(const CameraParams& other, double tol) const {
259 return dyno::equals_with_abs_tol(intrinsics_, other.intrinsics_, tol) &&
260 dyno::equals_with_abs_tol(distortion_coeff_, other.distortion_coeff_,
261 tol) &&
262 (image_size_.width == other.image_size_.width &&
263 image_size_.height == other.image_size_.height) &&
264 distortion_model_ == other.distortion_model_ &&
265 T_robot_camera_.equals(other.T_robot_camera_, tol) &&
266 utils::compareCvMatsUpToTol(K_, other.K_, tol) &&
267 utils::compareCvMatsUpToTol(
268 D_, other.D_); // NOTE: do not comapre with P matrix as currently
269 // not calculated
270}
