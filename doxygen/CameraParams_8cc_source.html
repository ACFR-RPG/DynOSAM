44template <>
45std::string to_string(const DistortionModel& distortion) {
46 switch (distortion) {
47 case DistortionModel::NONE:
48 return "None";
49 break;
50 case DistortionModel::RADTAN:
51 return "Radtan";
52 break;
53 case DistortionModel::EQUIDISTANT:
54 return "Equidistant";
55 break;
56 case DistortionModel::FISH_EYE:
57 return "Fish-eye";
58 break;
59 default:
60 return "Unknown distortion model";
61 break;
62 }
63}

65void declare_config(CameraParams& config) {
66 using namespace config;
67 name("Camera Params");
68
69 cv::Size image_size;
70 field(image_size, "resolution");
71
72 // camera to robot pose
73 std::vector<double> vector_pose;
74 field(vector_pose, "T_BS");
75 checkCondition(
76 vector_pose.size() == 16u,
77 "param 'T_BS' must be a 16 length vector in homogenous matrix form");
78 gtsam::Pose3 T_robot_camera = utils::poseVectorToGtsamPose3(vector_pose);
79
80 std::vector<double> intrinsics_v;
81 field(intrinsics_v, "intrinsics");
82 checkCondition(intrinsics_v.size() == 4,
83 "param 'intrinsics' must be a 4 length vector.");
84 CameraParams::IntrinsicsCoeffs intrinsics;
85 intrinsics.resize(4u);
86 // Move elements from one to the other.
87 std::copy_n(std::make_move_iterator(intrinsics_v.begin()), intrinsics.size(),
88 intrinsics.begin());
89
90 CameraParams::DistortionCoeffs distortion;
91 field(distortion, "distortion_coefficients");
92
93 std::string distortion_model, camera_model;
94 field(distortion_model, "distortion_model");
95 field(camera_model, "camera_model");
96
97 DistortionModel model =
98 CameraParams::stringToDistortion(distortion_model, camera_model);
99 if (model == DistortionModel::NONE) {
100 throw InvalidCameraCalibration(
101 "Invalid distortion model/camera model combination when constructing "
102 "camera params from yaml file - "
103 " distortion model: " +
104 distortion_model + ", camera model: " + camera_model);
105 }
106
107 config =
108 CameraParams(intrinsics, distortion, image_size, model, T_robot_camera);
109}

142void CameraParams::convertDistortionVectorToMatrix(
143 const DistortionCoeffs& distortion_coeffs, cv::Mat* distortion_coeffs_mat) {
144 *distortion_coeffs_mat = cv::Mat::zeros(1, distortion_coeffs.size(), CV_64F);
145 for (int k = 0; k < distortion_coeffs_mat->cols; k++) {
146 distortion_coeffs_mat->at<double>(0, k) = distortion_coeffs[k];
147 }
148}

150void CameraParams::convertIntrinsicsVectorToMatrix(
151 const IntrinsicsCoeffs& intrinsics, cv::Mat* camera_matrix) {
152 *camera_matrix = cv::Mat::eye(3, 3, CV_64F);
153 camera_matrix->at<double>(0, 0) = intrinsics[0];
154 camera_matrix->at<double>(1, 1) = intrinsics[1];
155 camera_matrix->at<double>(0, 2) = intrinsics[2];
156 camera_matrix->at<double>(1, 2) = intrinsics[3];
157}

159void CameraParams::convertKMatrixToIntrinsicsCoeffs(
160 const cv::Mat& K, IntrinsicsCoeffs& intrinsics) {
161 CHECK_EQ(K.type(), CV_64F) << "Cannot convert: Camera Matrix (K) is expected "
162 "to be of type CV_64F but is "
163 << utils::cvTypeToString(K.type());
164 CHECK(K.rows == 3 && K.cols) << "Cannot convert: Camera Matrix (K) is "
165 "expected to be of size (3 x 3) but is ["
166 << K.rows << " x " << K.cols << "]";
167 ;
168
169 intrinsics.resize(4);
170 intrinsics[0] = K.at<double>(0, 0);
171 intrinsics[1] = K.at<double>(1, 1);
172 intrinsics[2] = K.at<double>(0, 2);
173 intrinsics[3] = K.at<double>(1, 2);
174}

176DistortionModel CameraParams::stringToDistortion(
177 const std::string& distortion_model, const std::string& camera_model) {
178 std::string lower_case_distortion_model = distortion_model;
179 std::string lower_case_camera_model = camera_model;
180
181 std::transform(lower_case_distortion_model.begin(),
182 lower_case_distortion_model.end(),
183 lower_case_distortion_model.begin(), ::tolower);
184 std::transform(lower_case_camera_model.begin(), lower_case_camera_model.end(),
185 lower_case_camera_model.begin(), ::tolower);
186
187 if (lower_case_camera_model == "pinhole") {
188 if (lower_case_distortion_model == "none") {
189 return DistortionModel::NONE;
190 } else if ((lower_case_distortion_model == "plumb_bob") ||
191 (lower_case_distortion_model == "radial_tangential") ||
192 (lower_case_distortion_model == "radtan")) {
193 return DistortionModel::RADTAN;
194 } else if (lower_case_distortion_model == "equidistant") {
195 return DistortionModel::EQUIDISTANT;
196 } else if (lower_case_distortion_model == "kannala_brandt") {
197 return DistortionModel::FISH_EYE;
198 } else {
199 LOG(ERROR) << "Unrecognized distortion model for pinhole camera. Valid "
200 "pinhole distortion model options are 'none', 'radtan', "
201 "'equidistant', 'fish eye'.";
202 }
203 } else {
204 LOG(ERROR)
205 << "Unrecognized camera model. Valid camera models are 'pinhole'";
206 }
207 // Return no distortion model if invalid model is provided
208 return DistortionModel::NONE;
209}

211bool CameraParams::distortionModelToString(const DistortionModel& model,
212 std::string& distortion_model,
213 std::string& camera_model) {
214 switch (model) {
215 case DistortionModel::NONE:
216 return false;
217 case DistortionModel::RADTAN:
218 // in ROS only plumb_bob is defined so we use this
219 distortion_model = "plumb_bob";
220 camera_model = "pinhole";
221 return true;
222 case DistortionModel::EQUIDISTANT:
223 distortion_model = "equidistant";
224 camera_model = "pinhole";
225 return true;
226 case DistortionModel::FISH_EYE:
227 distortion_model = "kannala_brandt";
228 camera_model = "pinhole";
229 return true;
230 default:
231 LOG(WARNING) << "Cannot convert distortion model to string: "
232 << to_string(model);
233 return false;
234 }
235}

237bool CameraParams::equals(const CameraParams& other, double tol) const {
238 return dyno::equals_with_abs_tol(intrinsics_, other.intrinsics_, tol) &&
239 dyno::equals_with_abs_tol(distortion_coeff_, other.distortion_coeff_,
240 tol) &&
241 (image_size_.width == other.image_size_.width &&
242 image_size_.height == other.image_size_.height) &&
243 distortion_model_ == other.distortion_model_ &&
244 T_robot_camera_.equals(other.T_robot_camera_, tol) &&
245 utils::compareCvMatsUpToTol(K_, other.K_, tol) &&
246 utils::compareCvMatsUpToTol(
247 D_, other.D_); // NOTE: do not comapre with P matrix as currently
248 // not calculated
249}

253 out << "\nIntrinsics: \n- fx " << fx() << "\n- fy " << fy() << "\n- cu "
254 << cu() << "\n- cv " << cv()
