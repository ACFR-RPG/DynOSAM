65void declare_config(CameraParams& config) {
66 using namespace config;
67 name("Camera Params");
68
69 cv::Size image_size;
70 field(image_size, "resolution");
71
72 // camera to robot pose
73 std::vector<double> vector_pose;
74 field(vector_pose, "T_BS");
75 checkCondition(
76 vector_pose.size() == 16u,
77 "param 'T_BS' must be a 16 length vector in homogenous matrix form");
78 gtsam::Pose3 T_robot_camera = utils::poseVectorToGtsamPose3(vector_pose);
79
80 std::vector<double> intrinsics_v;
81 field(intrinsics_v, "intrinsics");
82 checkCondition(intrinsics_v.size() == 4,
83 "param 'intrinsics' must be a 4 length vector.");
84 CameraParams::IntrinsicsCoeffs intrinsics;
85 intrinsics.resize(4u);
86 // Move elements from one to the other.
87 std::copy_n(std::make_move_iterator(intrinsics_v.begin()), intrinsics.size(),
88 intrinsics.begin());
89
90 CameraParams::DistortionCoeffs distortion;
91 field(distortion, "distortion_coefficients");
92
93 std::string distortion_model, camera_model;
94 field(distortion_model, "distortion_model");
95 field(camera_model, "camera_model");
96
97 DistortionModel model =
98 CameraParams::stringToDistortion(distortion_model, camera_model);
99 if (model == DistortionModel::NONE) {
100 throw InvalidCameraCalibration(
101 "Invalid distortion model/camera model combination when constructing "
102 "camera params from yaml file - "
103 " distortion model: " +
104 distortion_model + ", camera model: " + camera_model);
105 }
106
107 config =
108 CameraParams(intrinsics, distortion, image_size, model, T_robot_camera);
109}

142void CameraParams::convertDistortionVectorToMatrix(
143 const DistortionCoeffs& distortion_coeffs, cv::Mat* distortion_coeffs_mat) {
144 *distortion_coeffs_mat = cv::Mat::zeros(1, distortion_coeffs.size(), CV_64F);
145 for (int k = 0; k < distortion_coeffs_mat->cols; k++) {
146 distortion_coeffs_mat->at<double>(0, k) = distortion_coeffs[k];
147 }
148}

150void CameraParams::convertIntrinsicsVectorToMatrix(
151 const IntrinsicsCoeffs& intrinsics, cv::Mat* camera_matrix) {
152 CHECK_GT(intrinsics.at(0), 0.0) << "fx cannot be zero";
153 CHECK_GT(intrinsics.at(1), 0.0) << "fy cannot be zero";
154 CHECK_GT(intrinsics.at(2), 0.0) << "cx cannot be zero";
155 CHECK_GT(intrinsics.at(3), 0.0) << "cy cannot be zero";
156
157 *camera_matrix = cv::Mat::eye(3, 3, CV_64F);
158 camera_matrix->at<double>(0, 0) = intrinsics[0];
159 camera_matrix->at<double>(1, 1) = intrinsics[1];
160 camera_matrix->at<double>(0, 2) = intrinsics[2];
161 camera_matrix->at<double>(1, 2) = intrinsics[3];
162}

164void CameraParams::convertKMatrixToIntrinsicsCoeffs(
165 const cv::Mat& K, IntrinsicsCoeffs& intrinsics) {
166 CHECK_EQ(K.type(), CV_64F) << "Cannot convert: Camera Matrix (K) is expected "
167 "to be of type CV_64F but is "
168 << utils::cvTypeToString(K.type());
169 CHECK(K.rows == 3 && K.cols) << "Cannot convert: Camera Matrix (K) is "
170 "expected to be of size (3 x 3) but is ["
171 << K.rows << " x " << K.cols << "]";
172 ;
173
174 intrinsics.resize(4);
175 intrinsics[0] = K.at<double>(0, 0);
176 intrinsics[1] = K.at<double>(1, 1);
177 intrinsics[2] = K.at<double>(0, 2);
178 intrinsics[3] = K.at<double>(1, 2);
179}

181DistortionModel CameraParams::stringToDistortion(
182 const std::string& distortion_model, const std::string& camera_model) {
183 std::string lower_case_distortion_model = distortion_model;
184 std::string lower_case_camera_model = camera_model;
185
186 std::transform(lower_case_distortion_model.begin(),
187 lower_case_distortion_model.end(),
188 lower_case_distortion_model.begin(), ::tolower);
189 std::transform(lower_case_camera_model.begin(), lower_case_camera_model.end(),
190 lower_case_camera_model.begin(), ::tolower);
191
192 if (lower_case_camera_model == "pinhole") {
193 if (lower_case_distortion_model == "none") {
194 return DistortionModel::NONE;
195 } else if ((lower_case_distortion_model == "plumb_bob") ||
196 (lower_case_distortion_model == "radial_tangential") ||
197 (lower_case_distortion_model == "radtan")) {
198 return DistortionModel::RADTAN;
199 } else if (lower_case_distortion_model == "equidistant") {
200 return DistortionModel::EQUIDISTANT;
201 } else if (lower_case_distortion_model == "kannala_brandt") {
202 return DistortionModel::FISH_EYE;
203 } else {
204 LOG(ERROR) << "Unrecognized distortion model for pinhole camera. Valid "
205 "pinhole distortion model options are 'none', 'radtan', "
206 "'equidistant', 'fish eye'.";
207 }
208 } else {
209 LOG(ERROR)
210 << "Unrecognized camera model. Valid camera models are 'pinhole'";
211 }
212 // Return no distortion model if invalid model is provided
213 return DistortionModel::NONE;
214}

216bool CameraParams::distortionModelToString(const DistortionModel& model,
217 std::string& distortion_model,
218 std::string& camera_model) {
219 switch (model) {
220 case DistortionModel::NONE:
221 return false;
222 case DistortionModel::RADTAN:
223 // in ROS only plumb_bob is defined so we use this
224 distortion_model = "plumb_bob";
225 camera_model = "pinhole";
226 return true;
227 case DistortionModel::EQUIDISTANT:
228 distortion_model = "equidistant";
229 camera_model = "pinhole";
230 return true;
231 case DistortionModel::FISH_EYE:
232 distortion_model = "kannala_brandt";
233 camera_model = "pinhole";
234 return true;
235 default:
236 LOG(WARNING) << "Cannot convert distortion model to string: "
237 << to_string(model);
238 return false;
239 }
240}

242bool CameraParams::equals(const CameraParams& other, double tol) const {
243 return dyno::equals_with_abs_tol(intrinsics_, other.intrinsics_, tol) &&
244 dyno::equals_with_abs_tol(distortion_coeff_, other.distortion_coeff_,
245 tol) &&
246 (image_size_.width == other.image_size_.width &&
247 image_size_.height == other.image_size_.height) &&
248 distortion_model_ == other.distortion_model_ &&
249 T_robot_camera_.equals(other.T_robot_camera_, tol) &&
250 utils::compareCvMatsUpToTol(K_, other.K_, tol) &&
251 utils::compareCvMatsUpToTol(
252 D_, other.D_); // NOTE: do not comapre with P matrix as currently
253 // not calculated
254}
