111 template <typename DERIVEDSTATUS>
112 void updateObservations(
113 const GenericTrackedStatusVector<DERIVEDSTATUS>& measurements) {
114 using DerivedMeasurement =
115 typename GenericTrackedStatusVector<DERIVEDSTATUS>::Value;
116 for (const DERIVEDSTATUS& status_measurement : measurements) {
117 const TrackedValueStatus<DerivedMeasurement>& derived_status =
118 static_cast<const TrackedValueStatus<DerivedMeasurement>&>(
119 status_measurement);
120 const TrackedValueStatus<MEASUREMENT>& status =
121 derived_status.template asType<MEASUREMENT>();
122 const MEASUREMENT& measurement = status_measurement.value();
123 const TrackletId tracklet_id = status.trackletId();
124 const FrameId frame_id = status.frameId();
125 const ObjectId object_id = status.objectId();
126 const bool is_static = status.isStatic();
127 addOrUpdateMapStructures(measurement, tracklet_id, frame_id, object_id,
128 is_static);
129 }
130 }

381 void addOrUpdateMapStructures(const Measurement& measurement,
382 TrackletId tracklet_id, FrameId frame_id,
383 ObjectId object_id, bool is_static) {
384 typename LandmarkNodeM::Ptr landmark_node = nullptr;
385 typename FrameNodeM::Ptr frame_node = nullptr;
386
387 CHECK((is_static && object_id == background_label) ||
388 (!is_static && object_id != background_label));
389
390 if (!landmarkExists(tracklet_id)) {
391 landmark_node = std::make_shared<LandmarkNodeM>(getptr());
392 CHECK_NOTNULL(landmark_node);
393 landmark_node->tracklet_id = tracklet_id;
394 landmark_node->object_id = object_id;
395 landmarks_.insert2(tracklet_id, landmark_node);
396 }
397
398 if (!frameExists(frame_id)) {
399 frame_node = std::make_shared<FrameNodeM>(getptr());
400 frame_node->frame_id = frame_id;
401 frames_.insert2(frame_id, frame_node);
402 }
403
404 landmark_node = getLandmark(tracklet_id);
405 frame_node = getFrame(frame_id);
406
407 CHECK(landmark_node);
408 CHECK(frame_node);
409
410 CHECK_EQ(landmark_node->tracklet_id, tracklet_id);
411 // this might fail of a tracklet get associated with a different object
412 CHECK_EQ(landmark_node->object_id, object_id);
413 CHECK_EQ(frame_node->frame_id, frame_id);
414
415 landmark_node->add(frame_node, measurement);
416
417 // add to frame
418 if (is_static) {
419 frame_node->static_landmarks.insert(landmark_node);
420 } else {
421 CHECK(object_id != background_label);
422
423 typename ObjectNodeM::Ptr object_node = nullptr;
424 if (!objectExists(object_id)) {
425 object_node = std::make_shared<ObjectNodeM>(getptr());
426 object_node->object_id = object_id;
427 objects_.insert2(object_id, object_node);
428 }
429
430 object_node = getObject(object_id);
431 CHECK(object_node);
432
433 object_node->dynamic_landmarks.insert(landmark_node);
434
435 frame_node->dynamic_landmarks.insert(landmark_node);
436 frame_node->objects_seen.insert(object_node);
437 }
438 }
