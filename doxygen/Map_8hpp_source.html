109 template <typename DERIVEDSTATUS>
110 void updateObservations(
111 const GenericTrackedStatusVector<DERIVEDSTATUS>& measurements) {
112 using DerivedMeasurement =
113 typename GenericTrackedStatusVector<DERIVEDSTATUS>::Value;
114 for (const DERIVEDSTATUS& status_measurement : measurements) {
115 const TrackedValueStatus<DerivedMeasurement>& derived_status =
116 static_cast<const TrackedValueStatus<DerivedMeasurement>&>(
117 status_measurement);
118 const TrackedValueStatus<MEASUREMENT>& status =
119 derived_status.template asType<MEASUREMENT>();
120 const MEASUREMENT& measurement = status_measurement.value();
121 const TrackletId tracklet_id = status.trackletId();
122 const FrameId frame_id = status.frameId();
123 const ObjectId object_id = status.objectId();
124 const bool is_static = status.isStatic();
125 addOrUpdateMapStructures(measurement, tracklet_id, frame_id, object_id,
126 is_static);
127 }
128 }

268 bool hasInitialObjectMotion(
269 FrameId frame_id, ObjectId object_id,
270 Motion3ReferenceFrame* motion_frame = nullptr) const {
271 auto frame_node = this->getFrame(frame_id);
272 if (!frame_node) {
273 return false;
274 }
275
276 // motion map has not been set
277 if (frame_node->motions) {
278 const auto& motions = *frame_node->motions;
279
280 // no motion for this object in map
281 if (!motions.exists(object_id)) {
282 return false;
283 }
284
285 if (motion_frame) {
286 *motion_frame = motions.at(object_id);
287 }
288 return true;
289 } else {
290 return false;
291 }
292 }

420 void addOrUpdateMapStructures(const Measurement& measurement,
421 TrackletId tracklet_id, FrameId frame_id,
422 ObjectId object_id, bool is_static) {
423 typename LandmarkNodeM::Ptr landmark_node = nullptr;
424 typename FrameNodeM::Ptr frame_node = nullptr;
425
426 CHECK((is_static && object_id == background_label) ||
427 (!is_static && object_id != background_label));
428
429 if (!landmarkExists(tracklet_id)) {
430 landmark_node = std::make_shared<LandmarkNodeM>(getptr());
431 CHECK_NOTNULL(landmark_node);
432 landmark_node->tracklet_id = tracklet_id;
433 landmark_node->object_id = object_id;
434 landmarks_.insert2(tracklet_id, landmark_node);
435 }
436
437 if (!frameExists(frame_id)) {
438 frame_node = std::make_shared<FrameNodeM>(getptr());
439 frame_node->frame_id = frame_id;
440 frames_.insert2(frame_id, frame_node);
441 }
442
443 landmark_node = getLandmark(tracklet_id);
444 frame_node = getFrame(frame_id);
445
446 CHECK(landmark_node);
447 CHECK(frame_node);
448
449 CHECK_EQ(landmark_node->tracklet_id, tracklet_id);
450 // this might fail of a tracklet get associated with a different object
451 CHECK_EQ(landmark_node->object_id, object_id);
452 CHECK_EQ(frame_node->frame_id, frame_id);
453
454 landmark_node->add(frame_node, measurement);
455
456 // add to frame
457 if (is_static) {
458 frame_node->static_landmarks.insert(landmark_node);
459 } else {
460 CHECK(object_id != background_label);
461
462 typename ObjectNodeM::Ptr object_node = nullptr;
463 if (!objectExists(object_id)) {
464 object_node = std::make_shared<ObjectNodeM>(getptr());
465 object_node->object_id = object_id;
466 objects_.insert2(object_id, object_node);
467 }
468
469 object_node = getObject(object_id);
470 CHECK(object_node);
471
472 object_node->dynamic_landmarks.insert(landmark_node);
473
474 frame_node->dynamic_landmarks.insert(landmark_node);
475 frame_node->objects_seen.insert(object_node);
476 }
477 }
