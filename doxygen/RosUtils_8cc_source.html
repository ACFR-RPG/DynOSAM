82template <>
83bool dyno::convert(const gtsam::Pose3& pose, geometry_msgs::msg::Pose& msg) {
84 const gtsam::Rot3& rotation = pose.rotation();
85 const gtsam::Quaternion& quaternion = rotation.toQuaternion();
86
87 // Position
88 msg.position.x = pose.x();
89 msg.position.y = pose.y();
90 msg.position.z = pose.z();
91
92 // Orientation
93 msg.orientation.w = quaternion.w();
94 msg.orientation.x = quaternion.x();
95 msg.orientation.y = quaternion.y();
96 msg.orientation.z = quaternion.z();
97 return true;
98}

100template <>
101bool dyno::convert(const geometry_msgs::msg::Pose& msg, gtsam::Pose3& pose) {
102 gtsam::Point3 translation(msg.position.x, msg.position.y, msg.position.z);
103
104 gtsam::Rot3 rotation(msg.orientation.w, msg.orientation.x, msg.orientation.y,
105 msg.orientation.z);
106
107 pose = gtsam::Pose3(rotation, translation);
108 return true;
109}

207rclcpp::Parameter ParameterDetails::get_param(
208 const rclcpp::Parameter& default_param) const {
209 const bool is_set = isSet();
210 bool has_default = default_param.get_type() != rclcpp::PARAMETER_NOT_SET;
211
212 if (!is_set) {
213 // if no default value we treat a not set parameter as a error as the use
214 // MUST override it via runtime configuration
215 if (!has_default) {
216 throw InvalidDefaultParameter(this->name());
217 } else {
218 return default_param;
219 }
220 }
221 return node_->get_parameter(this->name());
222}
