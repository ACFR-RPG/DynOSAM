46 YamlParser(const std::string& filepath) : fs_(), filepath_(filepath) {
47 throwExceptionIfPathInvalid(filepath_);
48 openFile(filepath, &fs_);
49 }
50 ~YamlParser() { closeFile(&fs_); }

52 template <class T>
53 bool getYamlParam(const std::string& id, T* output) const {
54 CHECK(!id.empty());
55 CHECK_NOTNULL(output);
56 const cv::FileNode& file_handle = fs_[id];
57 CHECK_NE(file_handle.type(), cv::FileNode::NONE)
58 << "Missing parameter: " << id.c_str()
59 << " in file: " << filepath_.c_str();
60 file_handle >> *output;
61 return true;
62 }

64 template <class T>
65 bool getYamlParam(const std::string& id, T* output, T default_value) const {
66 CHECK(!id.empty());
67 CHECK_NOTNULL(output);
68 const cv::FileNode& file_handle = fs_[id];
69 if(file_handle.type() == cv::FileNode::NONE) {
70 *output = default_value;
71 return false;
72 }
73 else {
74 file_handle >> *output;
75 return true;

100 template <class T>
101 void getNestedYamlParam(const std::string& id,
102 const std::string& id_2,
103 T* output,
104 T default_value) const {
105 CHECK(!id.empty());
106 CHECK(!id_2.empty());
107 const cv::FileNode& file_handle = fs_[id];
108 CHECK_NE(file_handle.type(), cv::FileNode::NONE)
109 << "Missing parameter: " << id.c_str()
110 << " in file: " << filepath_.c_str();
111 const cv::FileNode& file_handle_2 = file_handle[id_2];
112 if(file_handle_2.type() == cv::FileNode::NONE) {
113 *output = default_value;
114 }
115 else {
116 T v;
117 file_handle_2 >> v;
118 file_handle_2 >> *output;
119 }
120
121 }

123 private:
124 void openFile(const std::string& filepath, cv::FileStorage* fs) const {
125 CHECK(!filepath.empty()) << "Empty filepath!";
126 try {
127 CHECK_NOTNULL(fs)->open(filepath, cv::FileStorage::READ);
128 } catch (cv::Exception& e) {
129 throw DynosamException(
130 "Cannot open file: " + filepath + '\n' + "OpenCV error code: " + e.msg
131 + " (remember that the first line should be: %YAML:1.0)!"
132 );
133 }
134 LOG_IF(FATAL, !fs->isOpened())
135 << "Cannot open file in parseYAML: " << filepath
136 << " (remember that the first line should be: %YAML:1.0)";
137 }
138
139 inline void closeFile(cv::FileStorage* fs) const {
140 CHECK_NOTNULL(fs)->release();
141 }
