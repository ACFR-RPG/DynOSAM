35static bool internalReconstructInfo(gtsam::Key key, SymbolChar expected_chr, ObjectId& object_label, FrameId& frame_id) {
36 //assume motion/pose key is a labelled symbol
37 if(!checkIfLabeledSymbol(key)) {return false; }
38
39 const gtsam::LabeledSymbol as_labeled_symbol(key);
40 if(as_labeled_symbol.chr() != expected_chr) { return false;}
41
42 frame_id = static_cast<FrameId>(as_labeled_symbol.index());
43
44 char label = as_labeled_symbol.label();
45 object_label = label - '0';
46 return true;
47}

63std::string DynoLikeKeyFormatter(gtsam::Key key)
64{

66 if (asLabeledSymbol.chr() > 0 && asLabeledSymbol.label() > 0) {
67 //if used as motion
68 if(asLabeledSymbol.chr() == kObjectMotionSymbolChar || asLabeledSymbol.chr() == kObjectPoseSymbolChar) {
69 return (std::string) asLabeledSymbol;

71 return (std::string) asLabeledSymbol;

73
74 const gtsam::Symbol asSymbol(key);

107std::string DynoLikeKeyFormatterVerbose(gtsam::Key key) {
108 const gtsam::LabeledSymbol asLabeledSymbol(key);
109 if (asLabeledSymbol.chr() > 0 && asLabeledSymbol.label() > 0) {
110 //if used as motion
111 if(asLabeledSymbol.chr() == kObjectMotionSymbolChar) {
112 ObjectId object_label;
113 FrameId frame_id;
114 CHECK(reconstructMotionInfo(asLabeledSymbol, object_label, frame_id));
115
116 std::stringstream ss;
117 ss << "H: label" << object_label << ", frames: " << frame_id - 1 << " -> " << frame_id;
118 return ss.str();
119 }
120 else if(asLabeledSymbol.chr() == kObjectPoseSymbolChar) {
121 ObjectId object_label;
122 FrameId frame_id;
123 CHECK(reconstructPoseInfo(asLabeledSymbol, object_label, frame_id));
124
125 std::stringstream ss;
126 ss << "K: label" << object_label << ", frame: " << frame_id;
127 return ss.str();
128 }
129 return (std::string) asLabeledSymbol;
130 }
131
132 const gtsam::Symbol asSymbol(key);
133 if (asLabeledSymbol.chr() > 0) {
134 if(asLabeledSymbol.chr() == kDynamicLandmarkSymbolChar) {
135 const DynamicPointSymbol asDynamicPointSymbol(key);
136
137 FrameId frame_id = asDynamicPointSymbol.frameId();
138 TrackletId tracklet_id = asDynamicPointSymbol.trackletId();
139 std::stringstream ss;
140 ss << kDynamicLandmarkSymbolChar << ": frame " << frame_id << ", tracklet " << tracklet_id;
141 return ss.str();
142
143 }
144 else {
145 return (std::string) asSymbol;
146 }
147
148 }
149 else {
150 return std::to_string(key);
151 }
