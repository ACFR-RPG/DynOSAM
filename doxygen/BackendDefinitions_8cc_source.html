41bool ApplyFunctionalSymbol::operator()(gtsam::Key key) const {
42 const gtsam::Symbol sym(key);
43 switch (sym.chr()) {
44 case kPoseSymbolChar:
45 if (pose_func_) {
46 pose_func_(static_cast<FrameId>(sym.index()), sym);
47 }
48 return true;
49 case kObjectMotionSymbolChar: {
50 FrameId frame_id;
51 ObjectId object_id;
52 // attempt to get info about this key
53 bool valid = reconstructMotionInfo(key, object_id, frame_id);
54 // if valid and motion func registered, do call back
55 if (valid && object_motion_func_)
56 object_motion_func_(frame_id, object_id, gtsam::LabeledSymbol(key));
57 }
58 return true;
59 case kObjectPoseSymbolChar: {
60 FrameId frame_id;
61 ObjectId object_id;
62 // attempt to get info about this key
63 bool valid = reconstructPoseInfo(key, object_id, frame_id);
64 // if valid and motion func registered, do call back
65 if (valid && object_pose_func_)
66 object_pose_func_(frame_id, object_id, gtsam::LabeledSymbol(key));
67 }
68 return true;
69 case kStaticLandmarkSymbolChar: {
70 if (static_lmk_func_) {
71 static_lmk_func_(static_cast<TrackletId>(sym.index()), sym);
72 }
73 }
74 return true;
75 case kDynamicLandmarkSymbolChar: {
76 if (dynamic_lmk_func_) {
77 DynamicPointSymbol dps(key);
78 dynamic_lmk_func_(dps.trackletId(), dps);
79 }
80 }
81 return true;
82
83 default:
84 return false;
85 }
86}

124NoiseModels NoiseModels::fromBackendParams(
125 const BackendParams& backend_params) {
126 NoiseModels noise_models;
127
128 // odometry
129 gtsam::Vector6 odom_sigmas;
130 odom_sigmas.head<3>().setConstant(backend_params.odometry_rotation_sigma_);
131 odom_sigmas.tail<3>().setConstant(backend_params.odometry_translation_sigma_);
132 noise_models.odometry_noise =
133 gtsam::noiseModel::Diagonal::Sigmas(odom_sigmas);
134 CHECK(noise_models.odometry_noise);
135
136 // first pose prior (world frame)
137 noise_models.initial_pose_prior =
138 gtsam::noiseModel::Isotropic::Sigma(6u, 0.000001);
139 CHECK(noise_models.initial_pose_prior);
140
141 // landmark motion noise (needed for some formulations ie world-centric)
142 noise_models.landmark_motion_noise = gtsam::noiseModel::Isotropic::Sigma(
143 3u, backend_params.motion_ternary_factor_noise_sigma_);
144 CHECK(noise_models.landmark_motion_noise);
145
146 // smoothing factor noise model (can be any variant of the smoothing factor as
147 // long as the dimensions are 6, ie. pose)
148 gtsam::Vector6 object_constant_vel_sigmas;
149 object_constant_vel_sigmas.head<3>().setConstant(
150 backend_params.constant_object_motion_rotation_sigma_);
151 object_constant_vel_sigmas.tail<3>().setConstant(
152 backend_params.constant_object_motion_translation_sigma_);
153 noise_models.object_smoothing_noise =
154 gtsam::noiseModel::Diagonal::Sigmas(object_constant_vel_sigmas);
155 CHECK(noise_models.object_smoothing_noise);
156
157 // TODO: CHECKS that values are not zero!!!
158
159 // TODO: should now depricate if we're using covariance from frontend??
160 noise_models.static_point_noise = gtsam::noiseModel::Isotropic::Sigma(
161 3u, backend_params.static_point_noise_sigma);
162 noise_models.dynamic_point_noise = gtsam::noiseModel::Isotropic::Sigma(
163 3u, backend_params.dynamic_point_noise_sigma);
164
165 if (backend_params.use_robust_kernals_) {
166 LOG(INFO) << "Using robust huber loss function: "
167 << backend_params.k_huber_3d_points_;
168
169 if (backend_params.static_point_noise_as_robust) {
170 LOG(INFO) << "Making static point noise model robust!";
171 noise_models.static_point_noise = gtsam::noiseModel::Robust::Create(
172 gtsam::noiseModel::mEstimator::Huber::Create(
173 backend_params.k_huber_3d_points_),
174 noise_models.static_point_noise);
175 }
176
177 // TODO: JUST FOR TESTING!!!
178 if (backend_params.dynamic_point_noise_as_robust) {
179 LOG(INFO) << "Making dynamic point noise model robust!";
180 noise_models.dynamic_point_noise = gtsam::noiseModel::Robust::Create(
181 gtsam::noiseModel::mEstimator::Huber::Create(
182 backend_params.k_huber_3d_points_),
183 noise_models.dynamic_point_noise);
184 }
185
186 // TODO: not k_huber_3d_points_ not just used for 3d points
187 noise_models.landmark_motion_noise = gtsam::noiseModel::Robust::Create(
188 gtsam::noiseModel::mEstimator::Huber::Create(
189 backend_params.k_huber_3d_points_),
190 noise_models.landmark_motion_noise);
191 }
192
193 return noise_models;
194}

196void NoiseModels::print(const std::string& name) const {
197 auto print_impl = [](gtsam::SharedNoiseModel model,
198 const std::string& name) -> void {
199 if (model) {
200 model->print(name);
201 } else {
202 std::cout << "Noise model " << name << " is null!";
203 }
204 };
205
206 print_impl(initial_pose_prior, "Pose Prior ");
207 print_impl(odometry_noise, "VO ");
208 print_impl(landmark_motion_noise, "Landmark Motion ");
209 print_impl(object_smoothing_noise, "Object Smoothing ");
210 print_impl(dynamic_point_noise, "Dynamic Point ");
211 print_impl(static_point_noise, "Static Point ");
212}
