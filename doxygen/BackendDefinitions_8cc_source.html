42static bool internalReconstructInfo(gtsam::Key key, SymbolChar expected_chr,
43 ObjectId& object_label, FrameId& frame_id) {
44 // assume motion/pose key is a labelled symbol
45 if (!checkIfLabeledSymbol(key)) {
46 return false;
47 }
48
49 const gtsam::LabeledSymbol as_labeled_symbol(key);
50 if (as_labeled_symbol.chr() != expected_chr) {
51 return false;
52 }
53
54 frame_id = static_cast<FrameId>(as_labeled_symbol.index());
55
56 SymbolChar label = as_labeled_symbol.label();
57 object_label = label - '0';
58 return true;
59}

61bool checkIfLabeledSymbol(gtsam::Key key) {
62 const gtsam::LabeledSymbol asLabeledSymbol(key);
63 return (asLabeledSymbol.chr() > 0 && asLabeledSymbol.label() > 0);
64}

78bool ApplyFunctionalSymbol::operator()(gtsam::Key key) const {
79 const gtsam::Symbol sym(key);
80 switch (sym.chr()) {
81 case kPoseSymbolChar:
82 if (pose_func_) {
83 pose_func_(static_cast<FrameId>(sym.index()), sym);
84 }
85 return true;
86 case kObjectMotionSymbolChar: {
87 FrameId frame_id;
88 ObjectId object_id;
89 // attempt to get info about this key
90 bool valid = reconstructMotionInfo(key, object_id, frame_id);
91 // if valid and motion func registered, do call back
92 if (valid && object_motion_func_)
93 object_motion_func_(frame_id, object_id, gtsam::LabeledSymbol(key));
94 }
95 return true;
96 case kObjectPoseSymbolChar: {
97 FrameId frame_id;
98 ObjectId object_id;
99 // attempt to get info about this key
100 bool valid = reconstructPoseInfo(key, object_id, frame_id);
101 // if valid and motion func registered, do call back
102 if (valid && object_pose_func_)
103 object_pose_func_(frame_id, object_id, gtsam::LabeledSymbol(key));
104 }
105 return true;
106 case kStaticLandmarkSymbolChar: {
107 if (static_lmk_func_) {
108 static_lmk_func_(static_cast<TrackletId>(sym.index()), sym);
109 }
110 }
111 return true;
112 case kDynamicLandmarkSymbolChar: {
113 if (dynamic_lmk_func_) {
114 DynamicPointSymbol dps(key);
115 dynamic_lmk_func_(dps.trackletId(), dps);
116 }
117 }
118 return true;
119
120 default:
121 return false;
122 }
123}

161NoiseModels NoiseModels::fromBackendParams(
162 const BackendParams& backend_params) {
163 NoiseModels noise_models;
164
165 // odometry
166 gtsam::Vector6 odom_sigmas;
167 odom_sigmas.head<3>().setConstant(backend_params.odometry_rotation_sigma_);
168 odom_sigmas.tail<3>().setConstant(backend_params.odometry_translation_sigma_);
169 LOG(INFO) << "backend_params.odometry_rotation_sigma_ "
170 << backend_params.odometry_rotation_sigma_;
171 LOG(INFO) << "backend_params.odometry_translation_sigma_"
172 << backend_params.odometry_translation_sigma_;
173 LOG(INFO) << "Odom sigma: " << odom_sigmas;
174 noise_models.odometry_noise =
175 gtsam::noiseModel::Diagonal::Sigmas(odom_sigmas);
176 CHECK(noise_models.odometry_noise);
177
178 // first pose prior (world frame)
179 noise_models.initial_pose_prior =
180 gtsam::noiseModel::Isotropic::Sigma(6u, 0.000001);
181 CHECK(noise_models.initial_pose_prior);
182
183 // landmark motion noise (needed for some formulations ie world-centric)
184 noise_models.landmark_motion_noise = gtsam::noiseModel::Isotropic::Sigma(
185 3u, backend_params.motion_ternary_factor_noise_sigma_);
186 CHECK(noise_models.landmark_motion_noise);
187
188 // smoothing factor noise model (can be any variant of the smoothing factor as
189 // long as the dimensions are 6, ie. pose)
190 gtsam::Vector6 object_constant_vel_sigmas;
191 object_constant_vel_sigmas.head<3>().setConstant(
192 backend_params.constant_object_motion_rotation_sigma_);
193 object_constant_vel_sigmas.tail<3>().setConstant(
194 backend_params.constant_object_motion_translation_sigma_);
195 noise_models.object_smoothing_noise =
196 gtsam::noiseModel::Diagonal::Sigmas(object_constant_vel_sigmas);
197 CHECK(noise_models.object_smoothing_noise);
198
199 // TODO: CHECKS that values are not zero!!!
200
201 noise_models.static_point_noise = gtsam::noiseModel::Isotropic::Sigma(
202 3u, backend_params.static_point_noise_sigma_);
203 noise_models.dynamic_point_noise = gtsam::noiseModel::Isotropic::Sigma(
204 3u, backend_params.dynamic_point_noise_sigma_);
205
206 if (backend_params.use_robust_kernals_) {
207 LOG(INFO) << "Using robust huber loss function: "
208 << backend_params.k_huber_3d_points_;
209 noise_models.static_point_noise = gtsam::noiseModel::Robust::Create(
210 gtsam::noiseModel::mEstimator::Huber::Create(
211 backend_params.k_huber_3d_points_),
212 noise_models.static_point_noise);
213
214 noise_models.dynamic_point_noise = gtsam::noiseModel::Robust::Create(
215 gtsam::noiseModel::mEstimator::Huber::Create(
216 backend_params.k_huber_3d_points_),
217 noise_models.dynamic_point_noise);
218
219 // TODO: not k_huber_3d_points_ not just used for 3d points
220 noise_models.landmark_motion_noise = gtsam::noiseModel::Robust::Create(
221 gtsam::noiseModel::mEstimator::Huber::Create(
222 backend_params.k_huber_3d_points_),
223 noise_models.landmark_motion_noise);
224 }
225
226 return noise_models;
227}

229std::string DynoLikeKeyFormatter(gtsam::Key key) {
230 const gtsam::LabeledSymbol asLabeledSymbol(key);
231 if (asLabeledSymbol.chr() > 0 && asLabeledSymbol.label() > 0) {
232 // if used as motion
233 if (asLabeledSymbol.chr() == kObjectMotionSymbolChar ||
234 asLabeledSymbol.chr() == kObjectPoseSymbolChar) {
235 return (std::string)asLabeledSymbol;
236 }
237 return (std::string)asLabeledSymbol;
238 }
239
240 const gtsam::Symbol asSymbol(key);
241 if (asLabeledSymbol.chr() > 0) {
242 if (asLabeledSymbol.chr() == kDynamicLandmarkSymbolChar) {
243 const DynamicPointSymbol asDynamicPointSymbol(key);
244 return (std::string)asDynamicPointSymbol;
245 } else {
246 return (std::string)asSymbol;
247 }
248
249 } else {
250 return std::to_string(key);
251 }
252}

267std::string DynoLikeKeyFormatterVerbose(gtsam::Key key) {
268 const gtsam::LabeledSymbol asLabeledSymbol(key);
269 if (asLabeledSymbol.chr() > 0 && asLabeledSymbol.label() > 0) {
270 // if used as motion
271 if (asLabeledSymbol.chr() == kObjectMotionSymbolChar) {
272 ObjectId object_label;
273 FrameId frame_id;
274 CHECK(reconstructMotionInfo(asLabeledSymbol, object_label, frame_id));
275
276 std::stringstream ss;
277 ss << "H: label" << object_label << ", frames: " << frame_id - 1 << " -> "
278 << frame_id;
279 return ss.str();
280 } else if (asLabeledSymbol.chr() == kObjectPoseSymbolChar) {
281 ObjectId object_label;
282 FrameId frame_id;
283 CHECK(reconstructPoseInfo(asLabeledSymbol, object_label, frame_id));
284
285 std::stringstream ss;
286 ss << "K: label" << object_label << ", frame: " << frame_id;
287 return ss.str();
288 }
289 return (std::string)asLabeledSymbol;
290 }
291
292 const gtsam::Symbol asSymbol(key);
293 if (asLabeledSymbol.chr() > 0) {
294 if (asLabeledSymbol.chr() == kDynamicLandmarkSymbolChar) {
295 const DynamicPointSymbol asDynamicPointSymbol(key);
296
297 FrameId frame_id = asDynamicPointSymbol.frameId();
298 TrackletId tracklet_id = asDynamicPointSymbol.trackletId();
299 std::stringstream ss;
300 ss << kDynamicLandmarkSymbolChar << ": frame " << frame_id
301 << ", tracklet " << tracklet_id;
302 return ss.str();
303
304 } else {
305 return (std::string)asSymbol;
306 }
307
308 } else {
309 return std::to_string(key);
310 }
311}
