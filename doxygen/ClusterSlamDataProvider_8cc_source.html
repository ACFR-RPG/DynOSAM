212 void setIntrisics() {
213 std::ifstream fstream;
214 fstream.open(intrinsics_file_path_, std::ios::in);
215 if(!fstream) {
216 throw std::runtime_error("Could not open file " + intrinsics_file_path_ + " when trying to load Cluster Slam intrinsics!");
217 }

219 auto load_projection_matrix = [](std::ifstream& fstream, gtsam::Matrix34& projection_matrix) -> void {
220 //read the first 3 liens of the projection matrix and populate camera matrix
221 for(size_t i = 0; i < 3; i++) {
222 // std::vector<std::string> split_line;
223 //hmmm my getline function does not seem to work here (maybe because doubles so we'll do it manually!)
224 // CHECK(getLine(fstream, split_line));
225 // CHECK_EQ(split_line.size(), 4) << "Line was: " << container_to_string(split_line, ", ");
226
227 std::string s;
228 std::getline(fstream, s);
229
230 std::stringstream ss;
231 ss << s;
232
233 LOG(INFO) << ss.str();
234 for(size_t j = 0; j < 4; j++) {
235 double tmp;
236 ss >> tmp;
237 projection_matrix(i, j) = tmp;
238 }
239 }
240 }; //load_projection_matrix
241
242 //intrinscis file is
243 //3 x 4 projection matrix, over 3 lines
244 // new line
245 //3 x 4 projection matrix, over 3 lines
246 //populate camera 1 matrix
247 load_projection_matrix(fstream, projection_matrix_cam1_);
248
249 //read the line line
250 std::string s;
251 std::getline(fstream, s);
252 //populate camera 2 matrix
253 load_projection_matrix(fstream, projection_matrix_cam2_);
254
255 //set K matrices
256 K_cam_1 = projection_matrix_cam1_.topLeftCorner(3, 3);
257 K_cam_2 = projection_matrix_cam2_.topLeftCorner(3, 3);
258
259 double fx_left = K_cam_1(0, 0);
260 double fy_left = K_cam_1(1, 1);
261 double cu_left = K_cam_1(0, 2);
262 double cv_left = K_cam_1(1, 2);
263
264 double fx_right = K_cam_2(0, 0);
265 double fy_right = K_cam_2(1, 1);
266 double cu_right = K_cam_2(0, 2);
267 double cv_right = K_cam_2(1, 2);
268
269 //Projection_matrix = K * extrinsics
270 gtsam::Pose3 extrinsics_left = gtsam::Pose3(K_cam_1.inverse() * projection_matrix_cam1_);
271 gtsam::Pose3 extrinsics_right = gtsam::Pose3(K_cam_2.inverse() * projection_matrix_cam2_);
272
273 //I think this dataset has the projection matrix in the local left frame (i.e projection of 2 into 1)
274 //becuase the translation component of extrinsics_right is negative
275 //we want extrinsice to be in the world frame because this is what the camera params expect
276 //OR the dataset is in OpenCV convention...
277 extrinsics_right = extrinsics_right.inverse();
278
279 LOG(INFO) << "left e " << extrinsics_left;
280 LOG(INFO) << "right e " << extrinsics_right;
281
282 left_camera_params_ = CameraParams(
283 CameraParams::IntrinsicsCoeffs({fx_left, fy_left, cu_left, cv_left}),
284 CameraParams::DistortionCoeffs({0, 0, 0, 0}),
285 getRGB(0).size(),
286 DistortionModel::RADTAN,
287 extrinsics_left
288 );
289
290 CameraParams right_camera_params(
291 CameraParams::IntrinsicsCoeffs({fx_right, fy_right, cu_right, cv_right}),
292 CameraParams::DistortionCoeffs({0, 0, 0, 0}),
293 getRGB(0).size(),
294 DistortionModel::RADTAN,
295 extrinsics_right
296 );
297
298 stereo_camera_ = std::make_shared<StereoCamera>(left_camera_params_, right_camera_params);
299 stereo_matcher_ = std::make_shared<StereoMatcher>(stereo_camera_);
300
301 }

325 using LandmarksMap = gtsam::FastMap<int, Keypoint>;
326 using LandmarksMapPerFrame = gtsam::FastMap<FrameId, LandmarksMap>;
327
328 void loadLeftLandmarks(LandmarksMapPerFrame& detected_landmarks) {
329 std::vector<std::filesystem::path> files_in_directory = getAllFilesInDir(left_landmarks_folder_);
330
331 for (const std::filesystem::path& file_path : files_in_directory) {
332
333 //this should be the file name which we will then decompose into the frame id
334 //e.g. .../pose/0000.txt -> filename = 0000 -> frame id = int(0)
335 const std::string file_name = file_path.filename();
336 const int frame = std::stoi(file_name);
337
338 std::ifstream fstream;
339 fstream.open((std::string)file_path, std::ios::in);
340 if(!fstream) {
341 throw std::runtime_error("Could not open file " + (std::string)file_path + " when trying to load Cluster Slam landmarks information!");
342 }
343
344 //check we've never processed this frame before
345 CHECK(!detected_landmarks.exists(frame));
346
347 LandmarksMap detected_lmks_this_frame;
348
349 //expected size of each line (landamrk_id u v)
350 constexpr static size_t kExpectedSize = 3u;
351 while (!fstream.eof()) {
352 std::vector<std::string> split_line;
353 if(!getLine(fstream, split_line)) continue;
354
355 CHECK_EQ(split_line.size(), kExpectedSize) << "Line present in landmarks file does not have the right length - " << container_to_string(split_line);
356
357 const int landmark_id = std::stoi(split_line.at(0));
358 const double u = std::stod(split_line.at(1));
359 const double v = std::stod(split_line.at(2));
360
361 detected_lmks_this_frame.insert2(landmark_id, Keypoint(u, v));
362 }
363
364 detected_landmarks.insert2(frame, detected_lmks_this_frame);
365 }
366
367 VLOG(20) << "Loaded " << detected_landmarks.size() << " detected landmarks...";
368
369 }

371 //Landmark to cluster id mapping.
372 //the landmark id is the key in LandmarksMap
373 void loadLandMarkMapping(gtsam::FastMap<int, int>& landmark_mapping) {
374 std::ifstream fstream;
375 fstream.open(landmark_mapping_file_path_, std::ios::in);
376 if(!fstream) {
377 throw std::runtime_error("Could not open file " + landmark_mapping_file_path_ + " when trying to load Cluster Slam landmarks mapping!");
378 }
379
380 //expected size of each line (landamrk_id cluster_id)
381 constexpr static size_t kExpectedSize = 2u;
382 while (!fstream.eof()) {
383 std::vector<std::string> split_line;
384 if(!getLine(fstream, split_line)) continue;
385
386 CHECK_EQ(split_line.size(), kExpectedSize) << "Line present in landmark mapping file does not have the right length - " << container_to_string(split_line);
387
388 const int landmark_id = std::stoi(split_line.at(0));
389 //clusters start at 0 here, but that includes background label (which is 0)
390 //since the line ordering of the trajectories files determines the label
391 //the camera pose is always first (line = 0), so any cluster that is associated to an object
392 //and not the camera pose, should have idx > 0
393 const int cluster_id = std::stoi(split_line.at(1));
394
395 landmark_mapping.insert2(landmark_id, cluster_id);
396 }
397 }

399 //we're going to have to associate the detected bounding boxes
400 //with the actual landmark ids so that the gt labels match with the
401 //tracking labels....
402 //we will have to do this by taking all the landmarks of detected features
403 //and check which objects lies within the bounding box....
404 //this is almost the definitions of heinous but like, whatever....
405 //this function updates
406 //returns the new instance mask
407 //from the original mask, relabel the pixel values using the gt tracks
408 void associateDetectedBBWithObject(const cv::Mat& instance_mask, FrameId frame_id, cv::Mat& relabelled_mask) const {
409 const GroundTruthInputPacket& gt_packet = getGtPacket(frame_id);
410 ObjectIds object_ids = vision_tools::getObjectLabels(instance_mask);
411 const size_t n = object_ids.size();
412
413 instance_mask.copyTo(relabelled_mask);
414
415 if(n == 0) {
416 return;
417 }
418
419
420 //sort kps map into a map of clusters -> [kps....]
421 const LandmarksMap& kps_map = left_landmarks_map_.at(frame_id);
422 //cluster id (track id -> all keypoints for that cluster)
423 std::set<int> cluster_id_set;
424 gtsam::FastMap<int, Keypoints> clustered_kps;
425 for(const auto&[landmark_id, kp] : kps_map) {
426 const auto cluster_id = landmark_mapping_.at(landmark_id);
427 cluster_id_set.insert(cluster_id);
428 if(!clustered_kps.exists(cluster_id)) {
429 clustered_kps.insert2(cluster_id, Keypoints());
430 }
431 clustered_kps.at(cluster_id).push_back(kp);
432 }
433
434 std::vector<int> cluster_ids(cluster_id_set.begin(), cluster_id_set.end());
435
436 //want to assign object ids to clusters
437 const size_t m = cluster_ids.size();
438 CHECK_EQ(m, clustered_kps.size());
439
440 LOG(INFO) << n <<" " << m;
441
442 Eigen::MatrixXd cost;
443 cost.resize(n, m);
444
445 for(size_t rows = 0; rows < n; rows++) {
446 ObjectId object_id = object_ids.at(rows);
447
448 // get the binary mask just for this object, we will use this too find a bounding box
449 //and count how many kp's fall into this mask
450 const cv::Mat object_mask = relabelled_mask == object_id;
451
452 cv::Rect bb;
453 if(!vision_tools::findObjectBoundingBox(instance_mask, object_id, bb)) {
454 VLOG(10) << "No bb could be found for detected object " << object_id << " at frame " << frame_id << ". Skipping association and removing object from mask";
455 relabelled_mask.setTo(cv::Scalar(0), object_mask);
456 continue;
457 }
458
459
460 for(size_t cols = 0; cols < m; cols++) {
461 int cluster_id = cluster_ids.at(cols);
462 const Keypoints& kps = clustered_kps.at(cluster_id);
463
464 //(inverse) cost for the object id with this cluster id
465 //hack - make really small number so we divide by zero!!
466 double object_count = 0.000001;
467 for(const auto& kp : kps) {
468 cv::Point2f pt(utils::gtsamPointToCv(kp));
469 if(bb.contains(pt)) {
470 object_count++;
471 }
472 }
473 cost(rows, cols) = object_count;
474 }
475 }
476
477 //apply scaling to the costs to turn the cost function from an argmax to an argmin problem
478 //which the hungrian problem sovles
479 //ie we want the assignment that maximuses the number of kp's in the object insancec
480 Eigen::MatrixXd loged_costs = cost.unaryExpr([](double x) { return 1.0/x * 10; });
481 Eigen::VectorXi assignment;
482 internal::HungarianAlgorithm().solve(loged_costs, assignment);
483
484 // for(size_t i = 0; i < assignment.size(); i++) {
485 // int j = assignment[i];
486
487 // //the i-th object is assignment to the j-th cluster
488 // ObjectId object_id = object_ids.at(i);
489 // int assigned_custer = cluster_ids.at(j);
490
491 // const cv::Mat object_mask = relabelled_mask == object_id;
492 // relabelled_mask.setTo(cv::Scalar(assigned_custer), object_mask);
493 // }
494 ObjectIds old_labels = object_ids;
495 ObjectIds new_labels;
496 for(size_t i = 0; i < assignment.size(); i++) {
497 int j = assignment[i];
498
499 // //the i-th object is assignment to the j-th cluster
500 // ObjectId instance_id = instance_ids.at(i);
501 int assigned_custer = cluster_ids.at(j);
502 new_labels.push_back(assigned_custer);
503
504 // LOG(INFO) << "Relabelled " << instance_id << " to " << assigned_object_id;
505
506 // const cv::Mat object_mask = relabelled_mask == instance_id;
507 // relabelled_mask.setTo(cv::Scalar(assigned_object_id), object_mask);
508 }
509
510 vision_tools::relabelMasks(instance_mask, relabelled_mask, old_labels, new_labels);
511
512 }

515 void setGroundTruthPacket(GroundTruthPacketMap& ground_truth_packets) {
516 //load the poses
517 //first line contains camera pose
518 //consequative lines contain object ids which are indexed from 0, but we will index from 1
519 //expect the file name to give us the frame we are working with!!
520 std::vector<std::filesystem::path> files_in_directory = getAllFilesInDir(pose_folder_path_);
521
522 gtsam::Pose3 initial_pose = gtsam::Pose3::Identity();
523 bool initial_frame_set = false;
524
525 for (const std::filesystem::path& pose_file_path : files_in_directory) {
526
527 //this should be the file name which we will then decompose into the frame id
528 //e.g. .../pose/0000.txt -> filename = 0000 -> frame id = int(0)
529 const std::string file_name = pose_file_path.filename();
530 const int frame = std::stoi(file_name);
531 const gtsam::Pose3Vector poses = processPoseFile(pose_file_path, ground_truth_packets);
532
533 //check we've never processed this frame before
534 CHECK(!ground_truth_packets.exists(frame));
535
536 if(frame > 0) {
537 //check we have the previous frame!! (ie.e we process in order!)
538 CHECK(ground_truth_packets.exists(frame-1));
539 }
540
541 GroundTruthInputPacket gt_packet;
542 gt_packet.frame_id_ = frame;
543 gt_packet.timestamp_ = getTimestamp(frame);
544 // Quoting the dataset home page:
545 // For each trajectory file under pose/ directory, each line in each file represents
546 // the pose of each cluster except for the first line - that line represents the camera pose.
547 for(size_t i = 0; i < poses.size(); i++) {
548
549 //pose is in world coordinate convention so we put into camera convention (the camera_to_world.inverse())
550 // gtsam::Pose3 pose = camera_to_world.inverse() * poses.at(i);
551 gtsam::Pose3 pose = poses.at(i);
552
553 const static gtsam::Pose3 camera_to_world(gtsam::Rot3::RzRyRx(M_PI_2, 0, M_PI_2), gtsam::traits<gtsam::Point3>::Identity());
554
555 // LOG(INFO) << pose;
556
557 // pose = camera_to_world.inverse() * pose;
558
559 // LOG(INFO) << pose;
560
561
562 if(!initial_frame_set) {
563 //expect very first pose (first frame, and first pose within that frame) to be the first camera pose
564 initial_pose = pose;
565 initial_frame_set = true;
566 LOG(INFO) << pose;
567 }
568
569 // offset initial pose so we start at "0, 0, 0"
570 pose = initial_pose.inverse() * pose;
571
572
573 if(i == 0) {
574 gt_packet.X_world_ = pose;
575 }
576 else {
577 ObjectPoseGT object_pose_gt;
578 object_pose_gt.frame_id_ = frame;
579 object_pose_gt.object_id_ = i;
580 object_pose_gt.L_world_ = pose;
581 object_pose_gt.L_camera_ = gt_packet.X_world_.inverse() * object_pose_gt.L_world_;
582 gt_packet.object_poses_.push_back(object_pose_gt);
583 }
584 }
585
586 if(frame > 0) {
587 const GroundTruthInputPacket& previous_gt_packet = ground_truth_packets.at(frame - 1);
588 gt_packet.calculateAndSetMotions(previous_gt_packet);
589 }
590
591 ground_truth_packets.insert2(frame, gt_packet);
592
593 }

596 gtsam::Pose3Vector processPoseFile(const std::string& pose_file, GroundTruthPacketMap& ground_truth_packets) {
597 std::ifstream fstream;
598 fstream.open(pose_file, std::ios::in);
599 if(!fstream) {
600 throw std::runtime_error("Could not open file " + pose_file + " when trying to load Cluster Slam pose information!");
601 }
602
603 //expected size of each line (3 translation components + 4 for quaternion)
604 constexpr static size_t kExpectedSize = 7u;
605
606 //pose string expected to be in form x,y,z, qx, qy, qz, qw
607 auto construct_pose =[](const std::vector<std::string>& pose_string) -> gtsam::Pose3 {
608 const double x = std::stod(pose_string.at(0));
609 const double y = std::stod(pose_string.at(1));
610 const double z = std::stod(pose_string.at(2));
611
612 const double qx = std::stod(pose_string.at(3));
613 const double qy = std::stod(pose_string.at(4));
614 const double qz = std::stod(pose_string.at(5));
615 const double qw = std::stod(pose_string.at(6));
616
617 return gtsam::Pose3(
618 gtsam::Rot3(qw, qx, qy, qz),
619 gtsam::Point3(x, y, z)
620 );
621 };
622
623 gtsam::Pose3Vector poses;
624 while (!fstream.eof()) {
625 std::vector<std::string> split_line;
626 if(!getLine(fstream, split_line)) continue;
627
