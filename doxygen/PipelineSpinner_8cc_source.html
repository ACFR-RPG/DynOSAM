36void PipelineSpinner::registerPipeline(PipelineBase::Ptr pipeline, bool parallel_run) {
37 const std::string name = pipeline->getModuleName();
38 checkAndThrow(!pipelines_.exists(name), "Cannot register pipeline with name " + name + " as it has already been registered");
39
40 if(parallel_run) {
41 pipelines_.insert({name, std::move(std::make_unique<AsynchronousSpinnerPair>(pipeline))});
42 }
43 else {
44 pipelines_.insert({name, std::move(std::make_unique<SynchronousSpinnerPair>(pipeline))});
45 }
46
47 ordering_.push_back(name);
48}

76void PipelineSpinner::appyAll(std::function<void(SpinnerPair::UniquePtr&)> apply) {
77 for(const std::string& name : ordering_) {
78 SpinnerPair::UniquePtr& spinner_pair = pipelines_.at(name);
79 apply(spinner_pair);

98void PipelineSpinner::AsynchronousSpinnerPair::launch() {
99 CHECK(spinner_ == nullptr);
100
101 const std::string spinner_name = p_->getModuleName() + "-spinner";
102 spinner_ = std::make_unique<Spinner>(std::bind(&PipelineBase::spin, p_.get()), spinner_name);
103 LOG(INFO) << "Launching spinner: " << spinner_name;
