43void PipelineSpinner::registerPipeline(PipelineBase::Ptr pipeline,
44 bool parallel_run) {
45 const std::string name = pipeline->getModuleName();
46 checkAndThrow(!pipelines_.exists(name),
47 "Cannot register pipeline with name " + name +
48 " as it has already been registered");
49
50 if (parallel_run) {
51 pipelines_.insert(
52 {name, std::move(std::make_unique<AsynchronousSpinnerPair>(pipeline))});
53 } else {
54 pipelines_.insert(
55 {name, std::move(std::make_unique<SynchronousSpinnerPair>(pipeline))});
56 }
57
58 ordering_.push_back(name);
59}

61void PipelineSpinner::launchAll() {

69void PipelineSpinner::spinOnceAll() {

88void PipelineSpinner::appyAll(
89 std::function<void(SpinnerPair::UniquePtr&)> apply) {
90 for (const std::string& name : ordering_) {
91 SpinnerPair::UniquePtr& spinner_pair = pipelines_.at(name);
92 apply(spinner_pair);

114void PipelineSpinner::AsynchronousSpinnerPair::launch() {
115 CHECK(spinner_ == nullptr);
116
117 const std::string spinner_name = p_->getModuleName() + "-spinner";
118 spinner_ = std::make_unique<Spinner>(std::bind(&PipelineBase::spin, p_.get()),
119 spinner_name);
120 LOG(INFO) << "Launching spinner: " << spinner_name;
121}

131 p_->shutdown();
132 // then shutdown spinner
133 spinner_->shutdown();
