69void BackendInbuiltDisplayRos::spinOnce(
70 const BackendOutputPacket::ConstPtr& backend_output) {
71 publishPointCloud(static_tracked_points_pub_,
72 backend_output->static_landmarks,
73 backend_output->pose());
74 CloudPerObject clouds_per_obj = publishPointCloud(
75 dynamic_tracked_points_pub_, backend_output->dynamic_landmarks,
76 backend_output->pose());
77
78 publishObjectBoundingBox(object_aabb_pub_, nullptr, /* no Obb publisher */
79 clouds_per_obj, utils::fromRosTime(node_->now()),
80 "backend");
81 publishObjectPositions(
82 object_pose_pub_, backend_output->optimized_object_poses,
83 backend_output->getFrameId(), backend_output->getTimestamp(), "backend");
84
85 publishObjectPaths(
86 object_pose_path_pub_, backend_output->optimized_object_poses,
87 backend_output->getFrameId(), backend_output->getTimestamp(), "backend", 60);
88
89 {
90 nav_msgs::msg::Odometry odom_msg;
91 utils::convertWithHeader(backend_output->pose(), odom_msg,
92 backend_output->getTimestamp(), "world", "camera");
93 odometry_pub_->publish(odom_msg);
94 }
95
96 {
97 nav_msgs::msg::Path odom_path_msg;
98
99 for (const gtsam::Pose3& T_world_camera :
100 backend_output->optimized_camera_poses) {
101 // optimized camera traj
102 geometry_msgs::msg::PoseStamped pose_stamped;
103 utils::convertWithHeader(T_world_camera, pose_stamped,
104 backend_output->getTimestamp(), "world");
105
106 static std_msgs::msg::Header header;
107 header.stamp = utils::toRosTime(backend_output->getTimestamp());
108 header.frame_id = "world";
109 odom_path_msg.header = header;
110 odom_path_msg.poses.push_back(pose_stamped);
111 }
112
113 odometry_path_pub_->publish(odom_path_msg);
114 }
115}
