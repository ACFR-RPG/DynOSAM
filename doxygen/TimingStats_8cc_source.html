46void TimingStatsCollector::tocAndLog() {
47 if(is_valid_) {
48 auto toc = Timer::toc<std::chrono::nanoseconds>(tic_time_);
49 auto milliseconds = std::chrono::duration_cast<std::chrono::milliseconds>(toc);
50 collector_.AddSample( static_cast<double>(milliseconds.count()));
51 is_valid_ = false;
