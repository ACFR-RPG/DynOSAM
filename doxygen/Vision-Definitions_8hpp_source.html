118template <>
119inline std::string to_string(const FeatureTrackerInfo& info) {
120 std::stringstream ss;
121 ss << "FeatureTrackerInfo: \n"
122 << " - frame id: " << info.frame_id << "\n"
123 << " - timestamp: " << std::setprecision(15) << info.timestamp << "\n"
124 << "\t- # optical flow: " << info.static_track_optical_flow << "\n"
125 << "\t- # detections: " << info.static_track_detections << "\n";
126
127 for (const auto& [object_id, object_status] : info.dynamic_track) {
128 ss << "\t- Object: " << object_id << ": \n";
129 ss << "\t\t - num_track " << object_status.num_track << "\n";
130 ss << "\t\t - num_sampled " << object_status.num_sampled << "\n";
131
132 if (VLOG_IS_ON(20)) {
133 ss << "\t\t - num_previous_track " << object_status.num_previous_track
134 << "\n";
135 ss << "\t\t - num_outside_shrunken_image "
136 << object_status.num_outside_shrunken_image << "\n";
137 ss << "\t\t - num_zero_flow " << object_status.num_zero_flow << "\n";
138 ss << "\t\t - num_tracked_with_different_label "
139 << object_status.num_tracked_with_different_label << "\n";
140 ss << "\t\t - num_tracked_with_background_label "
141 << object_status.num_tracked_with_background_label << "\n";
142 }
143 };
144 return ss.str();
