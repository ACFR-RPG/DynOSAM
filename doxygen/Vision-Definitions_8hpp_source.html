96template<>
97inline std::string to_string(const FeatureTrackerInfo& info) {
98 std::stringstream ss;
99 ss << "FeatureTrackerInfo: \n"
100 << " - frame id: " << info.frame_id << "\n"
101 << " - timestamp: " << std::setprecision(15) << info.timestamp << "\n"
102 << "\t- # optical flow: " << info.static_track_optical_flow << "\n"
103 << "\t- # detections: " << info.static_track_detections << "\n";
104
105 for(const auto& [object_id, object_status] : info.dynamic_track) {
106 ss << "\t- Object: " << object_id << ": \n";
107 ss << "\t\t - num_track " << object_status.num_track << "\n";
108 ss << "\t\t - num_sampled " << object_status.num_sampled << "\n";
109
110 if(VLOG_IS_ON(20)) {
111 ss << "\t\t - num_previous_track " << object_status.num_previous_track << "\n";
112 ss << "\t\t - num_outside_shrunken_image " << object_status.num_outside_shrunken_image << "\n";
113 ss << "\t\t - num_zero_flow " << object_status.num_zero_flow << "\n";
114 ss << "\t\t - num_tracked_with_different_label " << object_status.num_tracked_with_different_label << "\n";
115 ss << "\t\t - num_tracked_with_background_label " << object_status.num_tracked_with_background_label << "\n";
116 }
117 };
