97 void finalize() {
98
99 totalsize = ytmparray.size();

101 std::sort(xtmparray.begin(), xtmparray.end());
102 std::sort(ytmparray.begin(), ytmparray.end());

123 u32 count(point xmin, point xmax, point ymin, point ymax) const {
124
125 if (!init)
126 return 0;
127
128 // If needed, swap arguments
129 if (xmax < xmin)
130 pswap(xmax, xmin);
131 if (ymax < ymin)
132 pswap(ymax, ymin);
133
134 u32 sum = 0;
135
136 const node * list[maxterminals];
137 u32 listcur = 0;
138 findnodes(&start, xmin, xmax, list, listcur);
139
140 const u32 ncount = listcur;
141
142 for (u32 k = 0; k < ncount; k++) {
143 const node * const n = list[k];
144 const u32 max = n->ycount;
145 if (!max)
146 continue;
147
148 const u32 lower = binarynext(n->ypoints, n->ycount, ymin);
149 const u32 upper = binarynext(n->ypoints, n->ycount, ymax + 1);
150
151 sum += upper - lower;
152 }
153
154 return sum;
155 }

318 // A binary search that returns the index if found, the next index if not
319 u32 binarynext(const pty * const arr, const u32 ycount, const point goal) const {
320
321 if (ycount == 0)
322 return 0;
323
324 const u32 max = ycount - 1;
325
326 // These have to be signed to avoid overflow. s64 to contain u32.
327 s64 t, left = 0, right = max;
328
329 do {
330 t = (left + right) / 2;
331
332 if (goal < arr[t].y)
333 right = t - 1;
334 else
335 left = t + 1;
336
337 } while (left <= right && goal != arr[t].y);
338
339 // We might've landed in the middle. Linearly get the earliest.
340 if (arr[t].y == goal) {
341 while (t && arr[t-1].y == goal)
342 t--;
343 return t;
344 }
345
346 if (arr[t].y < goal)
347 return t + 1;
348
349 return t;
350 }

352 // Same for X-sorted arrays
353 u32 binarynextx(const std::vector<ptx> &arr, const point goal) const {
354
355 if (arr.size() == 0)
356 return 0;
357
358 const u32 max = arr.size() - 1;
359
360 // These have to be signed to avoid overflow. s64 to contain u32.
361 s64 t, left = 0, right = max;
362
363 do {
364 t = (left + right) / 2;
365
366 if (goal < arr[t].x)
367 right = t - 1;
368 else
369 left = t + 1;
370
371 } while (left <= right && goal != arr[t].x);
372
373 // We might've landed in the middle. Linearly get the earliest.
374 if (arr[t].x == goal) {
375 while (t && arr[t-1].x == goal)
376 t--;
377 return t;
378 }
379
380 if (arr[t].x < goal)
381 return t + 1;
382
383 return t;
384 }

386 void build() {
387 if (totalsize < 1) {
388 return;
389 }
390#ifndef LESSRAM64
391 start.ypoints = (pty *) xcalloc(totalsize, sizeof(pty));
392 memcpy(start.ypoints, &ytmparray[0], totalsize * sizeof(pty));
393 start.ycount = totalsize;
394#endif
395 ytmparray.clear();
396
397 const u32 medianidx = totalsize / 2;
398 const point median = xtmparray[medianidx].x;
399
400 start.min = xtmparray[0].x;
401 start.max = xtmparray[totalsize - 1].x;
402
403 // Ok, divide it between everyone
404 start.left = build(start.min, median);
405
406 // If it's really small, it may not have a right branch..
407 if (median != start.max)
408 start.right = build(median + 1, start.max);
409
410#ifdef LESSRAM64
411 mergekids(start.ypoints, start.ycount, start.left, start.right);
412#endif
413 }

415 node *build(const point min, const point max) {
416
417// if (min > max)
418// abort();
419
420 const u32 lower = binarynextx(xtmparray, min);
421 const u32 upper = binarynextx(xtmparray, max + 1);
422
423 // Quick check: if nothing below me, no need to create anything below
424 if (lower == upper) {
425 return NULL;
426 }
427
428 node * const n = newnode();
429
430 n->min = xtmparray[lower].x;
431 n->max = xtmparray[upper-1].x;
432
433 // If no kids, create the array; otherwise, recurse
434 if (n->min == n->max) {
435
436 const u32 size = upper - lower;
437
438 n->ypoints = newpty(size);
439
440 u32 i;
441 for (i = 0; i < size; i++) {
442#ifdef LESSRAM64
443 n->ypoints[i].offset = lower + i;
444#else
445 n->ypoints[i].ptr = xtmparray[lower + i].ptr;
446#endif
447 n->ypoints[i].y = xtmparray[lower + i].y;
448 }
449
450 n->ycount = size;
451
452 std::sort(n->ypoints, n->ypoints + size);
453 } else {
454 const u32 median = (n->min + n->max) / 2;
455
456 n->left = build(n->min, median);
457 n->right = build(median + 1, n->max);
458
459 // For faster builds, we merge our kids' arrays into ours
460 mergekids(n->ypoints, n->ycount, n->left, n->right);
461 }
462
463 return n;
464 }

491 void findnodes(const node * const n, const point xmin, const point xmax,
492 const node * list[], u32 &listcur) const {
493 if (!n)
494 return;
495
496 // Fast outs
497 if (xmin > n->max)
498 return;
499 if (xmax < n->min)
500 return;
501 if (!n->ycount)
502 return;
503
504 // Prefetch nodes to cache
505 fetch(n->left);
506 fetch(n->right);
507
508 if (xmin <= n->min && xmax >= n->max) {
509 list[listcur] = n;
510 listcur++;
511
512// if (listcur >= maxterminals)
513// abort();
514
515 return;
516 }
517
518 findnodes(n->left, xmin, xmax, list, listcur);
519 findnodes(n->right, xmin, xmax, list, listcur);
520 }

522 void mergekids(pty * &arr, u32 &ycount, node * const __restrict__ left,
523 node * const __restrict__ right) const {
524
525 u32 l, r;
526 const u32 lmax = left ? left->ycount : 0;
527 const u32 rmax = right ? right->ycount : 0;
528
529 arr = newpty(lmax + rmax);
530 ycount = lmax + rmax;
531
532 u32 cur = 0;
533
534 for (l = 0, r = 0; l < lmax || r < rmax;) {
535 // Special cases first: if one array is out
536 if (l == lmax) {
537 memcpy(&arr[cur], &right->ypoints[r],
538 (rmax - r) * sizeof(pty));
539 break;
540 } else if (r == rmax) {
541 memcpy(&arr[cur], &left->ypoints[l],
542 (lmax - l) * sizeof(pty));
543 break;
544 } else {
545 if (left->ypoints[l].y <= right->ypoints[r].y) {
546 arr[cur] = left->ypoints[l];
547 cur++;
548 l++;
549 } else {
550 arr[cur] = right->ypoints[r];
551 cur++;
552 r++;
553 }
554 }
555 }

601 // A memory pool for nodes, to save on the housekeeping overhead,
602 // and hopefully gain a bit in cache advantages.
603 void initpools(const u32 totalcount, const point maxrange) {
604
605 const u32 amount = u32min(totalcount, maxrange);
606
607 // If we have N points, the likely amount of nodes is 2N - 1.
608 nodepoolcount = amount*2;
609 nodepoolgiven = 0;
610
611 const u32 log2ed = log2f(nodepoolcount) + 1;
612
613 // Pre-allocate n*log(n) ptys
614 ptypoolcount = totalcount * log2ed;
615 ptypoolgiven = 0;
616
617 // Given the estimated max, we can calculate the max number
618 // of terminal nodes - it's at most 2 per level (proof in the papers).
619 maxterminals = log2ed;
620 maxterminals *= 2;
621
622 nodepool = new node[nodepoolcount];
623 ptypool = (pty *) xcalloc(ptypoolcount, sizeof(pty));
624 }
625
626 node * newnode() {
627 if (nodepoolgiven < nodepoolcount) {
628 node * const out = &nodepool[nodepoolgiven];
629 nodepoolgiven++;
630 return out;
631 } else {
632 return new node;
633 }
634 }
635
636 pty * newpty(const u32 num) const {
637 if (ptypoolgiven + num <= ptypoolcount) {
638 pty * const out = &ptypool[ptypoolgiven];
639 ptypoolgiven += num;
640 return out;
641 } else {
642 return (pty *) xcalloc(num, sizeof(pty));
643 }
644 }
645
646 bool isnodepooled(const node * const n) const {
647
648 if (!nodepool)
649 return false;
650
651 if (n >= &nodepool[0] && n <= &nodepool[nodepoolcount - 1])
652 return true;
653 return false;
654 }

656 bool isptypooled(const pty * const p) const {
657
658 if (!ptypool)
659 return false;
660
661 if (p >= &ptypool[0] && p <= &ptypool[ptypoolcount - 1])
662 return true;
663 return false;
664 }
