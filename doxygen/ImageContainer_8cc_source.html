60ImageContainer::Ptr ImageContainer::RectifyImages(ImageContainer::Ptr images, const UndistorterRectifier& undistorter) {
61 //do deep copy (the underlying ImageWrapper class has a deep copy in its copy constructor so
62 //the image data is cloned)
63 ImageContainer::Ptr undistorted_images = std::make_shared<ImageContainer>(*images);
64
65 static constexpr size_t N = ImageContainer::N;
66 for (size_t i = 0; i < N; i++) {
67 internal::select_apply<N>(i, [&](auto I){
68 using ImageType = typename ImageContainer::ImageTypeStruct<I>;
69
70 //get reference to image and modify in place
71 //TODO: bug!!?
72 cv::Mat& distorted_image = undistorted_images->template get<ImageType>();
73 undistorter.undistortRectifyImage(distorted_image, distorted_image);
74
75 });
76 }
77
78 return undistorted_images;
79}

97std::string ImageContainer::toString() const {
98 std::stringstream ss;
99 ss << "Timestamp: " << timestamp_ << "\n";
100 ss << "Frame Id: " << frame_id_ << "\n";
101
102 const std::string image_config = isImageRGB() ? "RGB" : "Grey";
103 ss << "Configuration: " << image_config
104 << " Depth (" << hasDepth() << ") Semantic Mask (" << hasSemanticMask()
105 << ") Motion Mask (" << hasMotionMask()
106 << ") Class Segmentation (" << hasClassSegmentation() << ")" << "\n";
107

150void ImageContainer::validateSetup() const {
151 //check image sizes are the same
152 Base::validateSetup();
153
154 //TODO: shoudl eventually change to exception
155 CHECK(!getImage().empty()) << "RGBMono image must not be empty!";
156 CHECK(!getOpticalFlow().empty()) << "OPticalFlow image must not be empty!";
157
158 //must have at least one of semantic mask or motion mask
159 CHECK(!getSemanticMask().empty() || !getMotionMask().empty()) << "At least one of the masks (semantic or motion) must be valid. Both are empty";
160
161 //should only provide one mask (for clarity - which one to use if both given?)
162 if(hasSemanticMask()) {
163 CHECK(!hasMotionMask()) << "Semantic mask has been provided, so motion mask should be empty!";
164 }
165
166 if(hasMotionMask()) {
167 CHECK(!hasSemanticMask()) << "Motion mask has been provided, so semantic mask should be empty!";
168 }
169}
