41ImageContainer ImageContainer::clone() const {
42 ImageContainer container(this->frameId(), this->timestamp());
43
44 // explicit deep copies for all key image pairs
45 for (const auto& [k, v] : images_) {
46 container.images_.emplace(k, v.clone());
47 }
48
49 return container;
50}

139ImageContainerDeprecate::Ptr ImageContainerDeprecate::RectifyImages(
140 ImageContainerDeprecate::Ptr images,
141 const UndistorterRectifier& undistorter) {
142 // do deep copy (the underlying ImageWrapper class has a deep copy in its copy
143 // constructor so the image data is cloned)
144 ImageContainerDeprecate::Ptr undistorted_images =
145 std::make_shared<ImageContainerDeprecate>(*images);
146
147 static constexpr size_t N = ImageContainerDeprecate::N;
148 for (size_t i = 0; i < N; i++) {
149 internal::select_apply<N>(i, [&](auto I) {
150 using ImageType = typename ImageContainerDeprecate::ImageTypeStruct<I>;
151
152 // get reference to image and modify in place
153 // TODO: bug!!?
154 cv::Mat& distorted_image = undistorted_images->template get<ImageType>();
155 undistorter.undistortRectifyImage(distorted_image, distorted_image);
156 });
157 }
158
159 return undistorted_images;
160}

224void ImageContainerDeprecate::validateSetup() const {
225 // check image sizes are the same
226 Base::validateSetup();
227
228 // TODO: shoudl eventually change to exception
229 CHECK(!getImage().empty()) << "RGBMono image must not be empty!";
230 CHECK(!getOpticalFlow().empty()) << "OPticalFlow image must not be empty!";
231
232 // must have at least one of semantic mask or motion mask
233 CHECK(!getSemanticMask().empty() || !getMotionMask().empty())
234 << "At least one of the masks (semantic or motion) must be valid. Both "
235 "are empty";
236
237 // should only provide one mask (for clarity - which one to use if both
238 // given?)
239 if (hasSemanticMask()) {
240 CHECK(!hasMotionMask())
241 << "Semantic mask has been provided, so motion mask should be empty!";
242 }
243
244 if (hasMotionMask()) {
245 CHECK(!hasSemanticMask())
246 << "Motion mask has been provided, so semantic mask should be empty!";
247 }
248}
