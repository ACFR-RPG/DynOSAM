40ImageContainer ImageContainer::clone() const {
41 ImageContainer container(this->frameId(), this->timestamp());
42
43 // explicit deep copies for all key image pairs
44 for (const auto& [k, v] : images_) {
45 container.images_.emplace(k, v.clone());
46 }
47
48 return container;
49}

105std::string ImageContainer::toString() const {
106 std::stringstream ss;
107 ss << "[ ";
108 for (const auto& [key, key_image_pair] : images_) {
109 CHECK_NOTNULL(key_image_pair.ptr);
110 ss << "{" << key << ": " << key_image_pair.ptr->toString() << "} ";
111 }
112 ss << "]";
113 return ss.str();
114}
