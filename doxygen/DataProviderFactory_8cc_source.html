42DataProvider::Ptr DataProviderFactory::Create(const std::string& dataset_folder_path, const std::string& params_folder_path, DatasetType dataset_type) {
43 if(dataset_type == DatasetType::KITTI) {
44 LOG(INFO) << "Using KITTI dataset at path: " << dataset_folder_path;
45 KittiDataLoader::Params params = KittiDataLoader::Params::fromYaml(params_folder_path);
46 auto loader = std::make_shared<KittiDataLoader>(dataset_folder_path, params);
47
48 loader->setStartingFrame(FLAGS_starting_frame);
49 loader->setEndingFrame(FLAGS_ending_frame);
50 return loader;
51 }
52 else if(dataset_type == DatasetType::VIRTUAL_KITTI) {
53 LOG(INFO) << "Using Virtual KITTI dataset at path: " << dataset_folder_path;
54 VirtualKittiDataLoader::Params params = VirtualKittiDataLoader::Params::fromYaml(params_folder_path);
55 auto loader = std::make_shared<VirtualKittiDataLoader>(dataset_folder_path, params);
56 loader->setStartingFrame(FLAGS_starting_frame);
57 loader->setEndingFrame(FLAGS_ending_frame);
58 return loader;
59 }
60 else if(dataset_type == DatasetType::CLUSTER) {
61 LOG(INFO) << "Using Cluster (SLAM) dataset at path: " << dataset_folder_path;
62 auto loader = std::make_shared<ClusterSlamDataLoader>(dataset_folder_path);
63 loader->setStartingFrame(FLAGS_starting_frame);
64 loader->setEndingFrame(FLAGS_ending_frame);
65 return loader;
66 }
67 else if (dataset_type == DatasetType::OMD) {
68 LOG(INFO) << "Using Oxford Multi-motion Dataset dataset at path: " << dataset_folder_path;
69 auto loader = std::make_shared<OMDDataLoader>(dataset_folder_path);
70 loader->setStartingFrame(FLAGS_starting_frame);
71 loader->setEndingFrame(FLAGS_ending_frame);
72 return loader;
73 }
74 else if (dataset_type == DatasetType::ARIA) {
75 LOG(INFO) << "Using ARIA dataset at path: " << dataset_folder_path;
76 auto loader = std::make_shared<ProjectARIADataLoader>(dataset_folder_path);
77 loader->setStartingFrame(FLAGS_starting_frame);
78 loader->setEndingFrame(FLAGS_ending_frame);
79 return loader;
80 }
81 else {
82 throw std::runtime_error("Unable to construct Dataprovider - unknown dataset type: " + static_cast<int>(dataset_type));
83 }
84}
