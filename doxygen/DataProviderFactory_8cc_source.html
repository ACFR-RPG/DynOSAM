54DataProvider::Ptr DataProviderFactory::Create(
55 const std::string& dataset_folder_path,
56 const std::string& params_folder_path, DatasetType dataset_type) {
57 if (dataset_type == DatasetType::KITTI) {
58 LOG(INFO) << "Using KITTI dataset at path: " << dataset_folder_path;
59 KittiDataLoader::Params params =
60 KittiDataLoader::Params::fromYaml(params_folder_path);
61 auto loader =
62 std::make_shared<KittiDataLoader>(dataset_folder_path, params);
63
64 loader->setStartingFrame(FLAGS_starting_frame);
65 loader->setEndingFrame(FLAGS_ending_frame);
66 return loader;
67 } else if (dataset_type == DatasetType::VIRTUAL_KITTI) {
68 LOG(INFO) << "Using Virtual KITTI dataset at path: " << dataset_folder_path;
69 VirtualKittiDataLoader::Params params =
70 VirtualKittiDataLoader::Params::fromYaml(params_folder_path);
71 auto loader =
72 std::make_shared<VirtualKittiDataLoader>(dataset_folder_path, params);
73 loader->setStartingFrame(FLAGS_starting_frame);
74 loader->setEndingFrame(FLAGS_ending_frame);
75 return loader;
76 } else if (dataset_type == DatasetType::CLUSTER) {
77 LOG(INFO) << "Using Cluster (SLAM) dataset at path: "
78 << dataset_folder_path;
79 auto loader = std::make_shared<ClusterSlamDataLoader>(dataset_folder_path);
80 loader->setStartingFrame(FLAGS_starting_frame);
81 loader->setEndingFrame(FLAGS_ending_frame);
82 return loader;
83 } else if (dataset_type == DatasetType::OMD) {
84 LOG(INFO) << "Using Oxford Multi-motion Dataset dataset at path: "
85 << dataset_folder_path;
86 auto loader = std::make_shared<OMDDataLoader>(dataset_folder_path);
87 loader->setStartingFrame(FLAGS_starting_frame);
88 loader->setEndingFrame(FLAGS_ending_frame);
89 return loader;
90 } else if (dataset_type == DatasetType::ARIA) {
91 LOG(INFO) << "Using ARIA dataset at path: " << dataset_folder_path;
92 auto loader = std::make_shared<ProjectARIADataLoader>(dataset_folder_path);
93 loader->setStartingFrame(FLAGS_starting_frame);
94 loader->setEndingFrame(FLAGS_ending_frame);
95 return loader;
96 } else if (dataset_type == DatasetType::TARTAN_AIR_SHIBUYA) {
97 LOG(INFO) << "Using TARTAN_AIR_SHIBUYA dataset at path: "
98 << dataset_folder_path;
99 auto loader = std::make_shared<TartanAirShibuyaLoader>(dataset_folder_path);
100 loader->setStartingFrame(FLAGS_starting_frame);
101 loader->setEndingFrame(FLAGS_ending_frame);
102 return loader;
103 } else if (dataset_type == DatasetType::VIODE) {
104 LOG(INFO) << "Using VIODE dataset at path: " << dataset_folder_path;
105 auto loader = std::make_shared<ViodeLoader>(dataset_folder_path);
106 loader->setStartingFrame(FLAGS_starting_frame);
107 loader->setEndingFrame(FLAGS_ending_frame);
108 return loader;
109 } else {
110 throw std::runtime_error(
111 "Unable to construct Dataprovider - unknown dataset type: " +
112 static_cast<int>(dataset_type));
113 }
114}
