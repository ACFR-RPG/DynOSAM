74// expects DERIVEDFACTOR to be a NoiseModelFactor so that we can use
75// whitenedError
76template <typename DERIVEDFACTOR>
77gtsam::FactorIndices determineFactorOutliers(
78 const gtsam::NonlinearFactorGraph& graph, const gtsam::Values& values,
79 double confidence = 0.99) {
80 gtsam::FactorIndices outlier_indicies;
81 for (size_t k = 0; k < graph.size(); k++) {
82 if (!graph[k]) continue;
83
84 auto derived_factor = boost::dynamic_pointer_cast<DERIVEDFACTOR>(graph[k]);
85 if (derived_factor) {
86 double weighted_threshold =
87 0.5 *
88 chi_squared_quantile(
89 derived_factor->dim(),
90 confidence); // 0.5 derives from the error definition in gtsam
91 // NOTE: double casting as we expect DERIVEDFACTOR to be a
92 // NoiseModelFactor
93 gtsam::NoiseModelFactor::shared_ptr nm_factor =
94 boost::dynamic_pointer_cast<gtsam::NoiseModelFactor>(derived_factor);
95 CHECK(nm_factor);
96 // all noise models must be gaussian otherwise whittening will be
97 // reweighted
98 auto robust = boost::dynamic_pointer_cast<gtsam::noiseModel::Robust>(
99 nm_factor->noiseModel());
100 auto gaussian_nm_factor =
101 robust ? nm_factor->cloneWithNewNoiseModel(robust->noise())
102 : nm_factor;
103 CHECK(gaussian_nm_factor);
104
105 double error = gaussian_nm_factor->error(values);
106
107 if (error > weighted_threshold) {
108 outlier_indicies.push_back(k);
109 }
110 }
111 }
112 return outlier_indicies;
113}

115template <typename GRAPH>
116size_t getAssociatedFactors(
117 std::vector<typename GRAPH::const_iterator>& associated_factors,
118 const GRAPH& graph, gtsam::Key query_key) {
119 associated_factors.clear();
120
121 for (auto it = graph.begin(); it != graph.end(); it++) {
122 auto factor = *it;
123 if (factor->find(query_key) != factor->end()) {
124 associated_factors.push_back(it);
125 }
126 }
127 return associated_factors.size();
128}

130template <typename GRAPH>
131size_t getAssociatedFactors(gtsam::FactorIndices& associated_factors,
132 const GRAPH& graph, gtsam::Key query_key) {
133 associated_factors.clear();
134 std::vector<typename GRAPH::const_iterator> associated_factor_iters;
135 size_t result =
136 getAssociatedFactors(associated_factor_iters, graph, query_key);
137
138 for (const typename GRAPH::const_iterator& iter : associated_factor_iters) {
139 // index in graph
140 associated_factors.push_back(std::distance(graph.begin(), iter));
141 }
142 return result;
143}

240 template <typename CLIQUE>
241 static gtsam::KeySet getEliminatonOrder(
242 const gtsam::BayesTree<CLIQUE>& bayes_tree) {
243 gtsam::KeySet keys;
244 CliqueVisitor<CLIQUE> visitor =
245 [&keys](const boost::shared_ptr<CLIQUE>& clique) -> void {
246 auto conditional = clique->conditional();
247 // it is FACTOR::const_iterator
248 for (auto it = conditional->beginFrontals();
249 it != conditional->endFrontals(); it++) {
250 keys.insert(*it);
251 }
252 };
253
254 depthFirstTraversalEliminiationOrder(bayes_tree, visitor);
255
256 return keys;
257 }

261 gtsam::KeySet keys;
262 CliqueVisitor<CLIQUE> visitor =
263 [&keys](const boost::shared_ptr<CLIQUE>& clique) -> void {
264 if (clique->children.size() == 0u) {

266 // it is FACTOR::const_iterator

268 it != conditional->endFrontals(); it++) {
269 keys.insert(*it);
270 }
271 }
272 };
273
274 depthFirstTraversalEliminiationOrder(bayes_tree, visitor);
275
276 return keys;
277 }

288template <typename CLIQUE>
289std::pair<size_t, double> getCliqueSize(const gtsam::BayesTree<CLIQUE>& isam2) {
290 std::set<typename gtsam::BayesTree<CLIQUE>::sharedClique> isam2_cliques;
291 auto nodes = isam2.nodes();
292 for (const auto& it : nodes) {
293 isam2_cliques.insert(it.second);
294 }
295 size_t max_clique_size = 0;
296 size_t total_clique_size = 0;
297 for (auto& isam2_clique : isam2_cliques) {
298 size_t clique_size = isam2_clique->conditional()->frontals().size() +
299 isam2_clique->conditional()->parents().size();
300 total_clique_size += clique_size;
301 max_clique_size = std::max(max_clique_size, clique_size);
302 }
303 double avg_clique_size =
304 (double)total_clique_size / (double)isam2_cliques.size();
305 return std::make_pair(max_clique_size, avg_clique_size);
306}

308struct SparsityStats {
309 gtsam::Matrix matrix;
311 size_t nr_zero_elements{0};
313 size_t nr_elements{0};
315 size_t nnz_elements{0};
316
317 SparsityStats(){};
318
327 template <typename Derived>
328 SparsityStats(const Eigen::MatrixBase<Derived>& M)
329 : matrix(M.template cast<double>()), nr_zero_elements(0), nr_elements(0) {
330 nr_elements = M.rows() * M.cols();
331 for (int i = 0; i < M.rows(); ++i) {
332 for (int j = 0; j < M.cols(); ++j) {
333 if (std::fabs(M(i, j)) < 1e-15) {
334 nr_zero_elements++;
335 } else {
336 nnz_elements++;
337 }
338 }
339 }
340 }

365template <class CLIQUE>
366void dotBayesTree(
367 std::ostream& s, boost::shared_ptr<CLIQUE> clique,
368 const gtsam::KeyFormatter& keyFormatter,
369 const gtsam::FastMap<gtsam::Key, std::string>& colour_map = {},
370 int parentnum = 0) {
371 static int num = 0;
372 bool first = true;
373 std::stringstream out;
374 out << num;
375 std::string parent = out.str();
376 parent += "[label=\"";
377
378 // frontal key since this is what is used to add to nodes [j] (ie.e j is a
379 // frontal key)
380 bool keyInColourMap = false;
381 std::string colour;
382
383 for (auto key : clique->conditional_->frontals()) {
384 if (!first) parent += ", ";
385 first = false;
386 parent += keyFormatter(key);
387
388 keyInColourMap = colour_map.exists(key);
389 // only for the last key? Assumes all keys have the same colour?
390 if (keyInColourMap) colour = colour_map.at(key);
391 }
392
393 if (clique->parent()) {
394 parent += " : ";
395 s << parentnum << "->" << num << "\n";
396 }
397
398 first = true;
399 for (auto parentKey : clique->conditional_->parents()) {
400 if (!first) parent += ", ";
401 first = false;
402 parent += keyFormatter(parentKey);
403 }
404 // parent += "\"];\n";
405
406 if (keyInColourMap) {
407 parent += "\",color=" + colour;
408 parent += "];\n";
409 } else {
410 parent += "\"];\n";
411 }
412
413 s << parent;
414 parentnum = num;
415
416 for (auto c : clique->children) {
417 num++;
418 dotBayesTree(s, c, keyFormatter, colour_map, parentnum);
419 }
420}

422template <class CLIQUE>
423void saveBayesTree(
424 const gtsam::BayesTree<CLIQUE>& tree, const std::string& filename,
425 const gtsam::KeyFormatter& keyFormatter,
426 const gtsam::FastMap<gtsam::Key, std::string> colour_map = {}) {
427 std::ofstream of(filename.c_str());
428
429 if (tree.roots().empty())
430 throw std::invalid_argument(
431 "the root of Bayes tree has not been initialized!");
432 of << "digraph G{\n";
433 for (const auto& root : tree.roots())
434 dotBayesTree(of, root, keyFormatter, colour_map);
435 of << "}";
436 std::flush(of);
437}

439template <class CLIQUE>
440std::pair<SparsityStats, cv::Mat> computeRFactor(
441 const gtsam::BayesTree<CLIQUE>& bayes_tree) {
442 size_t dim = 0;
443 for (const auto& clique : bayes_tree) {
444 dim += clique->conditional()->rows();
445 }
446
447 gtsam::Matrix full_R = gtsam::Matrix::Zero(dim, dim);
448
449 // Step 2: Extract R blocks from the Bayes Tree and assemble them
450 size_t rowStart = 0;
451 for (const auto& clique : bayes_tree) {
452 auto conditional = clique->conditional();
453 if (conditional) {
454 gtsam::Matrix R_block = conditional->R(); // Extract R block
455 size_t blockSize = R_block.rows();
456
457 // Insert R_block into the full R matrix
458 full_R.block(rowStart, rowStart, blockSize, blockSize) = R_block;
459
460 rowStart += blockSize; // Move row index
461 }
462 }
463
464 size_t nnz = 0u;
465 cv::Mat R_img(cv::Size(full_R.cols(), full_R.rows()), CV_8UC3,
466 cv::viz::Color::white());
467 for (int i = 0; i < full_R.rows(); ++i) {
468 for (int j = 0; j < full_R.cols(); ++j) {
469 // only draw if non zero
470 if (std::fabs(full_R(i, j)) > 1e-15) {
471 R_img.at<cv::Vec3b>(i, j) = (cv::Vec3b)cv::viz::Color::black();
472 nnz++;
473 }
474 }
475 }
476 return std::make_pair(SparsityStats{}, R_img);
477}

748 static std::unordered_set<Key> gatherAdditionalKeysToReEliminate(
749 const BayesTree& bayesTree, const KeyVector& marginalizableKeys) {
750 const bool debug = true; // ISDEBUG("BayesTreeMarginalizationHelper");

752 std::unordered_set<const Clique*> additionalCliques =
753 gatherAdditionalCliquesToReEliminate(bayesTree, marginalizableKeys);

757 addCliqueToKeySet(clique, &additionalKeys);

778 static std::unordered_set<const Clique*> gatherAdditionalCliquesToReEliminate(
779 const BayesTree& bayesTree, const KeyVector& marginalizableKeys) {
780 std::unordered_set<const Clique*> additionalCliques;
781 std::unordered_set<Key> marginalizableKeySet(marginalizableKeys.begin(),
782 marginalizableKeys.end());
783 CachedSearch cachedSearch;
784
785 // Check each clique that contains a marginalizable key
786 for (const Clique* clique :
787 getCliquesContainingKeys(bayesTree, marginalizableKeySet)) {
788 // clique->print("Clique: ", DynoLikeKeyFormatter);
789 if (additionalCliques.count(clique)) {
790 // The clique has already been visited. This can happen when an
791 // ancestor of the current clique also contain some marginalizable
792 // varaibles and it's processed beore the current.
793 continue;
794 }
795
796 if (needsReelimination(clique, marginalizableKeySet, &cachedSearch)) {
797 // Add the current clique
798 additionalCliques.insert(clique);
799
800 // Then add the dependent cliques
801 gatherDependentCliques(clique, marginalizableKeySet, &additionalCliques,
802 &cachedSearch);
803 }
804 }
805 return additionalCliques;
806 }

830 struct CachedSearch {
831 std::unordered_map<const Clique*, bool> wholeMarginalizableCliques;
832 std::unordered_map<const Clique*, bool> wholeMarginalizableSubtrees;
833 };
834
840 static bool isWholeCliqueMarginalizable(
841 const Clique* clique, const std::unordered_set<Key>& marginalizableKeys,
842 CachedSearch* cache) {
843 auto it = cache->wholeMarginalizableCliques.find(clique);
844 if (it != cache->wholeMarginalizableCliques.end()) {
845 return it->second;
846 } else {
847 bool ret = true;
848 for (Key key : clique->conditional()->frontals()) {
849 if (!marginalizableKeys.count(key)) {
850 ret = false;
851 break;
852 }
853 }
854 cache->wholeMarginalizableCliques.insert({clique, ret});
855 return ret;
856 }
857 }
858
864 static bool isWholeSubtreeMarginalizable(
865 const Clique* subtree, const std::unordered_set<Key>& marginalizableKeys,
866 CachedSearch* cache) {
867 auto it = cache->wholeMarginalizableSubtrees.find(subtree);
868 if (it != cache->wholeMarginalizableSubtrees.end()) {
869 return it->second;
870 } else {
871 bool ret = true;
872 if (isWholeCliqueMarginalizable(subtree, marginalizableKeys, cache)) {
873 for (const sharedClique& child : subtree->children) {
874 if (!isWholeSubtreeMarginalizable(child.get(), marginalizableKeys,
875 cache)) {
876 ret = false;
877 break;
878 }
879 }
880 } else {
881 ret = false;
882 }
883 cache->wholeMarginalizableSubtrees.insert({subtree, ret});
884 return ret;
885 }
886 }
887
896 static bool needsReelimination(
897 const Clique* clique, const std::unordered_set<Key>& marginalizableKeys,
898 CachedSearch* cache) {
899 bool hasNonMarginalizableAhead = false;
900
901 // Check each frontal variable in order
902 for (Key key : clique->conditional()->frontals()) {
903 if (marginalizableKeys.count(key)) {
904 // If we've seen non-marginalizable variables before this one,
905 // we need to reeliminate
906 if (hasNonMarginalizableAhead) {
907 return true;
908 }
909
910 // Check if any child depends on this marginalizable key and the
911 // subtree rooted at that child contains non-marginalizables.
912 for (const sharedClique& child : clique->children) {
913 if (hasDependency(child.get(), key) &&
914 !isWholeSubtreeMarginalizable(child.get(), marginalizableKeys,
915 cache)) {
916 return true;
917 }
918 }
919 } else {
920 hasNonMarginalizableAhead = true;
921 }
922 }
923 return false;
924 }

933 static void gatherDependentCliques(
934 const Clique* rootClique,
935 const std::unordered_set<Key>& marginalizableKeys,
936 std::unordered_set<const Clique*>* additionalCliques,
937 CachedSearch* cache) {
938 std::vector<const Clique*> dependentChildren;
939 dependentChildren.reserve(rootClique->children.size());
940 for (const sharedClique& child : rootClique->children) {
941 if (additionalCliques->count(child.get())) {
942 // This child has already been visited. This can happen if the
943 // child itself contains a marginalizable variable and it's
944 // processed before the current rootClique.
945 continue;
946 }
947 if (hasDependency(child.get(), marginalizableKeys)) {
948 dependentChildren.push_back(child.get());
949 }
950 }
951 gatherDependentCliquesFromChildren(dependentChildren, marginalizableKeys,
952 additionalCliques, cache);
953 }
954
958 static void gatherDependentCliquesFromChildren(
959 const std::vector<const Clique*>& dependentChildren,
960 const std::unordered_set<Key>& marginalizableKeys,
961 std::unordered_set<const Clique*>* additionalCliques,
962 CachedSearch* cache) {
963 std::deque<const Clique*> descendants(dependentChildren.begin(),
964 dependentChildren.end());
965 while (!descendants.empty()) {
966 const Clique* descendant = descendants.front();
967 descendants.pop_front();
968
969 // If the subtree rooted at this descendant contains non-marginalizables,
970 // it must lie on a path from the root clique to a clique containing
971 // non-marginalizables at the leaf side.
972 if (!isWholeSubtreeMarginalizable(descendant, marginalizableKeys,
973 cache)) {
974 additionalCliques->insert(descendant);
975
976 // Add children of the current descendant to the set descendants.
977 for (const sharedClique& child : descendant->children) {
978 if (additionalCliques->count(child.get())) {
979 // This child has already been visited.
980 continue;
981 } else {
982 descendants.push_back(child.get());
983 }
984 }
985 }
986 }
987 }
988
995 static void addCliqueToKeySet(const Clique* clique,
996 std::unordered_set<Key>* additionalKeys) {
997 for (Key key : clique->conditional()->frontals()) {
998 additionalKeys->insert(key);
999 }
1000 }

1009 static bool hasDependency(const Clique* clique, Key key) {
1010 auto& conditional = clique->conditional();
1011 if (std::find(conditional->beginParents(), conditional->endParents(),
1012 key) != conditional->endParents()) {
1013 return true;
1014 } else {
1015 return false;
1016 }
1017 }
1018
1022 static bool hasDependency(const Clique* clique,
1023 const std::unordered_set<Key>& keys) {
1024 auto& conditional = clique->conditional();
1025 for (auto it = conditional->beginParents(); it != conditional->endParents();
1026 ++it) {
1027 if (keys.count(*it)) {
1028 return true;
1029 }
1030 }
1031
1032 return false;
1033 }
