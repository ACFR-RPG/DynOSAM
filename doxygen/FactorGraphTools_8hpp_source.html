68//expects DERIVEDFACTOR to be a NoiseModelFactor so that we can use whitenedError
69template<typename DERIVEDFACTOR>
70gtsam::FactorIndices determineFactorOutliers(
71 const gtsam::NonlinearFactorGraph& graph,
72 const gtsam::Values& values,
73 double confidence = 0.99)
74{
75 gtsam::FactorIndices outlier_indicies;
76 for (size_t k = 0; k < graph.size(); k++) {
77 if (!graph[k]) continue;
78
79 auto derived_factor = boost::dynamic_pointer_cast<DERIVEDFACTOR>(graph[k]);
80 if(derived_factor) {
81 double weighted_threshold = 0.5 * chi_squared_quantile(derived_factor->dim(), confidence); // 0.5 derives from the error definition in gtsam
82 //NOTE: double casting as we expect DERIVEDFACTOR to be a NoiseModelFactor
83 gtsam::NoiseModelFactor::shared_ptr nm_factor = boost::dynamic_pointer_cast<gtsam::NoiseModelFactor>(derived_factor);
84 CHECK(nm_factor);
85 // all noise models must be gaussian otherwise whittening will be reweighted
86 auto robust = boost::dynamic_pointer_cast<gtsam::noiseModel::Robust>(nm_factor->noiseModel());
87 auto gaussian_nm_factor = robust ? nm_factor-> cloneWithNewNoiseModel(robust->noise()) : nm_factor;
88 CHECK(gaussian_nm_factor);
89
90 double error = gaussian_nm_factor->error(values);
91
92 if(error > weighted_threshold ) {
93 outlier_indicies.push_back(k);
94 }
95 }
96 }
97 return outlier_indicies;
98}

101template<typename GRAPH>
102size_t getAssociatedFactors(std::vector<typename GRAPH::const_iterator>& associated_factors, const GRAPH& graph, gtsam::Key query_key) {
103 associated_factors.clear();
104
105 for(auto it = graph.begin(); it != graph.end(); it++) {
106 auto factor = *it;
107 if(factor->find(query_key) != factor->end()) {
108 associated_factors.push_back(it);
109 }
110 }
111 return associated_factors.size();
112}

114template<typename GRAPH>
115size_t getAssociatedFactors(gtsam::FactorIndices& associated_factors, const GRAPH& graph, gtsam::Key query_key) {
116 associated_factors.clear();
117 std::vector<typename GRAPH::const_iterator> associated_factor_iters;
118 size_t result = getAssociatedFactors(associated_factor_iters, graph, query_key);
119
120 for(const typename GRAPH::const_iterator& iter : associated_factor_iters) {
121 //index in graph
122 associated_factors.push_back(std::distance(graph.begin(), iter));
123 }
124 return result;
125}

218 template<typename CLIQUE>
219 static gtsam::KeySet getEliminatonOrder(const gtsam::BayesTree<CLIQUE>& bayes_tree) {
220 gtsam::KeySet keys;
221 CliqueVisitor<CLIQUE> visitor = [&keys](const boost::shared_ptr<CLIQUE>& clique) -> void {
222 auto conditional = clique->conditional();
223 //it is FACTOR::const_iterator
224 for(auto it = conditional->beginFrontals(); it != conditional->endFrontals(); it++) {
225 keys.insert(*it);
226 }
227 };

229 depthFirstTraversalEliminiationOrder(bayes_tree, visitor);

257struct SparsityStats {
258 gtsam::Matrix matrix;
259 size_t nr_zero_elements{0};
260 size_t nr_elements{0};
261
262 SparsityStats() {};
263 SparsityStats(const gtsam::Matrix& M) : matrix(M), nr_zero_elements(0), nr_elements(0) {
264 nr_elements = M.rows() * M.cols();
265 for (int i = 0; i < M.rows(); ++i) {
266 for (int j = 0; j < M.cols(); ++j) {
267 if (std::fabs(M(i, j)) < 1e-15) {
268 nr_zero_elements += 1;
269 }
270 }
271
272 }
273 }

290template <class CLIQUE>
291void dotBayesTree(std::ostream& s, boost::shared_ptr<CLIQUE> clique,
292 const gtsam::KeyFormatter& keyFormatter,
293 const gtsam::FastMap<gtsam::Key, std::string>& colour_map = {},
294 int parentnum = 0) {
295 static int num = 0;
296 bool first = true;
297 std::stringstream out;
298 out << num;
299 std::string parent = out.str();
300 parent += "[label=\"";
301
302 //frontal key since this is what is used to add to nodes [j] (ie.e j is a frontal key)
303 bool keyInColourMap = false;
304 std::string colour;
305
306 for (auto key : clique->conditional_->frontals()) {
307 if (!first) parent += ", ";
308 first = false;
309 parent += keyFormatter(key);
310
311 keyInColourMap = colour_map.exists(key);
312 //only for the last key? Assumes all keys have the same colour?
313 if(keyInColourMap) colour = colour_map.at(key);
314 }
315
316 if (clique->parent()) {
317 parent += " : ";
318 s << parentnum << "->" << num << "\n";
319 }
320
321 first = true;
322 for (auto parentKey : clique->conditional_->parents()) {
323 if (!first) parent += ", ";
324 first = false;
325 parent += keyFormatter(parentKey);
326 }
327 // parent += "\"];\n";
328
329 if(keyInColourMap) {
330 parent += "\",color=" + colour;
331 parent += "];\n";
332 }
333 else {
334 parent += "\"];\n";
335 }
336
337 s << parent;
338 parentnum = num;
339
340 for (auto c : clique->children) {
341 num++;
342 dotBayesTree(s, c, keyFormatter, colour_map, parentnum);
343 }
344}

346template <class CLIQUE>
347void saveBayesTree(const gtsam::BayesTree<CLIQUE>& tree, const std::string& filename,
348 const gtsam::KeyFormatter& keyFormatter, const gtsam::FastMap<gtsam::Key, std::string> colour_map = {})
349{
350 std::ofstream of(filename.c_str());
351
352 if (tree.roots().empty())
353 throw std::invalid_argument(
354 "the root of Bayes tree has not been initialized!");
355 of << "digraph G{\n";
356 for (const auto& root : tree.roots()) dotBayesTree(of, root, keyFormatter, colour_map);
357 of << "}";
358 std::flush(of);
359}
