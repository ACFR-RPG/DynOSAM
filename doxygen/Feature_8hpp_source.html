44struct functional_keypoint {
45 template <typename T = int>
46 static inline T u(const Keypoint& kp) {
47 return static_cast<T>(kp(0));
48 }
49
50 template <typename T = int>
51 static inline int v(const Keypoint& kp) {
52 return static_cast<T>(kp(1));
53 }
54
55 // ImageWrapperType should be a ImageType, eg ImageType::RGBMono etc...
56 template <typename ImageWrapperType,
57 typename AccessType = typename ImageWrapperType::OpenCVType>
58 static AccessType at(const Keypoint& kp,
59 const ImageWrapper<ImageWrapperType>& image_wrapper) {
60 return at<AccessType>(kp, static_cast<const cv::Mat&>(image_wrapper));
61 }
62
63 template <typename AccessType>
64 static AccessType at(const Keypoint& kp, const cv::Mat& img) {
65 const int x = functional_keypoint::u<int>(kp);
66 const int y = functional_keypoint::v<int>(kp);
67 return img.at<AccessType>(y, x);
68 }
69};
