46struct functional_keypoint {
47 template <typename T = int>
48 static inline T u(const Keypoint& kp) {
49 return static_cast<T>(kp(0));
50 }
51
52 template <typename T = int>
53 static inline int v(const Keypoint& kp) {
54 return static_cast<T>(kp(1));
55 }
56
57 // ImageWrapperType should be a ImageType, eg ImageType::RGBMono etc...
58 template <typename ImageWrapperType,
59 typename AccessType = typename ImageWrapperType::OpenCVType>
60 static AccessType at(const Keypoint& kp,
61 const ImageWrapper<ImageWrapperType>& image_wrapper) {
62 return at<AccessType>(kp, static_cast<const cv::Mat&>(image_wrapper));
63 }
64
65 template <typename AccessType>
66 static AccessType at(const Keypoint& kp, const cv::Mat& img) {
67 const int x = functional_keypoint::u<int>(kp);
68 const int y = functional_keypoint::v<int>(kp);
69 return img.at<AccessType>(y, x);
70 }
71};
