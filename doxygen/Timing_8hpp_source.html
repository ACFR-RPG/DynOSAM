46 // Stop timer and report duration in given time.
47 // Returns duration in milliseconds by default.
48 // call .count() on returned duration to have number of ticks.
49 template <typename T = std::chrono::milliseconds>
50 static T toc(const TimePoint& start)
51 {
52 return std::chrono::duration_cast<T>(Clock::now() - start);
53 }
54
60 static double now()
61 {
62 auto t_now = Clock::now().time_since_epoch();
63 return static_cast<double>(std::chrono::duration_cast<std::chrono::nanoseconds>(t_now).count()) / 1e9;
64 }
65
66 template <typename T = std::chrono::milliseconds>
67 static double toSeconds(const T& time)
68 {
69 auto seconds = std::chrono::duration_cast<std::chrono::seconds>(time);
70 return static_cast<double>(seconds.count());
71 }

80 static typename T::rep execution(F&& func, Args&&... args)
81 {
82 auto start = std::chrono::steady_clock::now();
83 std::forward<decltype(func)>(func)(std::forward<Args>(args)...);
84 auto duration = std::chrono::duration_cast<T>(std::chrono::steady_clock::now() - start);
85 return duration.count();
86 }
