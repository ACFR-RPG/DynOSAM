37FunctionalDetector::Ptr FunctionalDetector::Create<cv::GFTTDetector>(const TrackerParams& tracker_params) {
38 LOG(INFO) << "Creating cv::GFTTDetector";
39
40 auto feature_detector_ = cv::GFTTDetector::create(
41 tracker_params.max_nr_keypoints_before_anms,
42 tracker_params.gfft_params.quality_level,
43 tracker_params.min_distance_btw_tracked_and_detected_features,
44 tracker_params.gfft_params.block_size,
45 tracker_params.gfft_params.use_harris_corner_detector,
46 tracker_params.gfft_params.k);
47 auto functional_detector = [=](const cv::Mat& img, KeypointsCV& keypoints, const cv::Mat& mask) -> void {
48 CHECK_NOTNULL(feature_detector_)->detect(img, keypoints, mask);
49 };
50
51 return std::make_shared<FunctionalDetector>(functional_detector);
52}

70 };
71
72 return std::make_shared<FunctionalDetector>(functional_detector);

76FunctionalDetector::Ptr FunctionalDetector::FactoryCreate(const TrackerParams& tracker_params) {
77 using FDT = TrackerParams::FeatureDetectorType;
78 switch (tracker_params.feature_detector_type)
79 {
80 case FDT::GFTT:
81 return FunctionalDetector::Create<cv::GFTTDetector>(tracker_params);
82 case FDT::ORB_SLAM_ORB:
83 return FunctionalDetector::Create<ORBextractor>(tracker_params);
84 default:
85 LOG(ERROR) << "Unknown Feature detection type!";
86 return nullptr;
87 break;
88 }
89}

105void SparseFeatureDetector::detect(const cv::Mat& image, KeypointsCV& keypoints, int number_tracked, const cv::Mat& detection_mask) {
106 cv::Mat processed_image = image.clone();
107
108 //pre-process image if required
109 if(clahe_) clahe_->apply(processed_image, processed_image);
110
111 std::vector<cv::KeyPoint> raw_keypoints;
112 feature_detector_->detect(processed_image, raw_keypoints, detection_mask);
113
114 std::vector<cv::KeyPoint>& max_keypoints = raw_keypoints;
115 if(tracker_params_.use_anms) {
116
117 //calculate number of corners needed
118 int nr_corners_needed = std::max(
119 tracker_params_.max_features_per_frame - number_tracked, 0);
120
121 static constexpr float tolerance = 0.1;
122
123 const auto& anms_params = tracker_params_.anms_params;
124 Eigen::MatrixXd binning_mask = anms_params.binning_mask;
125
126 max_keypoints = non_maximum_supression_->suppressNonMax(
127 raw_keypoints,
128 nr_corners_needed,
129 tolerance,
130 processed_image.cols,
131 processed_image.rows,
132 anms_params.nr_horizontal_bins,
133 anms_params.nr_vertical_bins,
134 binning_mask);
135
136 }
137
138 if(tracker_params_.use_subpixel_corner_refinement && max_keypoints.size() > 0u) {
139 // Convert keypoints to points
140 std::vector<cv::Point2f> points;
141 cv::KeyPoint::convert(max_keypoints, points);
142
143 const auto& subpixel_corner_refinement_params = tracker_params_.subpixel_corner_refinement_params;
144 cv::cornerSubPix(
145 processed_image,
146 points,
147 subpixel_corner_refinement_params.window_size,
148 subpixel_corner_refinement_params.zero_zone,
149 subpixel_corner_refinement_params.criteria);
150
151 cv::KeyPoint::convert(points, max_keypoints);
152 }
153
154 keypoints = max_keypoints;
155}
