53 std::string toString() const {
54 std::stringstream ss;
55 for (auto&& x : bh::indexed(histogram_)) {
56 ss << "Bin [";
57 for (size_t i = 0; i < histogram_.rank(); ++i) {
58 ss << x.bin(i).lower() << ", " << x.bin(i).upper();
59 if (i < histogram_.rank() - 1) ss << "], [";
60 }
61 ss << "]: " << x.get() << "\n";
62 }
63 return ss.str();
64 }
