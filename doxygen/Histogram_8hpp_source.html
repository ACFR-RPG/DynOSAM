49 boost::histogram::axis::variable<>,
50 boost::histogram::axis::integer<> >;

54 // TODO: does this need to be && to enable perfect forwarning?
55 template <typename... Axis>
56 Histogram(const boost::histogram::histogram<Axis...>& histogram)
57 : histogram_(histogram) {}
58
59 std::string toString() const {
60 std::stringstream ss;
61 for (auto&& x : bh::indexed(histogram_)) {
62 ss << "Bin [";
63 for (size_t i = 0; i < histogram_.rank(); ++i) {
64 ss << x.bin(i).lower() << ", " << x.bin(i).upper();
65 if (i < histogram_.rank() - 1) ss << "], [";
66 }
67 ss << "]: " << x.get() << "\n";
68 }
69 return ss.str();
70 }
71
72 public:
73 ImplHistogram histogram_;
74 std::string name_;
75};
