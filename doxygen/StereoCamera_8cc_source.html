44StereoCamera::StereoCamera(Camera::ConstPtr left_camera,
45 Camera::ConstPtr right_camera)
46 : original_left_camera_(CHECK_NOTNULL(left_camera)),
47 original_right_camera_(CHECK_NOTNULL(right_camera)),
48 undistorted_rectified_stereo_camera_impl_(),
49 stereo_calibration_(nullptr),
50 left_cam_undistort_rectifier_(nullptr),
51 right_cam_undistort_rectifier_(nullptr),
52 stereo_baseline_(0.0) {
53 computeRectificationParameters(left_camera->getParams(),
54 right_camera->getParams(), R1_, R2_, P1_, P2_,
55 Q_, ROI1_, ROI2_);
56 calculateBaseLine(stereo_baseline_, Q_);
57
59 const auto left_undist_rect_cam_mat = utils::Cvmat2Cal3_S2(P1_);
60 stereo_calibration_.reset(new gtsam::Cal3_S2Stereo(
61 left_undist_rect_cam_mat.fx(), left_undist_rect_cam_mat.fy(),
62 left_undist_rect_cam_mat.skew(), left_undist_rect_cam_mat.px(),
63 left_undist_rect_cam_mat.py(), stereo_baseline_));
64
67 left_cam_undistort_rectifier_ = std::make_unique<UndistorterRectifier>(
68 P1_, left_camera->getParams(), R1_);
69 right_cam_undistort_rectifier_ = std::make_unique<UndistorterRectifier>(
70 P2_, right_camera->getParams(), R2_);
71
74 // TODO: is this right? In the original implementation (Kimera) we use the
75 // rectified pose of the left image
76 // which SHOULD still be local!!
77 undistorted_rectified_stereo_camera_impl_ =
78 gtsam::StereoCamera(gtsam::Pose3::Identity(), stereo_calibration_);
79}

94void StereoCamera::computeRectificationParameters(
95 const CameraParams& left_cam_params, const CameraParams& right_cam_params,
96 cv::Mat& R1, cv::Mat& R2, cv::Mat& P1, cv::Mat& P2, cv::Mat& Q,
97 cv::Rect& ROI1, cv::Rect& ROI2) {
98 // ! Extrinsics of the stereo (not rectified) relative pose between cameras
99 gtsam::Pose3 camL_Pose_camR = (left_cam_params.getExtrinsics())
100 .between(right_cam_params.getExtrinsics());
101
102 LOG(INFO) << camL_Pose_camR;
103
105 CHECK(!camL_Pose_camR.equals(gtsam::Pose3::Identity()))
106 << "Relative pose between cameras cannot be identity - camL_Pose_camR = "
107 << camL_Pose_camR;
108
109 // Get extrinsics in open CV format.
110 // NOTE: openCV pose convention is the opposite, that's why we have to
111 // invert
112 cv::Mat camL_Rot_camR, camL_Tran_camR;
113 std::tie(camL_Rot_camR, camL_Tran_camR) =
114 utils::Pose2cvmats(camL_Pose_camR.inverse());
115
116 LOG(INFO) << camL_Rot_camR;
117 LOG(INFO) << camL_Tran_camR;
118
119 const auto left_k = left_cam_params.getCameraMatrix();
120 const auto left_d = left_cam_params.getDistortionCoeffs();
121
122 const auto right_k = right_cam_params.getCameraMatrix();
123 const auto right_d = right_cam_params.getDistortionCoeffs();
124
125 const auto size = left_cam_params.imageSize();
126
127 // kAlpha is -1 by default, but that introduces invalid keypoints!
128 // here we should use kAlpha = 0 so we get only valid pixels...
129 // But that has an issue that it removes large part of the image, check:
130 // https://github.com/opencv/opencv/issues/7240 for this issue with kAlpha
131 // Setting to -1 to make it easy, but it should NOT be -1!
132 static constexpr int kAlpha = 0;
133 switch (left_cam_params.getDistortionModel()) {
134 case DistortionModel::RADTAN: {
135 cv::stereoRectify(
136 // Input
137 left_k, left_d, right_k, right_d, size, camL_Rot_camR, camL_Tran_camR,
138 // Output
139 R1, R2, P1, P2, Q, cv::CALIB_ZERO_DISPARITY, kAlpha, cv::Size(),
140 &ROI1, &ROI2);
141 } break;
142 case DistortionModel::EQUIDISTANT: {
143 cv::fisheye::stereoRectify(
144 // Input
145 left_k, left_d, right_k, right_d, size, camL_Rot_camR, camL_Tran_camR,
146 // Output
147 R1, R2, P1, P2, Q,
148 // TODO: Flag to maximise area???
149 cv::CALIB_ZERO_DISPARITY);
150 } break;
151 default: {
152 LOG(FATAL) << "Unknown DistortionModel: "
153 << to_string(left_cam_params.getDistortionModel());
154 }
155 }
156
157 LOG(INFO) << Q;
158}

160void StereoCamera::calculateBaseLine(Baseline& base_line, const cv::Mat& Q) {
161 // Calc baseline (see L.2700 and L.2616 in
162 // https://github.com/opencv/opencv/blob/master/modules/calib3d/src/calibration.cpp
163 // NOTE: OpenCV pose convention is the opposite, therefore the missing -1.0
164 CHECK_NE(Q.at<double>(3, 2), 0.0);
165 base_line = 1.0 / Q.at<double>(3, 2);
166 CHECK_GT(base_line, 0.0);
167}
