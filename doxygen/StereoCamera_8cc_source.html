95void StereoCamera::computeRectificationParameters(
96 const CameraParams& left_cam_params,
97 const CameraParams& right_cam_params,
98 cv::Mat& R1,
99 cv::Mat& R2,
100 cv::Mat& P1,
101 cv::Mat& P2,
102 cv::Mat& Q,
103 cv::Rect& ROI1,
104 cv::Rect& ROI2) {
105
106 // ! Extrinsics of the stereo (not rectified) relative pose between cameras
107 gtsam::Pose3 camL_Pose_camR =
108 (left_cam_params.getExtrinsics()).between(right_cam_params.getExtrinsics());
109
110
112 CHECK(!camL_Pose_camR.equals(gtsam::Pose3::Identity())) << "Relative pose between cameras cannot be identity - camL_Pose_camR = " << camL_Pose_camR;
113
114 // Get extrinsics in open CV format.
115 // NOTE: openCV pose convention is the opposite, that's why we have to
116 // invert
117 cv::Mat camL_Rot_camR, camL_Tran_camR;
118 std::tie(camL_Rot_camR, camL_Tran_camR) =
119 utils::Pose2cvmats(camL_Pose_camR.inverse());
120
121 LOG(INFO) << camL_Rot_camR;
122 LOG(INFO) << camL_Tran_camR;
123
124 const auto left_k = left_cam_params.getCameraMatrix();
125 const auto left_d = left_cam_params.getDistortionCoeffs();
126
127 const auto right_k = right_cam_params.getCameraMatrix();
128 const auto right_d = right_cam_params.getDistortionCoeffs();
129
130 const auto size = left_cam_params.imageSize();
131
132 // kAlpha is -1 by default, but that introduces invalid keypoints!
133 // here we should use kAlpha = 0 so we get only valid pixels...
134 // But that has an issue that it removes large part of the image, check:
135 // https://github.com/opencv/opencv/issues/7240 for this issue with kAlpha
136 // Setting to -1 to make it easy, but it should NOT be -1!
137 static constexpr int kAlpha = 0;
138 switch (left_cam_params.getDistortionModel()) {
139 case DistortionModel::RADTAN: {
140 cv::stereoRectify(
141 // Input
142 left_k,
143 left_d,
144 right_k,
145 right_d,
146 size,
147 camL_Rot_camR,
148 camL_Tran_camR,
149 // Output
150 R1,
151 R2,
152 P1,
153 P2,
154 Q,
155 cv::CALIB_ZERO_DISPARITY,
156 kAlpha,
157 cv::Size(),
158 &ROI1,
159 &ROI2);
160 } break;
161 case DistortionModel::EQUIDISTANT: {
162 cv::fisheye::stereoRectify(
163 // Input
164 left_k,
165 left_d,
166 right_k,
167 right_d,
168 size,
169 camL_Rot_camR,
170 camL_Tran_camR,
171 // Output
172 R1,
173 R2,
174 P1,
175 P2,
176 Q,
177 // TODO: Flag to maximise area???
178 cv::CALIB_ZERO_DISPARITY);
179 } break;
180 default: {
181 LOG(FATAL) << "Unknown DistortionModel: " << to_string(left_cam_params.getDistortionModel());
182 }
183 }
184
185 LOG(INFO) << Q;
186}

188void StereoCamera::calculateBaseLine(Baseline& base_line, const cv::Mat& Q) {
189 // Calc baseline (see L.2700 and L.2616 in
190 // https://github.com/opencv/opencv/blob/master/modules/calib3d/src/calibration.cpp
191 // NOTE: OpenCV pose convention is the opposite, therefore the missing -1.0
192 CHECK_NE(Q.at<double>(3, 2), 0.0);
193 base_line = 1.0 / Q.at<double>(3, 2);
194 CHECK_GT(base_line, 0.0);
195}
