74static constexpr int kInfiniteWindowSize = std::numeric_limits<int>::max();
75
76// If the window size is set to -1, the vector will grow infinitely, otherwise,
77// the vector has a fixed size.
78template <typename SampleType, typename SumType, int WindowSize>
79class Accumulator {
80 public:
81 using This = Accumulator<SampleType, SumType, WindowSize>;
82 Accumulator()
83 : sample_index_(0),
84 total_samples_(0),
85 sum_(0),
86 window_sum_(0),
87 min_(std::numeric_limits<SampleType>::max()),
88 max_(std::numeric_limits<SampleType>::lowest()),
89 most_recent_(0) {
90 CHECK_GT(WindowSize, 0);
91 if (WindowSize < kInfiniteWindowSize) {
92 samples_.reserve(WindowSize);
93 }
94 }
95
96 /* ------------------------------------------------------------------------ */
97 void Add(SampleType sample) {
98 most_recent_ = sample;
99 if (sample_index_ < WindowSize) {
100 samples_.push_back(sample);
101 window_sum_ += sample;
102 ++sample_index_;
103 } else {
104 SampleType& oldest = samples_.at(sample_index_++ % WindowSize);
105 window_sum_ += sample - oldest;
106 oldest = sample;
107 }
108 sum_ += sample;
109 ++total_samples_;
110 if (sample > max_) {
111 max_ = sample;
112 }
113 if (sample < min_) {
114 min_ = sample;
115 }
116 }
117
118 /* ------------------------------------------------------------------------ */
119 int total_samples() const { return total_samples_; }
120
121 /* ------------------------------------------------------------------------ */
122 SumType sum() const { return sum_; }
123
124 /* ------------------------------------------------------------------------ */
125 SumType Mean() const {
126 return (total_samples_ < 1) ? 0.0
127 : sum_ / static_cast<SumType>(total_samples_);
128 }
129
130 /* ------------------------------------------------------------------------ */
131 // Rolling mean is only used for fixed sized data for now. We don't need this
132 // function for our infinite accumulator at this point.
133 SumType RollingMean() const {
134 if (WindowSize < kInfiniteWindowSize) {
135 return window_sum_ / std::min(sample_index_, WindowSize);
136 } else {
137 return Mean();
138 }
139 }
140
141 /* ------------------------------------------------------------------------ */
142 SampleType GetMostRecent() const { return most_recent_; }
143
144 /* ------------------------------------------------------------------------ */
145 SumType max() const { return max_; }
146
147 /* ------------------------------------------------------------------------ */
148 SumType min() const { return min_; }
149
150 /* ------------------------------------------------------------------------ */
151 inline SumType median() const {
152 CHECK_GT(samples_.size(), 0);
153
154 return samples_.at(std::ceil(samples_.size() / 2) - 1);
155 }
156
157 /* ------------------------------------------------------------------------ */
158 // First quartile.
159 inline SumType q1() const {
160 CHECK_GT(samples_.size(), 0);
161 return samples_.at(std::ceil(samples_.size() / 4) - 1);
162 }
163
164 /* ------------------------------------------------------------------------ */
165 // Third quartile.
166 inline SumType q3() const {
167 CHECK_GT(samples_.size(), 0);
168 return samples_.at(std::ceil(samples_.size() * 3 / 4) - 1);
169 }
170
171 /* ------------------------------------------------------------------------ */
172 SumType LazyVariance() const {
173 if (samples_.size() < 2) {
174 return 0.0;
175 }
176
177 SumType var = static_cast<SumType>(0.0);
178 SumType mean = RollingMean();
179
180 for (unsigned int i = 0; i < samples_.size(); ++i) {
181 var += (samples_[i] - mean) * (samples_[i] - mean);
182 }
183
184 var /= samples_.size() - 1;
185 return var;
186 }
187
188 /* ------------------------------------------------------------------------ */
189 SumType StandardDeviation() const { return std::sqrt(LazyVariance()); }
190
191 // TODO: needs tests
192 // only includes values that are within mean +/- threshold * std
193 This OutlierRejectionStd(double threshold) {
194 const auto mean = Mean();
195 const auto std = StandardDeviation();
196
197 const auto min_value = mean - threshold * std;
198 const auto max_value = mean + threshold * std;
199
200 This filtered;
201 for (auto value : *this) {
202 if (value > min_value && value < max_value) {
203 filtered.Add(value);
204 }
205 }
206
207 return filtered;
208 }
209
210 /* ------------------------------------------------------------------------ */
211 const std::vector<SampleType>& GetAllSamples() const { return samples_; }
212
213 typename std::vector<SampleType>::iterator begin() {
214 return samples_.begin();
215 }
216 const typename std::vector<SampleType>::iterator begin() const {
217 return samples_.begin();
218 }
219
220 typename std::vector<SampleType>::iterator end() { return samples_.end(); }
221 const typename std::vector<SampleType>::iterator end() const {
222 return samples_.end();
223 }
224
225 const typename std::vector<SampleType>::const_iterator cbegin() const {
226 return samples_.cbegin();
227 }
228 const typename std::vector<SampleType>::const_iterator cend() const {
229 return samples_.cend();
230 }
231
232 private:
233 std::vector<SampleType> samples_;
234 int sample_index_;
235 int total_samples_;
236 SumType sum_;
237 SumType window_sum_;
238 SampleType min_;
239 SampleType max_;
240 SampleType most_recent_;
241};
