67static constexpr int kInfiniteWindowSize = std::numeric_limits<int>::max();
68
69// If the window size is set to -1, the vector will grow infinitely, otherwise,
70// the vector has a fixed size.
71template <typename SampleType, typename SumType, int WindowSize>
72class Accumulator {
73 public:
74 using This = Accumulator<SampleType, SumType, WindowSize>;
75 Accumulator()
76 : sample_index_(0),
77 total_samples_(0),
78 sum_(0),
79 window_sum_(0),
80 min_(std::numeric_limits<SampleType>::max()),
81 max_(std::numeric_limits<SampleType>::lowest()),
82 most_recent_(0) {
83 CHECK_GT(WindowSize, 0);
84 if (WindowSize < kInfiniteWindowSize) {
85 samples_.reserve(WindowSize);
86 }
87 }
88
89 /* ------------------------------------------------------------------------ */
90 void Add(SampleType sample) {
91 most_recent_ = sample;
92 if (sample_index_ < WindowSize) {
93 samples_.push_back(sample);
94 window_sum_ += sample;
95 ++sample_index_;
96 } else {
97 SampleType& oldest = samples_.at(sample_index_++ % WindowSize);
98 window_sum_ += sample - oldest;
99 oldest = sample;
100 }
101 sum_ += sample;
102 ++total_samples_;
103 if (sample > max_) {
104 max_ = sample;
105 }
106 if (sample < min_) {
107 min_ = sample;
108 }
109 }
110
111 /* ------------------------------------------------------------------------ */
112 int total_samples() const { return total_samples_; }
113
114 /* ------------------------------------------------------------------------ */
115 SumType sum() const { return sum_; }
116
117 /* ------------------------------------------------------------------------ */
118 SumType Mean() const {
119 return (total_samples_ < 1) ? 0.0 : sum_ / static_cast<SumType>(total_samples_);
120 }
121
122 /* ------------------------------------------------------------------------ */
123 // Rolling mean is only used for fixed sized data for now. We don't need this
124 // function for our infinite accumulator at this point.
125 SumType RollingMean() const {
126 if (WindowSize < kInfiniteWindowSize) {
127 return window_sum_ / std::min(sample_index_, WindowSize);
128 } else {
129 return Mean();
130 }
131 }
132
133 /* ------------------------------------------------------------------------ */
134 SampleType GetMostRecent() const { return most_recent_; }
135
136 /* ------------------------------------------------------------------------ */
137 SumType max() const { return max_; }
138
139 /* ------------------------------------------------------------------------ */
140 SumType min() const { return min_; }
141
142 /* ------------------------------------------------------------------------ */
143 inline SumType median() const {
144 CHECK_GT(samples_.size(), 0);
145
146 return samples_.at(std::ceil(samples_.size() / 2) - 1);
147 }
148
149 /* ------------------------------------------------------------------------ */
150 // First quartile.
151 inline SumType q1() const {
152 CHECK_GT(samples_.size(), 0);
153 return samples_.at(std::ceil(samples_.size() / 4) - 1);
154 }
155
156 /* ------------------------------------------------------------------------ */
157 // Third quartile.
158 inline SumType q3() const {
159 CHECK_GT(samples_.size(), 0);
160 return samples_.at(std::ceil(samples_.size() * 3 / 4) - 1);
161 }
162
163 /* ------------------------------------------------------------------------ */
164 SumType LazyVariance() const {
165 if (samples_.size() < 2) {
166 return 0.0;
167 }
168
169 SumType var = static_cast<SumType>(0.0);
170 SumType mean = RollingMean();
171
172 for (unsigned int i = 0; i < samples_.size(); ++i) {
173 var += (samples_[i] - mean) * (samples_[i] - mean);
174 }
175
176 var /= samples_.size() - 1;
177 return var;
178 }
179
180 /* ------------------------------------------------------------------------ */
181 SumType StandardDeviation() const { return std::sqrt(LazyVariance()); }
182
183 //TODO: needs tests
184 // only includes values that are within mean +/- threshold * std
185 This OutlierRejectionStd(double threshold) {
186 const auto mean = Mean();
187 const auto std = StandardDeviation();
188
189 const auto min_value = mean - threshold * std;
190 const auto max_value = mean + threshold * std;
191
192 This filtered;
193 for(auto value : *this) {
194 if(value > min_value && value < max_value) {
195 filtered.Add(value);
196 }
197 }
198
199 return filtered;
200 }
201
202 /* ------------------------------------------------------------------------ */
203 const std::vector<SampleType> &GetAllSamples() const { return samples_; }
204
205 typename std::vector<SampleType>::iterator begin() { return samples_.begin(); }
206 const typename std::vector<SampleType>::iterator begin() const { return samples_.begin(); }
207
208 typename std::vector<SampleType>::iterator end() { return samples_.end(); }
209 const typename std::vector<SampleType>::iterator end() const { return samples_.end(); }
210
211 const typename std::vector<SampleType>::const_iterator cbegin() const { return samples_.cbegin(); }
212 const typename std::vector<SampleType>::const_iterator cend() const { return samples_.cend(); }
213
214private:
215 std::vector<SampleType> samples_;
216 int sample_index_;
217 int total_samples_;
218 SumType sum_;
219 SumType window_sum_;
220 SampleType min_;
221 SampleType max_;
222 SampleType most_recent_;
223};
