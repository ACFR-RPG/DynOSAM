39template <typename INPUT, typename OUTPUT>
40PipelineBase::ReturnCode PipelineModule<INPUT, OUTPUT>::spinOnce() {
41 using ReturnCode = PipelineBase::ReturnCode;
42
43 if (isShutdown()) {
44 return ReturnCode::IS_SHUTDOWN;
45 }
46
47 // log get_input, process and get_output packet timing explicitly if VLOG >=
48 // 10
49 static constexpr int intermediate_timing_glog_verbosity = 5;
50 auto getInputPacketWrapped = [&]() -> InputConstSharedPtr {
51 // LOG(INFO) << "construct_intermediate_timers_as_stopped " <<
52 // construct_intermediate_timers_as_stopped;
53 utils::ChronoTimingStats timing(module_name_ + ".get_input",
54 intermediate_timing_glog_verbosity);
55 InputConstSharedPtr input = nullptr;
56 is_thread_working_ = false;
57 input = getInputPacket();
58 is_thread_working_ = true;
59 return input;
60 };
61
62 auto processWrapped = [&](InputConstSharedPtr input) -> OutputConstSharedPtr {
63 utils::ChronoTimingStats timing(module_name_ + ".process",
64 intermediate_timing_glog_verbosity);
65 OutputConstSharedPtr output = nullptr;
66 output = process(input);
67 return output;
68 };
69
70 auto pushOutputPacketWrapped = [&](OutputConstSharedPtr output) -> bool {
71 // Received a valid output, send to output queue
72 utils::ChronoTimingStats timing(module_name_ + ".push_output",
73 intermediate_timing_glog_verbosity);
74 const bool push_packet_result = pushOutputPacket(output);
75 return push_packet_result;
76 };
77
78 ReturnCode return_code;
79 utils::ChronoTimingStats timing_stats(module_name_);
80
81 InputConstSharedPtr input = getInputPacketWrapped();
82
83 if (input) {
84 // Transfer the ownership of input to the actual pipeline module.
85 // From this point on, you cannot use input, since process owns it.
86 OutputConstSharedPtr output = processWrapped(input);
87
88 if (output) {
89 if (!pushOutputPacketWrapped(output)) {
90 LOG_EVERY_N(WARNING, 100)
91 << "Module: " << module_name_ << " - Output push failed.";
92 is_thread_working_ = false;
93 return_code = ReturnCode::OUTPUT_PUSH_FAILURE;
94 } else {
95 emitProcessCallbacks(input, output);
96 is_thread_working_ = false;
97 return_code = ReturnCode::SUCCESS;
98 }
99 } else {
100 return_code = ReturnCode::PROCESSING_FAILURE;
101 is_thread_working_ = false;
102 }
103 } else {
104 is_thread_working_ = false;
105 return_code = ReturnCode::GET_PACKET_FAILURE;
106 }
107
108 return return_code;
109}

143template <typename INPUT, typename OUTPUT>
144bool MIMOPipelineModule<INPUT, OUTPUT>::pushOutputPacket(
145 const typename Base::OutputConstSharedPtr& output_packet) const {
146 auto tic_callbacks = utils::Timer::tic();
150 for (OutputQueue* queue : output_queues_) {
151 CHECK(queue);
152 queue->push(output_packet);
153 }
154 for (OutputCallback cb : output_callbacks_) {
155 cb(output_packet);
156 }
157
158 static constexpr auto kTimeLimitCallbacks = std::chrono::milliseconds(10);
159 auto callbacks_duration = utils::Timer::toc(tic_callbacks);
160 LOG_IF(WARNING, callbacks_duration > kTimeLimitCallbacks)
161 << "Pushing output packet to queues for module: " << this->module_name_
162 << " are taking very long! Current latency: "
163 << callbacks_duration.count() << " ms.";
164 return true;
165}

169template <typename INPUT, typename OUTPUT>
170typename SIMOPipelineModule<INPUT, OUTPUT>::InputConstSharedPtr
171SIMOPipelineModule<INPUT, OUTPUT>::getInputPacket() {
172 typename Base::InputConstSharedPtr input = nullptr;
173
174 bool queue_state;
175 if (parallel_run_) {
176 queue_state = input_queue->popBlocking(input);
177 } else {
178 queue_state = input_queue->pop(input);
179 }
180
181 if (queue_state) {
182 return input;
183 } else {
184 LOG_IF_EVERY_N(WARNING, !Base::isShutdown() && !hasWork(), 100)
185 << "Module: " << Base::module_name_ << " - didn't return an output.";
186 return nullptr;
187 }
188}
