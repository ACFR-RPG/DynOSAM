41bool isMoving(const gtsam::Pose3& prev_L_world, const gtsam::Pose3& curr_L_world, double tol_m = 0.4) {
42 gtsam::Vector3 t_diff = curr_L_world.translation() - prev_L_world.translation();
43 // L2 norm - ie. magnitude
44 double t_error = t_diff.norm();
45 return t_error > tol_m;
46}

48void ObjectPoseGT::setMotions(const ObjectPoseGT& previous_object_gt, const gtsam::Pose3& prev_X_world, const gtsam::Pose3& curr_X_world) {
49 checkAndThrow(previous_object_gt.frame_id_ == frame_id_ - 1, "Previous object gt frame is not at k-1. Current frame = " + std::to_string(frame_id_) + " and previous frame =" + std::to_string(previous_object_gt.frame_id_));
50 checkAndThrow(previous_object_gt.object_id_ == object_id_, "Previous object gt does not have the same object id");
51
52
53 // for t-1
54 const gtsam::Pose3 L_camera_prev = previous_object_gt.L_camera_;
55 const gtsam::Pose3 L_world_prev = previous_object_gt.L_world_;
56 //changed tol becuase seems to be numerical issues
57 // CHECK(L_world_prev.equals(prev_X_world * L_camera_prev));
58 CHECK(gtsam::assert_equal(L_world_prev, prev_X_world * L_camera_prev, 1e-5));
59 // for t
60 const gtsam::Pose3 L_camera_curr = L_camera_;
61 const gtsam::Pose3 L_world_curr = L_world_;
62 //changed tol becuase seems to be numerical issues
63 // CHECK(L_world_curr.equals(curr_X_world * L_camera_curr));
64 CHECK(gtsam::assert_equal(L_world_curr, curr_X_world * L_camera_curr, 1e-5));
65
66 // H between t-1 and t in the object frame ground truth
67 gtsam::Pose3 H_L_gt = L_world_prev.inverse() * L_world_curr;
68 prev_H_current_L_ = H_L_gt;
69
70 // H between t-1 and t in the camera frame at t-1
71 gtsam::Pose3 H_X_gt = L_camera_prev * H_L_gt * L_camera_prev.inverse();
72 prev_H_current_X_ = H_X_gt;
73
74 // H between t-1 and t in the world frame
75 // equ.5 (morris2023importance)
76 gtsam::Pose3 H_W_gt = L_world_curr * L_world_prev.inverse();
77 prev_H_current_world_ = H_W_gt;
78
79 MotionInfo motion_info;
80 motion_info.is_moving_ = isMoving(L_world_prev, L_world_curr);
81 motion_info.has_stopped_ = false; //default
82
83 if(previous_object_gt.motion_info_) {
84 //moving in the previous frame but not in this frame
85 if(previous_object_gt.motion_info_->is_moving_ && !motion_info.is_moving_) {
86 motion_info.has_stopped_ = true;
87 }
88 }
89
90 motion_info_ = motion_info;
91

157bool GroundTruthInputPacket::getObject(ObjectId object_id, ObjectPoseGT& object_pose_gt, int log_severity) const {
158 auto it_this = std::find_if(object_poses_.begin(), object_poses_.end(),
159 [=](const ObjectPoseGT& gt_object) { return gt_object.object_id_ == object_id; });
160 if(it_this == object_poses_.end()) {
161 std::stringstream ss;
162 ss << "Could not find object at frame " << this->frame_id_ << " for object Id " << object_id << " and packet " << *this;
163
164 if(log_severity == 1) {
165 LOG(WARNING) << ss.str();
166 }
167 else if(log_severity > 1) {
168 throw DynosamException(ss.str());
169 }
170 return false;
171 }
172
173 object_pose_gt = *it_this;
174 return true;
175

224bool GroundTruthInputPacket::findAssociatedObject(ObjectId label, const GroundTruthInputPacket& other, size_t& obj_idx, size_t& other_obj_idx) const {
225 auto it_this = std::find_if(object_poses_.begin(), object_poses_.end(),
226 [=](const ObjectPoseGT& gt_object) { return gt_object.object_id_ == label; });
227 if(it_this == object_poses_.end()) {
228 return false;
229 }
230
231 auto it_other = std::find_if(other.object_poses_.cbegin(), other.object_poses_.cend(),
232 [=](const ObjectPoseGT& gt_object) { return gt_object.object_id_ == label; });
233 if(it_other == other.object_poses_.cend()) {
234 return false;
235 }
236
237 obj_idx = std::distance(object_poses_.begin(), it_this);
238 other_obj_idx = std::distance(other.object_poses_.cbegin(), it_other);
239 return true;
240}

244size_t GroundTruthInputPacket::calculateAndSetMotions(const GroundTruthInputPacket& previous_object_packet, ObjectIds& motions_set) {
245 motions_set.clear();
246 if(previous_object_packet.frame_id_ != frame_id_ -1) { return false; }
247
248 //iterate over all objects and try to find assications in previous packet
249 for(ObjectPoseGT& object_pose_gt : object_poses_) {
250 const ObjectId label = object_pose_gt.object_id_;
251
252 ObjectPoseGT* current_object_pose_gt;
253 const ObjectPoseGT* previous_object_pose_gt;
254 if(findAssociatedObject(label, previous_object_packet, &current_object_pose_gt, &previous_object_pose_gt)) {
255 CHECK_NOTNULL(current_object_pose_gt);
256 CHECK_NOTNULL(previous_object_pose_gt);
257 CHECK_EQ(*current_object_pose_gt, object_pose_gt);
258
259 //set motions for current motion
260 current_object_pose_gt->setMotions(*previous_object_pose_gt, previous_object_packet.X_world_, X_world_);
261 motions_set.push_back(label);
262 }
263
264 }
265
266 return motions_set.size();
267}

271 return calculateAndSetMotions(previous_object_packet, motion_set);
