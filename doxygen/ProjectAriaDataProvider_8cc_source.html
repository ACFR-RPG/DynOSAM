124 void loadFlowImagesAndSize(std::vector<std::string>& images_paths, size_t& dataset_size) {
125 std::vector<std::filesystem::path> files_in_directory = getAllFilesInDir(optical_flow_folder_path_);
126 dataset_size = files_in_directory.size();
127 CHECK_GT(dataset_size, 0);
128
129 for (const std::filesystem::path& path : files_in_directory) {
130 const std::string file_path = path;
131 throwExceptionIfPathInvalid(file_path);
132 images_paths.push_back(file_path);
133
134 std::string timestamp_str = path.stem();
135 double timestamp_nano = std::stod(timestamp_str);
136 double timestamp_seconds = timestamp_nano / 1.0e+9;
137 times_.push_back(timestamp_seconds);
138 }
139 }
