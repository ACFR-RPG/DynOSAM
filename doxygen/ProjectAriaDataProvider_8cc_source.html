109 cv::Mat getInstanceMask(size_t idx) const {
110 CHECK_LT(idx, rgb_image_paths_.size());
111 CHECK_LT(idx, dataset_size_);
112
113 // relabel masks from 1 to N
114 static gtsam::FastMap<ObjectId, ObjectId> relabelled;
115 static ObjectId current_max_label = 1u;
116
117 cv::Mat mask;
118 loadMask(instance_masks_image_paths_.at(idx), mask);
119 CHECK(!mask.empty());
120
121 ObjectIds labels = vision_tools::getObjectLabels(mask);
122 const ObjectIds& old_labels = labels;
123 ObjectIds new_labels;
124 for (ObjectId old_label : labels) {
125 if (!relabelled.exists(old_label)) {
126 relabelled[old_label] = current_max_label;
127 current_max_label++;
128 }
129 ObjectId new_label = relabelled.at(old_label);
130 new_labels.push_back(new_label);
131 }
132
133 cv::Mat relabelled_mask;
134 vision_tools::relabelMasks(mask, relabelled_mask, old_labels, new_labels);
135
136 return relabelled_mask;
137 }

157 void loadFlowImagesAndSize(std::vector<std::string>& images_paths,
158 size_t& dataset_size) {
159 std::vector<std::filesystem::path> files_in_directory =
160 getAllFilesInDir(optical_flow_folder_path_);
161 dataset_size = files_in_directory.size();
162 CHECK_GT(dataset_size, 0);
163
164 for (const std::filesystem::path& path : files_in_directory) {
165 const std::string file_path = path;
166 throwExceptionIfPathInvalid(file_path);
167 images_paths.push_back(file_path);
168
169 std::string timestamp_str = path.stem();
170 double timestamp_nano = std::stod(timestamp_str);
171 double timestamp_seconds = timestamp_nano / 1.0e+9;
172 times_.push_back(timestamp_seconds);
173 }
174 }

176 void loadOtherImages() {
177 auto rgb_image_files = getAllFilesInDir(rgb_images_folder_path_);

179 getAllFilesInDir(right_rgb_images_folder_path_);
180 auto depth_image_files = getAllFilesInDir(depth_images_folder_path_);

182
183 for (auto f : rgb_image_files) {
184 rgb_image_paths_.push_back(f);

186

190 }
191 right_rgb_image_paths_.pop_back();
192
193 for (auto f : depth_image_files) {
194 depth_image_paths_.push_back(f);
195 }
196 for (auto f : instance_masks_image_files) {
197 instance_masks_image_paths_.push_back(f);

199
200 CHECK_EQ(rgb_image_paths_.size(), depth_image_paths_.size());
201 CHECK_EQ(right_rgb_image_paths_.size(), depth_image_paths_.size());
202 CHECK_EQ(depth_image_paths_.size(), instance_masks_image_paths_.size());

204 // dataset size is number optical flow which should be one less!!
