64bool createDirectory(const std::string& path) {
65 std::filesystem::path dir_path(path);
66
67 // Check if directory already exists
68 if (std::filesystem::exists(dir_path)) {
69 if (std::filesystem::is_directory(dir_path)) {
70 VLOG(10) << "Directory already exists: " << path;
71 return true;
72 } else {
73 VLOG(10) << "Path exists but is not a directory: " << path;
74 return false;
75 }
76 }
77
78 // Create directory if it doesn't exist
79 if (std::filesystem::create_directories(dir_path)) {
80 VLOG(10) << "Directory created: " << path;
81 return true;
82 } else {
83 LOG(WARNING) << "Failed to create directory: " << path;
84 return false;
85 }
86}

115bool OfstreamWrapper::WriteOutCsvWriter(const CsvWriter& csv,
116 const std::string& filename) {
117 // set append mode to false as we never want to write over the top of a csv

123void OfstreamWrapper::openLogFile(bool open_file_in_append_mode) {
124 CHECK(!filename_.empty());

187std::optional<size_t> EstimationModuleLogger::logObjectMotion(
188 FrameId frame_id, const MotionEstimateMap& motion_estimates,
189 const std::optional<GroundTruthPacketMap>& gt_packets) {
190 // if gt packet provided by no data exists at this frame
191 if (gt_packets && !gt_packets->exists(frame_id)) {
192 VLOG(100) << "No gt packet at frame id " << frame_id
193 << ". Unable to log object motion errors";
194 return {};
195 }
196
197 size_t number_logged = 0;
198 for (const auto& [object_id, motions] : motion_estimates) {
199 // use identity motion if no ground truth so that we keep the csv file
200 // format
201 gtsam::Pose3 gt_motion = gtsam::Pose3::Identity();
202 const gtsam::Pose3& estimate = motions;
203
204 // gt packet provided so only log if gt pose data found
205 if (gt_packets) {
206 if (gt_packets->exists(frame_id)) {
207 const GroundTruthInputPacket& gt_packet_k = gt_packets->at(frame_id);
208 // check object exists in this frame
209 ObjectPoseGT object_gt_k;
210 if (!gt_packet_k.getObject(object_id, object_gt_k)) {
211 // if no packet for this object found, continue and do not log
212 continue;
213 } else {
214 CHECK(object_gt_k.prev_H_current_world_);
215 gt_motion = *object_gt_k.prev_H_current_world_;
216 }
217 } else {
218 // gt packet has no entry for this frame id so skip
219 continue;
220 }
221 }
222
223 const auto& quat = estimate.rotation().toQuaternion();
224 const auto& gt_quat = gt_motion.rotation().toQuaternion();
225
226 *object_motion_csv_ << frame_id << object_id << estimate.x() << estimate.y()
227 << estimate.z() << quat.x() << quat.y() << quat.z()
228 << quat.w() << gt_motion.x() << gt_motion.y()
229 << gt_motion.z() << gt_quat.x() << gt_quat.y()
230 << gt_quat.z() << gt_quat.w();
231 number_logged++;
232 }
233 return number_logged;
234}

236// assume poses are in world?
237std::optional<size_t> EstimationModuleLogger::logObjectPose(
238 FrameId frame_id, const ObjectPoseMap& propogated_poses,
239 const std::optional<GroundTruthPacketMap>& gt_packets) {
240 // if gt packet provided by no data exists at this frame
241 if (gt_packets && !gt_packets->exists(frame_id)) {
242 VLOG(100) << "No gt packet at frame id " << frame_id
243 << ". Unable to log object pose errors";
244 return {};
245 }
246
247 size_t number_logged = 0;
248 // assume object poses get logged in frame order!!!
249 for (const auto& [object_id, poses_map] : propogated_poses) {
250 // do not draw if in current frame
251 if (!poses_map.exists(frame_id)) {
252 VLOG(10) << "Cannot log object pose (id=" << object_id << ") for frame "
253 << frame_id << " as it does not exist in the map";
254 continue;
255 }
256
257 // use identity if no ground truth so that we keep the csv file format
258 gtsam::Pose3 gt_L_world_k = gtsam::Pose3::Identity();
259 const gtsam::Pose3& L_world_k = poses_map.at(frame_id);
260
261 // gt packet provided so only log if gt pose data found
262 if (gt_packets) {
263 if (gt_packets->exists(frame_id)) {
264 const GroundTruthInputPacket& gt_packet_k = gt_packets->at(frame_id);
265 // check object exists in this frame
266 ObjectPoseGT object_gt_k;
267 if (!gt_packet_k.getObject(object_id, object_gt_k)) {
268 // if no packet for this object found, continue and do not log
269 continue;
270 } else {
271 gt_L_world_k = object_gt_k.L_world_;
272 }
273 } else {
274 // gt packet has no entry for this frame id so skip
275 continue;
276 }
277 }
278 // we have either skipped logging (if gt provided by the object gt pose was
279 // not found) or no gt was provided
280 const auto& gt_R_k = gt_L_world_k.rotation().toQuaternion();
281 // estimate
282 const auto R_world_k = L_world_k.rotation().toQuaternion();
283 // write out object pose with gt
284 *object_pose_csv_ << frame_id << object_id << L_world_k.x() << L_world_k.y()
285 << L_world_k.z() << R_world_k.x() << R_world_k.y()
286 << R_world_k.z() << R_world_k.w() << gt_L_world_k.x()
287 << gt_L_world_k.y() << gt_L_world_k.z() << gt_R_k.x()
288 << gt_R_k.y() << gt_R_k.z() << gt_R_k.w();
289
290 number_logged++;
291 }
292 return number_logged;
293}

295std::optional<size_t> EstimationModuleLogger::logCameraPose(
296 FrameId frame_id, const gtsam::Pose3& T_world_camera,
297 const std::optional<GroundTruthPacketMap>& gt_packets) {
298 gtsam::Pose3 gt_T_world_camera_k = gtsam::Pose3::Identity();
299 // if gt packet provided by no data exists at this frame
300 if (gt_packets && !gt_packets->exists(frame_id)) {
301 VLOG(100) << "No gt packet at frame id " << frame_id
302 << ". Unable to log object motions";
303 return {};
304 } else if (gt_packets && gt_packets->exists(frame_id)) {
305 const GroundTruthInputPacket& gt_packet_k = gt_packets->at(frame_id);
306 gt_T_world_camera_k = gt_packet_k.X_world_;
307 }
308
309 const auto& rot = T_world_camera.rotation().toQuaternion();
310 const auto& gt_rot = gt_T_world_camera_k.rotation().toQuaternion();
311 *camera_pose_csv_ << frame_id << T_world_camera.x() << T_world_camera.y()
312 << T_world_camera.z() << rot.x() << rot.y() << rot.z()
313 << rot.w() << gt_T_world_camera_k.x()
314 << gt_T_world_camera_k.y() << gt_T_world_camera_k.z()
315 << gt_rot.x() << gt_rot.y() << gt_rot.z() << gt_rot.w();
316 return 1;
317}

319void EstimationModuleLogger::logPoints(FrameId frame_id,
320 const gtsam::Pose3& T_world_local_k,
321 const StatusLandmarkVector& landmarks) {
322 for (const auto& status_lmks : landmarks) {
323 const TrackletId tracklet_id = status_lmks.trackletId();
324 ObjectId object_id = status_lmks.objectId();
325 Landmark lmk_world = status_lmks.value();
326
327 if (status_lmks.referenceFrame() == ReferenceFrame::LOCAL) {
328 lmk_world = T_world_local_k * status_lmks.value();
329 } else if (status_lmks.referenceFrame() == ReferenceFrame::OBJECT) {
330 throw DynosamException(
331 "Cannot log object point in the object reference frame");
332 }
333
334 *map_points_csv_ << frame_id << object_id << tracklet_id << lmk_world(0)
335 << lmk_world(1) << lmk_world(2);
336 }
337}

352 << this_object_bbx.bbx_position_.z() << q.w() << q.x()
353 << q.y() << q.z();
