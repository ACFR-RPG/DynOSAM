63bool createDirectory(const std::string& path) {
64 std::filesystem::path dir_path(path);
65
66 // Check if directory already exists
67 if (std::filesystem::exists(dir_path)) {
68 if (std::filesystem::is_directory(dir_path)) {
69 VLOG(10) << "Directory already exists: " << path;
70 return true;
71 } else {
72 VLOG(10) << "Path exists but is not a directory: " << path;
73 return false;
74 }
75 }
76
77 // Create directory if it doesn't exist
78 if (std::filesystem::create_directories(dir_path)) {
79 VLOG(10) << "Directory created: " << path;
80 return true;
81 } else {
82 LOG(WARNING) << "Failed to create directory: " << path;
83 return false;
84 }
85}

114bool OfstreamWrapper::WriteOutCsvWriter(const CsvWriter& csv,
115 const std::string& filename) {
116 // set append mode to false as we never want to write over the top of a csv
117 // file as this will upset the header
118 OfstreamWrapper ofsw(filename, false);
119 return csv.write(ofsw.ofstream_);
120}

122void OfstreamWrapper::openLogFile(bool open_file_in_append_mode) {
123 CHECK(!filename_.empty());
124 CHECK(!output_path_.empty());
125 LOG(INFO) << "Opening output file: " << filename_.c_str();
126 OpenFile((std::string)getFilePath(), &ofstream_, open_file_in_append_mode);
127}

129fs::path OfstreamWrapper::getFilePath() const {
130 fs::path fs_out_path(output_path_);
131 if (!fs::exists(fs_out_path))
132 throw std::runtime_error("OfstreamWrapper - Output path does not exist: " +
133 output_path_);
134
135 return fs_out_path / fs::path(filename_);
136}

303std::optional<size_t> EstimationModuleLogger::logCameraPose(
304 FrameId frame_id, const gtsam::Pose3& T_world_camera,
305 const std::optional<GroundTruthPacketMap>& gt_packets) {
306 gtsam::Pose3 gt_T_world_camera_k = gtsam::Pose3::Identity();
307 // if gt packet provided by no data exists at this frame
308 if (gt_packets && !gt_packets->exists(frame_id)) {
309 VLOG(100) << "No gt packet at frame id " << frame_id
310 << ". Unable to log object motions";
311 return {};
312 } else if (gt_packets && gt_packets->exists(frame_id)) {
313 const GroundTruthInputPacket& gt_packet_k = gt_packets->at(frame_id);
314 gt_T_world_camera_k = gt_packet_k.X_world_;
315 }
316
317 const auto& rot = T_world_camera.rotation().toQuaternion();
318 const auto& gt_rot = gt_T_world_camera_k.rotation().toQuaternion();
319 *camera_pose_csv_ << frame_id << T_world_camera.x() << T_world_camera.y()
320 << T_world_camera.z() << rot.x() << rot.y() << rot.z()
321 << rot.w() << gt_T_world_camera_k.x()
322 << gt_T_world_camera_k.y() << gt_T_world_camera_k.z()
323 << gt_rot.x() << gt_rot.y() << gt_rot.z() << gt_rot.w();
324 return 1;
325}

327void EstimationModuleLogger::logPoints(FrameId frame_id,
328 const gtsam::Pose3& T_world_local_k,
329 const StatusLandmarkVector& landmarks) {
330 for (const auto& status_lmks : landmarks) {
331 const TrackletId tracklet_id = status_lmks.trackletId();
332 ObjectId object_id = status_lmks.objectId();
333 Landmark lmk_world = status_lmks.value();
334
335 if (status_lmks.referenceFrame() == ReferenceFrame::LOCAL) {
336 lmk_world = T_world_local_k * status_lmks.value();
337 } else if (status_lmks.referenceFrame() == ReferenceFrame::OBJECT) {
338 throw DynosamException(
339 "Cannot log object point in the object reference frame");
340 }
341
342 *map_points_csv_ << frame_id << object_id << tracklet_id << lmk_world(0)
343 << lmk_world(1) << lmk_world(2);
344 }
345}

389bool EstimationModuleLogger::logObjectMotion(
390 CsvWriter& writer, const Motion3ReferenceFrame& motion, FrameId frame_id,
391 ObjectId object_id, const std::optional<GroundTruthPacketMap>& gt_packets) {
392 gtsam::Pose3 gt_motion = gtsam::Pose3::Identity();
393 CHECK_EQ(motion.style(), MotionRepresentationStyle::F2F)
394 << "Output evaluation expects motion in F2F representation";
395 const gtsam::Pose3& estimate = motion;
396
397 if (gt_packets) {
398 if (gt_packets->exists(frame_id)) {
399 const GroundTruthInputPacket& gt_packet_k = gt_packets->at(frame_id);
400
401 // check object exists in this frame
402 ObjectPoseGT object_gt_k;
403 if (!gt_packet_k.getObject(object_id, object_gt_k)) {
404 // if no packet for this object found, continue and do not log
405 // return false;
406 } else {
407 CHECK(object_gt_k.prev_H_current_world_);
408 gt_motion = *object_gt_k.prev_H_current_world_;
409 }
410 } else {
411 // gt packet has no entry for this frame id and the object ground truth is
412 // valid
413 // TODO: for now?
414 // return false;
415 }
416 }
417
418 const auto& quat = estimate.rotation().toQuaternion();
419 const auto& gt_quat = gt_motion.rotation().toQuaternion();
420
421 writer << frame_id << object_id << estimate.x() << estimate.y()
422 << estimate.z() << quat.x() << quat.y() << quat.z() << quat.w()
423 << gt_motion.x() << gt_motion.y() << gt_motion.z() << gt_quat.x()
424 << gt_quat.y() << gt_quat.z() << gt_quat.w();
425 return true;
426}

428bool EstimationModuleLogger::logObjectPose(
429 CsvWriter& writer, const gtsam::Pose3& pose, FrameId frame_id,
430 ObjectId object_id, const std::optional<GroundTruthPacketMap>& gt_packets) {
431 // use identity if no ground truth so that we keep the csv file format
432 gtsam::Pose3 gt_L_world_k = gtsam::Pose3::Identity();
433 const gtsam::Pose3& L_world_k = pose;
434
435 // gt packet provided so only log if gt pose data found
436 if (gt_packets) {
437 if (gt_packets->exists(frame_id)) {
438 const GroundTruthInputPacket& gt_packet_k = gt_packets->at(frame_id);
439 // check object exists in this frame
440 ObjectPoseGT object_gt_k;
441 if (!gt_packet_k.getObject(object_id, object_gt_k)) {
442 // if no packet for this object found, continue and do not log
443 // return false;
444 } else {
445 gt_L_world_k = object_gt_k.L_world_;
446 }
447 } else {
448 // gt packet has no entry for this frame id so skip
449 // TODO: for now?
450 // return false;
451 }
452 }
453 // we have either skipped logging (if gt provided by the object gt pose was
454 // not found) or no gt was provided
455 const auto& gt_R_k = gt_L_world_k.rotation().toQuaternion();
456 // estimate
457 const auto R_world_k = L_world_k.rotation().toQuaternion();
458 // write out object pose with gt
459 writer << frame_id << object_id << L_world_k.x() << L_world_k.y()
460 << L_world_k.z() << R_world_k.x() << R_world_k.y() << R_world_k.z()
461 << R_world_k.w() << gt_L_world_k.x() << gt_L_world_k.y()
462 << gt_L_world_k.z() << gt_R_k.x() << gt_R_k.y() << gt_R_k.z()
463 << gt_R_k.w();
464 return true;
465}
