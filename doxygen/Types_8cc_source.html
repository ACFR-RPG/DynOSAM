61std::string demangle(const char* name)
62{
63 // by default set to the original mangled name
64 std::string demangled_name = std::string(name);
65
66 // g++ version of demangle
67 char* demangled = nullptr;
68 int status = -1; // some arbitrary value to eliminate the compiler warning
69 demangled = abi::__cxa_demangle(name, nullptr, nullptr, &status);
70
71 demangled_name = (status == 0) ? std::string(demangled) : std::string(name);
72 std::free(demangled);
73
74 return demangled_name;
75}
