367void Frame::constructDynamicObservations() {
368 object_observations_.clear();
369 //assumes that the mask gets updated with the tracking label
370 const ObjectIds instance_labels = vision_tools::getObjectLabels(image_container_.get<ImageType::MotionMask>());
371
372 auto inlier_iterator = dynamic_features_.beginUsable();
373 for(const Feature::Ptr& dynamic_feature : inlier_iterator) {
374 CHECK(!dynamic_feature->isStatic());
375 CHECK(dynamic_feature->usable());
376
377 const ObjectId instance_label = dynamic_feature->objectId();
378 //this check is just for sanity!
379 CHECK(std::find(instance_labels.begin(), instance_labels.end(), instance_label) != instance_labels.end())
380 << "Missing " << instance_label << " in " << container_to_string(instance_labels);
381
382 if(object_observations_.find(instance_label) == object_observations_.end()) {
383 DynamicObjectObservation observation;
384 observation.tracking_label_ = -1;
385 observation.instance_label_ = instance_label;
386 object_observations_[instance_label] = observation;
387 }
388
389 object_observations_[instance_label].object_features_.push_back(dynamic_feature->trackletId());
390 }
391
392 // now construct image masks from tracking mask
393 // For each tracked object, find its id in the mask
394 // and draw it.
395 // We apply some eroding/dilation on it to make the resulting submask smoother
396 // so that we can more easily fit an rectangle to it
397 const cv::Mat& mask = image_container_.get<ImageType::MotionMask>();
398 for(auto& object_observation_pair : object_observations_) {
399 const ObjectId object_id = object_observation_pair.first;
400 DynamicObjectObservation& obs = object_observation_pair.second;
401 vision_tools::findObjectBoundingBox(mask, object_id, obs.bounding_box_);
402 }

407void Frame::moveObjectToStatic(ObjectId instance_label) {
408 auto it = object_observations_.find(instance_label);
409 CHECK(it != object_observations_.end());
410
411
412 DynamicObjectObservation& observation = it->second;
413 observation.marked_as_moving_ = false;
414 CHECK(observation.instance_label_ == instance_label);
415 //go through all features, move them to from dynamic structure and add them to static
416 for(TrackletId tracklet_id : observation.object_features_) {
417 CHECK(dynamic_features_.exists(tracklet_id));
418 Feature::Ptr dynamic_feature = dynamic_features_.getByTrackletId(tracklet_id);
419
420 if(!dynamic_feature->usable()) {continue;}
421
422 CHECK(!dynamic_feature->isStatic());
423 CHECK_EQ(dynamic_feature->trackletId(), tracklet_id);
424 CHECK_EQ(dynamic_feature->objectId(), instance_label);
425 dynamic_feature->keypointType(KeyPointType::STATIC);
426 dynamic_feature->objectId(background_label);
427 // dynamic_feature->tracking_label_ = background_label;
428
429 dynamic_features_.remove(tracklet_id);
430 //Jesse: no, do not move points (these are dense) to static - instrad we need to mark the AREA
431 //around the object as static and then retrack all points in there!!
432 // static_features_.add(dynamic_feature);
433 }
434
435 object_observations_.erase(it);
