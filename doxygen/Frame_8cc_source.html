172PointCloudLabelRGB::Ptr Frame::projectToDenseCloud(
173 const cv::Mat* detection_mask) const {
174 if (!image_container_.hasDepth()) {
175 return nullptr;
176 }
177
178 PointCloudLabelRGB::Ptr cloud = pcl::make_shared<PointCloudLabelRGB>();
179 const cv::Mat& depth_image = image_container_.depth();
180 const cv::Mat& motion_mask = image_container_.objectMotionMask();
181
182 if (detection_mask) {
183 CHECK(utils::cvSizeEqual(detection_mask->size(), depth_image.size()));
184 }
185
186 const int rows = depth_image.rows;
187 const int cols = depth_image.cols;
188
189 // Reserve memory for the cloud (approximate max size)
190 // cloud->points.reserve(rows * cols);
191 cloud->points.resize(rows * cols);
192
193 // api needs image ref
194 // in this function we will not actually change the depth image
195 cv::Mat& depth_image_ref = const_cast<cv::Mat&>(depth_image);
196 FunctionalParallelOpenCVMat process(
197 depth_image_ref, [&](cv::Mat& depth_image, int i, int j) {
198 const unsigned char* detection_ptr =
199 detection_mask ? detection_mask->ptr<unsigned char>(i) : nullptr;
200 const ObjectId* motion_mask_ptr = motion_mask.ptr<ObjectId>(i);
201 const Depth* depth_ptr = depth_image.ptr<Depth>(i);
202
203 if (detection_mask && detection_ptr[j] == 0) return;
204
205 const ObjectId object_id = motion_mask_ptr[j];
206 const Depth depth = depth_ptr[j];
207
208 double depth_thresh;
209 Color colour;
210 if (object_id == background_label) {
211 depth_thresh = max_background_threshold_;
212 colour = Color::black();
213 } else {
214 depth_thresh = max_object_threshold_;
215 colour = Color::uniqueId(object_id);
216 }
217
218 if (depth > depth_thresh || depth <= 0 || !std::isfinite(depth)) return;
219
220 // Back-projection
221 const Keypoint kp(j, i);
222 Landmark point;
223 // this call is probably very slow
224 camera_->backProject(kp, depth, &point);
225
226 cloud->points[i * cols + j] = PointLabelRGB(
227 static_cast<float>(point(0)), static_cast<float>(point(1)),
228 static_cast<float>(point(2)), static_cast<std::uint8_t>(colour.r),
229 static_cast<std::uint8_t>(colour.g),
230 static_cast<std::uint8_t>(colour.b),
231 static_cast<std::uint32_t>(object_id));
232 });
233 process.run();
234
235 cloud->width = cloud->points.size();
236 cloud->height = 1;
237 cloud->is_dense = false;
238 return cloud;
239}

469void Frame::constructDynamicObservations() {
470 object_observations_.clear();
471 // assumes that the mask gets updated with the tracking label
472 const ObjectIds instance_labels =
473 vision_tools::getObjectLabels(image_container_.objectMotionMask());
474
475 auto inlier_iterator = dynamic_features_.beginUsable();
476 for (const Feature::Ptr& dynamic_feature : inlier_iterator) {
477 CHECK(!dynamic_feature->isStatic());
478 CHECK(dynamic_feature->usable());
479
480 const ObjectId object_id = dynamic_feature->objectId();
481 // this check is just for sanity!
482 CHECK(std::find(instance_labels.begin(), instance_labels.end(),
483 object_id) != instance_labels.end())
484 << "Missing " << object_id << " in "
485 << container_to_string(instance_labels);
486
487 if (object_observations_.find(object_id) == object_observations_.end()) {
488 SingleDetectionResult observation;
489 observation.object_id = object_id;
490 object_observations_[object_id] = observation;
491 }
492
493 // object_observations_[object_id].object_features.push_back(
494 // dynamic_feature->trackletId());
495 }
496
497 // now construct image masks from tracking mask
498 // For each tracked object, find its id in the mask
499 // and draw it.
500 // We apply some eroding/dilation on it to make the resulting submask smoother
501 // so that we can more easily fit an rectangle to it
502 const cv::Mat& mask = image_container_.objectMotionMask();
503 for (auto& object_observation_pair : object_observations_) {
504 const ObjectId object_id = object_observation_pair.first;
505 SingleDetectionResult& obs = object_observation_pair.second;
506 vision_tools::findObjectBoundingBox(mask, object_id, obs.bounding_box);
507 }
508}
