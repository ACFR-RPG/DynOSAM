472// atttempt to mimic the copy/assignment behaviour of the cv::Mat class
473// everything is a shallow copy until specified with clone
474class ImageContainer {
475 private:
492 struct KeyImagePair {
493 const std::string key;
494 const std::string stored_type;
495 std::unique_ptr<ImageBase> ptr;
496
497 // only for when the type is externally known as the ImageBase contains no
498 // information about the (IMAGETYPE) type
499 KeyImagePair(const std::string& name, const std::string& type,
500 std::unique_ptr<ImageBase> image_wrapper)
501 : key(name), stored_type(type), ptr(std::move(image_wrapper)) {}
502
503 template <typename IMAGETYPE>
504 KeyImagePair(const std::string& name,
505 std::unique_ptr<ImageWrapper<IMAGETYPE>> image_wrapper)
506 : key(name),
507 stored_type(type_name<IMAGETYPE>()),
508 ptr(std::move(image_wrapper)) {}
509
510 // Copy constructor
511 KeyImagePair(const KeyImagePair& other)
512 : key(other.key),
513 stored_type(other.stored_type),
514 ptr(other.ptr ? other.ptr->shallowCopy() : nullptr) {}
515
516 // Copy assignment operator
517 KeyImagePair& operator=(const KeyImagePair& other) {
518 if (this != &other) {
519 // key and stored_type are const, so normally non-assignable,
520 // so this only works if you can guarantee they are the same or make
521 // them non-const. Otherwise, you may want to remove const qualifier on
522 // key and stored_type.
523 assert(this->stored_type == other.stored_type);
524 assert(this->key == other.key);
525
526 // For simplicity, assume non-const or only allow assignment when key is
527 // the same
528 ptr = other.ptr ? other.ptr->shallowCopy() : nullptr;
529 }
530 return *this;
531 }
532
533 // Move constructor
534 KeyImagePair(KeyImagePair&&) noexcept = default;
535 // Move assignment
536 KeyImagePair& operator=(KeyImagePair&&) noexcept = default;
537
538 template <typename IMAGETYPE>
539 static KeyImagePair Create(const std::string& key, const cv::Mat& image) {
540 return KeyImagePair(key,
541 std::make_unique<ImageWrapper<IMAGETYPE>>(image));
542 }
543
544 KeyImagePair clone() const {
545 CHECK_NOTNULL(ptr);
546 return KeyImagePair(key, stored_type, ptr ? ptr->deepCopy() : nullptr);
547 }
548
549 template <typename IMAGETYPE>
550 const ImageWrapper<IMAGETYPE>& cast() const {
551 return castImpl<IMAGETYPE, true>();
552 }
553
554 template <typename IMAGETYPE>
555 ImageWrapper<IMAGETYPE>& cast() {
556 return castImpl<IMAGETYPE, false>();
557 }
558
559 template <typename IMAGETYPE, bool IsConst>
560 decltype(auto) castImpl() const {
561 using WrapperType = ImageWrapper<IMAGETYPE>;
562 using PointerType =
563 std::conditional_t<IsConst, const WrapperType*, WrapperType*>;
564
565 PointerType casted = dynamic_cast<PointerType>(ptr.get());
566
567 if (!casted) {
568 const auto requested_type = type_name<IMAGETYPE>();
569 throw MismatchedImageWrapperTypes(requested_type, stored_type);
570 }
571
572 return *casted;
573 }
574 };

576 template <typename Container, typename IMAGETYPE>
577 static decltype(auto) atImpl(Container* image_container,
578 const std::string& key) {
579 if (!image_container->exists(key)) {
580 throw ImageKeyDoesNotExist(key);
581 }
582
583 decltype(auto) key_image = image_container->images_.at(key);
584 return key_image.template cast<IMAGETYPE>();
585 }

587 FrameId frame_id_;
588 Timestamp timestamp_;
589 gtsam::FastMap<std::string, KeyImagePair> images_;
590
591 public:
592 DYNO_POINTER_TYPEDEFS(ImageContainer)
593
594 static constexpr char kRGB[] = "rgb";
595 static constexpr char kOPticalFlow[] = "opticalflow";
596 static constexpr char kDepth[] = "depth";
597 static constexpr char kObjectMask[] = "objectmask";
598 static constexpr char kRightRgb[] = "rightrgb";
599
600 public:
601 ImageContainer(FrameId frame_id, Timestamp timestamp)
602 : frame_id_(frame_id), timestamp_(timestamp), images_() {}
603 ImageContainer() : frame_id_(0), timestamp_(InvalidTimestamp), images_() {}
604
605 ImageContainer(const ImageContainer& other)
606 : frame_id_(other.frame_id_), timestamp_(other.timestamp_) {
607 for (const auto& [k, v] : other.images_) {
608 images_.emplace(
609 k, v); // Uses KeyImagePair copy ctor above (ie. shallow image copy)
610 }
611 }
612
613 ImageContainer& operator=(const ImageContainer& other) {
614 if (this != &other) {
615 frame_id_ = other.frame_id_;
616 timestamp_ = other.timestamp_;
617 images_.clear();
618 for (const auto& [k, v] : other.images_) {
619 images_.emplace(k, v);
620 }
621 }
622 return *this;
623 }

626 ImageContainer& operator=(ImageContainer&&) noexcept = default;

629 ImageContainer& add(const std::string& key, const cv::Mat& image) {
630 if (exists(key)) {

633

635 images_.insert({key, std::move(key_image)});

637 return *this;
638 }
