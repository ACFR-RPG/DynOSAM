127 struct KeyImagePair {
128 const std::string key;
129 const std::string stored_type;
130 std::unique_ptr<ImageBase> ptr;

132 // only for when the type is externally known as the ImageBase contains no
133 // information about the (IMAGETYPE) type
134 KeyImagePair(const std::string& name, const std::string& type,
135 std::unique_ptr<ImageBase> image_wrapper)
136 : key(name), stored_type(type), ptr(std::move(image_wrapper)) {}
137
138 template <typename IMAGETYPE>
139 KeyImagePair(const std::string& name,
140 std::unique_ptr<ImageWrapper<IMAGETYPE>> image_wrapper)
141 : key(name),
142 stored_type(type_name<IMAGETYPE>()),
143 ptr(std::move(image_wrapper)) {}
144
145 // Copy constructor
146 KeyImagePair(const KeyImagePair& other)
147 : key(other.key),
148 stored_type(other.stored_type),
149 ptr(other.ptr ? other.ptr->shallowCopy() : nullptr) {}
150
151 // Copy assignment operator
152 KeyImagePair& operator=(const KeyImagePair& other) {
153 if (this != &other) {
154 // key and stored_type are const, so normally non-assignable,
155 // so this only works if you can guarantee they are the same or make
156 // them non-const. Otherwise, you may want to remove const qualifier on
157 // key and stored_type.
158 assert(this->stored_type == other.stored_type);
159 assert(this->key == other.key);
160
161 // For simplicity, assume non-const or only allow assignment when key is
162 // the same
163 ptr = other.ptr ? other.ptr->shallowCopy() : nullptr;
164 }
165 return *this;
166 }
167
168 // Move constructor
169 KeyImagePair(KeyImagePair&&) noexcept = default;
170 // Move assignment
171 KeyImagePair& operator=(KeyImagePair&&) noexcept = default;
172
173 template <typename IMAGETYPE>
174 static KeyImagePair Create(const std::string& key, const cv::Mat& image) {
175 return KeyImagePair(key,
176 std::make_unique<ImageWrapper<IMAGETYPE>>(image));
177 }
178
179 KeyImagePair clone() const {
180 CHECK_NOTNULL(ptr);
181 return KeyImagePair(key, stored_type, ptr ? ptr->deepCopy() : nullptr);
182 }

184 template <typename IMAGETYPE>
185 const ImageWrapper<IMAGETYPE>& cast() const {
186 return castImpl<IMAGETYPE, true>();
187 }

189 template <typename IMAGETYPE>
190 ImageWrapper<IMAGETYPE>& cast() {
191 return castImpl<IMAGETYPE, false>();
192 }
193
194 template <typename IMAGETYPE, bool IsConst>
195 decltype(auto) castImpl() const {
196 using WrapperType = ImageWrapper<IMAGETYPE>;
197 using PointerType =
198 std::conditional_t<IsConst, const WrapperType*, WrapperType*>;
199
200 PointerType casted = dynamic_cast<PointerType>(ptr.get());
201
202 if (!casted) {
203 const auto requested_type = type_name<IMAGETYPE>();
204 throw MismatchedImageWrapperTypes(requested_type, stored_type);
205 }
206
207 return *casted;
208 }
209 };

211 template <typename Container, typename IMAGETYPE>
212 static decltype(auto) atImpl(Container* image_container,
213 const std::string& key) {
214 if (!image_container->exists(key)) {
215 throw ImageKeyDoesNotExist(key);
216 }
217
218 decltype(auto) key_image = image_container->images_.at(key);
219 return key_image.template cast<IMAGETYPE>();
220 }
221
222 FrameId frame_id_;
223 Timestamp timestamp_;
224 gtsam::FastMap<std::string, KeyImagePair> images_;
225
226 public:
227 DYNO_POINTER_TYPEDEFS(ImageContainer)
228
229 static constexpr char kRGB[] = "rgb";
230 static constexpr char kOPticalFlow[] = "opticalflow";
231 static constexpr char kDepth[] = "depth";
232 static constexpr char kObjectMask[] = "objectmask";
233 static constexpr char kRightRgb[] = "rightrgb";
234
235 public:
236 ImageContainer(FrameId frame_id, Timestamp timestamp)
237 : frame_id_(frame_id), timestamp_(timestamp), images_() {}
238 ImageContainer() : frame_id_(0), timestamp_(InvalidTimestamp), images_() {}
239
240 ImageContainer(const ImageContainer& other)
241 : frame_id_(other.frame_id_), timestamp_(other.timestamp_) {
242 for (const auto& [k, v] : other.images_) {
243 images_.emplace(
244 k, v); // Uses KeyImagePair copy ctor above (ie. shallow image copy)
245 }
246 }
247
248 ImageContainer& operator=(const ImageContainer& other) {
249 if (this != &other) {
250 frame_id_ = other.frame_id_;
251 timestamp_ = other.timestamp_;
252 images_.clear();
253 for (const auto& [k, v] : other.images_) {
254 images_.emplace(k, v);
255 }
256 }
257 return *this;
258 }

260 ImageContainer(ImageContainer&&) noexcept = default;
261 ImageContainer& operator=(ImageContainer&&) noexcept = default;

263 template <typename IMAGETYPE>
264 ImageContainer& add(const std::string& key, const cv::Mat& image) {
265 if (exists(key)) {
266 throw ImageKeyAlreadyExists(key);
267 }
268
269 KeyImagePair key_image = KeyImagePair::Create<IMAGETYPE>(key, image);
270 images_.insert({key, std::move(key_image)});
271
272 return *this;
273 }
274
275 template <typename IMAGETYPE>
276 ImageContainer& replace(const std::string& key, const cv::Mat& image) {
277 if (!exists(key)) {
278 return this->add<IMAGETYPE>(key, image);
279 }
280
281 KeyImagePair key_image = KeyImagePair::Create<IMAGETYPE>(key, image);
282 images_.insert_or_assign(key, std::move(key_image));
283 return *this;
284 }
