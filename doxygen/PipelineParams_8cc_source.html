62DynoParams::DynoParams(const std::string& params_folder_path) {
63 pipeline_params_ = config::fromYamlFile<PipelineParams>(
64 params_folder_path + "PipelineParams.yaml");
65 camera_params_ = config::fromYamlFile<CameraParams>(params_folder_path +
66 "CameraParams.yaml");
67 frontend_params_ = config::fromYamlFile<FrontendParams>(
68 params_folder_path + "FrontendParams.yaml");
69 imu_params_ =
70 config::fromYamlFile<ImuParams>(params_folder_path + "ImuParams.yaml");
71
72 backend_type = static_cast<BackendType>(FLAGS_backend_updater_enum);
73}
