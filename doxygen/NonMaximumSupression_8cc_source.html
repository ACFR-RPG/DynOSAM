33std::vector<cv::KeyPoint> AdaptiveNonMaximumSuppression::suppressNonMax(
34 const std::vector<cv::KeyPoint>& keyPoints, const int& numRetPoints,
35 const float& tolerance, const int& cols, const int& rows,
36 const int& nr_horizontal_bins, const int& nr_vertical_bins,
37 const Eigen::MatrixXd& binning_mask) {
38 utils::TimingStatsCollector timer("anms_timer");
39
40 if (keyPoints.size() == 0) {
41 std::cout << "No keypoints for non-max suppression..." << std::endl;
42 return std::vector<cv::KeyPoint>();
43 }
44
45 // Sorting keypoints by deacreasing order of strength
46 // VLOG(5) << "Sorting keypoints in decreasing order of strength.";
47 std::vector<int> responseVector;
48 for (unsigned int i = 0; i < keyPoints.size(); i++) {
49 responseVector.push_back(keyPoints[i].response);
50 }
51 std::vector<int> Indx(responseVector.size());
52 std::iota(std::begin(Indx), std::end(Indx), 0);
53 cv::sortIdx(responseVector, Indx, cv::SortFlags::SORT_DESCENDING);
54 std::vector<cv::KeyPoint> keyPointsSorted;
55 for (unsigned int i = 0; i < keyPoints.size(); i++) {
56 keyPointsSorted.push_back(keyPoints[Indx[i]]);
57 }
58
59 std::vector<cv::KeyPoint> keypoints;
60 // VLOG(5) << "Starting Adaptive Non-Maximum Suppression.";
61 switch (anms_algorithm_type_) {
62 case AnmsAlgorithmType::TopN: {
63 // VLOG(1) << "Running TopN: " <<
64 // VIO::to_underlying(anms_algorithm_type_);
65 keypoints = anms::TopN(keyPoints, numRetPoints);
66 break;
67 };
68 case AnmsAlgorithmType::BrownANMS: {
69 // VLOG(1) << "Running BrownANMS: "
70 // << VIO::to_underlying(anms_algorithm_type_);
71 keypoints = anms::BrownANMS(keyPoints, numRetPoints);
72 break;
73 };
74 case AnmsAlgorithmType::SDC: {
75 // VLOG(1) << "Running SDC: " <<
76 // VIO::to_underlying(anms_algorithm_type_);
77 keypoints =
78 anms::Sdc(keyPointsSorted, numRetPoints, tolerance, cols, rows);
79 break;
80 };
81 case AnmsAlgorithmType::KdTree: {
82 // VLOG(1) << "Running KdTree: " <<
83 // VIO::to_underlying(anms_algorithm_type_);
84 keypoints =
85 anms::KdTree(keyPointsSorted, numRetPoints, tolerance, cols, rows);
86 break;
87 };
88 case AnmsAlgorithmType::RangeTree: {
89 keypoints =
90 anms::RangeTree(keyPointsSorted, numRetPoints, tolerance, cols, rows);
91 break;
92 };
93 case AnmsAlgorithmType::Ssc: {
94 // VLOG(1) << "Running SSC: " <<
95 // VIO::to_underlying(anms_algorithm_type_);
96 keypoints =
97 anms::Ssc(keyPointsSorted, numRetPoints, tolerance, cols, rows);
98 break;
99 };
100 case AnmsAlgorithmType::Binning: {
101 // VLOG(1) << "Running Binning: "
102 // << VIO::to_underlying(anms_algorithm_type_);
103 keypoints = binning(keyPointsSorted, numRetPoints, cols, rows,
104 nr_horizontal_bins, nr_vertical_bins, binning_mask);
105 break;
106 };
107 default: {
108 // VLOG(1) << "Unknown ANMS algorithm requested: "
109 // << VIO::to_underlying(anms_algorithm_type_);
110 break;
111 };
112 }
113 // VLOG(1) << "Non Maximum Suppression Timing [ms]: "
114 // << utils::Timer::toc(tic).count();
115 return keypoints;
116}

119std::vector<cv::KeyPoint> AdaptiveNonMaximumSuppression::binning(
120 const std::vector<cv::KeyPoint>& keyPoints, const int& numKptsToRetain,
121 const int& imgCols, const int& imgRows, const int& nr_horizontal_bins,
122 const int& nr_vertical_bins, const Eigen::MatrixXd& binning_mask) {
123 if (static_cast<size_t>(numKptsToRetain) > keyPoints.size()) {
124 return keyPoints;
125 }
126
127 float binRowSize = float(imgRows) / float(nr_vertical_bins);
128 float binColSize = float(imgCols) / float(nr_horizontal_bins);
129
130 // Note: features should be already sorted by score at this point from detect
131
132 // 0. count the number of valid bins (as specified by the user in the yaml
133 float nrActiveBins = binning_mask.sum(); // sum of 1's in binary mask
134
135 // 1. compute how many features we want to retain in each bin
136 // numRetPointsPerBin
137 const int numRetPointsPerBin =
138 std::round(float(numKptsToRetain) / float(nrActiveBins));
139
140 // 2. assign keypoints to bins and retain top numRetPointsPerBin for each bin
141 std::vector<cv::KeyPoint> binnedKpts; // binned keypoints we want to output
142 Eigen::MatrixXd nrKptsInBin = Eigen::MatrixXd::Zero(
143 nr_vertical_bins,
144 nr_horizontal_bins); // store number of kpts for each bin
145 for (size_t i = 0; i < keyPoints.size(); i++) {
146 const size_t binRowInd =
147 static_cast<size_t>(keyPoints[i].pt.y / binRowSize);
148 const size_t binColInd =
149 static_cast<size_t>(keyPoints[i].pt.x / binColSize);
150 // if bin is active and needs more keypoints
151 if (binning_mask(binRowInd, binColInd) == 1 &&
152 nrKptsInBin(binRowInd, binColInd) <
153 numRetPointsPerBin) { // if we need more kpts in that bin
154 binnedKpts.push_back(keyPoints[i]);
155 nrKptsInBin(binRowInd, binColInd) += 1;
156 }
157 }
158 return binnedKpts;
159}
