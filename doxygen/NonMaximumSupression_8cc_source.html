35std::vector<cv::KeyPoint> AdaptiveNonMaximumSuppression::suppressNonMax(
36 const std::vector<cv::KeyPoint>& keyPoints,
37 const int& numRetPoints,
38 const float& tolerance,
39 const int& cols,
40 const int& rows,
41 const int& nr_horizontal_bins,
42 const int& nr_vertical_bins,
43 const Eigen::MatrixXd& binning_mask) {
44
45 utils::TimingStatsCollector timer("anms_timer");
46
47 if (keyPoints.size() == 0) {
48 std::cout << "No keypoints for non-max suppression..." <<std::endl;
49 return std::vector<cv::KeyPoint>();
50 }
51
52 // Sorting keypoints by deacreasing order of strength
53// VLOG(5) << "Sorting keypoints in decreasing order of strength.";
54 std::vector<int> responseVector;
55 for (unsigned int i = 0; i < keyPoints.size(); i++) {
56 responseVector.push_back(keyPoints[i].response);
57 }
58 std::vector<int> Indx(responseVector.size());
59 std::iota(std::begin(Indx), std::end(Indx), 0);
60 cv::sortIdx(responseVector, Indx, cv::SortFlags::SORT_DESCENDING);
61 std::vector<cv::KeyPoint> keyPointsSorted;
62 for (unsigned int i = 0; i < keyPoints.size(); i++) {
63 keyPointsSorted.push_back(keyPoints[Indx[i]]);
64 }
65
66 std::vector<cv::KeyPoint> keypoints;
67// VLOG(5) << "Starting Adaptive Non-Maximum Suppression.";
68 switch (anms_algorithm_type_) {
69 case AnmsAlgorithmType::TopN: {
70 // VLOG(1) << "Running TopN: " << VIO::to_underlying(anms_algorithm_type_);
71 keypoints = anms::TopN(keyPoints, numRetPoints);
72 break;
73 };
74 case AnmsAlgorithmType::BrownANMS: {
75 // VLOG(1) << "Running BrownANMS: "
76 // << VIO::to_underlying(anms_algorithm_type_);
77 keypoints = anms::BrownANMS(keyPoints, numRetPoints);
78 break;
79 };
80 case AnmsAlgorithmType::SDC: {
81 // VLOG(1) << "Running SDC: " << VIO::to_underlying(anms_algorithm_type_);
82 keypoints =
83 anms::Sdc(keyPointsSorted, numRetPoints, tolerance, cols, rows);
84 break;
85 };
86 case AnmsAlgorithmType::KdTree: {
87 // VLOG(1) << "Running KdTree: " << VIO::to_underlying(anms_algorithm_type_);
88 keypoints =
89 anms::KdTree(keyPointsSorted, numRetPoints, tolerance, cols, rows);
90 break;
91 };
92 case AnmsAlgorithmType::RangeTree: {
93 std::cout << "Running RangeTree " << keyPointsSorted.size() << std::endl;
94 std::cout << numRetPoints << std::endl;
95 // VLOG(1) << "Running RangeTree: "
96 // << VIO::to_underlying(anms_algorithm_type_);
97 keypoints =
98 anms::RangeTree(keyPointsSorted, numRetPoints, tolerance, cols, rows);
99 break;
100 };
101 case AnmsAlgorithmType::Ssc: {
102 // VLOG(1) << "Running SSC: " << VIO::to_underlying(anms_algorithm_type_);
103 keypoints =
104 anms::Ssc(keyPointsSorted, numRetPoints, tolerance, cols, rows);
105 break;
106 };
107 case AnmsAlgorithmType::Binning: {
108 // VLOG(1) << "Running Binning: "
109 // << VIO::to_underlying(anms_algorithm_type_);
110 keypoints = binning(keyPointsSorted,
111 numRetPoints,
112 cols,
113 rows,
114 nr_horizontal_bins,
115 nr_vertical_bins,
116 binning_mask);
117 break;
118 };
119 default: {
120 // VLOG(1) << "Unknown ANMS algorithm requested: "
121 // << VIO::to_underlying(anms_algorithm_type_);
122 break;
123 };
124 }
125// VLOG(1) << "Non Maximum Suppression Timing [ms]: "
126// << utils::Timer::toc(tic).count();
127 return keypoints;
128}

131std::vector<cv::KeyPoint> AdaptiveNonMaximumSuppression::binning(
132 const std::vector<cv::KeyPoint>& keyPoints,
133 const int& numKptsToRetain,
134 const int& imgCols,
135 const int& imgRows,
136 const int& nr_horizontal_bins,
137 const int& nr_vertical_bins,
138 const Eigen::MatrixXd& binning_mask) {
139 if (static_cast<size_t>(numKptsToRetain) > keyPoints.size()) {
140 return keyPoints;
141 }
142
143 float binRowSize = float(imgRows) / float(nr_vertical_bins);
144 float binColSize = float(imgCols) / float(nr_horizontal_bins);
145
146 // Note: features should be already sorted by score at this point from detect
147
148 // 0. count the number of valid bins (as specified by the user in the yaml
149 float nrActiveBins = binning_mask.sum(); // sum of 1's in binary mask
150
151 // 1. compute how many features we want to retain in each bin
152 // numRetPointsPerBin
153 const int numRetPointsPerBin =
154 std::round(float(numKptsToRetain) / float(nrActiveBins));
155
156 // 2. assign keypoints to bins and retain top numRetPointsPerBin for each bin
157 std::vector<cv::KeyPoint> binnedKpts; // binned keypoints we want to output
158 Eigen::MatrixXd nrKptsInBin = Eigen::MatrixXd::Zero(
159 nr_vertical_bins,
160 nr_horizontal_bins); // store number of kpts for each bin
161 for (size_t i = 0; i < keyPoints.size(); i++) {
162 const size_t binRowInd =
163 static_cast<size_t>(keyPoints[i].pt.y / binRowSize);
164 const size_t binColInd =
165 static_cast<size_t>(keyPoints[i].pt.x / binColSize);
166 // if bin is active and needs more keypoints
167 if (binning_mask(binRowInd, binColInd) == 1 &&
168 nrKptsInBin(binRowInd, binColInd) <
169 numRetPointsPerBin) { // if we need more kpts in that bin
170 binnedKpts.push_back(keyPoints[i]);
171 nrKptsInBin(binRowInd, binColInd) += 1;
172 }
173 }
174 return binnedKpts;
175}
