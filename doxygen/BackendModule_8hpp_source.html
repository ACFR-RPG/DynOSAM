115template <class MODULE_TRAITS>
116class BackendModuleType : public BackendModule {
117 public:
118 using ModuleTraits = MODULE_TRAITS;
119 // A Dervied BackedInputPacket type (e.g. RGBDOutputPacketType)
120 using DerivedPacketType = typename ModuleTraits::DerivedPacketType;
121 using DerivedPacketTypeConstPtr =
122 typename ModuleTraits::DerivedPacketTypeConstPtr;
123 using MeasurementType = typename ModuleTraits::MeasurementType;
124 using This = BackendModuleType<ModuleTraits>;
125 using Base = BackendModule;
126
127 using MapType = Map<MeasurementType>;
128
129 DYNO_POINTER_TYPEDEFS(This)
130
131 using Base::SpinReturn;
132 // Define the input type to the derived input type, defined in the
133 // MODULE_TRAITS this is the derived Input packet that is passed to the
134 // boostrap/nominal Spin Impl functions that must be implemented in the
135 // derived class that does the provessing on this module
136 using InputConstPtr = DerivedPacketTypeConstPtr;
137 using OutputConstPtr = Base::OutputConstPtr;
138
139 BackendModuleType(const BackendParams& params, typename MapType::Ptr map,
140 ImageDisplayQueue* display_queue)
141 : Base(params, display_queue), map_(CHECK_NOTNULL(map)) {}
142
143 virtual ~BackendModuleType() {}
144
145 inline const typename MapType::Ptr getMap() { return map_; }
146
147 protected:
148 virtual SpinReturn boostrapSpinImpl(InputConstPtr input) = 0;
149 virtual SpinReturn nominalSpinImpl(InputConstPtr input) = 0;
150
151 typename MapType::Ptr map_;
152
153 private:
154 SpinReturn boostrapSpin(Base::BaseInputConstPtr base_input) override {
155 return boostrapSpinImpl(attemptCast(base_input));
156 }
157
158 SpinReturn nominalSpin(Base::BaseInputConstPtr base_input) override {
159 return nominalSpinImpl(attemptCast(base_input));
160 }
161
162 DerivedPacketTypeConstPtr attemptCast(Base::BaseInputConstPtr base_input) {
163 DerivedPacketTypeConstPtr deriverd_input =
164 safeCast<Base::BaseInput, DerivedPacketType>(base_input);
165 checkAndThrow((bool)deriverd_input,
166 "Failed to cast " + type_name<Base::BaseInput>() + " to " +
167 type_name<DerivedPacketType>() + " in BackendModuleType");
168 return deriverd_input;
169 }
170};
