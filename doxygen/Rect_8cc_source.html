31float calc_iou(const RectBase& A, const RectBase& B) {
32 const float iw = std::min(A.left() + A.width(), B.left() + B.width()) -
33 std::max(A.left(), B.left());
34 float iou = 0;
35 if (iw > 0) {
36 const float ih = std::min(A.top() + A.height(), B.top() + B.height()) -
37 std::max(A.top(), B.top());
38 if (ih > 0) {
39 const float ua =
40 A.width() * A.height() + B.width() * B.height() - iw * ih;
41 iou = iw * ih / ua;
42 }
43 }
44 return iou;
45}
