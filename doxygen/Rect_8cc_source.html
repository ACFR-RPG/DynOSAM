93template <typename T>
94float byte_track::Rect<T>::calcIoU(const Rect<T>& other) const {
95 const float box_area = (other.tlwh[2] + 1) * (other.tlwh[3] + 1);
96 const float iw = std::min(tlwh[0] + tlwh[2], other.tlwh[0] + other.tlwh[2]) -
97 std::max(tlwh[0], other.tlwh[0]) + 1;
98 float iou = 0;
99 if (iw > 0) {
100 const float ih =
101 std::min(tlwh[1] + tlwh[3], other.tlwh[1] + other.tlwh[3]) -
102 std::max(tlwh[1], other.tlwh[1]) + 1;
103 if (ih > 0) {
104 const float ua = (tlwh[0] + tlwh[2] - tlwh[0] + 1) *
105 (tlwh[1] + tlwh[3] - tlwh[1] + 1) +
106 box_area - iw * ih;
107 iou = iw * ih / ua;
108 }
109 }
110 return iou;
111}
