60void Spinner::shutdown() {
61 if (spin_thread_ && spin_thread_->joinable()) {
62 VLOG(10) << "Spinner " << name_ << " shutting down";
63 spin_thread_->join();
64 is_running_ = false;
65 } else {
66 // VLOG(10) << "Spinner " << name_ << "already shutdown";
67 }
