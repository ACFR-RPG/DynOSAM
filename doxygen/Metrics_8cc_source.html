34TRErrorPair TRErrorPair::CalculatePoseError(const gtsam::Pose3& M, const gtsam::Pose3& M_hat) {
35 const gtsam::Pose3 E = M.inverse() * M_hat;
36
37 // L2 norm - ie. magnitude
38 const double t_e = E.translation().norm();
39
40 const gtsam::Matrix33 rotation_error = E.rotation().matrix();
41 std::pair<gtsam::Unit3, double> axis_angle = gtsam::Rot3(rotation_error).axisAngle();
42
43 //need to wrap?
44 const double r_e = axis_angle.second;
45
46 return TRErrorPair(t_e, r_e);
47}

63TRErrorPair TRErrorPairVector::average() const
64{
65 if (Base::empty())
66 {
67 return TRErrorPair();
68 }
69
70 TRErrorPair error_pair;
71
72 double x = 0, y = 0;
73 for (const TRErrorPair& pair : *this)
74 {
75 error_pair.translation_ += pair.translation_;
76
77 y += std::sin(pair.rot_);
78 x += std::cos(pair.rot_);
79 }
80
81 const double length = static_cast<double>(this->size());
82 x /= length;
83 y /= length;
84 error_pair.rot_ = dyno::wrapTwoPi(std::atan2(y, x));
85 error_pair.translation_ /= length;
86
87 return error_pair;
88}
