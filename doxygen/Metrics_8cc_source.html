42TRErrorPair TRErrorPair::CalculatePoseError(const gtsam::Pose3& M,
43 const gtsam::Pose3& M_hat) {
44 const gtsam::Pose3 E = M.inverse() * M_hat;
45
46 // L2 norm - ie. magnitude
47 const double t_e = E.translation().norm();
48
49 const gtsam::Matrix33 rotation_error = E.rotation().matrix();
50 std::pair<gtsam::Unit3, double> axis_angle =
51 gtsam::Rot3(rotation_error).axisAngle();
52
53 // need to wrap?
54 const double r_e = axis_angle.second;
55
56 return TRErrorPair(t_e, r_e);
57}

59TRErrorPair TRErrorPair::CalculateRelativePoseError(
60 const gtsam::Pose3& M_ref, const gtsam::Pose3& M_curr,
61 const gtsam::Pose3& M_hat_ref, const gtsam::Pose3& M_hat_curr) {
62 // pose change between ref and current
63 const gtsam::Pose3 ref_T_curr = M_ref.inverse() * M_curr;
64 const gtsam::Pose3 ref_T_hat_curr = M_hat_ref.inverse() * M_hat_curr;
65 return CalculatePoseError(ref_T_curr, ref_T_hat_curr);
66}

72TRErrorPair TRErrorPairVector::average() const {
73 if (Base::empty()) {
74 return TRErrorPair();
75 }
76
77 TRErrorPair error_pair;
78
79 double x = 0, y = 0;
80 for (const TRErrorPair& pair : *this) {
81 error_pair.translation_ += pair.translation_;
82
83 y += std::sin(pair.rot_);
84 x += std::cos(pair.rot_);
85 }
86
87 const double length = static_cast<double>(this->size());
88 x /= length;
89 y /= length;
90 error_pair.rot_ = dyno::wrapTwoPi(std::atan2(y, x));
91 error_pair.translation_ /= length;
92
93 return error_pair;
94}
