118 template <class Rep = int64_t, class Period = std::milli>
119 const CameraParams& waitAndSetCameraParams(
120 const std::chrono::duration<Rep, Period>& time_to_wait =
121 std::chrono::duration<Rep, Period>(-1),
122 const std::string& topic = "image/camera_info") {
123 RCLCPP_INFO_STREAM(node_->get_logger(),
124 "Waiting for camera params on topic: " << topic);
125 // it seems rclcpp::Adaptors do not work yet with wait for message
126 sensor_msgs::msg::CameraInfo camera_info;
127 if (rclcpp::wait_for_message<sensor_msgs::msg::CameraInfo, Rep, Period>(
128 camera_info, node_, topic, time_to_wait)) {
129 using Adaptor =
130 rclcpp::TypeAdapter<dyno::CameraParams, sensor_msgs::msg::CameraInfo>;
131 CameraParams camera_params;
132 Adaptor::convert_to_custom(camera_info, camera_params);
133 RCLCPP_INFO_STREAM(node_->get_logger(), "Received camera params: "
134 << camera_params.toString());
135 camera_params_ = camera_params;
136 return *camera_params_;
137 } else {
138 const auto milliseconds =
139 std::chrono::duration_cast<std::chrono::milliseconds>(time_to_wait);
140 throw DynosamException("Failed to receive camera params on topic " +
141 topic + " (waited with timeout " +
142 std::to_string(milliseconds.count()) + " ms).");
143 }
144 }
145
146 CameraParams::Optional getCameraParams() const override {
147 return camera_params_;
148 }

176 template <typename IMAGETYPE>
177 const cv::Mat convertRosImage(
178 const sensor_msgs::msg::Image::ConstSharedPtr& img_msg) const {
179 const cv_bridge::CvImageConstPtr cvb_image = readRosImage(img_msg);
180
181 try {
182 const cv::Mat img = cvb_image->image;
183 image_traits<IMAGETYPE>::validate(img);
184 return img;
185
186 } catch (const InvalidImageTypeException& exception) {
187 RCLCPP_FATAL_STREAM(node_->get_logger(),
188 image_traits<IMAGETYPE>::name()
189 << " Image msg was of the wrong type (validate "
190 "failed with exception "
191 << exception.what() << "). "
192 << "ROS encoding type used was "
193 << cvb_image->encoding);
194 rclcpp::shutdown();
195 return cv::Mat();
196 }
197 }
198
199 protected:
200 rclcpp::Node::SharedPtr node_;
201 CameraParams::Optional camera_params_;
