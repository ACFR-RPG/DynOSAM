138 template <typename Index = int>
139 std::vector<Index> collectIds() const {
140 std::vector<Index> ids;
141 for (const auto& node : *this) {
142 // ducktyping for node to have a getId function
143 ids.push_back(static_cast<Index>(node->getId()));
144 }
145
146 return ids;
147 }

233 bool continuousAscendingIndex() const {
234 auto ids = collectIds();
235 if (ids.size() < 2) return true;
236
237 bool is_continuous = true;
238 for (size_t i = 1; i < ids.size(); i++) {
239 const auto prev = ids.at(i - 1);
240 const auto curr = ids.at(i);
241 is_continuous &= (prev + 1 == curr);
242 }
243 return is_continuous;
244 }
245
246 private:
247 template <typename Index>
248 static inline Index getIndexSafe(const NODE& node) {
249 return static_cast<Index>(node->getId());
250 }
