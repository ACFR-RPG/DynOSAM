140 template <typename Index = int>
141 std::vector<Index> collectIds() const {
142 std::vector<Index> ids;
143 for (const auto& node : *this) {
144 // ducktyping for node to have a getId function
145 ids.push_back(static_cast<Index>(node->getId()));
146 }
147
148 return ids;
149 }

235 bool continuousAscendingIndex() const {
236 auto ids = collectIds();
237 if (ids.size() < 2) return true;
238
239 bool is_continuous = true;
240 for (size_t i = 1; i < ids.size(); i++) {
241 const auto prev = ids.at(i - 1);
242 const auto curr = ids.at(i);
243 is_continuous &= (prev + 1 == curr);
244 }
245 return is_continuous;
246 }
247
248 private:
249 template <typename Index>
250 static inline Index getIndexSafe(const NODE& node) {
251 return static_cast<Index>(node->getId());
252 }

283template <typename ValueType>
284class StateQuery : public std::optional<ValueType> {
285 public:
286 using Base = std::optional<ValueType>;
287 enum Status { VALID, NOT_IN_MAP, WAS_IN_MAP, INVALID_MAP };
288 gtsam::Key key_;
289 Status status_;
290
291 StateQuery() {}
292 StateQuery(gtsam::Key key, const ValueType& v) : key_(key), status_(VALID) {
293 Base::emplace(v);
294 }
295 StateQuery(gtsam::Key key, Status status) : key_(key), status_(status) {}
296
297 const ValueType& get() const {
298 if (!Base::has_value())
299 throw DynosamException("StateQuery has no value for query type " +
300 type_name<ValueType>() + " with key" +
301 DynoLikeKeyFormatter(key_));
302 return Base::value();
303 }
304
305 bool isValid() const { return status_ == VALID; }
306
307 static StateQuery InvalidMap() {
308 return StateQuery(gtsam::Key{}, INVALID_MAP);
309 }
310 static StateQuery NotInMap(gtsam::Key key) {
311 return StateQuery(key, NOT_IN_MAP);
312 }
313 static StateQuery WasInMap(gtsam::Key key) {
314 return StateQuery(key, WAS_IN_MAP);
315 }
316};
