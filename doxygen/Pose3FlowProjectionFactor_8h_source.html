70 gtsam::Vector evaluateError(const gtsam::Point2& optical_flow, const gtsam::Pose3& camera_pose,
71 boost::optional<gtsam::Matrix&> J1 = boost::none,
72 boost::optional<gtsam::Matrix&> J2 = boost::none) const override
73 {
74
75 auto I = gtsam::traits<gtsam::Pose3>::Identity();
76 gtsam::PinholeCamera<Calibration> previous_camera(I, calibration_);
77
78 const double fx = calibration_.fx();
79 const double fy = calibration_.fy();
80 //project the ref keypoint into the world frame given the camera at k-1
81 gtsam::Point3 P_W = pose_previous_ * previous_camera.backproject(keypoint_previous_, depth_);
82
83 //camera at k using the same calibration
84 gtsam::PinholeCamera<Calibration> camera_current(I, calibration_);
85
86 gtsam::Point2 predicted_keypoint = keypoint_previous_ + optical_flow;
87 //project P_w into the current frame frame which should be the predicted one
88 gtsam::Point3 P_curr = camera_pose.inverse() * P_W;
89
90 try {
91 gtsam::Point2 predicted_keypoint_from_projection = camera_current.project2(P_curr);
92
93
94 if(J1) {
95 *J1 = gtsam::Matrix22::Identity();
96 }
97
98 if(J2) {
99 //point in the current camera frame
100 const double x = P_curr(0);
101 const double y = P_curr(1);
102 const double z = P_curr(2);
103 const double z_2 = z*z;
104
105 gtsam::Matrix26 H;
106 H(0,0) = x*y/z_2 *fx;
107 H(0,1) = -(1+(x*x/z_2)) *fx;
108 H(0,2) = y/z *fx;
109 H(0,3) = -1./z *fx;
110 H(0,4) = 0;
111 H(0,5) = x/z_2 *fx;
112
113 H(1,0) = (1+y*y/z_2) *fy;
114 H(1,1) = -x*y/z_2 *fy;
115 H(1,2) = -x/z *fy;
116 H(1,3) = 0;
117 H(1,4) = -1./z *fy;
118 H(1,5) = y/z_2 *fy;
119
120 *J2 = -1.0 * H;
121 }
122
123 return predicted_keypoint - predicted_keypoint_from_projection;
124 }
125 catch(gtsam::CheiralityException&) {
126 if (J1) *J1 = gtsam::Matrix::Zero(2,2);
127 if (J2) *J2 = gtsam::Matrix::Zero(2,6);
128 return gtsam::Vector2::Constant(2.0 * calibration_.fx());
129 }
130 }

137 Calibration calibration_;
