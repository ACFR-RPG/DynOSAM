65std::array<float, 3> rgbFromChromaHue(float chroma, float hue) {
66 const float hue_prime = hue * 6.0f;
67 const float x = chroma * (1.0f - std::abs(std::fmod(hue_prime, 2) - 1.0f));
68 if (hue_prime < 1) {
69 return {chroma, x, 0};
70 } else if (hue_prime < 2) {
71 return {x, chroma, 0};
72 } else if (hue_prime < 3) {
73 return {0, chroma, x};
74 } else if (hue_prime < 4) {
75 return {0, x, chroma};
76 } else if (hue_prime < 5) {
77 return {x, 0, chroma};
78 }
79 return {chroma, 0, x};
80}

82Color Color::fromHSV(float hue, float saturation, float value) {
83 hue = std::clamp(hue, 0.0f, 1.0f);
84 saturation = std::clamp(saturation, 0.0f, 1.0f);
85 value = std::clamp(value, 0.0f, 1.0f);
86 const float chroma = value * saturation;
87 const auto [r1, g1, b1] = rgbFromChromaHue(chroma, hue);
88 const float m = value - chroma;
89 return Color(static_cast<uint8_t>((r1 + m) * 255),
90 static_cast<uint8_t>((g1 + m) * 255),
91 static_cast<uint8_t>((b1 + m) * 255));
92}

94Color Color::fromHLS(float hue, float luminance, float saturation) {
95 hue = std::clamp(hue, 0.0f, 1.0f);
96 luminance = std::clamp(luminance, 0.0f, 1.0f);
97 saturation = std::clamp(saturation, 0.0f, 1.0f);
98 const float chroma = (1.0f - std::abs(2.0f * luminance - 1.0f)) * saturation;
99 const auto [r1, g1, b1] = rgbFromChromaHue(chroma, hue);
100 const float m = luminance - chroma / 2.0f;
101 return Color(static_cast<uint8_t>((r1 + m) * 255),
102 static_cast<uint8_t>((g1 + m) * 255),
103 static_cast<uint8_t>((b1 + m) * 255));
104}

126Color Color::spectrum(float value, const std::vector<Color>& colors) {
127 if (colors.empty()) {
128 return Color::black();
129 }
130 value = std::clamp(value, 0.0f, 1.0f);
131 const size_t num_steps = colors.size() - 1;
132 const size_t index = static_cast<size_t>(value * num_steps);
133 if (index >= num_steps) {
134 return colors.at(num_steps);
135 }
136 const float weight = value * num_steps - index;
137 return colors.at(index).blend(colors.at(index + 1), weight);
138}

150float exponentialOffsetId(size_t id, size_t ids_per_revolution) {
151 const size_t revolution = id / ids_per_revolution;
152 const float progress_along_revolution =
153 std::fmod(static_cast<float>(id) / ids_per_revolution, 1.f);
154 float offset = 0.0f;
155 if (ids_per_revolution < id + 1u) {
156 const size_t current_episode = std::floor(std::log2(revolution));
157 const size_t episode_start = std::exp2(current_episode);
158 const size_t current_subdivision = revolution - episode_start;
159 const float subdivision_step_size = 1.0f / (ids_per_revolution * 2 * episode_start);
160 offset = (2.0f * current_subdivision + 1) * subdivision_step_size;
161 }
162 return progress_along_revolution + offset;
163}

169Color Color::uniqueId(size_t id, float saturation, float value) {
170 constexpr static float phi = (1 + std::sqrt(5.0))/2.0;
171 auto n = (float)id * phi - std::floor((float)id * phi);
172
173 //want in 255 mode
174 float hue = std::floor(n * 256.0) / 255.0;
175 return Color::fromHSV(hue, saturation, value);
176}
