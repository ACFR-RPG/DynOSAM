42 using is_specialized = std::true_type;

45
46 static void convert_to_ros_message(const custom_type& source,
47 ros_message_type& destination) {
48 // intrisincs
49 const cv::Mat& K = source.getCameraMatrix();
50 std::copy(K.begin<double>(), K.end<double>(), destination.k.begin());
51
52 // distortion
53 const cv::Mat& D = source.getDistortionCoeffs();
54 destination.d.resize(D.total());
55 destination.d = std::vector<double>(D.begin<double>(), D.end<double>());
56
57 destination.height = source.ImageHeight();
58 destination.width = source.ImageWidth();
59
60 std::string camera_model;
61 CHECK(dyno::CameraParams::distortionModelToString(
62 source.getDistortionModel(), destination.distortion_model,
63 camera_model));
64 (void)camera_model;
65 }
66
67 static void convert_to_custom(const ros_message_type& source,
68 custom_type& destination) {
69 // bit gross but cv::Mat does not accept a const void* for data (which
70 // source.k.data()) is so to ensure the data is safe (ie. rather than doing
71 // a const conversion), I decide to copy the data into std::vector and then
72 // copy into cv::Mat. this is inefficient but this conversion happens so
73 // rarely and the data pakcet is tiny...
74 std::vector<double> k_vector(source.k.begin(), source.k.end());
75 cv::Mat K(3, 3, CV_64F, std::data(k_vector));
76 dyno::CameraParams::IntrinsicsCoeffs intrinsics;
77 // TODO: should make this a constructor...
78 dyno::CameraParams::convertKMatrixToIntrinsicsCoeffs(K, intrinsics);
79 dyno::CameraParams::DistortionCoeffs distortion(source.d.begin(),
80 source.d.end());
81 cv::Size size(source.width, source.height);
82 const std::string& distortion_model = source.distortion_model;
83
84 destination =
85 dyno::CameraParams(intrinsics, distortion, size, distortion_model);
86 }
87};
