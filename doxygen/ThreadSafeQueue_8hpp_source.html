102 int size() { return data_queue_.size(); }

104 void shutdown() {
105 std::unique_lock<std::mutex> mlock(mutex_);
106 // Even if the shared variable is atomic, it must be modified under the
107 // mutex in order to correctly publish the modification to the waiting
108 // threads.
109 shutdown_ = true;
110 mlock.unlock();
111 data_cond_.notify_all();
112 }

114 inline bool isShutdown() { return shutdown_; }

116 void resume() {
117 std::unique_lock<std::mutex> mlock(mutex_);
118 // Even if the shared variable is atomic, it must be modified under the
119 // mutex in order to correctly publish the modification to the waiting
120 // threads.
121 shutdown_ = false;
122 mlock.unlock();
123 data_cond_.notify_all();
124 }
125
129 bool empty() const {
130 std::lock_guard<std::mutex> lk(mutex_);
131 return data_queue_.empty();
132 }
133
134 // TODO: not tested
135 void clear() {
136 std::lock_guard<std::mutex> lk(mutex_);
137 InternalQueue empty;
138 std::swap(data_queue_, empty);
139 }

146 return shutdown_;
147 }

149 protected:
150 mutable std::mutex mutex_;
151 InternalQueue data_queue_;
152 std::condition_variable data_cond_;
153 std::atomic_bool shutdown_;

241template <typename T>
242bool ThreadsafeQueue<T>::push(T new_value) {
243 if (shutdown_) return false; // atomic, no lock needed.
244 std::shared_ptr<T> data(std::make_shared<T>(std::move(new_value)));
245 std::unique_lock<std::mutex> lk(mutex_);
246
247 if (queue_size > 0) {
248 // greater than or equal?
249 while (data_queue_.size() >= queue_size) {
250 data_queue_.pop();
251 }
252 }
253
254 data_queue_.push(data);
255 lk.unlock(); // Unlock before notify.
256 data_cond_.notify_one();
257 // Thread-safe so doesn't need external mutex
258 return true;
259}

261template <typename T>
262bool ThreadsafeQueue<T>::pushBlockingIfFull(T new_value,
263 size_t max_queue_size) {
264 if (shutdown_) return false; // atomic, no lock needed.
265 std::shared_ptr<T> data(std::make_shared<T>(std::move(new_value)));
266 std::unique_lock<std::mutex> lk(mutex_);
267 // Wait until the queue has space or shutdown requested.
268 data_cond_.wait(lk, [this, max_queue_size] {
269 return data_queue_.size() < max_queue_size || shutdown_;
270 });
271 if (shutdown_) return false;
272 data_queue_.push(data);
273 lk.unlock(); // Unlock before notify.
274 data_cond_.notify_one();
275 // Thread-safe so doesn't need external mutex.
276 return true;
277}

279template <typename T>
280bool ThreadsafeQueue<T>::popBlocking(T& value) {
281 std::unique_lock<std::mutex> lk(mutex_);
282 // Wait until there is data in the queue or shutdown requested.
283 data_cond_.wait(lk, [this] { return !data_queue_.empty() || shutdown_; });
284 // Return false in case shutdown is requested.
285 if (shutdown_) return false;
286 value = std::move(*data_queue_.front());
287 data_queue_.pop();
288 lk.unlock(); // Unlock before notify.
289 data_cond_.notify_one();
290 return true;
291}

305template <typename T>
306bool ThreadsafeQueue<T>::popBlockingWithTimeout(T& value, size_t duration_ms) {
307 std::unique_lock<std::mutex> lk(mutex_);
308 // Wait until there is data in the queue, shutdown is requested, or
309 // the given time is elapsed...
310 data_cond_.wait_for(lk, std::chrono::milliseconds(duration_ms),
311 [this] { return !data_queue_.empty() || shutdown_; });
312 // Return false in case shutdown is requested or the queue is empty (in which
313 // case, a timeout has happened).
314 if (shutdown_ || data_queue_.empty()) return false;
315 value = std::move(*data_queue_.front());
316 data_queue_.pop();
317 return true;
318}

345template <typename T>
346bool ThreadsafeQueue<T>::popAll(std::vector<T>& data, size_t duration_ms) {
347 std::unique_lock<std::mutex> lk(mutex_);
348 data.clear();
349 // Wait until there is data in the queue, shutdown is requested, or
350 // the given time is elapsed...
351 data_cond_.wait_for(lk, std::chrono::milliseconds(duration_ms),
352 [this] { return !data_queue_.empty() || shutdown_; });
353
354 while (!data_queue_.empty()) {
355 if (shutdown_) {
356 break;
357 }
358
359 T value = std::move(*data_queue_.front());
360 data.push_back(value);
361 data_queue_.pop();
362 }
363
364 return !data.empty();
