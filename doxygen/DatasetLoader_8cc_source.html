100void TimestampFile::onPathInit() {
101 std::ifstream times_stream((std::string)getAbsolutePath(), std::ios::in);
102
103 while (!times_stream.eof())
104 {
105 std::string s;
106 getline(times_stream, s);
107 if (!s.empty())
108 {
109 std::stringstream ss;
110 ss << s;
111 double t;
112 ss >> t;
113 times.push_back(t);
114 }
115 }
116 times_stream.close();
