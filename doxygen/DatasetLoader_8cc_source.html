100void TimestampFile::onPathInit() {
101 std::ifstream times_stream((std::string)getAbsolutePath(), std::ios::in);
102
103 while (!times_stream.eof()) {
104 std::string s;
105 getline(times_stream, s);
106 if (!s.empty()) {
107 std::stringstream ss;
108 ss << s;
109 double t;
110 ss >> t;
111 times.push_back(t);
112 }
113 }
114 times_stream.close();
115}
