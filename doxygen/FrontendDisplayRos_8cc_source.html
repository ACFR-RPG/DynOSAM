138void FrontendDisplayRos::processRGBDOutputpacket(
139 const RGBDInstanceOutputPacket::ConstPtr& rgbd_frontend_output) {
140 CHECK(rgbd_frontend_output);
141 publishPointCloud(static_tracked_points_pub_,
142 rgbd_frontend_output->static_landmarks_,
143 rgbd_frontend_output->T_world_camera_);
144
145 // TODO: there is a bunch of repeated code in this function and in
146 // groupObjectCloud we leave this as is becuase this function ALSO creates the
147 // coloured point cloud (which groupObjectCloud does not) eventually, refactor
148 // into one function or calcualte the coloured cloud in the
149 // RGBDInstanceOutputPacket
150 CloudPerObject clouds_per_obj = publishPointCloud(
151 dynamic_tracked_points_pub_, rgbd_frontend_output->dynamic_landmarks_,
152 rgbd_frontend_output->T_world_camera_);
153
154 publishObjectPositions(object_pose_pub_,
155 rgbd_frontend_output->propogated_object_poses_,
156 rgbd_frontend_output->getFrameId(),
157 rgbd_frontend_output->getTimestamp(), "frontend");
158
159 publishObjectPaths(object_pose_path_pub_,
160 rgbd_frontend_output->propogated_object_poses_,
161 rgbd_frontend_output->getFrameId(),
162 rgbd_frontend_output->getTimestamp(), "frontend", 60);
163
164 // object bounding box using linelist and id text in rviz
165 {
166 visualization_msgs::msg::MarkerArray
167 object_bbx_linelist_marker_array; // using linelist
168 visualization_msgs::msg::MarkerArray object_bbx_marker_array; // using cube
169 static visualization_msgs::msg::Marker delete_marker;
170 delete_marker.action = visualization_msgs::msg::Marker::DELETEALL;
171 object_bbx_linelist_marker_array.markers.push_back(delete_marker);
172 object_bbx_marker_array.markers.push_back(delete_marker);
173
174 for (const auto& [object_id, obj_cloud] : clouds_per_obj) {
175 pcl::PointXYZ centroid;
176 pcl::computeCentroid(obj_cloud, centroid);
177
178 const cv::Scalar colour = Color::uniqueId(object_id);
179
180 visualization_msgs::msg::Marker txt_marker;
181 txt_marker.header.frame_id = "world";
182 txt_marker.ns = "object_id";
183 txt_marker.id = object_id;
184 txt_marker.type = visualization_msgs::msg::Marker::TEXT_VIEW_FACING;
185 txt_marker.action = visualization_msgs::msg::Marker::ADD;
186 txt_marker.header.stamp = node_->now();
187 txt_marker.scale.z = 2.0;
188 txt_marker.color.r = colour(0) / 255.0;
189 txt_marker.color.g = colour(1) / 255.0;
190 txt_marker.color.b = colour(2) / 255.0;
191 txt_marker.color.a = 1;
192 txt_marker.text = "obj " + std::to_string(object_id);
193 txt_marker.pose.position.x = centroid.x;
194 txt_marker.pose.position.y = centroid.y - 2.0;
195 txt_marker.pose.position.z = centroid.z - 1.0;
196 object_bbx_linelist_marker_array.markers.push_back(txt_marker);
197
198 pcl::PointCloud<pcl::PointXYZRGB>::Ptr obj_cloud_ptr =
199 pcl::make_shared<pcl::PointCloud<pcl::PointXYZRGB> >(obj_cloud);
200 ObjectBBX aabb = findAABBFromCloud<pcl::PointXYZRGB>(obj_cloud_ptr);
201 ObjectBBX obb = findOBBFromCloud<pcl::PointXYZRGB>(obj_cloud_ptr);
202
203 visualization_msgs::msg::Marker marker;
204 marker.header.frame_id = "world";
205 marker.ns = "object_bbx_linelist";
206 marker.id = object_id;
207 marker.type = visualization_msgs::msg::Marker::LINE_LIST;
208 marker.action = visualization_msgs::msg::Marker::ADD;
209 marker.header.stamp = node_->now();
210 marker.scale.x = 0.1;
211
212 marker.pose.orientation.x = 0;
213 marker.pose.orientation.y = 0;
214 marker.pose.orientation.z = 0;
215 marker.pose.orientation.w = 1;
216
217 marker.color.r = colour(0) / 255.0;
218 marker.color.g = colour(1) / 255.0;
219 marker.color.b = colour(2) / 255.0;
220 marker.color.a = 1;
221
222 for (const pcl::PointXYZ& this_line_list_point :
223 findLineListPointsFromAABBMinMax(aabb.min_bbx_point_,
224 aabb.max_bbx_point_)) {
225 geometry_msgs::msg::Point p;
226 p.x = this_line_list_point.x;
227 p.y = this_line_list_point.y;
228 p.z = this_line_list_point.z;
229 marker.points.push_back(p);
230 }
231 object_bbx_linelist_marker_array.markers.push_back(marker);
232
233 visualization_msgs::msg::Marker bbx_marker;
234 bbx_marker.header.frame_id = "world";
235 bbx_marker.ns = "object_bbx";
236 bbx_marker.id = object_id;
237 bbx_marker.type = visualization_msgs::msg::Marker::CUBE;
238 bbx_marker.action = visualization_msgs::msg::Marker::ADD;
239 bbx_marker.header.stamp = node_->now();
240 bbx_marker.scale.x = obb.max_bbx_point_.x() - obb.min_bbx_point_.x();
241 bbx_marker.scale.y = obb.max_bbx_point_.y() - obb.min_bbx_point_.y();
242 bbx_marker.scale.z = obb.max_bbx_point_.z() - obb.min_bbx_point_.z();
243
244 bbx_marker.pose.position.x = obb.bbx_position_.x();
245 bbx_marker.pose.position.y = obb.bbx_position_.y();
246 bbx_marker.pose.position.z = obb.bbx_position_.z();
247
248 const gtsam::Quaternion& q = obb.orientation_.toQuaternion();
249 bbx_marker.pose.orientation.x = q.x();
250 bbx_marker.pose.orientation.y = q.y();
251 bbx_marker.pose.orientation.z = q.z();
252 bbx_marker.pose.orientation.w = q.w();
253
254 bbx_marker.color.r = colour(0) / 255.0;
255 bbx_marker.color.g = colour(1) / 255.0;
256 bbx_marker.color.b = colour(2) / 255.0;
257 bbx_marker.color.a = 0.2;
258
259 object_bbx_marker_array.markers.push_back(bbx_marker);
260 }
261 object_bbx_line_pub_->publish(object_bbx_linelist_marker_array);
262 object_bbx_pub_->publish(object_bbx_marker_array);
263 }

265 // Predict 3D object poses
266 gtsam::FastMap<ObjectId, std::vector<gtsam::Pose3> > obj_predicted_poses;
267 {
268 int prediction_length = 3;
269 const FrameId current_frame_id = rgbd_frontend_output->getFrameId();
270 const ObjectPoseMap& obj_poses =
271 rgbd_frontend_output->propogated_object_poses_;
272 const MotionEstimateMap& obj_motions =
273 rgbd_frontend_output->estimated_motions_;
274 for (const auto& [object_id, this_obj_traj] : obj_poses) {
275 if (this_obj_traj.exists(current_frame_id)) {
276 gtsam::Pose3 current_obj_pose = this_obj_traj.at(current_frame_id);
277 if (obj_motions.exists(object_id)) {
278 gtsam::Pose3 current_obj_motion = obj_motions.at(object_id);
279
280 gtsam::Pose3 last_object_pose = current_obj_pose;
281 std::vector<gtsam::Pose3> predicted_poses;
282
283 for (int i_prediction = 0; i_prediction < prediction_length;
284 i_prediction++) {
285 predicted_poses.push_back(last_object_pose);
286 last_object_pose = current_obj_motion * last_object_pose;
287 last_object_pose = current_obj_motion * last_object_pose;
288 // last_object_pose = current_obj_motion * last_object_pose;
289 }
290
291 obj_predicted_poses.insert2(object_id, predicted_poses);
292 }
293 }
294 }
