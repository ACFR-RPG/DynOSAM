55class CsvHeader : public std::vector<std::string> {
56 public:
57 template <typename... HeaderArgs,
58 typename = std::enable_if_t<
59 (std::is_convertible<HeaderArgs, std::string>::value && ...)>>
60 CsvHeader(const HeaderArgs&... args) {
61 std::vector<std::string>* result = this;
62 std::apply(
63 [result](const auto&... elems) {
64 result->reserve(sizeof...(elems));
65 (result->push_back(std::string(elems)), ...);
66 },
67 std::forward_as_tuple(args...));
68 }
69
70 std::string toString(const std::string& seperator = " ") const {
71 std::stringstream ss;
72 size_t i = 0;
73 for (; i < size() - 1u; i++) {
74 ss << this->at(i);
75 ss << seperator;
76 }
77 ss << this->at(i);
78 return ss.str();
79 }

93 auto it = std::find(this->begin(), this->end(), column_header);
94 if (it != this->end()) {
95 return std::distance(this->begin(), it);

97 return -1; // Element not found

183 class RowIterator {

195 RowIterator(std::istream* input_stream)
196 : input_stream_(CHECK_NOTNULL(input_stream)->good() ? input_stream
197 : nullptr) {
198 ++(*this);
199 }
200
201 RowIterator() : input_stream_(nullptr) {}
202
203 // Pre Increment
204 RowIterator& operator++() {
205 if (input_stream_) {
206 // read new row
207 row_ << *input_stream_;
208 // if end of file
209 if (!(*input_stream_)) {
210 input_stream_ = nullptr;
211 }
212 }
213 return *this;
214 }
215 // Post increment
216 RowIterator operator++(int) {
217 RowIterator tmp(*this);
218 ++(*this);
219 return tmp;
220 }
221
222 value_type const& operator*() const { return row_; }
223 value_type const* operator->() const { return &row_; }
224
225 bool operator==(RowIterator const& rhs) {
226 return ((this == &rhs) || ((this->input_stream_ == nullptr) &&
227 (rhs.input_stream_ == nullptr)));
228 }
229 bool operator!=(RowIterator const& rhs) { return !((*this) == rhs); }

231 private:
232 private:

234 // a non const value type
235 // is requited to be non_const so that we can update the row with new data
236 // during the pre-increment
