55/*kdtree algorithm*/
56template <typename T>
57struct PointCloud {
58 struct Point {
59 T x, y;
60 };
61 std::vector<Point> pts;
62 inline size_t kdtree_get_point_count() const {
63 return pts.size();
64 } // Must return the number of data points
65 // Returns the distance between the std::vector "p1[0:size-1]" and the data
66 // point with index "idx_p2" stored in the class:
67 inline T kdtree_distance(const T* p1, const size_t idx_p2,
68 size_t /*size*/) const {
69 const T d0 = p1[0] - pts[idx_p2].x;
70 const T d1 = p1[1] - pts[idx_p2].y;
71 return d0 * d0 + d1 * d1;
72 }
73 // Returns the dim'th component of the idx'th point in the class:
74 // Since this is inlined and the "dim" argument is typically an immediate
75 // value, the
76 // "if/else's" are actually solved at compile time.
77 inline T kdtree_get_pt(const size_t idx, int dim) const {
78 if (dim == 0)
79 return pts[idx].x;
80 else if (dim == 1)
81 return pts[idx].y;
82 return 0;
83 }
84 // Optional bounding-box computation: return false to default to a standard
85 // bbox computation loop.
86 // Return true if the BBOX was already computed by the class and returned in
87 // "bb" so it can be avoided to redo it again. Look at bb.size() to find out
88 // the expected dimensionality (e.g. 2 or 3 for point clouds)
89 template <class BBOX>
90 bool kdtree_get_bbox(BBOX& /*bb*/) const {
91 return false;
92 }
93};
