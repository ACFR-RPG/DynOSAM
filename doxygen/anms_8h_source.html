57/*kdtree algorithm*/
58template <typename T>
59struct PointCloud {
60 struct Point {
61 T x, y;
62 };
63 std::vector<Point> pts;
64 inline size_t kdtree_get_point_count() const {
65 return pts.size();
66 } // Must return the number of data points
67 // Returns the distance between the std::vector "p1[0:size-1]" and the data
68 // point with index "idx_p2" stored in the class:
69 inline T kdtree_distance(const T* p1,
70 const size_t idx_p2,
71 size_t /*size*/) const {
72 const T d0 = p1[0] - pts[idx_p2].x;
73 const T d1 = p1[1] - pts[idx_p2].y;
74 return d0 * d0 + d1 * d1;
75 }
76 // Returns the dim'th component of the idx'th point in the class:
77 // Since this is inlined and the "dim" argument is typically an immediate
78 // value, the
79 // "if/else's" are actually solved at compile time.
80 inline T kdtree_get_pt(const size_t idx, int dim) const {
81 if (dim == 0)
82 return pts[idx].x;
83 else if (dim == 1)
84 return pts[idx].y;
85 return 0;
86 }
87 // Optional bounding-box computation: return false to default to a standard
88 // bbox computation loop.
89 // Return true if the BBOX was already computed by the class and returned in
90 // "bb" so it can be avoided to redo it again. Look at bb.size() to find out
91 // the expected dimensionality (e.g. 2 or 3 for point clouds)
92 template <class BBOX>
93 bool kdtree_get_bbox(BBOX& /*bb*/) const {
94 return false;
95 }
96};
