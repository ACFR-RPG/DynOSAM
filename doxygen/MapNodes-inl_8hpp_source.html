150template <typename MEASUREMENT>
151void LandmarkNode<MEASUREMENT>::add(FrameNodePtr<MEASUREMENT> frame_node,
152 const MEASUREMENT& measurement) {
153 frames_seen_.insert(frame_node);
154
155 // add measurement to map
156 // first check that we dont already have a measurement at this frame
157 if (measurements_.exists(frame_node)) {
158 throw DynosamException("Unable to add new measurement to landmark node " +
159 std::to_string(this->getId()) + " at frame " +
160 std::to_string(frame_node->getId()) +
161 " as a measurement already exists at this frame!");
162 }
163
164 measurements_.insert2(frame_node, measurement);
165
166 CHECK_EQ(frames_seen_.size(), measurements_.size());
167}

179template <typename MEASUREMENT>
180const MEASUREMENT& LandmarkNode<MEASUREMENT>::getMeasurement(
181 FrameNodePtr<MEASUREMENT> frame_node) const {
182 CHECK_NOTNULL(frame_node);
183 if (!hasMeasurement(frame_node->frame_id)) {
184 throw DynosamException("Missing measurement in landmark node with id " +
185 std::to_string(tracklet_id) + " at frame " +
186 std::to_string(frame_node->frame_id));
187 }
188 return measurements_.at(frame_node);
189}

196 std::to_string(tracklet_id) + " at frame " +

198 }
199 return getMeasurement(this->map_ptr_->template getFrame(frame_id));

248template <typename MEASUREMENT>
249LandmarkNodePtrSet<MEASUREMENT>
250ObjectNode<MEASUREMENT>::getLandmarksSeenAtFrame(FrameId frame_id) const {
251 LandmarkNodePtrSet<MEASUREMENT> seen_lmks;
252
253 for (const auto& lmk : dynamic_landmarks) {
254 // all frames this lmk was seen in
255 const FrameNodePtrSet<MEASUREMENT>& frames = lmk->getSeenFrames();
256 // lmk was observed at this frame
257 if (frames.find(frame_id) != frames.end()) {
258 seen_lmks.insert(lmk);
259 }
260 }
261 return seen_lmks;
262}
