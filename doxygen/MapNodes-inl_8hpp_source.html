138template <typename MEASUREMENT>
139void LandmarkNode<MEASUREMENT>::add(FrameNodePtr<MEASUREMENT> frame_node,
140 const MEASUREMENT& measurement) {
141 frames_seen_.insert(frame_node);
142
143 // add measurement to map
144 // first check that we dont already have a measurement at this frame
145 if (measurements_.exists(frame_node)) {
146 throw DynosamException("Unable to add new measurement to landmark node " +
147 std::to_string(this->getId()) + " at frame " +
148 std::to_string(frame_node->getId()) +
149 " as a measurement already exists at this frame!");
150 }
151
152 measurements_.insert2(frame_node, measurement);
153
154 CHECK_EQ(frames_seen_.size(), measurements_.size());
155}

167template <typename MEASUREMENT>
168const MEASUREMENT& LandmarkNode<MEASUREMENT>::getMeasurement(
169 FrameNodePtr<MEASUREMENT> frame_node) const {
170 CHECK_NOTNULL(frame_node);
171 if (!hasMeasurement(frame_node->frame_id)) {
172 throw DynosamException("Missing measurement in landmark node with id " +
173 std::to_string(tracklet_id) + " at frame " +
174 std::to_string(frame_node->frame_id));
175 }
176 return measurements_.at(frame_node);
177}

180const MEASUREMENT& LandmarkNode<MEASUREMENT>::getMeasurement(
181 FrameId frame_id) const {
182 if (!seenAtFrame(frame_id)) {
183 throw DynosamException("Missing measurement in landmark node with id " +
184 std::to_string(tracklet_id) + " at frame " +
185 std::to_string(frame_id));
186 }
187 return getMeasurement(this->map_ptr_->template getFrame(frame_id));

241template <typename MEASUREMENT>
242LandmarkNodePtrSet<MEASUREMENT>
243ObjectNode<MEASUREMENT>::getLandmarksSeenAtFrame(FrameId frame_id) const {
244 LandmarkNodePtrSet<MEASUREMENT> seen_lmks;
245
246 for (const auto& lmk : dynamic_landmarks) {
247 // all frames this lmk was seen in
248 const FrameNodePtrSet<MEASUREMENT>& frames = lmk->getSeenFrames();
249 // lmk was observed at this frame
250 if (frames.find(frame_id) != frames.end()) {
251 seen_lmks.insert(lmk);
252 }
253 }
254 return seen_lmks;
255}
