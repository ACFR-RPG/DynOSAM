215bool Feature::stereoPoint(gtsam::StereoPoint2& stereo) const {
216 if (!hasRightKeypoint()) {
217 return false;
218 }
219
220 const Keypoint& L = this->keypoint();
221 const Keypoint& R = this->rightKeypoint();
222 stereo = gtsam::StereoPoint2(L(0), R(0), L(1));
223 return true;
224}

239void FeatureContainer::add(Feature::Ptr feature) {
240 CHECK(!exists(feature->trackletId()))
241 << "Feailure in FeatureContainer::add - Tracklet Id "
242 << feature->trackletId() << " already exists";
243 feature_map_[feature->trackletId()] = feature;
244
245 // TODO: what if object id is not valid!
246 if (!object_feature_map_.exists(feature->objectId())) {
247 object_feature_map_.insert2(feature->objectId(),
248 std::unordered_set<TrackletId>{});
249 }
250 object_feature_map_[feature->objectId()].insert(feature->trackletId());
251}

254// this will mess up any iterator that currently has a reference to any

256void FeatureContainer::remove(TrackletId tracklet_id) {
257 if (!exists(tracklet_id)) {
258 throw std::runtime_error("Cannot remove feature with tracklet id " +
259 std::to_string(tracklet_id) +
260 " as feature does not exist!");
261 }
262
263 auto feature = this->getByTrackletId(tracklet_id);
264 const auto object_id = feature->objectId();
265 // check object id exists in map
266 CHECK(object_feature_map_.exists(object_id));
267 // remove tracklet id from tracklet set and if the set is now empty
268 // remove object id entirely
269 auto& tracklets_per_object = object_feature_map_.at(object_id);
270 tracklets_per_object.erase(tracklet_id);
271
272 if (tracklets_per_object.empty()) {
273 object_feature_map_.erase(object_id);
274 }
275
276 // remove from main feature map
277 feature_map_.erase(tracklet_id);
278}

280void FeatureContainer::removeByObjectId(ObjectId object_id) {
281 // collect all tracklets
282 auto itr = FilterIterator(*this, [object_id](const Feature::Ptr& f) -> bool {
283 return f->objectId() == object_id;
284 });
285
286 TrackletIds tracklets_to_remove;
287 for (const auto& feature : itr) {
288 CHECK_EQ(feature->objectId(), object_id);
289 tracklets_to_remove.push_back(feature->trackletId());
290 }
291
292 for (const auto tracklet_id : tracklets_to_remove) {
293 this->remove(tracklet_id);
294 }
295
296 // remove from object_feature_map_
297 object_feature_map_.erase(object_id);
298}

389std::vector<cv::Point2f> FeatureContainer::toOpenCV(TrackletIds* tracklet_ids,
390 bool only_inliers) const {
391 if (tracklet_ids) tracklet_ids->clear();
392
393 std::vector<cv::Point2f> keypoints_cv;
394 for (const auto& feature : *this) {
395 const Keypoint& kp = feature->keypoint();
396
397 float x = static_cast<float>(kp(0));
398 float y = static_cast<float>(kp(1));
399
400 if (only_inliers && feature->usable()) {
401 keypoints_cv.push_back(cv::Point2f(x, y));
402 if (tracklet_ids) tracklet_ids->push_back(feature->trackletId());
403 } else if (!only_inliers) {
404 keypoints_cv.push_back(cv::Point2f(x, y));
405 if (tracklet_ids) tracklet_ids->push_back(feature->trackletId());
406 }
407 }
408 return keypoints_cv;
409}
