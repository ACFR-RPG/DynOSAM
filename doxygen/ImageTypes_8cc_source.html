70cv::Mat RGBMonoToRGB(const cv::Mat& mat) {
71 const auto channels = mat.channels();
72 if (channels == 3) {
73 return mat;
74 } else if (channels == 4) {
75 cv::Mat rgb;
76 cv::cvtColor(mat, rgb, cv::COLOR_RGBA2RGB);
77 return rgb;
78 } else if (channels == 1) {
79 // grey scale but we want rgb so that we can draw colours on it or whatever
80 cv::Mat rgb;
81 cv::cvtColor(mat, rgb, cv::COLOR_GRAY2RGB);
82 return rgb;
83 } else {
84 return mat;
85 }
86}

88cv::Mat RGBMonoToMono(const cv::Mat& mat) {
89 const auto channels = mat.channels();
90 if (channels == 1) {
91 return mat;
92 }
93 if (channels == 3) {
94 cv::Mat mono;
95 cv::cvtColor(mat, mono, cv::COLOR_RGB2GRAY);
96 return mono;
97 } else if (channels == 4) {
98 cv::Mat mono;
99 cv::cvtColor(mat, mono, cv::COLOR_RGBA2GRAY);
100 return mono;
101 } else {
102 return mat;
103 }
104}

124// TODO: should this use getDisparityVis?
125// depth is metric so we should actually just scale it!
126cv::Mat ImageType::Depth::toRGB(const ImageWrapper<Depth>& image) {
127 cv::Mat metric_depth_map = image.clone();
128 // convert to floats to allow patch nan's to work
129 metric_depth_map.convertTo(metric_depth_map, CV_32F);
130
131 static constexpr float max_value = 25.0;
132
133 cv::patchNaNs(metric_depth_map, max_value);
134 for (int i = 0; i < metric_depth_map.rows; ++i) {
135 for (int j = 0; j < metric_depth_map.cols; ++j) {
136 float& pixel = metric_depth_map.at<float>(i, j);
137 if (std::isinf(pixel)) {
138 pixel = 0.0; // Replace Inf with small value so image is not saturated
139 // to a large number
140 }
141 }
142 }
143
144 // // truncate to max value (e.g. there is some outlier in the depth value
145 // that is a super larger number) cv::threshold(metric_depth_map,
146 // metric_depth_map, max_value, max_value, cv::THRESH_TRUNC);
147
148 // //fix a pixel as the maximum depth to avoid flickering!!
149 // metric_depth_map.at<float>(1, 1) = max_value;
150
151 // Normalize the depth map to the range [0, 1]
152 cv::Mat normalised_depth_map;
153 cv::normalize(metric_depth_map, normalised_depth_map, 0.0, 1.0,
154 cv::NORM_MINMAX);
155 // // Convert the normalized depth map to an 8-bit image [0, 255]
156 cv::Mat depth_map_8u;
157 normalised_depth_map.convertTo(depth_map_8u, CV_8U, 255.0, 20.0);
158
159 return depth_map_8u;
160}
