50void StereoMatcher::denseStereoReconstruction(const cv::Mat& left_img,
51 const cv::Mat& right_img,
52 cv::Mat& depth_image) const {
53 CHECK_EQ(right_img.cols, left_img.cols);
54 CHECK_EQ(right_img.rows, left_img.rows);
55 CHECK_EQ(right_img.type(), left_img.type());
56
57 cv::Mat left_rectified, right_rectified;
58 stereo_camera_->undistortRectifyImages(left_rectified, right_rectified,
59 left_img, right_img);
60
61 cv::Mat disparity_img;
62 disparityRectifiedReconstruction(left_rectified, right_rectified,
63 disparity_img);
64
65 constexpr auto depth_type = ImageType::Depth::OpenCVType;
66 depth_image = cv::Mat::zeros(disparity_img.size(), depth_type);
67
68 // Check
69 // https://github.com/opencv/opencv/blob/master/samples/cpp/stereo_match.cpp
70 // only if output is CV_16S
71 cv::Mat disparity_float;
72 disparity_img.convertTo(disparity_float, CV_32F, 1.0f / 16.0f);
73 // disp_img.convertTo(floatDisp, CV_32F, 1.0f / 16.0f);
74 disparity_img = disparity_float;
75
76 const auto baseline = stereo_camera_->getBaseline();
77 const auto fx = stereo_camera_->getStereoCameraCalibration().fx();
78
79 // Get depth from disparity
80 for (int i = 0u; i < disparity_img.rows; i++) {
81 // Loop over rows
82 const float* disp_ptr = disparity_img.ptr<float>(i);
83 double* depth_ptr = depth_image.ptr<double>(i);
84
85 for (int j = 0u; j < disparity_img.cols; j++) {
86 // Loop over cols
87 const double depth = (baseline * fx) / static_cast<double>(disp_ptr[j]);
88 *(depth_ptr + j) = depth;
89 }
90 }
91}

93void StereoMatcher::disparityRectifiedReconstruction(
94 const cv::Mat& rectified_left_img, const cv::Mat& rectified_right_img,
95 cv::Mat& disparity_image, cv::Mat* disparity_viz) const {
96 CHECK_EQ(rectified_right_img.cols, rectified_left_img.cols);
97 CHECK_EQ(rectified_right_img.rows, rectified_left_img.rows);
98 CHECK_EQ(rectified_right_img.type(), rectified_left_img.type());
99
100 // input expects 8bit single channel
101 cv::Mat rectified_left_mono = ImageType::RGBMono::toMono(rectified_left_img);
102 cv::Mat rectified_right_mono =
103 ImageType::RGBMono::toMono(rectified_right_img);
104
105 CHECK_EQ(rectified_left_mono.channels(), 1);
106 CHECK_EQ(rectified_right_mono.channels(), 1);
107
108 // initially computed as 16SC1
109 cv_stereo_matcher_->compute(rectified_left_mono, rectified_right_mono,
110 disparity_image);
111 CHECK_EQ(disparity_image.type(), CV_16S);
112
113 if (dense_stereo_params_.median_blur_disparity_) {
114 cv::medianBlur(disparity_image, disparity_image, 5);
115 }
116
117 if (disparity_viz) {
118 utils::getDisparityVis(disparity_image, *disparity_viz);
119 }
120}

122cv::Ptr<cv::StereoMatcher> StereoMatcher::constructStereoMatcher(
123 StereoCamera::ConstPtr stereo_camera,
124 const StereoMatchingParams& /*stereo_param*/,
125 const DenseStereoParams& dense_stereo_params) {
126 // Setup stereo matcher
127 cv::Ptr<cv::StereoMatcher> cv_stereo_matcher;
128 if (dense_stereo_params.use_sgbm_) {
129 int mode;
130 if (dense_stereo_params.use_mode_HH_) {
131 LOG(INFO) << "Using mode = cv::StereoSGBM::MODE_HH for dense stereo "
132 "reconstruction";
133 // mode = cv::StereoSGBM::MODE_HH;
134 mode = cv::StereoSGBM::MODE_SGBM_3WAY;
135 } else {
136 LOG(INFO) << "Using mode = cv::StereoSGBM::MODE_SGBM for dense stereo "
137 "reconstruction";
138 mode = cv::StereoSGBM::MODE_SGBM;
139 }
140 cv_stereo_matcher = cv::StereoSGBM::create(
141 dense_stereo_params.min_disparity_,
142 dense_stereo_params.num_disparities_,
143 dense_stereo_params.sad_window_size_, dense_stereo_params.p1_,
144 dense_stereo_params.p2_, dense_stereo_params.disp_12_max_diff_,
145 dense_stereo_params.pre_filter_cap_,
146 dense_stereo_params.uniqueness_ratio_,
147 dense_stereo_params.speckle_window_size_,
148 dense_stereo_params.speckle_range_, mode);
149 } else {
150 LOG(INFO) << "Using StereoBM for dense stereo reconstruction";
151 cv::Ptr<cv::StereoBM> sbm =
152 cv::StereoBM::create(dense_stereo_params.num_disparities_,
153 dense_stereo_params.sad_window_size_);
154
155 sbm->setPreFilterType(dense_stereo_params.pre_filter_type_);
156 sbm->setPreFilterSize(dense_stereo_params.pre_filter_size_);
157 sbm->setPreFilterCap(dense_stereo_params.pre_filter_cap_);
158 sbm->setMinDisparity(dense_stereo_params.min_disparity_);
159 sbm->setTextureThreshold(dense_stereo_params.texture_threshold_);
160 sbm->setUniquenessRatio(dense_stereo_params.uniqueness_ratio_);
161 sbm->setSpeckleRange(dense_stereo_params.speckle_range_);
162 sbm->setSpeckleWindowSize(dense_stereo_params.speckle_window_size_);
163 const cv::Rect& roi1 = stereo_camera->getROI1();
164 const cv::Rect& roi2 = stereo_camera->getROI2();
165 if (!roi1.empty() && !roi2.empty()) {
166 sbm->setROI1(roi1);
167 sbm->setROI2(roi2);
168 } else {
169 LOG(WARNING) << "ROIs are empty.";
170 }
171
172 cv_stereo_matcher = sbm;
173 }
174
175 CHECK_NOTNULL(cv_stereo_matcher);
176 return cv_stereo_matcher;
177}
