49void StereoMatcher::denseStereoReconstruction(
50 const cv::Mat& left_img,
51 const cv::Mat& right_img,
52 cv::Mat& depth_image) const
53{
54
55 CHECK_EQ(right_img.cols, left_img.cols);
56 CHECK_EQ(right_img.rows, left_img.rows);
57 CHECK_EQ(right_img.type(), left_img.type());
58
59 cv::Mat left_rectified, right_rectified;
60 stereo_camera_->undistortRectifyImages(left_rectified, right_rectified, left_img, right_img);
61
62 cv::Mat disparity_img;
63 disparityRectifiedReconstruction(left_rectified, right_rectified, disparity_img);
64
65 constexpr auto depth_type = ImageType::Depth::OpenCVType;
66 depth_image = cv::Mat::zeros(disparity_img.size(), depth_type);
67
68 // Check
69 // https://github.com/opencv/opencv/blob/master/samples/cpp/stereo_match.cpp
70 // only if output is CV_16S
71 cv::Mat disparity_float;
72 disparity_img.convertTo(disparity_float, CV_32F, 1.0f / 16.0f);
73 // disp_img.convertTo(floatDisp, CV_32F, 1.0f / 16.0f);
74 disparity_img = disparity_float;
75
76 const auto baseline = stereo_camera_->getBaseline();
77 const auto fx = stereo_camera_->getStereoCameraCalibration().fx();
78
79 // Get depth from disparity
80 for (int i = 0u; i < disparity_img.rows; i++) {
81 // Loop over rows
82 const float* disp_ptr = disparity_img.ptr<float>(i);
83 double* depth_ptr = depth_image.ptr<double>(i);
84
85 for (int j = 0u; j < disparity_img.cols; j++) {
86 // Loop over cols
87 const double depth = (baseline * fx) / static_cast<double>(disp_ptr[j]);
88 *(depth_ptr + j) = depth;
89 }
90 }
91
92}

94void StereoMatcher::disparityRectifiedReconstruction(const cv::Mat& rectified_left_img,
95 const cv::Mat& rectified_right_img,
96 cv::Mat& disparity_image,
97 cv::Mat* disparity_viz) const
98{
99 CHECK_EQ(rectified_right_img.cols, rectified_left_img.cols);
100 CHECK_EQ(rectified_right_img.rows, rectified_left_img.rows);
101 CHECK_EQ(rectified_right_img.type(), rectified_left_img.type());
102
103 //input expects 8bit single channel
104 cv::Mat rectified_left_mono = ImageType::RGBMono::toMono(rectified_left_img);
105 cv::Mat rectified_right_mono = ImageType::RGBMono::toMono(rectified_right_img);
106
107 CHECK_EQ(rectified_left_mono.channels(), 1);
108 CHECK_EQ(rectified_right_mono.channels(), 1);
109
110 // initially computed as 16SC1
111 cv_stereo_matcher_->compute(
112 rectified_left_mono, rectified_right_mono, disparity_image);
113 CHECK_EQ(disparity_image.type(), CV_16S);
114
115 if (dense_stereo_params_.median_blur_disparity_) {
116 cv::medianBlur(disparity_image, disparity_image, 5);
117 }
118
119 if(disparity_viz) {
120 utils::getDisparityVis(disparity_image, *disparity_viz);
121 }
122
123}

125cv::Ptr<cv::StereoMatcher> StereoMatcher::constructStereoMatcher(
126 StereoCamera::ConstPtr stereo_camera,
127 const StereoMatchingParams& /*stereo_param*/,
128 const DenseStereoParams& dense_stereo_params)
129{
130 // Setup stereo matcher
131 cv::Ptr<cv::StereoMatcher> cv_stereo_matcher;
132 if (dense_stereo_params.use_sgbm_) {
133 int mode;
134 if (dense_stereo_params.use_mode_HH_) {
135 LOG(INFO) << "Using mode = cv::StereoSGBM::MODE_HH for dense stereo reconstruction";
136 mode = cv::StereoSGBM::MODE_HH;
137 } else {
138 LOG(INFO) << "Using mode = cv::StereoSGBM::MODE_SGBM for dense stereo reconstruction";
139 mode = cv::StereoSGBM::MODE_SGBM;
140 }
141 cv_stereo_matcher =
142 cv::StereoSGBM::create(dense_stereo_params.min_disparity_,
143 dense_stereo_params.num_disparities_,
144 dense_stereo_params.sad_window_size_,
145 dense_stereo_params.p1_,
146 dense_stereo_params.p2_,
147 dense_stereo_params.disp_12_max_diff_,
148 dense_stereo_params.pre_filter_cap_,
149 dense_stereo_params.uniqueness_ratio_,
150 dense_stereo_params.speckle_window_size_,
151 dense_stereo_params.speckle_range_,
152 mode);
153 } else {
154 LOG(INFO) << "Using StereoBM for dense stereo reconstruction";
155 cv::Ptr<cv::StereoBM> sbm =
156 cv::StereoBM::create(dense_stereo_params.num_disparities_,
157 dense_stereo_params.sad_window_size_);
158
159 sbm->setPreFilterType(dense_stereo_params.pre_filter_type_);
160 sbm->setPreFilterSize(dense_stereo_params.pre_filter_size_);
161 sbm->setPreFilterCap(dense_stereo_params.pre_filter_cap_);
162 sbm->setMinDisparity(dense_stereo_params.min_disparity_);
163 sbm->setTextureThreshold(dense_stereo_params.texture_threshold_);
164 sbm->setUniquenessRatio(dense_stereo_params.uniqueness_ratio_);
165 sbm->setSpeckleRange(dense_stereo_params.speckle_range_);
166 sbm->setSpeckleWindowSize(dense_stereo_params.speckle_window_size_);
167 const cv::Rect& roi1 = stereo_camera->getROI1();
168 const cv::Rect& roi2 = stereo_camera->getROI2();
169 if (!roi1.empty() && !roi2.empty()) {
170 sbm->setROI1(roi1);
171 sbm->setROI2(roi2);
172 } else {
173 LOG(WARNING) << "ROIs are empty.";
174 }
175
176 cv_stereo_matcher = sbm;
177 }
178
179 CHECK_NOTNULL(cv_stereo_matcher);
180 return cv_stereo_matcher;
181}
