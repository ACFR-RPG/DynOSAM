77inline void ThreadsafeImuBuffer::addMeasurement(
78 const Timestamp& timestamp_nanoseconds,
79 const ImuAccGyr& imu_measurement) {
80 // Enforce strict time-wise ordering.
81 ImuMeasurement last_value;
82 if (buffer_.getNewestValue(&last_value)) {
83 CHECK_GT(timestamp_nanoseconds, last_value.timestamp_)
84 << "Timestamps not strictly increasing.";
85 }
86 buffer_.addValue(
87 timestamp_nanoseconds,
88 ImuMeasurement(timestamp_nanoseconds, imu_measurement));
89
90 // Notify possibly waiting consumers.
91 cv_new_measurement_.notify_all();
92}
93
94inline void ThreadsafeImuBuffer::addMeasurements(
95 const Timestamps& timestamps_nanoseconds,
96 const ImuAccGyrs& imu_measurements) {
97 CHECK_EQ(timestamps_nanoseconds.cols(), imu_measurements.cols());
98 size_t num_samples = timestamps_nanoseconds.cols();
99 CHECK_GT(num_samples, 0u);
100
101 for (size_t idx = 0u; idx < num_samples; ++idx) {
102 addMeasurement(timestamps_nanoseconds(idx), imu_measurements.col(idx));
103 }
104}
