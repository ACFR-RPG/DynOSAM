314 template <typename Ref, typename Curr>
315 void constructTrackletInliers(
316 TrackletIds& inliers, TrackletIds& outliers,
317 const GenericCorrespondences<Ref, Curr>& correspondences,
318 const std::vector<int>& ransac_inliers, const TrackletIds tracklets) {
319 CHECK_EQ(correspondences.size(), tracklets.size());
320 CHECK(ransac_inliers.size() <= correspondences.size());
321 for (int inlier_idx : ransac_inliers) {
322 const auto& corres = correspondences.at(inlier_idx);
323 inliers.push_back(corres.tracklet_id_);
324 }
325
326 determineOutlierIds(inliers, tracklets, outliers);
327 CHECK_EQ((inliers.size() + outliers.size()), tracklets.size());
328 CHECK_EQ(inliers.size(), ransac_inliers.size());
329 }
330
331 protected:
332 const Params params_;
333 const CameraParams camera_params_;
334};
