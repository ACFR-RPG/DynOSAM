317 template <typename Ref, typename Curr>
318 void constructTrackletInliers(
319 TrackletIds& inliers, TrackletIds& outliers,
320 const GenericCorrespondences<Ref, Curr>& correspondences,
321 const std::vector<int>& ransac_inliers, const TrackletIds tracklets) {
322 CHECK_EQ(correspondences.size(), tracklets.size());
323 CHECK(ransac_inliers.size() <= correspondences.size());
324 for (int inlier_idx : ransac_inliers) {
325 const auto& corres = correspondences.at(inlier_idx);
326 inliers.push_back(corres.tracklet_id_);
327 }
328
329 determineOutlierIds(inliers, tracklets, outliers);
330 CHECK_EQ((inliers.size() + outliers.size()), tracklets.size());
331 CHECK_EQ(inliers.size(), ransac_inliers.size());
332 }
333
334 protected:
335 const Params params_;
336 const CameraParams camera_params_;
