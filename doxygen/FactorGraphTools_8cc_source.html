87SparsityStats computeCholeskySparsityStats(
88 gtsam::GaussianFactorGraph::shared_ptr gaussian_fg,
89 const std::optional<gtsam::Ordering>& ordering) {
90 gtsam::Ordering order;
91 if (ordering) {
92 order = *ordering;
93 } else {
94 order = gtsam::Ordering::Colamd(*gaussian_fg);
95 }
96
97 auto [conditional, joint_factor] =
98 gtsam::EliminateCholesky(*gaussian_fg, order);
99
100 const gtsam::VerticalBlockMatrix::constBlock R = conditional->R();
101 return SparsityStats(R);
102}

243cv::Mat drawBlockJacobians(gtsam::GaussianFactorGraph::shared_ptr gaussian_fg,
244 const gtsam::Ordering& ordering,
245 const DrawBlockJacobiansOptions& options) {
246 const gtsam::JacobianFactor jacobian_factor(*gaussian_fg, ordering);
247 const gtsam::Matrix J = jacobian_factor.jacobian().first;
248
249 // each cv::mat will be the drawing of a column block (associated with one
250 // variable) in the jacobian Each block will have dimensions determiend by dim
251 // of the variable (determiend by the key) and number of rows of the full
252 // jacobian matrix, J
253 std::vector<std::pair<gtsam::Key, cv::Mat>> column_blocks;
254
255 for (gtsam::Key key : ordering) {
256 // this will have rows = J.rows(), cols = dimension of the variable
257 const gtsam::VerticalBlockMatrix::constBlock Ablock =
258 jacobian_factor.getA(jacobian_factor.find(key));
259 const int var_dimensions = Ablock.cols();
260 CHECK_EQ(Ablock.rows(), J.rows());
261
262 // drawn image of each vertical block
263 cv::Mat b_img(cv::Size(var_dimensions, Ablock.rows()), CV_8UC3,
264 cv::viz::Color::white());
265 for (int i = 0; i < Ablock.rows(); ++i) {
266 for (int j = 0; j < var_dimensions; ++j) {
267 // only draw if non zero
268 if (std::fabs(Ablock(i, j)) > 1e-15) {
269 const auto colour = options.colour_selector(key);
270 b_img.at<cv::Vec3b>(i, j) =
271 cv::Vec3b(colour[0], colour[1], colour[2]);
272 }
273 }
274 }
275
276 column_blocks.push_back(std::make_pair(key, b_img));
277 }
278
279 CHECK(!options.desired_size.empty()); // cannot be empty
280
281 const cv::Size& desired_size = options.desired_size;
282
283 const int current_cols = J.cols();
284 const int desired_cols = desired_size.width;
285 const int desired_rows = desired_size.height;
286
287 // define a ratio which which to scale each block (viz) individually
288 // since we're going to add things like spacing we want to scale
289 // up each block so its visible but dont want to scale up the line spacing
290 // etc. the ratio ensures we get close to the final desired size of the TOTAL
291 // image
292 double ratio;
293 // want positive ratio
294 if (current_cols < desired_cols) {
295 ratio = (double)desired_cols / (double)current_cols;
296 } else {
297 ratio = (double)current_cols / (double)desired_cols;
298 }
299
300 auto scale_and_draw_label = [&options, &ratio, &desired_rows](
301 cv::Mat& current_block, gtsam::Key key) {
302 // we all each jacobiab horizintally so only need to scale the columns
303 const int scaled_cols = (ratio * (int)current_block.cols) + 10;
304
305 // scale current block to the desired size with INTER_NEAREST - this keeps
306 // the pixels looking clear and visible
307 cv::resize(current_block, current_block,
308 cv::Size(scaled_cols, desired_rows), 0, 0, cv::INTER_NEAREST);
309
310 if (options.draw_label) {
311 // draw text info
312 cv::Mat text_box(cv::Size(current_block.cols, options.text_box_height),
313 CV_8UC3, cv::viz::Color::white());
314
315 constexpr static double kFontScale = 0.5;
316 constexpr static int kFontFace = cv::FONT_HERSHEY_DUPLEX;
317 constexpr static int kThickness = 1;
318 // draw text mid way in box
319 cv::putText(text_box, options.label_formatter(key),
320 cv::Point(2, text_box.rows / 2), kFontFace, kFontScale,
321 cv::viz::Color::black(), kThickness);
322 // add text box ontop of jacobian block
323 cv::vconcat(text_box, current_block, current_block);
324 }
325 }; // end scale_and_drawinternal::_label
326
327 if (column_blocks.size() == 1) {
328 gtsam::Key key = column_blocks.at(0).first;
329 cv::Mat current_block = column_blocks.at(0).second;
330 scale_and_draw_label(current_block, key);
331 return current_block;
332 }
333
334 // the final drawn image
335 cv::Mat concat_column_blocks;
336
337 for (size_t i = 1; i < column_blocks.size(); i++) {
338 gtsam::Key key = column_blocks.at(i).first;
339 cv::Mat current_block = column_blocks.at(i).second;
340 scale_and_draw_label(current_block, key);
341
342 if (i == 1) {
343 cv::Mat first_block = column_blocks.at(0).second;
344 gtsam::Key key_0 = column_blocks.at(0).first;
345 scale_and_draw_label(first_block, key_0);
346 const cv::Mat vert_line(cv::Size(2, first_block.rows), CV_8UC3,
347 cv::viz::Color::black());
348 // concat first block with vertical loine
349 cv::hconcat(first_block, vert_line, first_block);
350 // concat current jacobian block with other blocks
351 cv::hconcat(first_block, current_block, concat_column_blocks);
352 } else {
353 // concat current jacobian block with other blocks
354 cv::hconcat(concat_column_blocks, current_block, concat_column_blocks);
355 }
356
357 // if not last, add vertical line
358 if (i < column_blocks.size() - 1) {
359 // concat blocks with vertial line
360 const cv::Mat vert_line(cv::Size(2, concat_column_blocks.rows), CV_8UC3,
361 cv::viz::Color::black());
362 cv::hconcat(concat_column_blocks, vert_line, concat_column_blocks);
363 }
364 }
365 return concat_column_blocks;
366}

435void NonlinearFactorGraphManager::writeDynosamGraphFile(
436 const std::string& filename) const {
437 std::ofstream of(filename.c_str());
438
439 LOG(INFO) << "Writing dynosam graph file: " << filename;
440
441 using namespace factor_graph_tools;
442
443 // specalisations for dynosam factor graph
444 // gtsam does not have a write function for the base factor class so we dont
445 // actually know the types contained within the NonLinearFactorGraph so we
446 // have to test them
447 auto write_factor = [](std::ofstream& os,
448 const gtsam::NonlinearFactor::shared_ptr& nl_factor) {
449 // careful of mixing boost and std shared ptrs/casting here
450 auto prior_factor =
451 boost::dynamic_pointer_cast<gtsam::PriorFactor<gtsam::Pose3>>(
452 nl_factor);
453 if (prior_factor)
454 seralizeFactorToGraphFileFormat<gtsam::PriorFactor<gtsam::Pose3>>(
455 os, *prior_factor, "SE3_PRIOR_FACTOR");
456
457 // TODO: should give a different name for odom/smoothing factors?
458 // does not affect the graph structure but provides metadata?
459 auto between_factor =
460 boost::dynamic_pointer_cast<gtsam::BetweenFactor<gtsam::Pose3>>(
461 nl_factor);
462 if (between_factor)
463 seralizeFactorToGraphFileFormat<gtsam::BetweenFactor<gtsam::Pose3>>(
464 os, *between_factor, "SE3_BETWEEN_FACTOR");
465
466 auto motion_factor =
467 boost::dynamic_pointer_cast<LandmarkMotionTernaryFactor>(nl_factor);
468 if (motion_factor)
469 seralizeFactorToGraphFileFormat<LandmarkMotionTernaryFactor>(
470 os, *motion_factor, "SE3_MOTION_FACTOR");
471
472 auto pose_to_point_factor = boost::dynamic_pointer_cast<
473 gtsam::PoseToPointFactor<gtsam::Pose3, Landmark>>(nl_factor);
474 if (pose_to_point_factor)
475 seralizeFactorToGraphFileFormat<
476 gtsam::PoseToPointFactor<gtsam::Pose3, Landmark>>(
477 os, *pose_to_point_factor, "POSE_TO_POINT_FACTOR");
478 };
479
480 auto write_value = [](std::ofstream& os, const gtsam::Values& value,
481 gtsam::Key key) {
482 SymbolChar chr = DynoChrExtractor(key);
483
484 if (chr == InvalidDynoSymbol) {
485 throw std::runtime_error(
486 "Cannot write value to (dynosam) graph file as the associated key is "
487 "not a valid dynosam key!");
488 }
489
490 switch (chr) {
491 case kPoseSymbolChar:
492 seralizeValueToGraphFileFormat<gtsam::Pose3>(
493 os, value.at<gtsam::Pose3>(key), key, "SE3_POSE_VALUE");
494 break;
495 case kObjectMotionSymbolChar:
496 seralizeValueToGraphFileFormat<gtsam::Pose3>(
497 os, value.at<gtsam::Pose3>(key), key, "SE3_MOTION_VALUE");
498 break;
499 case kStaticLandmarkSymbolChar:
500 seralizeValueToGraphFileFormat<gtsam::Point3>(
501 os, value.at<gtsam::Point3>(key), key, "POINT3_STATIC_VALUE");
502 break;
503 case kDynamicLandmarkSymbolChar:
504 seralizeValueToGraphFileFormat<gtsam::Point3>(
505 os, value.at<gtsam::Point3>(key), key, "POINT3_DYNAMIC_VALUE");
506 break;
507 default:
508 LOG(FATAL) << "Should never reach here if DynoChrExtractor works!!";
509 }
510 };
511
512 gtsam::KeySet seen_keys;
513 for (const auto& factor : *this) {
514 write_factor(of, factor);
515 for (const auto key : *factor) {
516 // we have not seen this key before write it out to avoid duplication
517 if (seen_keys.find(key) == seen_keys.end()) {
518 write_value(of, values_, key);
519 seen_keys.insert(key);
520 }
521 }
522 }
523
524 std::flush(of);
525}

527bool ISAM2Helper::marginalizeKeys(gtsam::ISAM2Result* result,
528 gtsam::ISAM2* smoother,
529 const gtsam::KeyVector& marginalize_keys) {
530 CHECK_NOTNULL(smoother);
531
532 // checks all keys are in the smoother
533 gtsam::KeyVector keys_to_marginalize;
534 for (const auto& key : marginalize_keys) {
535 if (smoother->valueExists(key)) keys_to_marginalize.push_back(key);
536 }
537
538 if (keys_to_marginalize.empty()) {
539 return false;
540 }
541
542 // create ordering constraints; forcing ISAM2 to put marginalizable variables
543 // at the beginning
544 boost::optional<gtsam::FastMap<gtsam::Key, int>> constrainedKeys = {};
545 createOrderingConstraints(keys_to_marginalize, *smoother, constrainedKeys);
546
547 if (!constrainedKeys) {
548 return false;
549 }
550 // gather additional keys to re-eliminate
551 std::unordered_set<gtsam::Key> additionalKeys =
552 Base::gatherAdditionalKeysToReEliminate(*smoother, keys_to_marginalize);
553 gtsam::KeyList additionalMarkedKeys(additionalKeys.begin(),
554 additionalKeys.end());
555 // update ISAM
556 const auto isam_result = smoother->update(
557 gtsam::NonlinearFactorGraph(), gtsam::Values(), gtsam::FactorIndices(),
558 constrainedKeys, {}, additionalMarkedKeys);
559 // marginalize leaf keys!
560 gtsam::FastList<gtsam::Key> leafKeys(keys_to_marginalize.begin(),
561 keys_to_marginalize.end());
562 smoother->marginalizeLeaves(leafKeys);
563
564 if (result) *result = isam_result;
565 return true;
566}

568void ISAM2Helper::createOrderingConstraints(
569 const gtsam::KeyVector& marginalizableKeys, const ISAM2& smoother,
570 boost::optional<gtsam::FastMap<gtsam::Key, int>>& constrainedKeys) {
571 if (marginalizableKeys.size() > 0) {
572 // find the set difference to identify keys we dont want to marginalize
573 gtsam::KeySet all_keys_sorted = smoother.getLinearizationPoint().keySet();
574 gtsam::KeySet marginalizableKeys_sorted(marginalizableKeys.begin(),
575 marginalizableKeys.end());
576
577 gtsam::KeyVector other_variables;
578 std::set_difference(
579 all_keys_sorted.begin(), all_keys_sorted.end(),
580 marginalizableKeys_sorted.begin(), marginalizableKeys_sorted.end(),
581 std::inserter(other_variables, other_variables.begin()));
582
583 constrainedKeys = FastMap<Key, int>();
584 // Generate ordering constraints so that the marginalizable variables will
585 // be eliminated first Set all variables to Group1
586 for (gtsam::Key key : other_variables) {
587 constrainedKeys->operator[](key) = 1;
588 }
589 // Set marginalizable variables to Group0
590 for (gtsam::Key key : marginalizableKeys_sorted) {
591 constrainedKeys->operator[](key) = 0;
592 }
593 }
594}
