109std::pair<SparsityStats, cv::Mat> computeRFactor(gtsam::GaussianFactorGraph::shared_ptr gaussian_fg, const gtsam::Ordering ordering) {
110 //TODO: why do we need to specify the ordering twice?
111 //I think we should probaly construct the graph with native ordeing but then elimate with provided ordering?
112 gtsam::JacobianFactor jacobian_factor(*gaussian_fg);
113
114 gtsam::GaussianConditional::shared_ptr conditional;
115 gtsam::JacobianFactor::shared_ptr joint_factor;
116
119 std::tie(conditional, joint_factor) = jacobian_factor.eliminate(ordering);
120 CHECK(conditional);
121
122 const gtsam::VerticalBlockMatrix::constBlock R = conditional->R();
123
124 //drawn image
125 cv::Mat R_img(cv::Size(R.cols(), R.rows()), CV_8UC3, cv::viz::Color::white());
126 for (int i = 0; i < R.rows(); ++i) {
127 for (int j = 0; j < R.cols(); ++j) {
128 //only draw if non zero
129 if (std::fabs(R(i, j)) > 1e-15) {
130 R_img.at<cv::Vec3b>(i, j) = (cv::Vec3b)cv::viz::Color::black();
131 }
132 }
133 }
134
135 cv::imshow("R", R_img);
136 cv::waitKey(0);

141cv::Mat drawBlockJacobians(gtsam::GaussianFactorGraph::shared_ptr gaussian_fg, const gtsam::Ordering& ordering, const DrawBlockJacobiansOptions& options) {
142 const gtsam::JacobianFactor jacobian_factor(*gaussian_fg, ordering);
143 const gtsam::Matrix J = jacobian_factor.jacobian().first;
144
145 //each cv::mat will be the drawing of a column block (associated with one variable) in the jacobian
146 //Each block will have dimensions determiend by dim of the variable (determiend by the key)
147 //and number of rows of the full jacobian matrix, J
148 std::vector<std::pair<gtsam::Key, cv::Mat>> column_blocks;
149
150
151 for(gtsam::Key key : ordering) {
152 //this will have rows = J.rows(), cols = dimension of the variable
153 const gtsam::VerticalBlockMatrix::constBlock Ablock = jacobian_factor.getA(jacobian_factor.find(key));
154 const int var_dimensions = Ablock.cols();
155 CHECK_EQ(Ablock.rows(), J.rows());
156
157 //drawn image of each vertical block
158 cv::Mat b_img(cv::Size(var_dimensions, Ablock.rows()), CV_8UC3, cv::viz::Color::white());
159 for (int i = 0; i < Ablock.rows(); ++i) {
160 for (int j = 0; j < var_dimensions; ++j) {
161 //only draw if non zero
162 if (std::fabs(Ablock(i, j)) > 1e-15) {
163 const auto colour = options.colour_selector(key);
164 b_img.at<cv::Vec3b>(i, j) = cv::Vec3b(colour[0], colour[1], colour[2]);
165 }
166 }
167 }
168
169 column_blocks.push_back(std::make_pair(key, b_img));
170 }
171
172 CHECK(!options.desired_size.empty()); //cannot be empty
173
174 const cv::Size& desired_size = options.desired_size;
175
176 const int current_cols = J.cols();
177 const int desired_cols = desired_size.width;
178 const int desired_rows = desired_size.height;
179
180 //define a ratio which which to scale each block (viz) individually
181 //since we're going to add things like spacing we want to scale
182 //up each block so its visible but dont want to scale up the line spacing etc.
183 //the ratio ensures we get close to the final desired size of the TOTAL image
184 double ratio;
185 //want positive ratio
186 if(current_cols < desired_cols) {
187 ratio = (double)desired_cols/(double)current_cols;
188 }
189 else {
190 ratio = (double)current_cols/(double)desired_cols;
191 }
192
193 auto scale_and_draw_label = [&options, &ratio, &desired_rows](cv::Mat& current_block, gtsam::Key key) {
194 //we all each jacobiab horizintally so only need to scale the columns
195 const int scaled_cols = ratio * (int)current_block.cols;
196
197 //scale current block to the desired size with INTER_NEAREST - this keeps the pixels looking clear and visible
198 cv::resize(current_block, current_block, cv::Size(scaled_cols, desired_rows), 0, 0, cv::INTER_NEAREST);
199
200 if(options.draw_label) {
201 //draw text info
202 cv::Mat text_box(cv::Size(current_block.cols, options.text_box_height), CV_8UC3, cv::viz::Color::white());
203
204 constexpr static double kFontScale = 0.5;
205 constexpr static int kFontFace = cv::FONT_HERSHEY_DUPLEX;
206 constexpr static int kThickness = 1;
207 //draw text mid way in box
208 cv::putText(text_box, options.label_formatter(key), cv::Point(2, text_box.rows/2), kFontFace, kFontScale, cv::viz::Color::black(), kThickness);
209 //add text box ontop of jacobian block
210 cv::vconcat(text_box, current_block, current_block);
211 }
212 }; //end scale_and_drawinternal::_label
213
214
215 if(column_blocks.size() == 1) {
216 gtsam::Key key = column_blocks.at(0).first;
217 cv::Mat current_block = column_blocks.at(0).second;
218 scale_and_draw_label(current_block, key);
219 return current_block;
220 }
221
222 //the final drawn image
223 cv::Mat concat_column_blocks;
224
225 for(size_t i = 1; i < column_blocks.size(); i++) {
226 gtsam::Key key = column_blocks.at(i).first;
227 cv::Mat current_block = column_blocks.at(i).second;
228 scale_and_draw_label(current_block, key);
229
230
231 if(i == 1) {
232 cv::Mat first_block = column_blocks.at(0).second;
233 gtsam::Key key_0 = column_blocks.at(0).first;
234 scale_and_draw_label(first_block, key_0);
235 const cv::Mat vert_line(cv::Size(2, first_block.rows), CV_8UC3, cv::viz::Color::black());
236 //concat first block with vertical loine
237 cv::hconcat(first_block, vert_line, first_block);
238 //concat current jacobian block with other blocks
239 cv::hconcat(first_block, current_block, concat_column_blocks);
240 }
241 else {
242 //concat current jacobian block with other blocks
243 cv::hconcat(concat_column_blocks, current_block, concat_column_blocks);
244 }
245
246 //if not last, add vertical line
247 if(i < column_blocks.size() - 1) {
248 //concat blocks with vertial line
249 const cv::Mat vert_line(cv::Size(2, concat_column_blocks.rows), CV_8UC3, cv::viz::Color::black());
250 cv::hconcat(concat_column_blocks, vert_line, concat_column_blocks);
251 }
252 }
253 return concat_column_blocks;
254}

315void NonlinearFactorGraphManager::writeDynosamGraphFile(const std::string& filename) const {
316 std::ofstream of(filename.c_str());
317
318 LOG(INFO) << "Writing dynosam graph file: " << filename;
319
320 using namespace factor_graph_tools;
321
322 //specalisations for dynosam factor graph
323 //gtsam does not have a write function for the base factor class so we dont actually know
324 //the types contained within the NonLinearFactorGraph so we have to test them
325 auto write_factor =[](std::ofstream& os, const gtsam::NonlinearFactor::shared_ptr& nl_factor) {
326 //careful of mixing boost and std shared ptrs/casting here
327 auto prior_factor = boost::dynamic_pointer_cast<gtsam::PriorFactor<gtsam::Pose3>>(nl_factor);
328 if(prior_factor) seralizeFactorToGraphFileFormat<gtsam::PriorFactor<gtsam::Pose3>>(os, *prior_factor, "SE3_PRIOR_FACTOR");
329
330
331 //TODO: should give a different name for odom/smoothing factors?
332 //does not affect the graph structure but provides metadata?
333 auto between_factor = boost::dynamic_pointer_cast<gtsam::BetweenFactor<gtsam::Pose3>>(nl_factor);
334 if(between_factor) seralizeFactorToGraphFileFormat<gtsam::BetweenFactor<gtsam::Pose3>>(os, *between_factor, "SE3_BETWEEN_FACTOR");
335
336 auto motion_factor = boost::dynamic_pointer_cast<LandmarkMotionTernaryFactor>(nl_factor);
337 if(motion_factor) seralizeFactorToGraphFileFormat<LandmarkMotionTernaryFactor>(os, *motion_factor, "SE3_MOTION_FACTOR");
338
339 auto pose_to_point_factor = boost::dynamic_pointer_cast<gtsam::PoseToPointFactor<gtsam::Pose3, Landmark>>(nl_factor);
340 if(pose_to_point_factor) seralizeFactorToGraphFileFormat<gtsam::PoseToPointFactor<gtsam::Pose3, Landmark>>(os, *pose_to_point_factor, "POSE_TO_POINT_FACTOR");
341 };
342
343 auto write_value = [](std::ofstream& os, const gtsam::Values& value, gtsam::Key key) {
344 SymbolChar chr = DynoChrExtractor(key);
345
346 if(chr == InvalidDynoSymbol) {
347 throw std::runtime_error("Cannot write value to (dynosam) graph file as the associated key is not a valid dynosam key!");
348 }
349
350 switch(chr) {
351 case kPoseSymbolChar:
352 seralizeValueToGraphFileFormat<gtsam::Pose3>(os, value.at<gtsam::Pose3>(key),key, "SE3_POSE_VALUE");
353 break;
354 case kObjectMotionSymbolChar:
355 seralizeValueToGraphFileFormat<gtsam::Pose3>(os, value.at<gtsam::Pose3>(key), key, "SE3_MOTION_VALUE");
356 break;
357 case kStaticLandmarkSymbolChar:
358 seralizeValueToGraphFileFormat<gtsam::Point3>(os, value.at<gtsam::Point3>(key), key, "POINT3_STATIC_VALUE");
359 break;
360 case kDynamicLandmarkSymbolChar:
361 seralizeValueToGraphFileFormat<gtsam::Point3>(os, value.at<gtsam::Point3>(key), key, "POINT3_DYNAMIC_VALUE");
362 break;
363 default:
364 LOG(FATAL) << "Should never reach here if DynoChrExtractor works!!";
365 }
366 };
367
368 gtsam::KeySet seen_keys;
369 for(const auto& factor : *this) {
370 write_factor(of, factor);
371 for(const auto key : *factor) {
372 //we have not seen this key before write it out to avoid duplication
373 if(seen_keys.find(key) == seen_keys.end()) {
374 write_value(of, values_, key);
375 seen_keys.insert(key);
376 }
377 }
378 }
379
380 std::flush(of);
381}
