81// Static functions to query the stats collectors:
82size_t Statistics::GetHandle(std::string const& tag) {
83 std::lock_guard<std::mutex> lock(Instance().mutex_);
84 // Search for an existing tag.
85 map_t::iterator i = Instance().tag_map_.find(tag);
86 if (i == Instance().tag_map_.end()) {
87 // If it is not there, create a tag.
88 size_t handle = Instance().stats_collectors_.size();
89 Instance().tag_map_[tag] = handle;
90 Instance().stats_collectors_.push_back(StatisticsMapValue());
91 // Track the maximum tag length to help printing a table of values later.
92 Instance().max_tag_length_ =
93 std::max(Instance().max_tag_length_, tag.size());
94 return handle;
95 } else {
96 return i->second;
97 }
98}

100// Return true if a handle has been initialized for a specific tag.

104 std::lock_guard<std::mutex> lock(Instance().mutex_);
105 bool tag_found = Instance().tag_map_.count(tag);

109std::string Statistics::GetTag(size_t handle) {
110 std::lock_guard<std::mutex> lock(Instance().mutex_);
111 std::string tag;
112 // Perform a linear search for the tag.
113 for (typename map_t::value_type current_tag : Instance().tag_map_) {
114 if (current_tag.second == handle) {
115 return current_tag.first;
116 }
117 }
118 return tag;
119}

134std::vector<std::string> Statistics::getTagByModule(std::string const& query_module) {
135 std::vector<std::string> modules;
136 std::lock_guard<std::mutex> lock(Instance().mutex_);
137 for(const auto&[tag, _] : GetStatsCollectors()) {
138
139 std::optional<std::string> module = getModuleNameFromTag(tag);
140
141 //if query module is empty match with those tags in the global namespace
142 if(query_module.empty() && !module) {
143 modules.push_back(tag);
144 }
145 else if(module && *module == query_module) {
146 modules.push_back(tag);
147 }
148 }
149 return modules;
150}

279std::string Statistics::SecondsToTimeString(double seconds) {
280 double secs = fmod(seconds, 60);
281 int minutes = (seconds / 60);
282 int hours = (seconds / 3600);
283 minutes = minutes - (hours * 60);
284
285 char buffer[256];
286 snprintf(buffer,
287 sizeof(buffer),
288#ifdef SM_TIMING_SHOW_HOURS
289 "%02d:"
290#endif
291#ifdef SM_TIMING_SHOW_MINUTES
292 "%02d:"
293#endif
294 "%09.6f",
295#ifdef SM_TIMING_SHOW_HOURS
296 hours,
297#endif
298#ifdef SM_TIMING_SHOW_MINUTES
299 minutes,
300#endif
301 secs);
302 return buffer;
303}

305void Statistics::Print(std::ostream& out) { // NOLINT
306 const map_t& tag_map = Instance().tag_map_;
307
308 if (tag_map.empty()) {
309 return;
310 }
311
312 out << "Statistics\n";
313
314 out.width((std::streamsize)Instance().max_tag_length_);
315 out.setf(std::ios::left, std::ios::adjustfield);
316 out << "-----------";
317 out.width(7);
318 out.setf(std::ios::right, std::ios::adjustfield);
319 out << "#\t";
320 out << "Log Hz\t";
321 out << "{avg +- std }\t";
322 out << "[min,max]\n";
323
324 for (const typename map_t::value_type& t : tag_map) {
325 size_t i = t.second;
326 out.width((std::streamsize)Instance().max_tag_length_);
327 out.setf(std::ios::left, std::ios::adjustfield);
328 // Print Name of tag
329 out << t.first << "\t";
330
331 out.setf(std::ios::right, std::ios::adjustfield);
332 // Print #
333 out << std::setw(5) << GetNumSamples(i) << "\t";
334 if (GetNumSamples(i) > 0) {
335 out << std::showpoint << GetHz(i) << "\t";
336 double mean = GetMean(i);
337 double stddev = sqrt(GetVariance(i));
338 out << "{" << std::showpoint << mean;
339 out << " +- ";
340 out << std::showpoint << stddev << "}\t";
341
342 double min_value = GetMin(i);
343 double max_value = GetMax(i);
344
345 //out.width(5);
346 out << std::noshowpoint << "[" << min_value << "," << max_value << "]";
347 }
348 out << std::endl;
349 }

352void Statistics::WriteAllSamplesToCsvFile(const std::string& path) {
353 const map_t& tag_map = Instance().tag_map_;
354 if (tag_map.empty()) {
355 return;
356 }
357
358 static const CsvHeader header("label", "samples");
359 CsvWriter writer(header);
360
361
362 VLOG(1) << "Writing statistics to file: " << path;
363 for (const map_t::value_type& tag : tag_map) {
364 const size_t& index = tag.second;
365 if (GetNumSamples(index) > 0) {
366 const std::string& label = tag.first;
367
368 // // Add header to csv file. tag.first is the stats label.
369 writer << label;
370
371 // Each row has all samples.
372 std::stringstream ss;
373 const std::vector<double>& samples = GetAllSamples(index);
374 for (const auto& sample : samples) {
375 //make space separated so not to confuse the CsvWriter
376 //treat all samples as a single column
377 ss << ' ' << sample;
378 }
379 writer << ss.str();
380 }
381 }
382
383 writer.write(path);
384}

405void Statistics::WriteToYamlFile(const std::string& path) {
406 std::ofstream output_file(path);
407
408 if (!output_file) {
409 LOG(ERROR) << "Could not write statistics: Unable to open file: " << path;
410 return;
411 }
412
413 const map_t& tag_map = Instance().tag_map_;
414 if (tag_map.empty()) {
415 return;
416 }
417
418 VLOG(1) << "Writing statistics to file: " << path;
419 for (const map_t::value_type& tag : tag_map) {
420 const size_t index = tag.second;
421
422 if (GetNumSamples(index) > 0) {
423 std::string label = tag.first;
424
425 // We do not want colons or hashes in a label, as they might interfere
426 // with reading the yaml later.
427 std::replace(label.begin(), label.end(), ':', '_');
428 std::replace(label.begin(), label.end(), '#', '_');
429
430 output_file << label << ":\n";
431 output_file << " samples: " << GetNumSamples(index) << "\n";
432 output_file << " mean: " << GetMean(index) << "\n";
433 output_file << " stddev: " << sqrt(GetVariance(index)) << "\n";
434 output_file << " min: " << GetMin(index) << "\n";
435 output_file << " max: " << GetMax(index) << "\n";
436 output_file << " median: " << GetMedian(index) << "\n";
437 output_file << " q1: " << GetQ1(index) << "\n";
438 output_file << " q3: " << GetQ3(index) << "\n";
439 }
440 output_file << "\n";
441 }

444void Statistics::WritePerModuleSummariesToCsvFile(const std::string& folder_path) {
445 const map_t& tag_map = Instance().tag_map_;
446 if (tag_map.empty()) {
447 return;
448 }
449
450 static const CsvHeader header("label", "num samples", "log Hz", "mean", "stddev", "min", "max");
451 std::map<std::string, CsvWriter::Ptr> module_to_writer;
452
453 CsvWriter global_writer(header);
454
455 for (const map_t::value_type& tag : tag_map) {
456 std::optional<std::string> module_name = getModuleNameFromTag(tag.first);
457
458 // writer to use
459 CsvWriter* writer;
460 if(module_name) {
461 //tag is not in global namespace so use the specificied namespace
462 //if we dont have a writer for this module, make one
463 if(module_to_writer.find(*module_name) == module_to_writer.end()) {
464 module_to_writer[*module_name] = std::make_shared<CsvWriter>(header);
465 }
466
467 //set writer to use
468 writer = module_to_writer.at(*module_name).get();
469 }
470 else {
471 //no module name so use the global namespace
472 writer = &global_writer;
473 }
474 CHECK_NOTNULL(writer);
475 SummaryWriterHelper(*writer, tag);
476
477 }
478
479 //write out all summaries
480 for(auto& [module_name, writer] : module_to_writer) {
481 std::string full_path = fs::path(folder_path) / std::string("stats_" + module_name + ".csv");
482 LOG(INFO) << "Writing out csv stats module: " << full_path;
483 writer->write(full_path);
484 }
485
486 {
487 std::string full_path = fs::path(folder_path) / "stats_global.csv";
488 LOG(INFO) << "Writing out csv stats module: " << full_path;
489 global_writer.write(full_path);
490 }
491
492}
