87// Static functions to query the stats collectors:
88size_t Statistics::GetHandle(std::string const& tag) {
89 std::lock_guard<std::mutex> lock(Instance().mutex_);
90 // Search for an existing tag.
91 map_t::iterator i = Instance().tag_map_.find(tag);
92 if (i == Instance().tag_map_.end()) {
93 // If it is not there, create a tag.
94 size_t handle = Instance().stats_collectors_.size();
95 Instance().tag_map_[tag] = handle;
96 Instance().stats_collectors_.push_back(StatisticsMapValue());
97 // Track the maximum tag length to help printing a table of values later.
98 Instance().max_tag_length_ =
99 std::max(Instance().max_tag_length_, tag.size());
100 return handle;
101 } else {
102 return i->second;
103 }
104}

106// Return true if a handle has been initialized for a specific tag.

109bool Statistics::HasHandle(std::string const& tag) {
110 std::lock_guard<std::mutex> lock(Instance().mutex_);
111 bool tag_found = Instance().tag_map_.count(tag);
112 return tag_found;

115std::string Statistics::GetTag(size_t handle) {
116 std::lock_guard<std::mutex> lock(Instance().mutex_);
117 std::string tag;
118 // Perform a linear search for the tag.
119 for (typename map_t::value_type current_tag : Instance().tag_map_) {
120 if (current_tag.second == handle) {
121 return current_tag.first;
122 }
123 }
124 return tag;
125}

140std::vector<std::string> Statistics::getTagByModule(
141 std::string const& query_module) {
142 std::vector<std::string> modules;
143 std::lock_guard<std::mutex> lock(Instance().mutex_);
144 for (const auto& [tag, _] : GetStatsCollectors()) {
145 std::optional<std::string> module = getModuleNameFromTag(tag);
146
147 // if query module is empty match with those tags in the global namespace
148 if (query_module.empty() && !module) {
149 modules.push_back(tag);
150 } else if (module && *module == query_module) {
151 modules.push_back(tag);
152 }
153 }
154 return modules;
155}

283std::string Statistics::SecondsToTimeString(double seconds) {
284 double secs = fmod(seconds, 60);
285 int minutes = (seconds / 60);
286 int hours = (seconds / 3600);
287 minutes = minutes - (hours * 60);
288
289 char buffer[256];
290 snprintf(buffer, sizeof(buffer),
291#ifdef SM_TIMING_SHOW_HOURS
292 "%02d:"
293#endif
294#ifdef SM_TIMING_SHOW_MINUTES
295 "%02d:"
296#endif
297 "%09.6f",
298#ifdef SM_TIMING_SHOW_HOURS
299 hours,
300#endif
301#ifdef SM_TIMING_SHOW_MINUTES
302 minutes,
303#endif
304 secs);
305 return buffer;
306}

308void Statistics::Print(std::ostream& out) { // NOLINT
309 const map_t& tag_map = Instance().tag_map_;
310
311 if (tag_map.empty()) {
312 return;
313 }
314
315 out << "Statistics\n";
316
317 out.width((std::streamsize)Instance().max_tag_length_);
318 out.setf(std::ios::left, std::ios::adjustfield);
319 out << "-----------";
320 out.width(7);
321 out.setf(std::ios::right, std::ios::adjustfield);
322 out << "#\t";
323 out << "Log Hz\t";
324 out << "{avg +- std }\t";
325 out << "[min,max]\n";
326
327 for (const typename map_t::value_type& t : tag_map) {
328 size_t i = t.second;
329 out.width((std::streamsize)Instance().max_tag_length_);
330 out.setf(std::ios::left, std::ios::adjustfield);
331 // Print Name of tag
332 out << t.first << "\t";
333
334 out.setf(std::ios::right, std::ios::adjustfield);
335 // Print #
336 out << std::setw(5) << GetNumSamples(i) << "\t";
337 if (GetNumSamples(i) > 0) {
338 out << std::showpoint << GetHz(i) << "\t";
339 double mean = GetMean(i);
340 double stddev = sqrt(GetVariance(i));
341 out << "{" << std::showpoint << mean;
342 out << " +- ";
343 out << std::showpoint << stddev << "}\t";
344
345 double min_value = GetMin(i);
346 double max_value = GetMax(i);
347
348 // out.width(5);
349 out << std::noshowpoint << "[" << min_value << "," << max_value << "]";
350 }
351 out << std::endl;
352 }

355void Statistics::WriteAllSamplesToCsvFile(const std::string& path) {
356 const map_t& tag_map = Instance().tag_map_;
357 if (tag_map.empty()) {
358 return;
359 }
360
361 static const CsvHeader header("label", "samples");
362 CsvWriter writer(header);
363
364 VLOG(1) << "Writing statistics to file: " << path;
365 for (const map_t::value_type& tag : tag_map) {
366 const size_t& index = tag.second;
367 if (GetNumSamples(index) > 0) {
368 const std::string& label = tag.first;
369
370 // // Add header to csv file. tag.first is the stats label.
371 writer << label;
372
373 // Each row has all samples.
374 std::stringstream ss;
375 const std::vector<double>& samples = GetAllSamples(index);
376 for (const auto& sample : samples) {
377 // make space separated so not to confuse the CsvWriter
378 // treat all samples as a single column
379 ss << ' ' << sample;
380 }
381 writer << ss.str();
382 }
383 }
384
385 writer.write(path);
386}

405void Statistics::WriteToYamlFile(const std::string& path) {
406 std::ofstream output_file(path);
407
408 if (!output_file) {
409 LOG(ERROR) << "Could not write statistics: Unable to open file: " << path;
410 return;
411 }
412
413 const map_t& tag_map = Instance().tag_map_;
414 if (tag_map.empty()) {
415 return;
416 }
417
418 VLOG(1) << "Writing statistics to file: " << path;
419 for (const map_t::value_type& tag : tag_map) {
420 const size_t index = tag.second;
421
422 if (GetNumSamples(index) > 0) {
423 std::string label = tag.first;
424
425 // We do not want colons or hashes in a label, as they might interfere
426 // with reading the yaml later.
427 std::replace(label.begin(), label.end(), ':', '_');
428 std::replace(label.begin(), label.end(), '#', '_');
429
430 output_file << label << ":\n";
431 output_file << " samples: " << GetNumSamples(index) << "\n";
432 output_file << " mean: " << GetMean(index) << "\n";
433 output_file << " stddev: " << sqrt(GetVariance(index)) << "\n";
434 output_file << " min: " << GetMin(index) << "\n";
435 output_file << " max: " << GetMax(index) << "\n";
436 output_file << " median: " << GetMedian(index) << "\n";
437 output_file << " q1: " << GetQ1(index) << "\n";
438 output_file << " q3: " << GetQ3(index) << "\n";
439 }
440 output_file << "\n";
441 }

444void Statistics::WritePerModuleSummariesToCsvFile(
445 const std::string& folder_path) {
446 const map_t& tag_map = Instance().tag_map_;
447 if (tag_map.empty()) {
448 return;
449 }
450
451 static const CsvHeader header("label", "num samples", "log Hz", "mean",
452 "stddev", "min", "max");
453 std::map<std::string, CsvWriter::Ptr> module_to_writer;
454
455 CsvWriter global_writer(header);
456
457 for (const map_t::value_type& tag : tag_map) {
458 std::optional<std::string> module_name = getModuleNameFromTag(tag.first);
459
460 // writer to use
461 CsvWriter* writer;
462 if (module_name) {
463 // tag is not in global namespace so use the specificied namespace
464 // if we dont have a writer for this module, make one
465 if (module_to_writer.find(*module_name) == module_to_writer.end()) {
466 module_to_writer[*module_name] = std::make_shared<CsvWriter>(header);
467 }
468
469 // set writer to use
470 writer = module_to_writer.at(*module_name).get();
471 } else {
472 // no module name so use the global namespace
473 writer = &global_writer;
474 }
475 CHECK_NOTNULL(writer);
476 SummaryWriterHelper(*writer, tag);
477 }
478
479 // write out all summaries
480 for (auto& [module_name, writer] : module_to_writer) {
481 std::string full_path =
482 fs::path(folder_path) / std::string("stats_" + module_name + ".csv");
483 LOG(INFO) << "Writing out csv stats module: " << full_path;
484 writer->write(full_path);
485 }
486
487 {
488 std::string full_path = fs::path(folder_path) / "stats_global.csv";
489 LOG(INFO) << "Writing out csv stats module: " << full_path;
490 global_writer.write(full_path);
491 }
492}
