55std::vector<Track::Ptr> ByteTracker::update(
56 const std::vector<DetectionBase::Ptr> &input_detections, size_t frame_id) {
57 CHECK_GE(frame_id, frame_id_);
58 // if first call of update, check incoming frame is the same as the starting frame
59 // TODO: (jesse) this is super weird logic - there is a dependancy on ByteTrackerParams::params::starting_frame
60 // but nothing inside this class enforces consistnecy, yet we check that the frame calls incrementing!
61 // if(count_ == 0) CHECK_EQ(frame_id, frame_id_);
62 // // if not first call of update, (when frame id should be set t)
63 // if(count_ > 0) CHECK_EQ(frame_id, frame_id_ + 1u);
64
65 //update frame count
66 frame_id_ = frame_id;
67 count_++;
68
70
71 // Sort new tracks from detection by score
72 std::vector<DetectionBase::Ptr> detections;
73 std::vector<DetectionBase::Ptr> low_score_detections;
74 for (const auto &detection : input_detections) {
75 if (detection->score() >= track_thresh_)
76 detections.push_back(detection);
77 else
78 low_score_detections.push_back(detection);
79 }
80
81 // Sort existing tracks by confirmed status
82 std::vector<Track::Ptr> confirmed_tracks;
83 std::vector<Track::Ptr> unconfirmed_tracks;
84
85 for (const auto &track : tracked_tracks_) {
86 if (!track->is_confirmed())
87 unconfirmed_tracks.push_back(track);
88 else
89 confirmed_tracks.push_back(track);
90 }
91
92 std::vector<Track::Ptr> track_pool;
93 track_pool = joint_tracks(confirmed_tracks, lost_tracks_);
94
95 // Predict current pose by KF
96 for (auto &track : track_pool) track->predict();
97
100 auto [matched_tracks, unmatched_tracked_tracks, unmatched_detections] =
101 iou_association(track_pool, detections);
102
104 auto new_lost_tracks = low_score_association(
105 matched_tracks, low_score_detections, unmatched_tracked_tracks);
106
108 auto removed_tracks =
109 init_new_tracks(matched_tracks, unconfirmed_tracks, unmatched_detections);
110
112 for (auto &lost_track : lost_tracks_) {
113 if (frame_id_ - lost_track->get_frame_id() > max_time_lost_) {
114 removed_tracks.push_back(lost_track);
115 }
116 }
117
118 lost_tracks_ = sub_tracks(
119 joint_tracks(sub_tracks(lost_tracks_, matched_tracks), new_lost_tracks),
120 removed_tracks);
121
122 std::tie(tracked_tracks_, lost_tracks_) =
123 remove_duplicate_tracks(matched_tracks, lost_tracks_);
124
125 std::vector<Track::Ptr> output_tracks;
126 for (const auto &track : tracked_tracks_) {
127 if (track->is_confirmed()) output_tracks.push_back(track);
128 }
129
130 return output_tracks;
131}

140std::tuple<std::vector<Track::Ptr>, std::vector<Track::Ptr>,
141 std::vector<DetectionBase::Ptr>>
142ByteTracker::iou_association(const std::vector<Track::Ptr> &track_pool,
143 const std::vector<DetectionBase::Ptr> &detections) {
144 auto [matches, unmatched_tracks, unmatched_detections] =
145 linear_assignment(track_pool, detections, match_thresh_);
146
147 std::vector<Track::Ptr> matched_tracks;
148 for (const auto &match : matches) {
149 const auto track = match.first;
150 const auto detection = match.second;
151 track->update(detection, frame_id_);
152 matched_tracks.push_back(track);
153 }
154
155 std::vector<Track::Ptr> unmatched_tracked_tracks;
156 for (const auto &unmatch : unmatched_tracks) {
157 if (unmatch->get_track_state() == TrackState::Tracked) {
158 unmatched_tracked_tracks.push_back(unmatch);
159 }
160 }
161 return {std::move(matched_tracks), std::move(unmatched_tracked_tracks),
162 std::move(unmatched_detections)};
163}

165std::vector<Track::Ptr> ByteTracker::low_score_association(
166 std::vector<Track::Ptr> &matched_tracks,
167 const std::vector<DetectionBase::Ptr> &low_score_detections,
168 const std::vector<Track::Ptr> &unmatched_tracked_tracks) {
169 auto [matches, unmatched_tracks, unmatch_detection] =
170 linear_assignment(unmatched_tracked_tracks, low_score_detections, 0.5);
171
172 for (const auto &match : matches) {
173 const auto track = match.first;
174 const auto detection = match.second;
175 track->update(detection, frame_id_);
176 matched_tracks.push_back(track);
177 }
178
179 std::vector<Track::Ptr> new_lost_tracks;
180 for (const auto &track : unmatched_tracks) {
181 if (track->get_track_state() != TrackState::Lost) {
182 track->mark_as_lost();
183 new_lost_tracks.push_back(track);
184 }
185 }
186 return new_lost_tracks;
187}

189std::vector<Track::Ptr> ByteTracker::init_new_tracks(
190 std::vector<Track::Ptr> &matched_tracks,
191 const std::vector<Track::Ptr> &unconfirmed_tracks,
192 const std::vector<DetectionBase::Ptr> &unmatched_detections) {
193 // Deal with unconfirmed tracks, usually tracks with only one beginning frame
194 auto [matches, unmatched_unconfirmed_tracks, new_detections] =
195 linear_assignment(unconfirmed_tracks, unmatched_detections, 0.7);
196
197 for (const auto &match : matches) {
198 match.first->update(match.second, frame_id_);
199 matched_tracks.push_back(match.first);
200 }
201
202 std::vector<Track::Ptr> new_removed_tracks;
203 for (const auto &track : unmatched_unconfirmed_tracks) {
204 new_removed_tracks.push_back(track);
205 }
206
207 // Add new tracks
208 for (const auto &detection : new_detections) {
209 if (detection->score() < track_thresh_) continue;
210 track_id_count_++;
211 Track::Ptr new_track =
212 std::make_shared<Track>(detection, frame_id_, track_id_count_);
213 if (detection->score() >= high_thresh_) {
214 new_track->mark_as_confirmed();
215 }
216 matched_tracks.push_back(new_track);
217 }
218 return new_removed_tracks;
219}

221std::vector<Track::Ptr> ByteTracker::joint_tracks(
222 const std::vector<Track::Ptr> &a_tlist,
223 const std::vector<Track::Ptr> &b_tlist) const {
224 std::set<int> exists;
225 std::vector<Track::Ptr> res;
226 for (auto &track : a_tlist) {
227 exists.emplace(track->get_track_id());
228 res.push_back(track);
229 }
230 for (auto &track : b_tlist) {
231 if (exists.count(track->get_track_id()) == 0) res.push_back(track);
232 }
233 return res;
234}

236std::vector<Track::Ptr> ByteTracker::sub_tracks(
237 const std::vector<Track::Ptr> &a_tlist,
238 const std::vector<Track::Ptr> &b_tlist) const {
239 std::map<int, Track::Ptr> tracks;
240 for (auto &track : a_tlist) tracks.emplace(track->get_track_id(), track);
241 for (auto &track : b_tlist) tracks.erase(track->get_track_id());
242
243 std::vector<Track::Ptr> res;
244 for (auto &[_, track] : tracks) res.push_back(track);
245 return res;
246}

248std::tuple<std::vector<Track::Ptr>, std::vector<Track::Ptr>>
249ByteTracker::remove_duplicate_tracks(
250 const std::vector<Track::Ptr> &a_tracks,
251 const std::vector<Track::Ptr> &b_tracks) const {
252 if (a_tracks.empty() || b_tracks.empty()) return {a_tracks, b_tracks};
253
254 std::vector<std::vector<float>> ious;
255 ious.resize(a_tracks.size());
256 for (size_t i = 0; i < ious.size(); i++) ious[i].resize(b_tracks.size());
257 for (size_t ai = 0; ai < a_tracks.size(); ai++) {
258 for (size_t bi = 0; bi < b_tracks.size(); bi++) {
259 ious[ai][bi] = 1 - calc_iou(b_tracks[bi]->predicted_rect_,
260 a_tracks[ai]->predicted_rect_);
261 }
262 }
263
264 std::vector<bool> a_overlapping(a_tracks.size(), false),
265 b_overlapping(b_tracks.size(), false);
266 for (size_t ai = 0; ai < ious.size(); ai++) {
267 for (size_t bi = 0; bi < ious[ai].size(); bi++) {
268 if (ious[ai][bi] < 0.15) {
269 const int timep =
270 a_tracks[ai]->get_frame_id() - a_tracks[ai]->get_start_frame_id();
271 const int timeq =
272 b_tracks[bi]->get_frame_id() - b_tracks[bi]->get_start_frame_id();
273 if (timep > timeq) {
274 b_overlapping[bi] = true;
275 } else {
276 a_overlapping[ai] = true;
277 }
278 }
279 }
280 }
281
282 std::vector<Track::Ptr> a_tracks_out;
283 for (size_t ai = 0; ai < a_tracks.size(); ai++) {
284 if (!a_overlapping[ai]) a_tracks_out.push_back(a_tracks[ai]);
285 }
286
287 std::vector<Track::Ptr> b_tracks_out;
288 for (size_t bi = 0; bi < b_tracks.size(); bi++) {
289 if (!b_overlapping[bi]) b_tracks_out.push_back(b_tracks[bi]);
290 }
291 return {std::move(a_tracks_out), std::move(b_tracks_out)};
292}

294std::tuple<std::vector<std::pair<Track::Ptr, DetectionBase::Ptr>>,
295 std::vector<Track::Ptr>, std::vector<DetectionBase::Ptr>>
296ByteTracker::linear_assignment(const std::vector<Track::Ptr> &tracks,
297 const std::vector<DetectionBase::Ptr> &detections,
298 float thresh) const {
299 if (tracks.empty() || detections.empty()) return {{}, tracks, detections};
300
301 size_t n_rows = tracks.size();
302 size_t n_cols = detections.size();
303 std::vector<float> cost_matrix(n_rows * n_cols);
304 for (size_t i = 0; i < n_rows; i++) {
305 for (size_t j = 0; j < n_cols; j++) {
306 cost_matrix[i * n_cols + j] =
307 1 - calc_iou(detections[j]->rect(), tracks[i]->predicted_rect_);
308 }
309 }
310
311 std::vector<std::pair<Track::Ptr, DetectionBase::Ptr>> matches;
312 std::vector<Track::Ptr> a_unmatched;
313 std::vector<DetectionBase::Ptr> b_unmatched;
314
315 auto [rowsol, colsol, _] =
316 exec_lapjv(std::move(cost_matrix), n_rows, n_cols, true, thresh);
317 for (size_t i = 0; i < rowsol.size(); i++) {
318 if (rowsol[i] >= 0)
319 matches.push_back({tracks[i], detections[rowsol[i]]});
320 else
321 a_unmatched.push_back(tracks[i]);
322 }
323
324 for (size_t i = 0; i < colsol.size(); i++) {
325 if (colsol[i] < 0) b_unmatched.push_back(detections[i]);
326 }
327 return {std::move(matches), std::move(a_unmatched), std::move(b_unmatched)};
328}

330std::tuple<std::vector<int>, std::vector<int>, double> ByteTracker::exec_lapjv(
331 std::vector<float> &&cost, size_t n_rows, size_t n_cols, bool extend_cost,
332 float cost_limit, bool return_cost) const {
333 std::vector<int> rowsol(n_rows);
334 std::vector<int> colsol(n_cols);
335
336 if (n_rows != n_cols && !extend_cost) {
337 throw std::runtime_error("The `extend_cost` variable should set True");
338 }
339
340 size_t n = 0;
341 std::vector<float> cost_c;
342 if (extend_cost || cost_limit < std::numeric_limits<float>::max()) {
343 n = n_rows + n_cols;
344 cost_c.resize(n * n);
345 if (cost_limit < std::numeric_limits<float>::max()) {
346 cost_c.assign(cost_c.size(), cost_limit / 2.0);
347 } else {
348 cost_c.assign(cost_c.size(),
349 *std::max_element(cost.begin(), cost.end()) + 1);
350 }
351 // Assign cost to top-left corner
352 for (size_t i = 0; i < n_rows; i++) {
353 for (size_t j = 0; j < n_cols; j++) {
354 cost_c[i * n + j] = cost[i * n_cols + j];
355 }
356 }
357 // Set bottom-right corner to 0
358 for (size_t i = n_rows; i < n; i++) {
359 for (size_t j = n_cols; j < n; j++) {
360 cost_c[i * n + j] = 0;
361 }
362 }
363 } else {
364 n = n_rows;
365 cost_c = std::move(cost);
366 }
367
368 std::vector<int> x_c(n), y_c(n);
369
370 int ret = lapjv_internal(n, cost_c, x_c, y_c);
371 if (ret != 0) {
372 throw std::runtime_error("The result of lapjv_internal() is invalid.");
373 }
374
375 double opt = 0.0;
376
377 if (n != n_rows) {
378 for (size_t i = 0; i < n; i++) {
379 if (x_c[i] >= static_cast<int>(n_cols)) x_c[i] = -1;
380 if (y_c[i] >= static_cast<int>(n_rows)) y_c[i] = -1;
381 }
382 for (size_t i = 0; i < n_rows; i++) {
383 rowsol[i] = x_c[i];
384 }
385 for (size_t i = 0; i < n_cols; i++) {
386 colsol[i] = y_c[i];
387 }
388
389 if (return_cost) {
390 for (size_t i = 0; i < rowsol.size(); i++) {
391 if (rowsol[i] != -1) {
392 opt += cost_c[i * n_cols + rowsol[i]];
393 }
394 }
395 }
396 } else if (return_cost) {
397 for (size_t i = 0; i < rowsol.size(); i++) {
398 opt += cost_c[i * n_cols + rowsol[i]];
399 }
400 }
401
402 return {std::move(rowsol), std::move(colsol), opt};
403}
