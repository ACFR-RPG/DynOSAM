41CsvReader::Row::Row(const std::string& line, const CsvHeader& header,
42 const char delimiter)
43 : line_(line), header_(header), delimiter_(delimiter) {
44 pos_data_.emplace_back(-1);
45 std::string::size_type pos = 0;
46 while ((pos = line_.find(delimiter_, pos)) != std::string::npos) {
47 pos_data_.emplace_back(pos);
48 ++pos;
49 }
50 // This checks for a trailing comma with no data after it.
51 pos = line_.size();
52 pos_data_.emplace_back(pos);
53
54 // if usable header provided (one of non zero size, ie. non default header)
55 // check that the number of elements in the parsed input string matches the
56 // expected number of cols in the header
57 if (usableHeader() && this->size() != header_.size()) {
58 std::stringstream ss;
59 ss << "Invalid row construction using a valid CsvHeader: input row has "
60 "length "
61 << this->size() << " and header has size " << header_.size()
62 << " with column values - " << header_.toString();
63 throw std::runtime_error(ss.str());
64 }
65}

67std::string CsvReader::Row::operator[](std::size_t index) const {
68 size_t size = pos_data_[index + 1] - (pos_data_[index] + 1);
69 std::string_view view = std::string_view(&line_[pos_data_[index] + 1], size);
70 std::string val = {view.begin(), view.end()};
71
72 // removes all the leading and trailing white spaces.
73 boost::algorithm::trim(val);
74 return val;
75}

104CsvWriter& CsvWriter::add(const std::string& str) {
105 std::string str_cpy = str;
106 // if " character was found, escape it
107 size_t position = str_cpy.find("\"", 0);
108 bool foundQuotationMarks = position != std::string::npos;
109 while (position != std::string::npos) {
110 str_cpy.insert(position, "\"");
111 position = str_cpy.find("\"", position + 2);
112 }
113 if (foundQuotationMarks) {
114 str_cpy = "\"" + str_cpy + "\"";
115 } else if (str_cpy.find(this->seperator_) != std::string::npos) {
116 // if seperator_ was found and string was not escapted before, surround
117 // string with "
118 str_cpy = "\"" + str_cpy + "\"";
119 }
120 return this->add<std::string>(str_cpy);
121}

133bool CsvWriter::write(const std::string& filename, bool append) const {
134 std::ofstream file;
135 bool appendNewLine = false;
136 if (append) {
137 // check if last char of the file is newline
138 std::ifstream fin;
139 fin.open(filename);
140 if (fin.is_open()) {
141 fin.seekg(-1, std::ios_base::end); // go to end of file
142 int lastChar = fin.peek();
143 if (lastChar != -1 &&
144 lastChar !=
145 '\n') // if file is not empry and last char is not new line char
146 appendNewLine = true;
147 }
148 file.open(filename.c_str(), std::ios::out | std::ios::app);
149 } else {
150 file.open(filename.c_str(), std::ios::out | std::ios::trunc);
151 }
152 if (!file.is_open()) return false;
153 if (append && appendNewLine) file << std::endl;
154
155 return write(file);
156}
