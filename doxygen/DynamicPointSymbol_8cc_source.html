31std::uint64_t CantorPairingFunction::pair(const Pair& input) {
32 const auto k1 = (input.first);
33 const auto k2 = (input.second);
34 return ((k1+k2)*(k1+k2+1)/2) + k2;
35}

37CantorPairingFunction::Pair CantorPairingFunction::depair(const std::uint64_t z) {
38 std::uint64_t w = static_cast<std::uint64_t>(floor(((sqrt((z*8)+1))-1)/2));
39 std::uint64_t t = static_cast<std::uint64_t>((w*(w+1))/2);
40
41 std::uint64_t k2 = z - t;
42 std::uint64_t k1 = w - k2;
43 return std::make_pair(k1, k2);
44}

95std::uint64_t DynamicPointSymbol::constructIndex(TrackletId tracklet_id, FrameId frame_id) {
96 if(tracklet_id == -1) {
97 throw std::invalid_argument("DynamicPointSymbol cannot be constructed from invalid tracklet id (-1)");
98 }
99
100 return CantorPairingFunction::pair({tracklet_id, frame_id});
101}

103void DynamicPointSymbol::recover(std::uint64_t z, TrackletId& tracklet_id, FrameId& frame_id) {
104 const auto result = CantorPairingFunction::depair(z);
105 tracklet_id = static_cast<TrackletId>(result.first);
106 frame_id = static_cast<FrameId>(result.second);
107}
