88const double kNumSecondsPerNanosecond = 1.e-9;
89
90struct StatisticsMapValue {
91 // this is the same for EVERY value, do we really want this?
92 static const int kWindowSize = 1000;
93
94 inline StatisticsMapValue() {
95 time_last_called_ = std::chrono::system_clock::now();
96 }
97
98 inline void AddValue(double sample) {
99 std::chrono::time_point<std::chrono::system_clock> now =
100 std::chrono::system_clock::now();
101 double dt = static_cast<double>(
102 std::chrono::duration_cast<std::chrono::nanoseconds>(
103 now - time_last_called_)
104 .count()) *
105 kNumSecondsPerNanosecond;
106 time_last_called_ = now;
107
108 values_.Add(sample);
109 time_deltas_.Add(dt);
110 }
111 inline double GetLastDeltaTime() const {
112 if (time_deltas_.total_samples()) {
113 return time_deltas_.GetMostRecent();
114 } else {
115 return 0;
116 }
117 }
118 inline double GetLastValue() const {
119 if (values_.total_samples()) {
120 return values_.GetMostRecent();
121 } else {
122 return 0;
123 }
124 }
125 inline double Sum() const { return values_.sum(); }
126 int TotalSamples() const { return values_.total_samples(); }
127 double Mean() const { return values_.Mean(); }
128 double RollingMean() const { return values_.RollingMean(); }
129 double Max() const { return values_.max(); }
130 double Min() const { return values_.min(); }
131 double Median() const { return values_.median(); }
132 double Q1() const { return values_.q1(); }
133 double Q3() const { return values_.q3(); }
134 double LazyVariance() const { return values_.LazyVariance(); }
135 double MeanCallsPerSec() const {
136 double mean_dt = time_deltas_.Mean();
137 if (mean_dt != 0) {
138 return 1.0 / mean_dt;
139 } else {
140 return -1.0;
141 }
142 }
143
144 double MeanDeltaTime() const { return time_deltas_.Mean(); }
145 double RollingMeanDeltaTime() const { return time_deltas_.RollingMean(); }
146 double MaxDeltaTime() const { return time_deltas_.max(); }
147 double MinDeltaTime() const { return time_deltas_.min(); }
148 double LazyVarianceDeltaTime() const { return time_deltas_.LazyVariance(); }
149 const std::vector<double>& GetAllValues() const {
150 return values_.GetAllSamples();
151 }
152
153 private:
154 // Create an accumulator with specified window size.
155 Accumulator<double, double, kWindowSize> values_;
156 Accumulator<double, double, kWindowSize> time_deltas_;
157 std::chrono::time_point<std::chrono::system_clock> time_last_called_;
158};
