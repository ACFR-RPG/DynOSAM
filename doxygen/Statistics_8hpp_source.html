81const double kNumSecondsPerNanosecond = 1.e-9;
82
83struct StatisticsMapValue {
84 //this is the same for EVERY value, do we really want this?
85 static const int kWindowSize = 1000;
86
87 inline StatisticsMapValue() {
88 time_last_called_ = std::chrono::system_clock::now();
89 }
90
91 inline void AddValue(double sample) {
92 std::chrono::time_point<std::chrono::system_clock> now =
93 std::chrono::system_clock::now();
94 double dt = static_cast<double>(
95 std::chrono::duration_cast<std::chrono::nanoseconds>(
96 now - time_last_called_)
97 .count()) *
98 kNumSecondsPerNanosecond;
99 time_last_called_ = now;
100
101 values_.Add(sample);
102 time_deltas_.Add(dt);
103 }
104 inline double GetLastDeltaTime() const {
105 if (time_deltas_.total_samples()) {
106 return time_deltas_.GetMostRecent();
107 } else {
108 return 0;
109 }
110 }
111 inline double GetLastValue() const {
112 if (values_.total_samples()) {
113 return values_.GetMostRecent();
114 } else {
115 return 0;
116 }
117 }
118 inline double Sum() const { return values_.sum(); }
119 int TotalSamples() const { return values_.total_samples(); }
120 double Mean() const { return values_.Mean(); }
121 double RollingMean() const { return values_.RollingMean(); }
122 double Max() const { return values_.max(); }
123 double Min() const { return values_.min(); }
124 double Median() const { return values_.median(); }
125 double Q1() const { return values_.q1(); }
126 double Q3() const { return values_.q3(); }
127 double LazyVariance() const { return values_.LazyVariance(); }
128 double MeanCallsPerSec() const {
129 double mean_dt = time_deltas_.Mean();
130 if (mean_dt != 0) {
131 return 1.0 / mean_dt;
132 } else {
133 return -1.0;
134 }
135 }
136
137 double MeanDeltaTime() const { return time_deltas_.Mean(); }
138 double RollingMeanDeltaTime() const { return time_deltas_.RollingMean(); }
139 double MaxDeltaTime() const { return time_deltas_.max(); }
140 double MinDeltaTime() const { return time_deltas_.min(); }
141 double LazyVarianceDeltaTime() const { return time_deltas_.LazyVariance(); }
142 const std::vector<double>& GetAllValues() const { return values_.GetAllSamples(); }
143
144private:
145 // Create an accumulator with specified window size.
146 Accumulator<double, double, kWindowSize> values_;
147 Accumulator<double, double, kWindowSize> time_deltas_;
148 std::chrono::time_point<std::chrono::system_clock> time_last_called_;
149};
