50bool FeatureTrackerBase::predictKeypointsGivenRotation(
51 std::vector<cv::Point2f>& predicted_pts_k,
52 const std::vector<cv::Point2f>& pts_km1, const gtsam::Rot3& R_km1_k) const {
53 // Handle case when rotation is small: just copy prev_kps
54 // Removed bcs even small rotations lead to huge optical flow at the borders
55 // of the image.
56 // Keep because then you save a lot of computation.
57 static constexpr double kSmallRotationTol = 1e-4;
58 if (std::abs(1.0 - std::abs(R_km1_k.toQuaternion().w())) <
59 kSmallRotationTol) {
60 predicted_pts_k = pts_km1;
61 return true;
62 }
63
64 const gtsam::Matrix K = camera_->getParams().getCameraMatrixEigen();
65 const gtsam::Matrix K_inv = K.inverse();
66
67 const cv::Matx33f K_cv = utils::gtsamMatrix3ToCvMat(K);
68 const cv::Matx33f K_inv_cv = utils::gtsamMatrix3ToCvMat(K_inv);
69 // R is a relative rotation which takes a vector from the last frame to
70 // the current frame.
71 const cv::Matx33f R = utils::gtsamMatrix3ToCvMat(R_km1_k.matrix());
72 // Get bearing vector for kpt, rotate knowing frame to frame rotation,
73 // get keypoints again
74 const cv::Matx33f H = K_cv * R * K_inv_cv;
75
76 const size_t& n_kps = pts_km1.size();
77 predicted_pts_k.reserve(n_kps);
78 for (size_t i = 0u; i < n_kps; ++i) {
79 // Create homogeneous keypoints.
80 const auto& prev_kpt = pts_km1[i];
81 cv::Vec3f p1(prev_kpt.x, prev_kpt.y, 1.0f);
82
83 cv::Vec3f p2 = H * p1;
84
85 // Project predicted bearing vectors to 2D again and re-homogenize.
86 cv::Point2f new_kpt;
87 if (p2[2] > 0.0f) {
88 new_kpt = cv::Point2f(p2[0] / p2[2], p2[1] / p2[2]);
89 } else {
90 LOG(WARNING) << "Landmark behind the camera:\n"
91 << "- p1: " << p1 << '\n'
92 << "- p2: " << p2;
93 new_kpt = prev_kpt;
94 }
95 // Check that keypoints remain inside the image boundaries!
96 if (isWithinShrunkenImage(new_kpt)) {
97 predicted_pts_k.push_back(new_kpt);
98 } else {
99 // Otw copy-paste previous keypoint.
100 predicted_pts_k.push_back(prev_kpt);
101 }
102 }
103
104 return true;
105}

131cv::Mat FeatureTrackerBase::computeImageTracks(
132 const Frame& previous_frame, const Frame& current_frame,
133 const ImageTracksParams& config) const {
134 const ImageWrapper<ImageType::RGBMono>& img_wrapper =
135 current_frame.image_container_.rgb();
136 cv::Mat img_rgb = img_wrapper.toRGB().clone();
137 const cv::Mat& object_mask =
138 current_frame.image_container_.objectMotionMask();
139
140 const bool& debug = config.isDebug();
141 const bool& show_intermediate_tracking = config.showIntermediateTracking();
142
143 const int static_point_thickness = config.featureThickness();
144
145 static const cv::Scalar red(Color::red().bgra());
146 static const cv::Scalar green(Color::green().bgra());
147 static const cv::Scalar blue(Color::blue().bgra());
148
149 int num_static_tracks = 0;
150 // Add all keypoints in cur_frame with the tracks.
151 for (const Feature::Ptr& feature : current_frame.static_features_) {
152 const Keypoint& px_cur = feature->keypoint();
153 const auto pc_cur = utils::gtsamPointToCv(px_cur);
154 if (!feature->usable() &&
155 show_intermediate_tracking) { // Untracked landmarks are red.
156 cv::circle(img_rgb, pc_cur, static_point_thickness, red, 2);
157 } else {
158 const Feature::Ptr& prev_feature =
159 previous_frame.static_features_.getByTrackletId(
160 feature->trackletId());
161 if (prev_feature) {
162 // If feature was in previous frame, display tracked feature with
163 // green circle/line:
164 cv::circle(img_rgb, pc_cur, static_point_thickness, green, 1);
165
166 // draw the optical flow arrow
167 const auto pc_prev = utils::gtsamPointToCv(prev_feature->keypoint());
168 cv::arrowedLine(img_rgb, pc_prev, pc_cur, green, 1);
169
170 num_static_tracks++;
171
172 } else if (debug &&
173 show_intermediate_tracking) { // New feature tracks are blue.
174 cv::circle(img_rgb, pc_cur, 6, blue, 1);
175 }
176 }
177 }
178
179 for (const Feature::Ptr& feature : current_frame.dynamic_features_) {
180 const Keypoint& px_cur = feature->keypoint();
181 if (!feature->usable()) { // Untracked landmarks are red.
182 // cv::circle(img_rgb, utils::gtsamPointToCv(px_cur), 1, red, 2);
183 } else {
184 const Feature::Ptr& prev_feature =
185 previous_frame.dynamic_features_.getByTrackletId(
186 feature->trackletId());
187 if (prev_feature) {
188 // If feature was in previous frame, display tracked feature with
189 // green circle/line:
190 // cv::circle(img_rgb, utils::gtsamPointToCv(px_cur), 6, green, 1);
191 const Keypoint& px_prev = prev_feature->keypoint();
192 const cv::Scalar colour = Color::uniqueId(feature->objectId()).bgra();
193 cv::arrowedLine(img_rgb, utils::gtsamPointToCv(px_prev),
194 utils::gtsamPointToCv(px_cur), colour, 1, 8, 0, 0.1);
195 cv::circle(img_rgb, utils::gtsamPointToCv(px_cur), 2, colour, -1);
196 } else { // New feature tracks are blue.
197 // cv::circle(img_rgb, utils::gtsamPointToCv(px_cur), 1, blue, 1);
198 }
199 }
200 }
201
202 const int bbox_thickness = config.bboxThickness();
203
204 std::vector<ObjectId> objects_to_print;
205 for (const auto& object_observation_pair :
206 current_frame.object_observations_) {
207 const ObjectId object_id = object_observation_pair.first;
208 const cv::Rect& bb = object_observation_pair.second.bounding_box;
209
210 // TODO: if its marked as moving!!
211 if (bb.empty()) {
212 continue;
213 }
214
215 objects_to_print.push_back(object_id);
216
217 if (config.drawObjectBoundingBox()) {
218 const cv::Scalar colour = Color::uniqueId(object_id).bgra();
219 const std::string label = "object " + std::to_string(object_id);
220 utils::drawLabeledBoundingBox(img_rgb, label, colour, bb, bbox_thickness);
221 }
222 }
223
224 if (config.drawObjectMask()) {
225 constexpr static float kAlpha = 0.7;
226 utils::labelMaskToRGB(object_mask, img_rgb, img_rgb, kAlpha);
227 }
228
229 // draw text info
230 std::stringstream ss;
231 ss << "Frame ID: " << current_frame.getFrameId() << " | ";
232 ss << "VO tracks: " << num_static_tracks << " | ";
233 ss << "Objects: ";
234
235 if (objects_to_print.empty()) {
236 ss << "None";
237 } else {
238 ss << "[";
239 for (size_t i = 0; i < objects_to_print.size(); ++i) {
240 ss << objects_to_print[i];
241 if (i != objects_to_print.size() - 1) {
242 ss << ", "; // Add comma between elements
243 }
244 }
245 ss << "]";
246 }
247
248 constexpr static double kFontScale = 0.6;
249 constexpr static int kFontFace = cv::FONT_HERSHEY_SIMPLEX;
250 constexpr static int kThickness = 1;
251
252 if (config.showFrameInfo()) {
253 // taken from ORB-SLAM2 ;)
254 int base_line;
255 cv::Size text_size = cv::getTextSize(ss.str(), kFontFace, kFontScale,
256 kThickness, &base_line);
257 cv::Mat image_text = cv::Mat(img_rgb.rows + text_size.height + 10,
258 img_rgb.cols, img_rgb.type());
259 img_rgb.copyTo(
260 image_text.rowRange(0, img_rgb.rows).colRange(0, img_rgb.cols));
261 image_text.rowRange(img_rgb.rows, image_text.rows) =
262 cv::Mat::zeros(text_size.height + 10, img_rgb.cols, img_rgb.type());
263 cv::putText(image_text, ss.str(), cv::Point(5, image_text.rows - 5),
264 kFontFace, kFontScale, cv::Scalar(255, 255, 255), kThickness);
265 return image_text;
266 } else {
267 return img_rgb;
268 }
269}

271bool FeatureTrackerBase::drawStereoMatches(cv::Mat& output_image,
272 const Frame& current_frame) const {
273 // for now only static tracks
274 if (!current_frame.image_container_.hasRightRgb()) {
275 return false;
276 }
277
278 const ImageWrapper<ImageType::RGBMono>& left_img_wrapper =
279 current_frame.image_container_.rgb();
280 cv::Mat img_rgb_left = left_img_wrapper.toRGB().clone();
281
282 const ImageWrapper<ImageType::RGBMono>& right_img_wrapper =
283 current_frame.image_container_.rightRgb();
284 cv::Mat img_rgb_right = right_img_wrapper.toRGB().clone();
285
286 // Stack side by side
287 cv::Mat canvas;
288 cv::hconcat(img_rgb_left, img_rgb_right, canvas);
289
290 int w1 = img_rgb_left.cols;
291
292 auto itr = current_frame.static_features_.beginUsable();
293 for (const auto& feature : itr) {
294 if (!feature->hasRightKeypoint()) {
295 continue;
296 }
297
298 const auto kp_left = utils::gtsamPointToCv(feature->keypoint());
299 const auto kp_right = utils::gtsamPointToCv(feature->rightKeypoint()) +
300 cv::Point2f((float)w1, 0.0f);
301
302 cv::circle(canvas, kp_left, 4, cv::Scalar(0, 0, 255), cv::FILLED,
303 cv::LINE_AA);
304 cv::circle(canvas, kp_right, 4, cv::Scalar(0, 0, 255), cv::FILLED,
305 cv::LINE_AA);
306 cv::line(canvas, kp_left, kp_right, cv::Scalar(0, 0, 255), 1, cv::LINE_AA);
307 }
308
309 output_image = canvas;
310 return true;
311}

313bool FeatureTrackerBase::isWithinShrunkenImage(const Keypoint& kp) const {
314 const auto shrunken_row = params_.shrink_row;
315 const auto shrunken_col = params_.shrink_col;
316
317 const int predicted_col = functional_keypoint::u(kp);
318 const int predicted_row = functional_keypoint::v(kp);
319
320 const auto image_rows = img_size_.height;
321 const auto image_cols = img_size_.width;
322 return (predicted_row > shrunken_row &&
323 predicted_row < (image_rows - shrunken_row) &&
324 predicted_col > shrunken_col &&
325 predicted_col < (image_cols - shrunken_col));
326}
