51cv::Mat FeatureTrackerBase::computeImageTracks(
52 const Frame& previous_frame, const Frame& current_frame,
53 const ImageTracksParams& config) const {
54 cv::Mat img_rgb;
55
56 const cv::Mat& rgb = current_frame.image_container_.get<ImageType::RGBMono>();
57 rgb.copyTo(img_rgb);
58
59 const bool debug = config.is_debug;
60 const bool show_frame_info = debug && config.show_frame_info;
61 const bool show_intermediate_tracking =
62 debug && config.show_intermediate_tracking;
63
64 static const cv::Scalar red(Color::red().bgra());
65 static const cv::Scalar green(Color::green().bgra());
66 static const cv::Scalar blue(Color::blue().bgra());
67
68 constexpr static int kFeatureThicknessDebug = 5;
69 constexpr static int kFeatureThickness = 4;
70 // constexpr static int kFeatureThickness = 7;
71 int static_point_thickness =
72 debug ? kFeatureThicknessDebug : kFeatureThickness;
73
74 int num_static_tracks = 0;
75 // Add all keypoints in cur_frame with the tracks.
76 for (const Feature::Ptr& feature : current_frame.static_features_) {
77 const Keypoint& px_cur = feature->keypoint();
78 const auto pc_cur = utils::gtsamPointToCv(px_cur);
79 if (!feature->usable() &&
80 show_intermediate_tracking) { // Untracked landmarks are red.
81 cv::circle(img_rgb, pc_cur, static_point_thickness, red, 2);
82 } else {
83 const Feature::Ptr& prev_feature =
84 previous_frame.static_features_.getByTrackletId(
85 feature->trackletId());
86 if (prev_feature) {
87 // If feature was in previous frame, display tracked feature with
88 // green circle/line:
89 cv::circle(img_rgb, pc_cur, static_point_thickness, green, 1);
90
91 // draw the optical flow arrow
92 const auto pc_prev = utils::gtsamPointToCv(prev_feature->keypoint());
93 cv::arrowedLine(img_rgb, pc_prev, pc_cur, green, 1);
94
95 num_static_tracks++;
96
97 } else if (debug &&
98 show_intermediate_tracking) { // New feature tracks are blue.
99 cv::circle(img_rgb, pc_cur, 6, blue, 1);
100 }
101 }
102 }
103
104 for (const Feature::Ptr& feature : current_frame.dynamic_features_) {
105 const Keypoint& px_cur = feature->keypoint();
106 if (!feature->usable()) { // Untracked landmarks are red.
107 // cv::circle(img_rgb, utils::gtsamPointToCv(px_cur), 1, red, 2);
108 } else {
109 const Feature::Ptr& prev_feature =
110 previous_frame.dynamic_features_.getByTrackletId(
111 feature->trackletId());
112 if (prev_feature) {
113 // If feature was in previous frame, display tracked feature with
114 // green circle/line:
115 // cv::circle(img_rgb, utils::gtsamPointToCv(px_cur), 6, green, 1);
116 const Keypoint& px_prev = prev_feature->keypoint();
117 const cv::Scalar colour = Color::uniqueId(feature->objectId()).bgra();
118 cv::arrowedLine(img_rgb, utils::gtsamPointToCv(px_prev),
119 utils::gtsamPointToCv(px_cur), colour, 1);
120 } else { // New feature tracks are blue.
121 // cv::circle(img_rgb, utils::gtsamPointToCv(px_cur), 1, blue, 1);
122 }
123 }
124 }
125
126 std::vector<ObjectId> objects_to_print;
127 for (const auto& object_observation_pair :
128 current_frame.object_observations_) {
129 const ObjectId object_id = object_observation_pair.first;
130 const cv::Rect& bb = object_observation_pair.second.bounding_box_;
131
132 // TODO: if its marked as moving!!
133 if (bb.empty()) {
134 continue;
135 }
136
137 objects_to_print.push_back(object_id);
138 const cv::Scalar colour = Color::uniqueId(object_id).bgra();
139
140 const std::string label = "object " + std::to_string(object_id);
141 utils::drawLabeledBoundingBox(img_rgb, label, colour, bb);
142 }
143
144 // draw text info
145 std::stringstream ss;
146 ss << "Frame ID: " << current_frame.getFrameId() << " | ";
147 ss << "VO tracks: " << num_static_tracks << " | ";
148 ss << "Objects: ";
149
150 if (objects_to_print.empty()) {
151 ss << "None";
152 } else {
153 ss << "[";
154 for (size_t i = 0; i < objects_to_print.size(); ++i) {
155 ss << objects_to_print[i];
156 if (i != objects_to_print.size() - 1) {
157 ss << ", "; // Add comma between elements
158 }
159 }
160 ss << "]";
161 }
162
163 constexpr static double kFontScale = 0.6;
164 constexpr static int kFontFace = cv::FONT_HERSHEY_SIMPLEX;
165 constexpr static int kThickness = 1;
166
167 if (debug && show_frame_info) {
168 // taken from ORB-SLAM2 ;)
169 int base_line;
170 cv::Size text_size = cv::getTextSize(ss.str(), kFontFace, kFontScale,
171 kThickness, &base_line);
172 cv::Mat image_text = cv::Mat(img_rgb.rows + text_size.height + 10,
173 img_rgb.cols, img_rgb.type());
174 img_rgb.copyTo(
175 image_text.rowRange(0, img_rgb.rows).colRange(0, img_rgb.cols));
176 image_text.rowRange(img_rgb.rows, image_text.rows) =
177 cv::Mat::zeros(text_size.height + 10, img_rgb.cols, img_rgb.type());
178 cv::putText(image_text, ss.str(), cv::Point(5, image_text.rows - 5),
179 kFontFace, kFontScale, cv::Scalar(255, 255, 255), kThickness);
180 return image_text;
181 } else {
182 return img_rgb;
183 }
184}

186bool FeatureTrackerBase::isWithinShrunkenImage(const Keypoint& kp) const {
187 const auto shrunken_row = params_.shrink_row;
188 const auto shrunken_col = params_.shrink_col;
189
190 const int predicted_col = functional_keypoint::u(kp);
191 const int predicted_row = functional_keypoint::v(kp);
192
193 const auto image_rows = img_size_.height;
194 const auto image_cols = img_size_.width;
195 return (predicted_row > shrunken_row &&
196 predicted_row < (image_rows - shrunken_row) &&
197 predicted_col > shrunken_col &&
198 predicted_col < (image_cols - shrunken_col));
199}
