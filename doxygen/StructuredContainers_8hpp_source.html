55// T is the pointer to the type that is expected to be iterated over
56// such that we have struct iterator_traits<T*> or struct iterator_traits<const T*>;
57// see https://en.cppreference.com/w/cpp/iterator/iterator_traits
58
59//usually defined as filter_iterator_detail<Container::pointer> could also be <filter_iterator_detail<Iter::point>
60template<typename T>
61struct filter_iterator_detail {
63 using value_type = typename std::iterator_traits<T>::value_type;
64 using reference_type = typename std::iterator_traits<T>::reference;
65 using pointer = typename std::iterator_traits<T>::pointer;
66 using difference_type = typename std::iterator_traits<T>::difference_type;
67 using iterator_category = std::forward_iterator_tag; //i guess? only forward is defined (++iter) right now
68};
69
70//container is the actual container type and Iter is the iterator type (eg container::iterator or container::const_iterator)
71//in this way, filter_iterator can act as either a regular iterator or a const iterator
72//we expect container to also be iterable
73//ref type should also be BaseDetail::reference_type?
74//check to ensure container types and iter types are the same?
75//container must at least contain (const and non const versions of end and begin which must return actual iterators which satisfy the
76//forward iterator category)
77//container must also at least have the definitions for Container::iterator, Container::const_iterator, Container::const_reference
78//and Iter must be a properly defined iterator such that it satisfies all the conditions for a forward_iterator (e.f Iter::value_type, Iter::reference etc...)
79template<typename Container, typename Iter, typename FilterFunction = std::function<bool(typename Container::const_reference)>>
80struct filter_iterator_base : public filter_iterator_detail<typename Iter::pointer> {
81public:
82
83 using BaseDetail = filter_iterator_detail<typename Iter::pointer>;
84 using iterator = Iter;
85 using container = Container;
86 using typename BaseDetail::value_type;
87 using typename BaseDetail::reference_type;
88 using typename BaseDetail::pointer;
89 using typename BaseDetail::difference_type;
90 using typename BaseDetail::iterator_category;
91
92 filter_iterator_base(Container& container, const FilterFunction& filter_func) : filter_iterator_base(container, filter_func,container.begin()) {}
93
94 reference_type operator*() { return *it_; }
95 reference_type operator->() { return *it_; }
96
97 bool operator==(const filter_iterator_base& other) const {
98 return it_ == other.it_;
99 }
100 bool operator!=(const filter_iterator_base& other) const { return it_ != other.it_; }
101
102 bool operator==(const iterator& other) const {
103 return it_ == other;
104 }
105 bool operator!=(const iterator& other) const { return it_ != other; }
106
107
108 Container& getContainer() { return container_; }
109 const Container& getContainer() const { return container_; }
110
111 bool operator()(typename Container::const_reference arg) const {
112 return filter_func_(arg);
113 }
114
115
116 //preincrement (++iter)
117 filter_iterator_base& operator++() {
118 do {
119 ++it_;
120 }
121 while(is_invalid());
122 return *this;
123 }
124
125 //TODO??
126 filter_iterator_base& operator++(int x) {
127 do {
128 it_+=x;
129 }
130 while(is_invalid());
131 return *this;
132 }
133
134 //allows the iterator to be used as a enhanced for loop
135 filter_iterator_base begin() { return filter_iterator_base(container_, filter_func_, container_.begin()); }
136 filter_iterator_base end() { return filter_iterator_base(container_, filter_func_, container_.end()); }
137
138 const filter_iterator_base begin() const { return filter_iterator_base(container_, filter_func_, container_.begin()); }
139 const filter_iterator_base end() const { return filter_iterator_base(container_, filter_func_, container_.end()); }
140
141private:
142 bool is_invalid() const {
143 return it_ != container_.end() && !filter_func_(*it_);
144 }
145
146 void find_next_valid() {
147 while(is_invalid()) {
148 this->operator++();
149 }
150 }
151
152protected:
153 filter_iterator_base(Container& container, const FilterFunction& filter_func, iterator it)
154 : container_(container), filter_func_(filter_func), it_(it)
155 { find_next_valid(); }
156
157protected:
158 Container& container_;
159 FilterFunction filter_func_;
160 mutable iterator it_;
161
162};
