58// T is the pointer to the type that is expected to be iterated over
59// such that we have struct iterator_traits<T*> or struct iterator_traits<const
60// T*>; see https://en.cppreference.com/w/cpp/iterator/iterator_traits
61
62// usually defined as filter_iterator_detail<Container::pointer> could also be
63// <filter_iterator_detail<Iter::point>
64template <typename T>
65struct filter_iterator_detail {
67 using value_type = typename std::iterator_traits<T>::value_type;
68 using reference_type = typename std::iterator_traits<T>::reference;
69 using pointer = typename std::iterator_traits<T>::pointer;
70 using difference_type = typename std::iterator_traits<T>::difference_type;
71 using iterator_category =
72 std::forward_iterator_tag; // i guess? only forward is defined (++iter)
73 // right now
74};

76// container is the actual container type and Iter is the iterator type (eg
77// container::iterator or container::const_iterator) in this way,
78// filter_iterator can act as either a regular iterator or a const iterator we
79// expect container to also be iterable ref type should also be
80// BaseDetail::reference_type? check to ensure container types and iter types
81// are the same? container must at least contain (const and non const versions
82// of end and begin which must return actual iterators which satisfy the forward
83// iterator category) container must also at least have the definitions for
84// Container::iterator, Container::const_iterator, Container::const_reference
85// and Iter must be a properly defined iterator such that it satisfies all the
86// conditions for a forward_iterator (e.f Iter::value_type, Iter::reference
87// etc...)
88template <typename Container, typename Iter,
89 typename FilterFunction =
90 std::function<bool(typename Container::const_reference)>>
91struct filter_iterator_base
92 : public filter_iterator_detail<typename Iter::pointer> {
93 public:
94 using BaseDetail = filter_iterator_detail<typename Iter::pointer>;
95 using iterator = Iter;
96 using container = Container;
97 using typename BaseDetail::difference_type;
98 using typename BaseDetail::iterator_category;
99 using typename BaseDetail::pointer;
100 using typename BaseDetail::reference_type;
101 using typename BaseDetail::value_type;
102
103 filter_iterator_base(Container& container, const FilterFunction& filter_func)
104 : filter_iterator_base(container, filter_func, container.begin()) {}
105
106 reference_type operator*() { return *it_; }
107 reference_type operator->() { return *it_; }
108
109 bool operator==(const filter_iterator_base& other) const {
110 return it_ == other.it_;
111 }
112 bool operator!=(const filter_iterator_base& other) const {
113 return it_ != other.it_;
114 }
115
116 bool operator==(const iterator& other) const { return it_ == other; }
117 bool operator!=(const iterator& other) const { return it_ != other; }
118
119 Container& getContainer() { return container_; }
120 const Container& getContainer() const { return container_; }
121
122 bool operator()(typename Container::const_reference arg) const {
123 return filter_func_(arg);
124 }
125
126 // preincrement (++iter)
127 filter_iterator_base& operator++() {
128 do {
129 ++it_;
130 } while (is_invalid());
131 return *this;
132 }
133
134 // TODO??
135 filter_iterator_base& operator++(int x) {
136 do {
137 it_ += x;
138 } while (is_invalid());
139 return *this;
140 }
141
142 // allows the iterator to be used as a enhanced for loop
143 filter_iterator_base begin() {
144 return filter_iterator_base(container_, filter_func_, container_.begin());
145 }
146 filter_iterator_base end() {
147 return filter_iterator_base(container_, filter_func_, container_.end());
148 }
149
150 const filter_iterator_base begin() const {
151 return filter_iterator_base(container_, filter_func_, container_.begin());
152 }
153 const filter_iterator_base end() const {
154 return filter_iterator_base(container_, filter_func_, container_.end());
155 }
156
157 private:
158 bool is_invalid() const {
159 return it_ != container_.end() && !filter_func_(*it_);
160 }
161
162 void find_next_valid() {
163 while (is_invalid()) {
164 this->operator++();
165 }
166 }
167
168 protected:
169 filter_iterator_base(Container& container, const FilterFunction& filter_func,
170 iterator it)
171 : container_(container), filter_func_(filter_func), it_(it) {
172 find_next_valid();
173 }
174
175 protected:
176 Container& container_;
177 FilterFunction filter_func_;
178 mutable iterator it_;
179};

220template <typename T>
221class FrameRangeData {
222 public:
223 using This = FrameRangeData<T>;
224 using FrameRangeT = FrameRange<T>;
225 using FrameRangeTVector = std::set<typename FrameRangeT::Ptr>;
226
227 const typename FrameRangeT::ConstPtr find(FrameId frame_id) const {
228 typename FrameRangeT::Ptr active_range = getActiveRange();
229 if (!active_range) {
230 return nullptr;
231 }
232
233 // sanity check
234 CHECK(active_range->is_active);
235 if (active_range->contains(frame_id)) {
236 return active_range;
237 } else {
238 CHECK_GE(ranges.size(), 1u);
239 // iterate over ranges
240 for (const typename FrameRangeT::Ptr& range : ranges) {
241 if (range->contains(frame_id)) {
242 return range;
243 }
244 }
245 }
246 return nullptr;
247 }
248
249 const typename FrameRangeT::ConstPtr startNewActiveRange(FrameId frame_id,
250 const T& data) {
251 typename FrameRangeT::Ptr old_active_range = getActiveRange();
252 auto new_range = std::make_shared<FrameRangeT>();
253 new_range->start = frame_id;
254 // dont set end (yet) but make active
255 new_range->is_active = true;
256 new_range->data = data;
257
258 return updateRanges(new_range, old_active_range);
259 }
260
261 auto begin() { return ranges.begin(); }
262 auto end() { return ranges.end(); }
263
264 auto begin() const { return ranges.begin(); }
265 auto end() const { return ranges.end(); }
266
267 typename FrameRangeT::Ptr getActiveRange() const {
268 if (!active_range) {
269 // no range means no data
270 CHECK_EQ(ranges.size(), 0u);
271 return nullptr;
272 }
273 CHECK(active_range->is_active);
274 return active_range;
275 }
276
277 private:
278 typename FrameRangeT::Ptr updateRanges(
279 typename FrameRangeT::Ptr new_range,
280 typename FrameRangeT::Ptr old_active_range = nullptr) {
281 CHECK_NOTNULL(new_range);
282 if (old_active_range) {
283 // modify existing range so that the end is the start of the next (new
284 // range)
285 old_active_range->end = new_range->start;
286 old_active_range->is_active = false;
287 }
288
289 // ranges.push_back(new_range);
290 ranges.insert(new_range);
291 // set new active range
292 active_range = new_range;
293 return new_range;
294 }
295
296 private:
297 FrameRangeTVector ranges;
298 typename FrameRangeT::Ptr active_range;
299};

301template <typename V, typename T>
302class MultiFrameRangeData {
303 public:
304 using This = MultiFrameRangeData<V, T>;
305 using FrameRangeT = FrameRange<T>;
306 using FrameRangeDataTVector = FrameRangeData<T>;
307
308 MultiFrameRangeData() {}
309
310 const typename FrameRangeT::ConstPtr find(V query, FrameId frame_id) const {
311 if (!ranges.exists(query)) {
312 return nullptr;
313 }
314
315 return ranges.at(query).find(frame_id);
316 }
317 const typename FrameRangeT::ConstPtr startNewActiveRange(V query,
318 FrameId frame_id,
319 const T& data) {
320 if (!ranges.exists(query)) {
321 ranges.insert2(query, FrameRangeDataTVector{});
322 }
323
324 return ranges.at(query).startNewActiveRange(frame_id, data);
325 }
326
327 bool exists(V query) const { return ranges.exists(query); }
328
329 const FrameRangeDataTVector& at(V query) const {
330 if (!ranges.exists(query)) {
331 DYNO_THROW_MSG(std::out_of_range) << "Query " << query << " out of range";
332 }
333
334 return ranges.at(query);
335 }
336
337 private:
338 gtsam::FastMap<V, FrameRangeDataTVector> ranges;
339};
