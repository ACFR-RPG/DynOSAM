38// Safely downcast and deal with errors. This works on raw types (not pointers)
39// NOTE: this will create a copy
40template <class Base, class Derived>
41inline Derived safeCast(const Base& base) {
42 try {
43 return dynamic_cast<const Derived&>(base);
44 } catch (const std::bad_cast& e) {
45 LOG(ERROR) << "Seems that you are casting an object that is not "
46 "the one you expected!";
47 LOG(FATAL) << e.what();
48 } catch (...) {
49 LOG(FATAL) << "Exception caught when dynamic casting.";
50 }
51}

53// Safely downcast shared pointers. Will not create copies of underlying data,
54// but returns a new pointer.
55template <typename Base, typename Derived>
56inline std::shared_ptr<Derived> safeCast(std::shared_ptr<Base> base_ptr) {
57 CHECK(base_ptr);
58 try {
59 return std::dynamic_pointer_cast<Derived>(base_ptr);
60 } catch (const std::bad_cast& e) {
61 LOG(ERROR) << "Seems that you are casting an object that is not "
62 "the one you expected!";
63 LOG(FATAL) << e.what();
64 } catch (...) {
65 LOG(FATAL) << "Exception caught when dynamic casting.";
66 }
67}

69// new pointer
70template <typename Base, typename Derived>
71inline std::shared_ptr<const Derived> safeCast(
72 std::shared_ptr<const Base> base_ptr) {
73 CHECK(base_ptr);
74 std::shared_ptr<Base> base = std::const_pointer_cast<Base>(base_ptr);
75 return safeCast<Base, Derived>(base);
76}

78// Safely downcast unique pointers.
79// NOTE: pass by rvalue because this prevents the copy of the pointer and
80// is faster. All others don't transfer ownership (aren't using move types)
81// and so don't need to pass by rvalue.
82template <typename Base, typename Derived>
83inline std::unique_ptr<Derived> safeCast(std::unique_ptr<Base>&& base_ptr) {
84 std::unique_ptr<Derived> derived_ptr;
85 Derived* tmp = nullptr;
86 try {
87 tmp = dynamic_cast<Derived*>(base_ptr.get());
88 } catch (const std::bad_cast& e) {
89 LOG(ERROR) << "Seems that you are casting an object that is not "
90 "the one you expected!";
91 LOG(FATAL) << e.what();
92 } catch (...) {
93 LOG(FATAL) << "Exception caught when dynamic casting.";
94 }
95 if (!tmp) return nullptr;
96 base_ptr.release();
97 derived_ptr.reset(tmp);
98 return derived_ptr;
99}
