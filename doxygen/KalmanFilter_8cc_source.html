33KalmanFilter::KalmanFilter(float std_weight_position, float std_weight_velocity)
34 : std_weight_position_(std_weight_position),
35 std_weight_velocity_(std_weight_velocity) {
36 motion_mat_ = Eigen::MatrixXf::Identity(8, 8);
37 motion_mat_.block<4, 4>(0, 4) = Eigen::Matrix4f::Identity();
38 update_mat_ = Eigen::MatrixXf::Identity(4, 8);
39}

41void KalmanFilter::initiate(const RectBase &measurement) {
42 mean_.block<1, 4>(0, 0) = rect_to_xyah(measurement);
43 mean_.block<1, 4>(0, 4) = Eigen::Vector4f::Zero();
44
45 Matrix<1, 8> std;
46 std(0) = 2 * std_weight_position_ * measurement.height();
47 std(1) = 2 * std_weight_position_ * measurement.height();
48 std(2) = 1e-2;
49 std(3) = 2 * std_weight_position_ * measurement.height();
50 std(4) = 10 * std_weight_velocity_ * measurement.height();
51 std(5) = 10 * std_weight_velocity_ * measurement.height();
52 std(6) = 1e-5;
53 std(7) = 10 * std_weight_velocity_ * measurement.height();
54
55 Matrix<1, 8> tmp = std.array().square();
56 covariance_ = tmp.asDiagonal();
57}

59TlwhRect KalmanFilter::predict(bool mean_eight_to_zero) {
60 if (mean_eight_to_zero) mean_[7] = 0;
61 Matrix<1, 8> std;
62 std(0) = std_weight_position_ * mean_(3);
63 std(1) = std_weight_position_ * mean_(3);
64 std(2) = 1e-2;
65 std(3) = std_weight_position_ * mean_(3);
66 std(4) = std_weight_velocity_ * mean_(3);
67 std(5) = std_weight_velocity_ * mean_(3);
68 std(6) = 1e-5;
69 std(7) = std_weight_velocity_ * mean_(3);
70
71 Matrix<1, 8> tmp = std.array().square();
72 Matrix<8, 8> motion_cov = tmp.asDiagonal();
73
74 mean_ = motion_mat_ * mean_.transpose();
75 covariance_ =
76 motion_mat_ * covariance_ * motion_mat_.transpose() + motion_cov;
77 return xyah_to_tlwh(mean_.block<1, 4>(0, 0));
78}

80TlwhRect KalmanFilter::update(const RectBase &measurement) {
81 Matrix<1, 4> projected_mean;
82 Matrix<4, 4> projected_cov;
83 project(projected_mean, projected_cov);
84
85 Eigen::Matrix<float, 4, 8> B =
86 (covariance_ * update_mat_.transpose()).transpose();
87 Eigen::Matrix<float, 8, 4> kalman_gain =
88 (projected_cov.llt().solve(B)).transpose();
89 Eigen::Matrix<float, 1, 4> innovation =
90 rect_to_xyah(measurement) - projected_mean;
91
92 const auto tmp = innovation * kalman_gain.transpose();
93 mean_ = (mean_.array() + tmp.array()).matrix();
94 covariance_ =
95 covariance_ - kalman_gain * projected_cov * kalman_gain.transpose();
96 return xyah_to_tlwh(mean_.block<1, 4>(0, 0));
97}

99void KalmanFilter::project(Matrix<1, 4> &projected_mean,
100 Matrix<4, 4> &projected_covariance) {
101 Matrix<1, 4> std{std_weight_position_ * mean_(3),
102 std_weight_position_ * mean_(3), 1e-1,
103 std_weight_position_ * mean_(3)};
104
105 projected_mean = update_mat_ * mean_.transpose();
106 projected_covariance = update_mat_ * covariance_ * update_mat_.transpose();
107
108 Eigen::Matrix<float, 4, 4> diag = std.asDiagonal();
109 projected_covariance += diag.array().square().matrix();
110}

112KalmanFilter::Matrix<1, 4> KalmanFilter::rect_to_xyah(
113 const RectBase &rect) const {
114 return Matrix<1, 4>{rect.left() + rect.width() / 2,
115 rect.top() + rect.height() / 2,
116 rect.width() / rect.height(), rect.height()};
117}

119TlwhRect KalmanFilter::xyah_to_tlwh(const Matrix<1, 4> &xyah) const {
120 float xyah_width = xyah(2) * xyah(3);
121 return TlwhRect{xyah(1) - xyah(3) / 2, xyah(0) - xyah_width / 2, xyah_width,
122 xyah(3)};
123}
