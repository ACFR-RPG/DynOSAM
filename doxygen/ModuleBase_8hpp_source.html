50template <typename INPUT, typename OUTPUT>
51class ModuleBase {
52 public:
53 using Input = INPUT;
54 using Output = OUTPUT;
55 using InputConstPtr = typename INPUT::ConstPtr;
56 using OutputConstPtr =
57 typename OUTPUT::ConstPtr;
58 using This = ModuleBase<INPUT, OUTPUT>;
59
60 DYNO_POINTER_TYPEDEFS(This)
61
62 enum class State {
63 Boostrap = 0u,
64 Nominal = 1u
65 };
66
67 using SpinReturn = std::pair<State, OutputConstPtr>;
68
69 using InputCallback = std::function<void(InputConstPtr)>;
70 using OutputCallback = std::function<void(OutputConstPtr)>;
71
72 ModuleBase(const std::string& name)
73 : name_(name), module_state_(State::Boostrap) {}
74 virtual ~ModuleBase() = default;
75
76 OutputConstPtr spinOnce(InputConstPtr input) {
77 CHECK(input);
78 validateInput(input);
79
80 SpinReturn spin_return{State::Boostrap, nullptr};
81
82 if (input_callback_) input_callback_(input);
83
84 switch (module_state_) {
85 case State::Boostrap: {
86 spin_return = boostrapSpin(input);
87 } break;
88 case State::Nominal: {
89 spin_return = nominalSpin(input);
90 } break;
91 default: {
92 LOG(FATAL) << "Unrecognized state in module " << name_;
93 break;
94 }
95 }
96
97 if (output_callback_) output_callback_(spin_return.second);
98
99 module_state_ = spin_return.first;
100 return spin_return.second;
101 }
102
109 inline void registerInputCallback(const InputCallback& input_callback) {
110 input_callback_ = input_callback;
111 }
112
119 inline void registerOutputCallback(const OutputCallback& output_callback) {
120 output_callback_ = output_callback;
121 }
122
123 protected:
133 virtual void validateInput(const InputConstPtr& input) const = 0;
134
135 virtual SpinReturn boostrapSpin(InputConstPtr input) = 0;
136 virtual SpinReturn nominalSpin(InputConstPtr input) = 0;
137
138 private:
139 const std::string name_;
140 std::atomic<State> module_state_;
141 std::atomic<State> previous_module_state_;
142
143 InputCallback input_callback_;
144 OutputCallback output_callback_;
145};
