41FrontendDSDRos::FrontendDSDRos(const DisplayParams params,
42 rclcpp::Node::SharedPtr node)
43 : FrontendDisplay(), DSDRos(params, node) {
44 tracking_image_pub_ =
45 image_transport::create_publisher(node.get(), "tracking_image");
46
47 auto ground_truth_node = node->create_sub_node("ground_truth");
48 dsd_ground_truth_transport_ =

54 ground_truth_node->create_publisher<nav_msgs::msg::Path>("odometry_path",
55 1);

58void FrontendDSDRos::spinOnce(
59 const FrontendOutputPacketBase::ConstPtr& frontend_output) {
60 // publish debug imagery
61 tryPublishDebugImagery(frontend_output);
62
63 // publish ground truth

65
66 // publish odometry
67 tryPublishVisualOdometry(frontend_output);

77void FrontendDSDRos::tryPublishDebugImagery(
78 const FrontendOutputPacketBase::ConstPtr& frontend_output) {
79 if (!frontend_output->debug_imagery_) return;
80
81 const DebugImagery& debug_imagery = *frontend_output->debug_imagery_;
82 if (debug_imagery.tracking_image.empty()) return;
83
84 std_msgs::msg::Header hdr;
85 sensor_msgs::msg::Image::SharedPtr msg =
86 cv_bridge::CvImage(hdr, "bgr8", debug_imagery.tracking_image)
87 .toImageMsg();
88 tracking_image_pub_.publish(msg);
89}

91void FrontendDSDRos::tryPublishGroundTruth(
92 const FrontendOutputPacketBase::ConstPtr& frontend_output) {
93 if (!frontend_output->gt_packet_ || !frontend_output->debug_imagery_) return;
94
95 const DebugImagery& debug_imagery = *frontend_output->debug_imagery_;
96 const cv::Mat& rgb_image = debug_imagery.rgb_viz;
97 const auto timestamp = frontend_output->getTimestamp();
98 const auto frame_id = frontend_output->getFrameId();
99
100 if (rgb_image.empty()) return;
101
102 // collect gt poses and motions
103 ObjectPoseMap poses;
104 MotionEstimateMap motions;
105
106 const GroundTruthInputPacket& gt_packet = frontend_output->gt_packet_.value();
107
108 for (const auto& object_pose_gt : gt_packet.object_poses_) {
109 // check we have a gt motion here
110 // in the case that we dont, this might be the first time the object
111 // appears...
112 if (!object_pose_gt.prev_H_current_world_) {
113 continue;
114 }
115
116 poses.insert22(object_pose_gt.object_id_, gt_packet.frame_id_,
117 object_pose_gt.L_world_);
118
119 ReferenceFrameValue<gtsam::Pose3> gt_motion(
120 *object_pose_gt.prev_H_current_world_, ReferenceFrame::GLOBAL);
121 motions.insert({object_pose_gt.object_id_, gt_motion});
122 }
123
124 // will this result in confusing tf's since the gt object and estimated
125 // objects use the same link?
126 DSDTransport::Publisher publisher =
127 dsd_ground_truth_transport_->addObjectInfo(
128 motions, poses, params_.world_frame_id, frame_id, timestamp);
129 publisher.publishObjectOdometry();
130
131 // publish ground truth odom
132 const gtsam::Pose3& T_world_camera = gt_packet.X_world_;
133 nav_msgs::msg::Odometry odom_msg;
134 utils::convertWithHeader(T_world_camera, odom_msg, timestamp,
135 params_.world_frame_id, params_.camera_frame_id);
136 vo_ground_truth_publisher_->publish(odom_msg);
137
138 // odom path gt
139 // make static variable since we dont build up the path anywhere else
140 // and just append the last gt camera pose to the path msg
141 static nav_msgs::msg::Path gt_odom_path_msg;
142 static std_msgs::msg::Header header;
143
144 geometry_msgs::msg::PoseStamped pose_stamped;
145 utils::convertWithHeader(T_world_camera, pose_stamped, timestamp,
146 params_.world_frame_id);
147
148 header.stamp = utils::toRosTime(timestamp);
149 header.frame_id = params_.world_frame_id;
150 gt_odom_path_msg.header = header;
151 gt_odom_path_msg.poses.push_back(pose_stamped);
152
153 vo_path_ground_truth_publisher_->publish(gt_odom_path_msg);
154}

164void FrontendDSDRos::processRGBDOutputpacket(
165 const RGBDInstanceOutputPacket::ConstPtr& rgbd_packet) {
166 // publish path
167 // why the camera poses are only in the RGBDInstanceOutputPacket and not in
168 // the base... I have no idea :)
169 this->publishVisualOdometryPath(rgbd_packet->camera_poses_,
170 rgbd_packet->getTimestamp());
171
172 // publish static cloud
173 CHECK(rgbd_packet);
174 this->publishStaticPointCloud(rgbd_packet->static_landmarks_,
175 rgbd_packet->T_world_camera_);
176
177 // publish and collect dynamic cloud
178 CloudPerObject clouds_per_obj = this->publishDynamicPointCloud(
179 rgbd_packet->dynamic_landmarks_, rgbd_packet->T_world_camera_);
180
181 const auto& object_motions = rgbd_packet->estimated_motions_;
182 const auto& object_poses = rgbd_packet->propogated_object_poses_;
183
184 DSDTransport::Publisher object_poses_publisher = dsd_transport_.addObjectInfo(
185 object_motions, object_poses, params_.world_frame_id,
186 rgbd_packet->getFrameId(), rgbd_packet->getTimestamp());
187 object_poses_publisher.publishObjectOdometry();
188 object_poses_publisher.publishObjectTransforms();
189 object_poses_publisher.publishObjectPaths();
190}
