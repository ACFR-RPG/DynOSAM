58Frame::Ptr FeatureTracker::track(FrameId frame_id, Timestamp timestamp,
59 const ImageContainer& image_container,
60 std::set<ObjectId>& object_keyframes,
61 const std::optional<gtsam::Rot3>& R_km1_k) {
62 // take "copy" of tracking_images which is then given to the frame
63 // this will mean that the tracking images (input) are not necessarily the
64 // same as the ones inside the returned frame
65 utils::TimingStatsCollector tracking_timer("tracking_timer");
66 ImageContainer input_images = image_container;
67
68 info_ = FeatureTrackerInfo(); // clear the info
69 info_.frame_id = frame_id;
70 info_.timestamp = timestamp;
71
72 if (initial_computation_) {
73 // intitial computation
74 const cv::Size& other_size =
75 static_cast<const cv::Mat&>(input_images.rgb()).size();
76 CHECK(!previous_frame_);
77 CHECK_EQ(img_size_.width, other_size.width);
78 CHECK_EQ(img_size_.height, other_size.height);
79 initial_computation_ = false;
80 } else {
81 if (params_.use_propogate_mask) {
82 utils::TimingStatsCollector timer("propogate_mask");
83 propogateMask(input_images);
84 }
85 CHECK(previous_frame_);
86 CHECK_EQ(previous_frame_->frame_id_, frame_id - 1u)
87 << "Incoming frame id must be consequative";
88 }
89
90 // from some experimental testing 10 pixles is a good boarder to add around
91 // objects when the image is 640x480 assuming we have some scaling factor r,
92 // width/height * r = 10 and for 640/480, r = (approx) 7.51 for images not
93 // this size we will try and keep the same ratio as this seemed to work well
94 double image_ratio =
95 static_cast<double>(img_size_.width * img_size_.height) / (640.0 * 480.0);
96 static constexpr double kScalingFactorR = 7.51;
97 // desired boarder thickness in pixels for a 640 x 480 image
98 const int scaled_boarder_thickness =
99 std::round(image_ratio * 640.0 / 480.0 * kScalingFactorR);
100 // create detection mask around the boarder of each dynamic object with some
101 // thickness this prevents static and dynamic points being detected around the
102 // edge of the dynamic object as there are lots of inconsistencies here the
103 // detection mask is in the opencv mask form: CV_8UC1 where white pixels (255)
104 // are valid and black pixels (0) should not be detected on
105 static constexpr bool kUseAsFeatureDetectionMask = true;
106 vision_tools::ObjectBoundaryMaskResult boundary_mask_result;
107 vision_tools::computeObjectMaskBoundaryMask(boundary_mask_result,
108 input_images.objectMotionMask(),
109 kUseAsFeatureDetectionMask);
110
111 FeatureContainer static_features;
112 {
113 VLOG(20) << "Starting static track";
114 utils::TimingStatsCollector static_track_timer("static_feature_track");
115 static_features = static_feature_tracker_->trackStatic(
116 previous_frame_, input_images, info_,
117 boundary_mask_result.boundary_mask, R_km1_k);
118 }
119
120 FeatureContainer dynamic_features;
121 {
122 VLOG(20) << "Starting dynamic track";
123 utils::TimingStatsCollector dynamic_track_timer("dynamic_feature_track");
124 trackDynamic(frame_id, input_images, dynamic_features, object_keyframes,
125 boundary_mask_result);
126 }
127
128 previous_tracked_frame_ = previous_frame_; // Update previous frame (previous
129 // to the newly created frame)
130
131 auto new_frame =
132 std::make_shared<Frame>(frame_id, timestamp, camera_, input_images,
133 static_features, dynamic_features, info_);
134
135 // update depth threshold information
136 new_frame->setMaxBackgroundDepth(frontend_params_.max_background_depth);
137 new_frame->setMaxObjectDepth(frontend_params_.max_object_depth);
138
139 VLOG(1) << "Tracked on frame " << frame_id << " t= " << std::setprecision(15)
140 << timestamp << ", object ids "
141 << container_to_string(new_frame->getObjectIds());
142 previous_frame_ = new_frame;
143 boarder_detection_mask_ = boundary_mask_result.boundary_mask;
144
145 return new_frame;
146}

148bool FeatureTracker::stereoTrack(FeaturePtrs& stereo_features,
149 FeatureContainer& left_features,
150 const cv::Mat& left_image,
151 const cv::Mat& right_image,
152 const double& virtual_baseline) const {
153 utils::TimingStatsCollector timing("stereo_track_timer");
154 TrackletIds tracklets_ids;
155 // collect left feature points to cv::point2f
156 std::vector<cv::Point2f> left_feature_points =
157 left_features.toOpenCV(&tracklets_ids, true);
158
159 if (left_feature_points.size() < 8) {
160 LOG(WARNING) << "Not enough left feature points for stereo matching...";
161 return false;
162 }
163
164 std::vector<cv::Point2f> right_feature_points;
165 std::vector<uchar> klt_status;
166 std::vector<float> err;
167
168 const cv::Mat& left_rgb = left_image;
169 cv::Mat left_mono = ImageType::RGBMono::toMono(left_rgb);
170 CHECK(!left_mono.empty());
171 cv::Mat right_mono = ImageType::RGBMono::toMono(right_image);
172 CHECK(!right_mono.empty());
173
174 right_feature_points = left_feature_points;
175
176 cv::calcOpticalFlowPyrLK(left_mono, right_mono, left_feature_points,
177 right_feature_points, klt_status, err,
178 cv::Size(21, 21), 5);
179 CHECK_EQ(klt_status.size(), tracklets_ids.size());
180 TrackletIds good_stereo_tracklets;
181
182 std::vector<cv::Point2f> pts_left_tracked, pts_right_tracked;
183 for (size_t i = 0; i < klt_status.size(); ++i) {
184 auto tracklet_id = tracklets_ids.at(i);
185 // LOG(INFO) << tracklet_id;
186 if (klt_status[i]) {
187 pts_left_tracked.push_back(left_feature_points[i]);
188 pts_right_tracked.push_back(right_feature_points[i]);
189 good_stereo_tracklets.push_back(tracklet_id);
190 }
191 }
192 LOG(INFO) << "Stereo KLT tracked: " << pts_left_tracked.size() << " points";
193
194 // need more than 8 points for fundamental matrix calc with ransac
195 TrackletIds inlier_stereo_tracklets;
196 if (pts_left_tracked.size() < 8) {
197 LOG(WARNING)
198 << "Not enough stereo matches to perform fundamental matrix calc";
199 return false;
200 } else {
201 std::vector<uchar> epipolar_inliers;
202 cv::Mat F =
203 cv::findFundamentalMat(pts_left_tracked, pts_right_tracked,
204 cv::FM_RANSAC, 1.0, 0.99, epipolar_inliers);
205 CHECK_EQ(epipolar_inliers.size(), good_stereo_tracklets.size());
206
207 std::vector<cv::Point2f> pts_left_inlier, pts_right_inlier;
208 for (size_t i = 0; i < epipolar_inliers.size(); ++i) {
209 auto tracklet_id = good_stereo_tracklets.at(i);
210 if (epipolar_inliers[i]) {
211 pts_left_inlier.push_back(pts_left_tracked[i]);
212 pts_right_inlier.push_back(pts_right_tracked[i]);
213
214 CHECK(left_features.getByTrackletId(tracklet_id))
215 << "Somehow tracklet id " << tracklet_id << " is missing!";
216 inlier_stereo_tracklets.push_back(tracklet_id);
217 }
218 }
219
220 LOG(INFO) << "After epipolar filtering: " << inlier_stereo_tracklets.size()
221 << " inliers";
222 const auto& fx = camera_->getParams().fx();
223
224 for (size_t i = 0; i < inlier_stereo_tracklets.size(); i++) {
225 auto inlier_stereo_track = inlier_stereo_tracklets.at(i);
226 Feature::Ptr feature = left_features.getByTrackletId(inlier_stereo_track);
227 CHECK(feature);
228 CHECK(feature->usable());
229
230 double uL = static_cast<double>(pts_left_inlier[i].x);
231 double vL = static_cast<double>(pts_left_inlier[i].y);
232 double uR = static_cast<double>(pts_right_inlier[i].x);
233
234 double disparity = uL - uR;
235 // Reject near-zero disparity
236 // this will also mean far away points.... multi-view triangulation
237 // across frames is needed here... fall back on depth map...
238 if (disparity <= 1.0) {
239 // feature->markOutlier();
240 }
241
242 double depth = fx * virtual_baseline / disparity;
243 // TODO: no max depth
244 feature->depth(depth);
245 feature->rightKeypoint(uR);
246
247 stereo_features.push_back(feature);
248 }
249
250 TrackletIds outlier_stereo_tracklets;
251 determineOutlierIds(inlier_stereo_tracklets, tracklets_ids,
252 outlier_stereo_tracklets);
253
254 left_features.markOutliers(outlier_stereo_tracklets);
255 return true;
256 }

259void FeatureTracker::trackDynamic(
260 FrameId frame_id, const ImageContainer& image_container,
261 FeatureContainer& dynamic_features, std::set<ObjectId>& object_keyframes,
262 const vision_tools::ObjectBoundaryMaskResult& boundary_mask_result) {
263 // first dectect dynamic points
264 const cv::Mat& rgb = image_container.rgb();
265 // flow is going to take us from THIS frame to the next frame (which does not
266 // make sense for a realtime system)
267 const cv::Mat& flow = image_container.opticalFlow();
268 const cv::Mat& motion_mask = image_container.objectMotionMask();
269
270 TrackletIdManager& tracked_id_manager = TrackletIdManager::instance();
271
272 std::set<ObjectId> instance_labels;
273 dynamic_features.clear();
274
275 gtsam::FastMap<ObjectId, FeatureContainer> tracks_per_object;
276 const cv::Mat& detection_mask = boundary_mask_result.boundary_mask;
277 // internal detection mask that is appended with new invalid pixels
278 // this builds the static detection mask over the existing input mask
279 cv::Mat detection_mask_impl;
280 // If we are provided with an external detection/feature mask, initalise the
281 // detection mask with this and add more invalid sections to it
282 if (!detection_mask.empty()) {
283 CHECK_EQ(motion_mask.rows, detection_mask.rows);
284 CHECK_EQ(motion_mask.cols, detection_mask.cols);
285 detection_mask_impl = detection_mask.clone();
286 } else {
287 detection_mask_impl = cv::Mat(motion_mask.size(), CV_8U, cv::Scalar(255));
288 }
289 CHECK_EQ(detection_mask_impl.type(), CV_8U);
290
291 // creating tracking mask, pixel level indicator (1....N) of dynamic feature
292 // location this is different to the detection_mask_impl which is a binary
293 // mask (0/255) and indicates the location of all features (static and
294 // dynamic) and is used to avoid detecting features near existing ones
295 cv::Mat dynamic_tracking_mask =
296 cv::Mat(detection_mask_impl.size(), CV_8U, cv::Scalar(0));
297
298 if (previous_frame_) {
299 const cv::Mat& previous_motion_mask =
300 previous_frame_->image_container_.objectMotionMask();
301 utils::TimingStatsCollector tracked_dynamic_features(
302 "tracked_dynamic_features");
303 for (Feature::Ptr previous_dynamic_feature :
304 previous_frame_->usableDynamicFeaturesBegin()) {
305 const TrackletId tracklet_id = previous_dynamic_feature->trackletId();
306 const size_t age = previous_dynamic_feature->age();
307
308 const Keypoint kp = previous_dynamic_feature->predictedKeypoint();
309 const int x = functional_keypoint::u(kp);
310 const int y = functional_keypoint::v(kp);
311 const ObjectId predicted_label = motion_mask.at<ObjectId>(y, x);
312
313 if (!detection_mask_impl.empty()) {
314 const unsigned char valid_detection =
315 detection_mask_impl.at<unsigned char>(y, x);
316 if (valid_detection == 0) {
317 continue;
318 }
319 }
320
321 ObjectId previous_label = previous_dynamic_feature->objectId();
322 CHECK_NE(previous_label, background_label);
323 CHECK_GT(previous_label, 0);
324
325 PerObjectStatus& object_tracking_info =
326 info_.getObjectStatus(predicted_label);
327 object_tracking_info.num_previous_track++;
328
329 // true if predicted label not on the background
330 const bool is_predicted_object_label =
331 predicted_label != background_label;
332 // true if predicted label the same as the previous label of the tracked
333 // point
334 const bool is_precited_same_as_previous =
335 predicted_label == previous_label;
336
337 // update stats
338 if (!is_predicted_object_label)
339 object_tracking_info.num_tracked_with_background_label++;
340 if (!is_precited_same_as_previous)
341 object_tracking_info.num_tracked_with_different_label++;
342
343 // only include point if it is contained, it is not static and the
344 // previous label is the same as the predicted label
345 if (camera_->isKeypointContained(kp) && is_predicted_object_label &&
346 is_precited_same_as_previous) {
347 size_t new_age = age + 1;
348 double flow_xe = static_cast<double>(flow.at<cv::Vec2f>(y, x)[0]);
349 double flow_ye = static_cast<double>(flow.at<cv::Vec2f>(y, x)[1]);
350
351 OpticalFlow flow(flow_xe, flow_ye);
352 const Keypoint predicted_kp =
353 Feature::CalculatePredictedKeypoint(kp, flow);
354
355 if (!isWithinShrunkenImage(predicted_kp)) {
356 object_tracking_info.num_outside_shrunken_image++;
357 continue;
358 }
359
360 if (flow_xe == 0 || flow_ye == 0) {
361 object_tracking_info.num_zero_flow++;
362 continue;
363 }
364
365 // limit point tracking of a certain age
366 TrackletId tracklet_to_use = tracklet_id;
367 if (new_age > params_.max_dynamic_feature_age) {
368 tracklet_to_use = tracked_id_manager.getTrackletIdCount();
369 tracked_id_manager.incrementTrackletIdCount();
370 new_age = 0;
371 }
372
373 Feature::Ptr feature = std::make_shared<Feature>();
374 (*feature)
375 .objectId(predicted_label)
376 .frameId(frame_id)
377 .keypointType(KeyPointType::DYNAMIC)
378 .age(new_age)
379 .trackletId(tracklet_to_use)
380 .keypoint(kp)
381 .measuredFlow(flow)
382 .predictedKeypoint(predicted_kp);
383
384 if (!tracks_per_object.exists(feature->objectId())) {
385 tracks_per_object.insert2(feature->objectId(), FeatureContainer{});
386 }
387 tracks_per_object.at(feature->objectId()).add(feature);
388 instance_labels.insert(feature->objectId());
389 object_tracking_info.num_track++;
390
391 // add zero fill to detection mask to indicate the existance of a
392 // tracked point at this feature location
393 cv::circle(
394 detection_mask_impl, cv::Point2f(x, y),
395 params_.min_distance_btw_tracked_and_detected_dynamic_features,
396 cv::Scalar(0), cv::FILLED);
397
398 // fill tracking mask with tracked points, labelled with the object
399 // label (j) to indicate places on object with keypoints
400 cv::circle(
401 dynamic_tracking_mask, cv::Point2f(x, y),
402 params_.min_distance_btw_tracked_and_detected_dynamic_features,
403 cv::Scalar(predicted_label), cv::FILLED);
404 }
405 }
406 // merge features
407 for (const auto& [_, features_j] : tracks_per_object) {
408 dynamic_features += features_j;
409 }
410 }
411
412 requiresSampling(object_keyframes, info_, image_container, tracks_per_object,
413 boundary_mask_result, dynamic_tracking_mask);
414
415 std::set<ObjectId> objects_sampled;
416 sampleDynamic(frame_id, image_container,
417 object_keyframes, // indicates which objects to sample!!
418 dynamic_features, objects_sampled, detection_mask_impl);
419}

421void FeatureTracker::sampleDynamic(FrameId frame_id,
422 const ImageContainer& image_container,
423 const std::set<ObjectId>& objects_to_sample,
424 FeatureContainer& dynamic_features,
425 std::set<ObjectId>& objects_sampled,
426 const cv::Mat& detection_mask) {
427 struct KeypointData {
428 OpticalFlow flow;
429 Keypoint predicted_kp;
430 };
431
432 const cv::Mat& rgb = image_container.rgb();
433 // flow is going to take us from THIS frame to the next frame (which does not
434 // make sense for a realtime system)
435 const cv::Mat& flow = image_container.opticalFlow();
436 const cv::Mat& motion_mask = image_container.objectMotionMask();
437
438 TrackletIdManager& tracked_id_manager = TrackletIdManager::instance();
439
440 // container to store keypoints per object
441 tbb::concurrent_hash_map<ObjectId, KeypointsCV> sampled_keypoints;
442 const int rows = rgb.rows;
443 const int cols = rgb.cols;
444
445 std::vector<KeypointData> cached_keypoint_data;
446 cached_keypoint_data.resize(rows * cols);
447 // TODO: since we're looping over the whole image here anyway why dont we also
448 // use this to create the dense point cloud image and then pass it to the
449 // frame!!!
450 std::mutex mutex;
451 tbb::parallel_for(0, rows, [&](int i) {
452 const unsigned char* detection_ptr = detection_mask.ptr<unsigned char>(i);
453 const ObjectId* motion_ptr = motion_mask.ptr<ObjectId>(i);
454 const cv::Vec2f* flow_ptr = flow.ptr<cv::Vec2f>(i);
455
456 for (int j = 0; j < cols; j++) {
457 if (detection_ptr[j] == 0) continue; // Skip invalid pixels
458
459 ObjectId object_id = motion_ptr[j];
460
461 // skip if this object does not need to be sampled
462 if (objects_to_sample.find(object_id) == objects_to_sample.end()) {
463 continue;
464 }
465
466 if (object_id == background_label) continue;
467
468 double flow_xe = static_cast<double>(flow_ptr[j][0]);
469 double flow_ye = static_cast<double>(flow_ptr[j][1]);
470 if (flow_xe == 0 || flow_ye == 0) {
471 const std::lock_guard<std::mutex> lock(mutex);
472 info_.getObjectStatus(object_id).num_zero_flow++;
473 continue;
474 }
475
476 OpticalFlow flow(flow_xe, flow_ye);
477 Keypoint keypoint(j, i);
478 Keypoint predicted_kp =
479 Feature::CalculatePredictedKeypoint(keypoint, flow);
480
481 if (isWithinShrunkenImage(keypoint)) {
482 int cache_index = i * cols + j;
483
484 // Directly assign instead of creating a new object
485 cached_keypoint_data[cache_index] = {flow, predicted_kp};
486
487 KeypointCV opencv_keypoint = utils::gtsamPointToKeyPoint(keypoint);
488 opencv_keypoint.class_id = cache_index;
489
490 tbb::concurrent_hash_map<ObjectId, KeypointsCV>::accessor acc;
491 if (sampled_keypoints.insert(acc, object_id)) {
492 acc->second = KeypointsCV{}; // Initialize with an empty vector
493 }
494 acc->second.push_back(opencv_keypoint); // Add keypoint safely
495
496 // is this going to be slow?
497 // need the statusObject to exit by the time we get to the next tbb loop
498 // so we can get the number of tracks
499 const std::lock_guard<std::mutex> lock(mutex);
500 info_.getObjectStatus(object_id).num_sampled++;
501 // object_tracking_info.num_sampled++;
502
503 } else {
504 // const std::lock_guard<std::mutex> lock(mutex);
505 // PerObjectStatus& object_tracking_info =
506 // info_.getObjectStatus(object_id);
507 // object_tracking_info.num_outside_shrunken_image++;
508 }
509 }
510 });
511
512 const int& max_features_to_track = params_.max_dynamic_features_per_frame;
513 static constexpr float tolerance = 0.01;
514 Eigen::MatrixXd binning_mask;
515
516 // for(const auto& [object_id, opencv_keypoints] : sampled_keypoints) {
517 tbb::parallel_for_each(
518 sampled_keypoints.begin(), sampled_keypoints.end(), [&](auto& entry) {
519 auto& [object_id, opencv_keypoints] = entry;
520
521 const PerObjectStatus& object_tracking_info =
522 info_.getObjectStatus(object_id);
523 const int& number_tracked = object_tracking_info.num_track;
524
525 int nr_corners_needed =
526 std::max(max_features_to_track - number_tracked, 0);
527
528 std::vector<KeypointCV>& max_keypoints = opencv_keypoints;
529
530 const size_t sampled_size = max_keypoints.size();
531
532 // TODO: Ssc better but maybe bad alloc????
533 AdaptiveNonMaximumSuppression non_maximum_supression(
534 AnmsAlgorithmType::RangeTree);
535 max_keypoints = non_maximum_supression.suppressNonMax(
536 opencv_keypoints, nr_corners_needed, tolerance, img_size_.width,
537 img_size_.height, 5, 5, binning_mask);
538
539 VLOG(10) << "Kps: " << max_keypoints.size() << " for j=" << object_id
540 << " after ANMS (originally " << sampled_size << ")";
541 {
542 const std::lock_guard<std::mutex> lock(mutex);
543 info_.getObjectStatus(object_id).num_sampled = max_keypoints.size();
544 }
545
546 for (const KeypointCV& cv_keypoint : max_keypoints) {
547 Keypoint keypoint = utils::cvKeypointToGtsam(cv_keypoint);
548 int cache_index = cv_keypoint.class_id;
549 // recover cached data
550 const KeypointData& cached_data = cached_keypoint_data[cache_index];
551
552 CHECK(isWithinShrunkenImage(keypoint));
553 TrackletId tracklet_id;
554 {
555 const std::lock_guard<std::mutex> lock(mutex);
556 tracklet_id = tracked_id_manager.getAndIncrementTrackletId();
557 }
558
559 Feature::Ptr feature = std::make_shared<Feature>();
560 (*feature)
561 .objectId(object_id)
562 .frameId(frame_id)
563 .keypointType(KeyPointType::DYNAMIC)
564 .age(0)
565 .trackletId(tracklet_id)
566 .keypoint(keypoint)
567 .measuredFlow(cached_data.flow)
568 .predictedKeypoint(cached_data.predicted_kp);
569
570 {
571 const std::lock_guard<std::mutex> lock(mutex);
572 dynamic_features.add(feature);
573 objects_sampled.insert(feature->objectId());
574 }
575 }
576 });
577}

579void FeatureTracker::requiresSampling(
580 std::set<ObjectId>& objects_to_sample, const FeatureTrackerInfo& info,
581 const ImageContainer& image_container,
582 const gtsam::FastMap<ObjectId, FeatureContainer>& features_per_object,
583 const vision_tools::ObjectBoundaryMaskResult& boundary_mask_result,
584 const cv::Mat& dynamic_tracking_mask) const {
585 VLOG(20) << "Starting sampling check";
586 ObjectIds detected_objects =
587 vision_tools::getObjectLabels(image_container.objectMotionMask());
588
589 {
590 // sanity check assert
591 CHECK(equals_with_abs_tol(detected_objects,
592 boundary_mask_result.objects_detected))
593 << "Explicit detected objects " << container_to_string(detected_objects)
594 << " != boundary mask result: "
595 << container_to_string(boundary_mask_result.objects_detected)
596 << " this could happen if the object mask changes dramatically...!!";
597 }
598
599 if (!previous_frame_) {
600 if (!detected_objects.empty()) {
601 VLOG(5) << "All objects sampled as first frame";
602 objects_to_sample.insert(detected_objects.begin(),
603 detected_objects.end());
604 }
605 return;
606 }
607
608 const int& max_dynamic_point_age = params_.max_dynamic_feature_age;
609 // bascially how early we want to retrack points based on their expiry
610 // it takes a few frames for the feature to end up in the backend (ie. at
611 // least twice, to ensure a valid track) so we want to track new points
612 // earlier than that to ensure we dont have a frame with NO points
613 const auto& age_buffer = std::max(3, params_.dynamic_feature_age_buffer);
614 const auto& min_dynamic_tracks = params_.min_dynamic_tracks;
615 const auto& min_iou = params_.min_dynamic_mask_iou;
616 const size_t expiry_age =
617 static_cast<size_t>(max_dynamic_point_age - age_buffer);
618 CHECK_GT(expiry_age, 0u);
619
620 for (size_t i = 0; i < detected_objects.size(); i++) {
621 ObjectId object_id = detected_objects.at(i);
622
623 // object is tracked and therefore should exist in the previous frame!
624 if (info.dynamic_track.exists(object_id)) {
625 const auto& per_object_status = info.dynamic_track.at(object_id);
626
627 if (!features_per_object.exists(object_id)) {
628 LOG(WARNING) << "Object " << object_id
629 << " found in mask and info at k=" << info.frame_id
630 << " but missing tracked features. Skipping...";
631 continue;
632 }
633 const FeatureContainer& per_object_tracks =
634 features_per_object.at(object_id);
635 size_t num_tracked = per_object_status.num_track;
636 CHECK_EQ(num_tracked, per_object_tracks.size());
637 // if more than 80% of points on the object are going to expire within the
638 // next (at least 3) frames
639 size_t are_geriatric = 0u;
640 for (const auto& feature : per_object_tracks) {
641 size_t age = feature->age();
642 if (age > expiry_age) {
643 are_geriatric++;
644 }
645 }
646 const bool many_old_points =
647 (double)are_geriatric / (double)num_tracked > 0.8;
648 // if we have less than N tracks
649 const bool too_few_tracks = num_tracked < min_dynamic_tracks;
650 // eventually also area based tings
651
652 // bounding box of the whole mask, representing the object detected in the
653 // actual image
654 const cv::Rect& detection_bb =
655 boundary_mask_result.inner_boarder_object_bounding_boxes.at(i);
656
657 // bounding box of the tracked feature points on the object
658 cv::Rect tracked_bb =
659 cv::boundingRect(per_object_tracks.toOpenCV(nullptr, true));
660
661 double iou = utils::calculateIoU(detection_bb, tracked_bb);
662 const bool small_iou = iou < min_iou;
663
664 const bool needs_sampling =
665 many_old_points || too_few_tracks || small_iou;
666
667 if (needs_sampling) {
668 objects_to_sample.insert(object_id);
669
670 VLOG(5) << "Object " << info_string(info.frame_id, object_id)
671 << " requires sampling";
672
673 VLOG_IF(5, many_old_points) << "Sampling reason: too many old points";
674 VLOG_IF(5, too_few_tracks) << "Sampling reason: too few points";
675 VLOG_IF(5, small_iou) << "Sampling reason: IoU too small";
676 }
677 } else {
678 objects_to_sample.insert(object_id);
679 VLOG(5) << "Object " << info_string(info.frame_id, object_id)
680 << " requires sampling. Sampling reason: new object";
681 }
682 }

705 std::sort(instance_labels.begin(), instance_labels.end());
706 instance_labels.erase(
707 std::unique(instance_labels.begin(), instance_labels.end()),
708 instance_labels.end());

710 // the tracklet id associated with that label
711 std::vector<TrackletIds> object_features(instance_labels.size());

767 label_duplicates.at(k)++;

800 const double flow_xe =

822 functional_keypoint::u(predicted_kp)) =
823 instance_labels[i];
