49Frame::Ptr FeatureTracker::track(FrameId frame_id, Timestamp timestamp, const ImageContainer& image_container) {
50 //take "copy" of tracking_images which is then given to the frame
51 //this will mean that the tracking images (input) are not necessarily the same as the ones inside the returned frame
52 ImageContainer input_images = image_container;
53
54 info_ = FeatureTrackerInfo(); //clear the info
55 info_.frame_id = frame_id;
56 info_.timestamp = timestamp;
57
58
59 if(initial_computation_) {
60 //intitial computation
61 const cv::Size& other_size = input_images.get<ImageType::RGBMono>().size();
62 CHECK(!previous_frame_);
63 CHECK_EQ(img_size_.width, other_size.width);
64 CHECK_EQ(img_size_.height, other_size.height);
65 initial_computation_ = false;
66 }
67 else {
68
69 if(params_.use_propogate_mask) {
70 propogateMask(input_images);
71 }
72 CHECK(previous_frame_);
73 CHECK_EQ(previous_frame_->frame_id_, frame_id - 1u) << "Incoming frame id must be consequative";
74 }
75
76 //TODO: figure out some better way of scaling this as it scales with the size of the image...
77 //TODO: and make parameter...
78 static constexpr auto kDetectionBoaderThickness = 10; //in pixels
79 //create detection mask around the boarder of each dynamic object with some thickness
80 //this prevents static and dynamic points being detected around the edge of the dynamic object
81 //as there are lots of inconsistencies here
82 //the detection mask is in the opencv mask form: CV_8UC1 where white pixels (255) are valid
83 //and black pixels (0) should not be detected on
84 cv::Mat boarder_detection_mask;
85 vision_tools::computeObjectMaskBoundaryMask(
86 input_images.get<ImageType::MotionMask>(),
87 boarder_detection_mask,
88 kDetectionBoaderThickness,
89 true
90 );
91
92
93 FeatureContainer static_features;
94 {
95 utils::TimingStatsCollector static_track_timer("static_feature_track");
96 static_features = static_feature_tracker_->trackStatic(previous_frame_, input_images, info_, boarder_detection_mask);
97 }
98
99 FeatureContainer dynamic_features;
100 {
101 utils::TimingStatsCollector dynamic_track_timer("dynamic_feature_track");
102 trackDynamic(
103 frame_id,
104 input_images,
105 dynamic_features,
106 boarder_detection_mask);
107 }
108
109 previous_tracked_frame_ = previous_frame_; // Update previous frame (previous to the newly created frame)
110
111
112 auto new_frame = std::make_shared<Frame>(
113 frame_id,
114 timestamp,
115 camera_,
116 input_images,
117 static_features,
118 dynamic_features,
119 info_);
120
121 //update depth threshold information
122 new_frame->setMaxBackgroundDepth(frontend_params_.max_background_depth);
123 new_frame->setMaxObjectDepth(frontend_params_.max_object_depth);
124
125
126 LOG(INFO) << "Tracked on frame " << frame_id << " t= " << std::setprecision(15) << timestamp << ", object ids " << container_to_string(new_frame->getObjectIds());
127 previous_frame_ = new_frame;
128
129 return new_frame;
130
131}

137void FeatureTracker::trackDynamic(FrameId frame_id, const ImageContainer& image_container, FeatureContainer& dynamic_features, const cv::Mat& detection_mask) {
138 // first dectect dynamic points
139 const cv::Mat& rgb = image_container.get<ImageType::RGBMono>();
140 //flow is going to take us from THIS frame to the next frame (which does not make sense for a realtime system)
141 const cv::Mat& flow = image_container.get<ImageType::OpticalFlow>();
142 const cv::Mat& motion_mask = image_container.get<ImageType::MotionMask>();
143
144
145 TrackletIdManager& tracked_id_manager = TrackletIdManager::instance();
146
147 ObjectIds instance_labels;
148 dynamic_features.clear();
149
150 const auto step_size = params_.semantic_mask_step_size;
151
152 OccupandyGrid2D grid(step_size,
153 std::ceil(static_cast<double>(img_size_.width)/step_size),
154 std::ceil(static_cast<double>(img_size_.height)/step_size));
155
156
157 if (previous_frame_)
158 {
159 const cv::Mat& previous_motion_mask = previous_frame_->image_container_.get<ImageType::MotionMask>();
160 utils::TimingStatsCollector tracked_dynamic_features("tracked_dynamic_features");
161 for (Feature::Ptr previous_dynamic_feature : previous_frame_->usableDynamicFeaturesBegin())
162 {
163 const TrackletId tracklet_id = previous_dynamic_feature->trackletId();
164 const size_t age = previous_dynamic_feature->age();
165
166 const Keypoint kp = previous_dynamic_feature->predictedKeypoint();
167 const int x = functional_keypoint::u(kp);
168 const int y = functional_keypoint::v(kp);
169 // ObjectId predicted_label = functional_keypoint::at<ObjectId>(kp, motion_mask);
170 const ObjectId predicted_label = motion_mask.at<ObjectId>(y,x);
171
172
173 if(!detection_mask.empty()) {
174 const unsigned char valid_detection = detection_mask.at<unsigned char>(y,x);
175 if(valid_detection == 0) {
176 continue;
177 }
178 }
179
180 // const Keypoint previous_kp = previous_dynamic_feature->keypoint();
181 // ObjectId previous_label = functional_keypoint::at<ObjectId>(previous_kp, previous_motion_mask);
182 ObjectId previous_label = previous_dynamic_feature->objectId();
183 CHECK_NE(previous_label, background_label);
184 CHECK_GT(previous_label, 0);
185
186
187 PerObjectStatus& object_tracking_info = info_.getObjectStatus(predicted_label);
188 object_tracking_info.num_previous_track++;
189
190 //true if predicted label not on the background
191 const bool is_predicted_object_label = predicted_label != background_label;
192 //true if predicted label the same as the previous label of the tracked point
193 const bool is_precited_same_as_previous = predicted_label == previous_label;
194
195 //update stats
196 if(!is_predicted_object_label) object_tracking_info.num_tracked_with_background_label++;
197 if(!is_precited_same_as_previous) object_tracking_info.num_tracked_with_different_label++;
198
199
200 //only include point if it is contained, it is not static and the previous label is the same as the predicted label
201 if(camera_->isKeypointContained(kp) && is_predicted_object_label && is_precited_same_as_previous) {
202 size_t new_age = age + 1;
203 double flow_xe = static_cast<double>(flow.at<cv::Vec2f>(y, x)[0]);
204 double flow_ye = static_cast<double>(flow.at<cv::Vec2f>(y, x)[1]);
205
206 OpticalFlow flow(flow_xe, flow_ye);
207 const Keypoint predicted_kp = Feature::CalculatePredictedKeypoint(kp, flow);
208
209 if(!isWithinShrunkenImage(predicted_kp)) {
210 object_tracking_info.num_outside_shrunken_image++;
211 continue;
212 }
213
214 if(flow_xe == 0 || flow_ye == 0) {
215 object_tracking_info.num_zero_flow++;
216 continue;
217 }
218
219 //limit point tracking of a certain age
220 TrackletId tracklet_to_use = tracklet_id;
221 if(new_age > 25u) {
222 tracklet_to_use = tracked_id_manager.getTrackletIdCount();
223 tracked_id_manager.incrementTrackletIdCount();
224 new_age = 0;
225 }
226
227 // // save correspondences
228 Feature::Ptr feature = std::make_shared<Feature>();
229 (*feature)
230 .objectId(predicted_label)
231 .frameId(frame_id)
232 .keypointType(KeyPointType::DYNAMIC)
233 .age(new_age)
234 .trackletId(tracklet_to_use)
235 .keypoint(kp)
236 .measuredFlow(flow)
237 .predictedKeypoint(predicted_kp);
238
239
240 dynamic_features.add(feature);
241 instance_labels.push_back(feature->objectId());
242
243 object_tracking_info.num_track++;
244
245 const size_t cell_idx = grid.getCellIndex(kp);
246 grid.occupancy_[cell_idx] = true;
247 }
248 }
249 }
250
251 int step = params_.semantic_mask_step_size;
252 for (int i = 0; i < rgb.rows - step; i = i + step)
253 {
254 for (int j = 0; j < rgb.cols - step; j = j + step)
255 {
256
257 if(!detection_mask.empty()) {
258 const unsigned char valid_detection = detection_mask.at<unsigned char>(i, j);
259 // //marked invalid by the detection mask
260 if(valid_detection == 0) {
261 continue;
262 }
263 }
264
265 const ObjectId label = motion_mask.at<ObjectId>(i, j);
266
267 if (label == background_label)
268 {
269 continue;
270 }
271
272
273 PerObjectStatus& object_tracking_info = info_.getObjectStatus(label);
274
275
276 double flow_xe = static_cast<double>(flow.at<cv::Vec2f>(i, j)[0]);
277 double flow_ye = static_cast<double>(flow.at<cv::Vec2f>(i, j)[1]);
278
279 //TODO: close to zero?
280 if(flow_xe == 0 || flow_ye == 0) {
281 object_tracking_info.num_zero_flow++;
282 continue;
283 }
284
285 OpticalFlow flow(flow_xe, flow_ye);
286 Keypoint keypoint(j, i);
287 const Keypoint predicted_kp = Feature::CalculatePredictedKeypoint(keypoint, flow);
288 const size_t cell_idx = grid.getCellIndex(keypoint);
289
290 // //TODO: this is a problem for the omd dataset?
291 // if(grid.isOccupied(cell_idx)) {continue;}
292
293 // if ((predicted_kp(0) < rgb.cols && predicted_kp(0) > 0 && predicted_kp(1) < rgb.rows && predicted_kp(1) > 0))
294 if(isWithinShrunkenImage(keypoint) && !grid.isOccupied(cell_idx))
295 {
296 // save correspondences
297 auto tracklet_id = tracked_id_manager.getTrackletIdCount();
298 tracked_id_manager.incrementTrackletIdCount();
299 Feature::Ptr feature = std::make_shared<Feature>();
300 (*feature)
301 .objectId(label)
302 .frameId(frame_id)
303 .keypointType(KeyPointType::DYNAMIC)
304 .age(0)
305 .trackletId(tracklet_id)
306 .keypoint(keypoint)
307 .measuredFlow(flow)
308 .predictedKeypoint(predicted_kp);
309
310
311 dynamic_features.add(feature);
312 instance_labels.push_back(feature->objectId());
313
314 object_tracking_info.num_sampled++;
315 }
316 else {
317 object_tracking_info.num_outside_shrunken_image++;
318 }
319 }
320 }
321
322 //bit of a hack -> make sure we have enough features on each object (around 100)
323 //for no reason other than there is weird behaviour when a car disappears (or is disappearing)
324 //usually becuase the depth is so bad and so the object jumps
325 //of course this number depends on the size of the frame etc etc
326 //this is just a tempory fix!!
327 // ObjectIds objects_removed;
328 // for(const auto& [object_id, status ]:info_.dynamic_track) {
329 // //cannot jus use the num tracked as on the first frame this will be zero
330 // //should track object age??
331 // if(status.num_track + status.num_sampled < 100) {
332 // VLOG(15) << "Removing object id " << object_id << " from tracks as not enough features! (" << (status.num_track + status.num_sampled) << " < 100)";
333 // dynamic_features.removeByObjectId(object_id);
334 // objects_removed.push_back(object_id);
335 // }
336 // }
337
338 // //remove from info?
339 // for (auto object_id : objects_removed) {
340 // info_.dynamic_track.erase(object_id);
341 // }

347void FeatureTracker::propogateMask(ImageContainer& image_container) {
348 if(!previous_frame_) return;
349
350
351 const cv::Mat& previous_rgb = previous_frame_->image_container_.get<ImageType::RGBMono>();
352 const cv::Mat& previous_mask = previous_frame_->image_container_.get<ImageType::MotionMask>();
353 const cv::Mat& previous_flow = previous_frame_->image_container_.get<ImageType::OpticalFlow>();
354
355 // note reference
356 cv::Mat& current_mask = image_container.get<ImageType::MotionMask>();
357
358 ObjectIds instance_labels;
359 for(const Feature::Ptr& dynamic_feature : previous_frame_->usableDynamicFeaturesBegin()) {
360 CHECK(dynamic_feature->objectId() != background_label);
361 instance_labels.push_back(dynamic_feature->objectId());
362 }
363
364 CHECK_EQ(instance_labels.size(), previous_frame_->numDynamicUsableFeatures());
365 std::sort(instance_labels.begin(), instance_labels.end());
366 instance_labels.erase(std::unique(instance_labels.begin(), instance_labels.end()), instance_labels.end());
367 //each row is correlated with a specific instance label and each column is the tracklet id associated with that label
368 std::vector<TrackletIds> object_features(instance_labels.size());
369
370 // collect the predicted labels and semantic labels in vector
371
372 //TODO: inliers?
373 for (const Feature::Ptr& dynamic_feature : previous_frame_->usableDynamicFeaturesBegin())
374 {
375 CHECK(Feature::IsNotNull(dynamic_feature));
376 for (size_t j = 0; j < instance_labels.size(); j++)
377 {
378 // save object label for object j with feature i
379 if (dynamic_feature->objectId() == instance_labels[j])
380 {
381 object_features[j].push_back(dynamic_feature->trackletId());
382 CHECK(dynamic_feature->objectId() != background_label);
383 break;
384 }
385 }
386 }
387
388 // check each object label distribution in the coming frame
389 for (size_t i = 0; i < object_features.size(); i++)
390 {
391 //labels at the current mask using the predicted keypoint from the previous frame
392 //each iteration is per label so temp_label should correspond to features within the same object
393 ObjectIds temp_label;
394 for (size_t j = 0; j < object_features[i].size(); j++)
395 {
396 //feature at k-1
397 Feature::Ptr feature = previous_frame_->dynamic_features_.getByTrackletId(object_features[i][j]);
398 CHECK(Feature::IsNotNull(feature));
399 //kp at k
400 const Keypoint& predicted_kp = feature->predictedKeypoint();
401 const int u = functional_keypoint::u(predicted_kp);
402 const int v = functional_keypoint::v(predicted_kp);
403 // ensure u and v are sitll inside the CURRENT frame
404 if (u < previous_rgb.cols && u > 0 && v < previous_rgb.rows && v > 0)
405 {
406 //add instance label at predicted keypoint
407 temp_label.push_back(current_mask.at<ObjectId>(v, u));
408 }
409 }
410
411 //this is a lovely magic number inherited from some old code :)
412 if (temp_label.size() < 150)
413 {
414 LOG(WARNING) << "not enoug points to track object " << instance_labels[i] << " points size - "
415 << temp_label.size();
416 //TODO:mark has static!!???
417 continue;
418 }
419
420 // find label that appears most in LabTmp()
421 // (1) count duplicates
422 std::map<int, int> label_duplicates;
423 //k is object label
424 for (int k : temp_label)
425 {
426 if (label_duplicates.find(k) == label_duplicates.end())
427 {
428 label_duplicates.insert({ k, 0 });
429 }
430 else
431 {
432 label_duplicates.at(k)++;
433 }
434 }
435 // (2) and sort them by descending order by number of times an object appeared (ie. by pair.second)
436 std::vector<std::pair<int, int>> sorted;
437 for (auto k : label_duplicates)
438 {
439 sorted.push_back(std::make_pair(k.first, k.second));
440 }
441
442 auto sort_pair_int = [](const std::pair<int, int>& a, const std::pair<int, int>& b) -> bool {
443 return (a.second > b.second);
444 };
445 std::sort(sorted.begin(), sorted.end(), sort_pair_int);
446
447 // recover the missing mask (time consuming!)
448 // LOG(INFO) << sorted[0].first << " " << sorted[0].second << " " << instance_labels[i];
449 // if (sorted[0].second < 30)
450 // {
451 // LOG(WARNING) << "not enoug points to track object " << instance_labels[i] << " points size - "
452 // << sorted[0].second;
453 // //TODO:mark has static!!
454 // continue;
455 // }
456 if (sorted[0].first == 0) //?
457 // if (sorted[0].first == instance_labels[i]) //?
458 {
459 for (int j = 0; j < previous_rgb.rows; j++)
460 {
461 for (int k = 0; k < previous_rgb.cols; k++)
462 {
463 if (previous_mask.at<ObjectId>(j, k) == instance_labels[i])
464 {
465 const double flow_xe = static_cast<double>(previous_flow.at<cv::Vec2f>(j, k)[0]);
466 const double flow_ye = static_cast<double>(previous_flow.at<cv::Vec2f>(j, k)[1]);
467
468 if(flow_xe == 0 || flow_ye == 0) {
469 continue;
470 }
471
472 OpticalFlow flow(flow_xe, flow_ye);
473 //x, y
474 Keypoint kp(k, j);
475 const Keypoint predicted_kp = Feature::CalculatePredictedKeypoint(kp, flow);
476
477 if(!isWithinShrunkenImage(predicted_kp)) {
478 continue;
479 }
480
481 if ((predicted_kp(0) < previous_rgb.cols && predicted_kp(0) > 0 && predicted_kp(1) < previous_rgb.rows && predicted_kp(1) > 0))
482 {
483 current_mask.at<ObjectId>(functional_keypoint::v(predicted_kp), functional_keypoint::u(predicted_kp)) = instance_labels[i];
484 // current_rgb
485 // updated_mask_points++;
486 }
487 }
488 }
489 }
490 }
491 }
492}
