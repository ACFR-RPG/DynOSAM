71 void dynamicPointUpdateCallback(

79
80 inline bool isDynamicTrackletInMap(
81 const typename MapTraitsType::LandmarkNodePtr& lmk_node) const override {
82 const TrackletId tracklet_id = lmk_node->tracklet_id;
83 return is_dynamic_tracklet_in_map_.exists(tracklet_id);
84 }
85
86 protected:
87 AccessorTypePointer createAccessor(
88 const SharedFormulationData& shared_data) const override {
89 return std::make_shared<WorldPoseAccessor>(shared_data, this->map());
90 }
91
92 std::string loggerPrefix() const override { return "wcpe"; }
93
94 protected:
95 // we need a separate way of tracking if a dynamic tracklet is in the map,
96 // since each point is modelled uniquely simply used as an O(1) lookup, the
97 // value is not actually used. If the key exists, we assume that the tracklet
98 // is in the map
99 gtsam::FastMap<TrackletId, bool>
100 is_dynamic_tracklet_in_map_;
104};
