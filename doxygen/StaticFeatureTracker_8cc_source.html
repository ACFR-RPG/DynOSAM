68// if previous frame is null, assume that it is the first frame... and that
69// frames are always processed in order!
70FeatureContainer ExternalFlowFeatureTracker::trackStatic(
71 Frame::Ptr previous_frame, const ImageContainer& image_container,
72 FeatureTrackerInfo& tracker_info, const cv::Mat&,
73 const std::optional<gtsam::Rot3>&) {
74 const ImageWrapper<ImageType::RGBMono>& rgb_wrapper = image_container.rgb();
75 const cv::Mat& rgb = rgb_wrapper.toRGB();
76 cv::Mat mono = ImageType::RGBMono::toMono(rgb_wrapper);
77 CHECK(!mono.empty());
78
79 const cv::Mat& motion_mask = image_container.objectMotionMask();
80 CHECK(!motion_mask.empty());
81
82 cv::Mat descriptors;
83 KeypointsCV detected_keypoints;
84 (*orb_detector_)(mono, cv::Mat(), detected_keypoints, descriptors);
85
86 // assign tracked features to grid and add to static features
87 FeatureContainer static_features;
88
89 const size_t& min_tracks =
90 static_cast<size_t>(params_.max_features_per_frame);
91 const FrameId frame_k = image_container.frameId();
92
93 // appy tracking (ie get correspondences)
94 // TODO: only track frames that have been tracked for some time?
95 if (previous_frame) {
96 // TODO: for now assume consequative frames
97 const FrameId frame_k_1 = previous_frame->getFrameId();
98 CHECK_EQ(frame_k_1 + 1u, frame_k);
99
100 for (Feature::Ptr previous_feature : previous_frame->static_features_) {
101 const size_t tracklet_id = previous_feature->trackletId();
102 const size_t age = previous_feature->age();
103 const Keypoint kp = previous_feature->predictedKeypoint();
104
105 // check kp contained before we do a static grid look up to ensure we
106 // don't go out of bounds
107 if (!camera_->isKeypointContained(kp)) {
108 continue;
109 }
110
111 const int x = functional_keypoint::u(kp);
112 const int y = functional_keypoint::v(kp);
113 const size_t cell_idx = static_grid_.getCellIndex(kp);
114 const ObjectId instance_label = motion_mask.at<ObjectId>(y, x);
115
116 if (static_grid_.isOccupied(cell_idx)) continue;
117
118 if (previous_feature->usable() && instance_label == background_label) {
119 size_t new_age = age + 1;
120 Feature::Ptr feature = constructStaticFeature(
121 image_container, kp, new_age, tracklet_id, frame_k);
122 if (feature) {
123 static_features.add(feature);
124 static_grid_.occupancy_[cell_idx] = true;
125 }
126 }
127 }
128 }
129
130 // number features tracked with optical flow
131 const auto n_optical_flow = static_features.size();
132 tracker_info.static_track_optical_flow = n_optical_flow;
133
134 TrackletIdManager& tracked_id_manager = TrackletIdManager::instance();
135
136 if (static_features.size() < min_tracks) {
137 // iterate over new observations
138 for (size_t i = 0; i < detected_keypoints.size(); i++) {
139 if (static_features.size() >= min_tracks) {
140 break;
141 }
142
143 const KeypointCV& kp_cv = detected_keypoints[i];
144 const int& x = kp_cv.pt.x;
145 const int& y = kp_cv.pt.y;
146
147 // if not already tracked with optical flow
148 if (motion_mask.at<int>(y, x) != background_label) {
149 continue;
150 }
151
152 Keypoint kp(x, y);
153 const size_t cell_idx = static_grid_.getCellIndex(kp);
154 if (!static_grid_.isOccupied(cell_idx)) {
155 const size_t age = 0;
156 size_t tracklet_id = tracked_id_manager.getTrackletIdCount();
157 Feature::Ptr feature = constructStaticFeature(image_container, kp, age,
158 tracklet_id, frame_k);
159 if (feature) {
160 tracked_id_manager.incrementTrackletIdCount();
161 static_grid_.occupancy_[cell_idx] = true;
162 static_features.add(feature);
163 }
164 }
165 }
166 }
167
168 static_grid_.reset();
169
170 size_t total_tracks = static_features.size();
171 tracker_info.static_track_detections = total_tracks - n_optical_flow;
172 return static_features;
173}

175Feature::Ptr ExternalFlowFeatureTracker::constructStaticFeature(
176 const ImageContainer& image_container, const Keypoint& kp, size_t age,
177 TrackletId tracklet_id, FrameId frame_id) const {
178 // implicit double -> int cast for pixel location
179 const int x = functional_keypoint::u(kp);
180 const int y = functional_keypoint::v(kp);
181
182 const cv::Mat& rgb = image_container.rgb();
183 const cv::Mat& motion_mask = image_container.objectMotionMask();
184 const cv::Mat& optical_flow = image_container.opticalFlow();
185
186 CHECK(!optical_flow.empty());
187 CHECK(!motion_mask.empty());
188
189 if (motion_mask.at<int>(y, x) != background_label) {
190 return nullptr;
191 }
192
193 // check flow
194 double flow_xe = static_cast<double>(optical_flow.at<cv::Vec2f>(y, x)[0]);
195 double flow_ye = static_cast<double>(optical_flow.at<cv::Vec2f>(y, x)[1]);
196
197 if (!(flow_xe != 0 && flow_ye != 0)) {
198 return nullptr;
199 }
200
201 OpticalFlow flow(flow_xe, flow_ye);
202
203 // check predicted flow is within image
204 Keypoint predicted_kp = Feature::CalculatePredictedKeypoint(kp, flow);
205 if (!camera_->isKeypointContained(predicted_kp)) {
206 return nullptr;
207 }
208
209 Feature::Ptr feature = std::make_shared<Feature>();
210 (*feature)
211 .objectId(background_label)
212 .frameId(frame_id)
213 .keypointType(KeyPointType::STATIC)
214 .age(age)
215 .trackletId(tracklet_id)
216 .keypoint(kp)
217 .measuredFlow(flow)
218 .predictedKeypoint(predicted_kp);
219 return feature;
220}

244FeatureContainer KltFeatureTracker::trackStatic(
245 Frame::Ptr previous_frame, const ImageContainer& image_container,
246 FeatureTrackerInfo& tracker_info, const cv::Mat& detection_mask,
247 const std::optional<gtsam::Rot3>& R_km1_k) {
248 // tracked features and new features
249 FeatureContainer new_tracks_and_detections;
250
251 cv::Mat current_equialized_greyscale;
252 equalizeImage(image_container, current_equialized_greyscale);
253
254 if (!previous_frame) {
255 FeatureContainer previous_inliers;
256 detectFeatures(current_equialized_greyscale, image_container,
257 previous_inliers, new_tracks_and_detections, detection_mask);
258
259 tracker_info.static_track_detections = new_tracks_and_detections.size();
260
261 return new_tracks_and_detections;
262 } else {
263 // we have previous tracks
264 // we should have already calculated the processed rgb image from the
265 // previous frame
266 cv::Mat previous_equialized_greyscale;
267 equalizeImage(previous_frame->image_container_,
268 previous_equialized_greyscale);
269
270 FeatureContainer previous_inliers;
271 auto iter = previous_frame->static_features_.beginUsable();
272 for (const auto& inlier_feature : iter) {
273 previous_inliers.add(inlier_feature);
274 }
275
276 // if we dont actually have any previous tracks
277 // this may be in cases where we have an IMU and so we have some odometry
278 // but no feature tracks from the previous frame!
279 if (previous_inliers.empty()) {
280 FeatureContainer previous_inliers;
281 detectFeatures(current_equialized_greyscale, image_container,
282 previous_inliers, new_tracks_and_detections,
283 detection_mask);
284 tracker_info.static_track_detections = new_tracks_and_detections.size();
285 return new_tracks_and_detections;
286 }
287
288 // Tracklet ids associated with the set of previous inliers that are now
289 // outliers
290 TrackletIds previous_outliers;
291
292 // track features from the previous frame and detect new ones if necessary
293 CHECK(trackPoints(
294 current_equialized_greyscale, previous_equialized_greyscale,
295 image_container, previous_inliers, new_tracks_and_detections,
296 previous_outliers, tracker_info, detection_mask, R_km1_k));
297
298 // after tracking, mark features in the older frame as outliers
299 // TODO: (jesse) actually not sure we HAVE to do this, but better to keep
300 // things as consisent as possible
301 previous_frame->static_features_.markOutliers(previous_outliers);
302
303 return new_tracks_and_detections;
304 }

330bool KltFeatureTracker::detectFeatures(const cv::Mat& processed_img,
331 const ImageContainer& image_container,
332 const FeatureContainer& current_features,
333 FeatureContainer& new_features,
334 const cv::Mat& detection_mask) {
335 const FrameId frame_k = image_container.frameId();
336 const cv::Mat& motion_mask = image_container.objectMotionMask();
337
338 // internal detection mask that is appended with new invalid pixels
339 // this builds the static detection mask over the existing input mask
340 cv::Mat detection_mask_impl;
341 // If we are provided with an external detection/feature mask, initalise the
342 // detection mask with this and add more invalid sections to it
343 if (!detection_mask.empty()) {
344 CHECK_EQ(motion_mask.rows, detection_mask.rows);
345 CHECK_EQ(motion_mask.cols, detection_mask.cols);
346 detection_mask_impl = detection_mask.clone();
347 } else {
348 detection_mask_impl = cv::Mat(motion_mask.size(), CV_8U, cv::Scalar(255));
349 }
350 CHECK_EQ(detection_mask_impl.type(), CV_8U);
351 // create mask from object mask so that all pixels > 0 are ignored (by setting
352 // the value in the new mask to 0 at these locations) start with an invalid
353 // mask
354 cv::Mat object_feature_mask = cv::Mat::zeros(motion_mask.size(), CV_8U);
355 // Set background pixels (0 in motion_mask) -> valid (1 in
356 // object_feature_mask)
357 object_feature_mask.setTo(255, motion_mask == 0);
358 // combine with existing mask information (from current features)
359 cv::bitwise_and(detection_mask_impl, object_feature_mask,
360 detection_mask_impl);
361
362 // slow
363 // add mask over objects detected in the scene
364 // TODO: should just be a masking operation but treating all non-zero pixels
365 // as 1 (ie make binary) and then inveverting the mask so that object pixels
366 // (originally 1) become 0, indicating they should not be used! for (int i =
367 // 0; i < motion_mask.rows; i++) {
368 // for (int j = 0; j < motion_mask.cols; j++) {
369 // const ObjectId label = motion_mask.at<ObjectId>(i, j);
370
371 // if (label != background_label) {
372 // cv::circle(
373 // detection_mask_impl, cv::Point2f(j, i),
374 // params_.min_distance_btw_tracked_and_detected_static_features,
375 // cv::Scalar(0), cv::FILLED);
376 // }
377 // }
378 // }
379
380 // add mask over current static features
381 for (const auto& feature : current_features) {
382 const Keypoint kp = feature->keypoint();
383 CHECK(feature->usable());
384 cv::circle(detection_mask_impl, cv::Point2f(kp(0), kp(1)),
385 params_.min_distance_btw_tracked_and_detected_static_features,
386 cv::Scalar(0), cv::FILLED);
387 }
388
389 std::vector<cv::Point2f> detected_points;
390 {
391 utils::TimingStatsCollector timer("static_feature_track.detect_raw");
392 detected_points = detectRawFeatures(processed_img, current_features.size(),
393 detection_mask_impl);
394 }
395
396 for (const cv::Point2f& detected_point : detected_points) {
397 Keypoint kp(static_cast<double>(detected_point.x),
398 static_cast<double>(detected_point.y));
399 const int x = functional_keypoint::u(kp);
400 const int y = functional_keypoint::v(kp);
401
402 if (!(camera_->isKeypointContained(kp) && isWithinShrunkenImage(kp))) {
403 continue;
404 }
405
406 // with the detection mask this should never happen
407 if (motion_mask.at<int>(y, x) != background_label) {
408 continue;
409 }
410
411 Feature::Ptr feature = constructNewStaticFeature(kp, frame_k);
412 if (feature) {
413 new_features.add(feature);
414 }
415 }
416
417 return true;
418}

420bool KltFeatureTracker::trackPoints(const cv::Mat& current_processed_img,
421 const cv::Mat& previous_processed_img,
422 const ImageContainer& image_container,
423 const FeatureContainer& previous_features,
424 FeatureContainer& tracked_features,
425 TrackletIds& outlier_previous_features,
426 FeatureTrackerInfo& tracker_info,
427 const cv::Mat& detection_mask,
428 const std::optional<gtsam::Rot3>& R_km1_k) {
429 if (current_processed_img.empty() || previous_processed_img.empty() ||
430 previous_features.empty()) {
431 return false;
432 }
433
434 outlier_previous_features.clear();
435
436 const cv::Mat& motion_mask = image_container.objectMotionMask();
437 const FrameId frame_k = image_container.frameId();
438
439 std::vector<uchar> klt_status;
440 std::vector<float> err;
441 // All tracklet ids from the set of previous features to track
442 TrackletIds tracklet_ids;
443
444 // cannot just get inliers (becuase in reality this is)
445 std::vector<cv::Point2f> previous_pts =
446 previous_features.toOpenCV(&tracklet_ids, true);
447 CHECK_EQ(previous_pts.size(), previous_features.size());
448 CHECK_EQ(previous_pts.size(), tracklet_ids.size());
449
450 static const cv::Size klt_window_size(21, 21); // Window size for KLT
451 static const int klt_max_level = 3; // Max pyramid levels for KLT
452 static const cv::TermCriteria klt_criteria = cv::TermCriteria(
453 cv::TermCriteria::EPS | cv::TermCriteria::COUNT, 30, 0.03);
454
455 // used as flags argument for calcOpticalFlowPyrLK - initially starts as
456 // default (0) flag
457 int klt_flags = 0;
458 std::vector<cv::Point2f> current_points;
459 if (R_km1_k) {
460 predictKeypointsGivenRotation(current_points, previous_pts, *R_km1_k);
461 klt_flags = cv::OPTFLOW_USE_INITIAL_FLOW;
462 } else {
463 // as per documentation the vector must have the same size as the input
464 current_points.resize(previous_pts.size());
465 }
466 CHECK_EQ(current_points.size(), previous_pts.size());
467
468 {
469 // utils::TimingStatsCollector timer("static_feature_track.calc_LK");
470 // cv::cuda::GpuMat gpu_prev_img(previous_processed_img);
471 // cv::cuda::GpuMat gpu_current_img(current_processed_img);
472
473 // cv::cuda::GpuMat d_points1(previous_pts); // upload points
474 // cv::cuda::GpuMat d_points2(current_points); // output points
475 // cv::cuda::GpuMat d_status; // status of each point
476 // cv::cuda::GpuMat d_err; // error for each point
477
478 // lk_cuda_tracker_->calc(gpu_prev_img, gpu_current_img, d_points1,
479 // d_points2,
480 // d_status, d_err);
481
482 // // Download results back to CPU
483 // d_points2.download(current_points);
484 // d_status.download(status);
485
486 cv::calcOpticalFlowPyrLK(previous_processed_img, current_processed_img,
487 previous_pts, current_points, klt_status, err,
488 klt_window_size, klt_max_level, klt_criteria,
489 klt_flags);
490
491 // if we used OPTFLOW_USE_INITIAL_FLOW check that we actually got good flow
492 if (klt_flags == cv::OPTFLOW_USE_INITIAL_FLOW) {
493 static constexpr int kMinSuccessTracks = 10;
494 int succ_num = 0;
495 for (size_t i = 0; i < klt_status.size(); i++) {
496 if (klt_status[i]) succ_num++;
497 }
498 if (succ_num < kMinSuccessTracks) {
499 LOG(WARNING) << "Using initial flow for KLT tracking failed: only "
500 << succ_num << " tracked!";
501 cv::calcOpticalFlowPyrLK(previous_processed_img, current_processed_img,
502 previous_pts, current_points, klt_status, err,
503 klt_window_size, klt_max_level, klt_criteria);
504 }
505 }
506
507 // check flow back
508 std::vector<cv::Point2f> reverse_previous_feature_points = current_points;
509 std::vector<uchar> klt_reverse_status;
510 cv::calcOpticalFlowPyrLK(current_processed_img, previous_processed_img,
511 current_points, reverse_previous_feature_points,
512 klt_reverse_status, err, cv::Size(21, 21), 5);
513 CHECK_EQ(klt_reverse_status.size(), tracklet_ids.size());
514
515 auto distance = [](const cv::Point2f& pt1,
516 const cv::Point2f& pt2) -> float {
517 float dx = pt1.x - pt2.x;
518 float dy = pt1.y - pt2.y;
519 return std::sqrt(dx * dx + dy * dy);
520 };
521 // update klt status based on result from flow
522 for (size_t i = 0; i < klt_status.size(); i++) {
523 const bool both_status_good =
524 klt_status.at(i) && klt_reverse_status.at(i);
525 const bool within_distance =
526 distance(previous_pts.at(i), reverse_previous_feature_points.at(i)) <=
527 0.5;
528
529 if (both_status_good && within_distance) {
530 klt_status.at(i) = 1;
531 } else {
532 klt_status.at(i) = 0;
533 }
534 }
535 }
536
537 CHECK_EQ(previous_pts.size(), current_points.size());
538 CHECK_EQ(klt_status.size(), current_points.size());
539
540 std::vector<cv::Point2f> good_current, good_previous;
541 TrackletIds good_tracklets;
542 // can also look at the err?
543 for (size_t i = 0; i < klt_status.size(); i++) {
544 if (klt_status[i]) {
545 good_current.push_back(current_points.at(i));
546 good_previous.push_back(previous_pts.at(i));
547 good_tracklets.push_back(tracklet_ids.at(i));
548 }
549 }
550
551 // Geometric verification using RANSAC
552 const cv::Mat geometric_verification_mask =
553 geometricVerification(good_previous, good_current);
554 std::vector<cv::Point2f> verified_current, verified_previous;
555 TrackletIds verified_tracklets;
556 for (int i = 0; i < geometric_verification_mask.rows; ++i) {
557 if (geometric_verification_mask.at<uchar>(i)) {
558 verified_current.push_back(good_current.at(i));
559 verified_previous.push_back(good_previous.at(i));
560 verified_tracklets.push_back(good_tracklets.at(i));
561 }
562 }
563
564 CHECK_EQ(verified_tracklets.size(), verified_current.size());
565
566 // add to tracked features
567 for (size_t i = 0; i < verified_tracklets.size(); i++) {
568 TrackletId tracklet_id = verified_tracklets.at(i);
569
570 const Feature::Ptr previous_feature =
571 previous_features.getByTrackletId(tracklet_id);
572 // TODO: check this is the same as the previos kp to guarnatee order?
573
574 CHECK(previous_feature->usable());
575
576 const cv::Point2f kp_cv = verified_current.at(i);
577 Keypoint kp(static_cast<double>(kp_cv.x), static_cast<double>(kp_cv.y));
578
579 const int x = functional_keypoint::u(kp);
580 const int y = functional_keypoint::v(kp);
581
582 if (motion_mask.at<int>(y, x) != background_label) {
583 continue;
584 }
585
586 if (!(camera_->isKeypointContained(kp) && isWithinShrunkenImage(kp))) {
587 continue;
588 }
589 Feature::Ptr feature = constructStaticFeatureFromPrevious(
590 kp, previous_feature, tracklet_id, frame_k);
591 if (feature) {
592 tracked_features.add(feature);
593 }
594 }
595
596 // Get the outliers associated with the previous_features container by taking
597 // the set difference between the verified and total tracklets NOTE: verified
598 // tracklets are not necessary the same as the tracklets in tracked_features
599 // as tracked features may excluse some features (e.g. if not in the shrunken
600 // image) or (will eventually) have new tracklets after a new detection takes
601 // place we just want the set difference between the original features and
602 // ones we KNOW are outliers
603 {
604 utils::TimingStatsCollector timer("static_feature_track.find_outliers");
605 determineOutlierIds(verified_tracklets, tracklet_ids,
606 outlier_previous_features);
607 }
608
609 const auto& n_tracked = tracked_features.size();
610 tracker_info.static_track_optical_flow = n_tracked;
611
612 if (tracked_features.size() <
613 static_cast<size_t>(params_.min_features_per_frame)) {
614 utils::TimingStatsCollector timer("static_feature_track.detect");
615 // if we do not have enough features, detect more on the current image
616 detectFeatures(current_processed_img, image_container, tracked_features,
617 tracked_features, detection_mask);
618
619 const auto n_detected = tracked_features.size() - n_tracked;
620 tracker_info.static_track_detections += n_detected;
621 }
622
623 return true;
624}

626cv::Mat KltFeatureTracker::geometricVerification(
627 const std::vector<cv::Point2f>& good_old,
628 const std::vector<cv::Point2f>& good_new) const {
629 if (good_old.size() >= 4) { // Minimum number of points required for RANSAC
630 cv::Mat mask;
