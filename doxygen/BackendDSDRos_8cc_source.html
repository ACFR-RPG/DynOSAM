39void BackendDSDRos::spinOnce(
40 const BackendOutputPacket::ConstPtr& backend_output) {
41 // publish vo and path
42 constexpr static bool kPublishOdomAsTf = false;
43 this->publishVisualOdometry(backend_output->pose(),
44 backend_output->getTimestamp(), kPublishOdomAsTf);
45 this->publishVisualOdometryPath(backend_output->optimized_camera_poses,
46 backend_output->getTimestamp());
47
48 // publish static cloud
49 this->publishStaticPointCloud(backend_output->static_landmarks,
50 backend_output->pose());
51
52 // publish dynamic cloud
53 this->publishDynamicPointCloud(backend_output->dynamic_landmarks,
54 backend_output->pose());
55
56 const auto& object_motions = backend_output->optimized_object_motions;
57 const auto& object_poses = backend_output->optimized_object_poses;
58
59 // publish objects
60 DSDTransport::Publisher object_poses_publisher = dsd_transport_.addObjectInfo(
61 object_motions, object_poses, params_.world_frame_id,
62 backend_output->getFrameId(), backend_output->getTimestamp());
63 object_poses_publisher.publishObjectOdometry();
64 object_poses_publisher.publishObjectPaths();
65}
