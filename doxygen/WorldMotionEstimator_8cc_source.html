47StateQuery<gtsam::Pose3> WorldMotionAccessor::getObjectPose(
48 FrameId frame_id, ObjectId object_id) const {
49 const auto object_poses = getObjectPoses(frame_id);
50 if (object_poses.exists(object_id)) {
51 return StateQuery<gtsam::Pose3>(ObjectPoseSymbol(object_id, frame_id),
52 object_poses.at(object_id));
53 }
54 return StateQuery<gtsam::Pose3>::InvalidMap();
55}

57EstimateMap<ObjectId, gtsam::Pose3> WorldMotionAccessor::getObjectPoses(
58 FrameId frame_id) const {
59 EstimateMap<ObjectId, gtsam::Pose3> object_poses;
60 // for each object, go over the cached object poses and check if that object
61 // has a pose at the query frame
62 for (const auto& [object_id, pose] :
63 object_pose_cache_.collectByFrame(frame_id)) {
64 object_poses.insert2(object_id, ReferenceFrameValue<gtsam::Pose3>(
65 pose, ReferenceFrame::GLOBAL));
66 }
67 return object_poses;
68}

70void WorldMotionAccessor::refreshPoseCache() {
71 // this is pretty slow!!
72 // update object_pose_cache_ with new values
73 // this means we have to start again at the first frame and update all the
74 // poses!!
75 ObjectPoseMap object_poses;
76 const auto frames = map()->getFrames();
77 auto frame_itr = frames.begin();
78
79 const auto& gt_packet_map = hooks().ground_truth_packets_request();
80
81 // advance itr one so we're now at the second frame
82 std::advance(frame_itr, 1);
83 for (auto itr = frame_itr; itr != frames.end(); itr++) {
84 auto prev_itr = itr;
85 std::advance(prev_itr, -1);
86 CHECK(prev_itr != frames.end());
87
88 const auto [frame_id_k, frame_k_ptr] = *itr;
89 const auto [frame_id_k_1, frame_k_1_ptr] = *prev_itr;
90 CHECK_EQ(frame_id_k_1 + 1, frame_id_k);
91
92 // collect all object centoids from the latest estimate
93 gtsam::FastMap<ObjectId, gtsam::Point3> centroids_k =
94 this->computeObjectCentroids(frame_id_k);
95 gtsam::FastMap<ObjectId, gtsam::Point3> centroids_k_1 =
96 this->computeObjectCentroids(frame_id_k_1);
97 // collect motions from k-1 to k
98 MotionEstimateMap motion_estimates = this->getObjectMotions(frame_id_k);
99
100 // construct centroid vectors in object id order
101 gtsam::Point3Vector object_centroids_k_1, object_centroids_k;
102 // we may not have a pose for every motion, e.g. if there is a new object at
103 // frame k, it wont have a motion yet! if we have a motion we MUST have a
104 // pose at the previous frame!
105 for (const auto& [object_id, _] : motion_estimates) {
106 CHECK(centroids_k.exists(object_id));
107 CHECK(centroids_k_1.exists(object_id));
108
109 object_centroids_k.push_back(centroids_k.at(object_id));
110 object_centroids_k_1.push_back(centroids_k_1.at(object_id));
111 }
112
113 PropogatePoseResult propogation_result;
114 if (FLAGS_init_object_pose_from_gt) {
115 if (!gt_packet_map)
116 LOG(WARNING) << "FLAGS_init_object_pose_from_gt is true but gt_packet "
117 "map not provided!";
118 dyno::propogateObjectPoses(
119 object_poses, motion_estimates, object_centroids_k_1,
120 object_centroids_k, frame_id_k, gt_packet_map, &propogation_result);
121 } else {
122 dyno::propogateObjectPoses(object_poses, motion_estimates,
123 object_centroids_k_1, object_centroids_k,
124 frame_id_k, std::nullopt, &propogation_result);
125 }
126
127 if (VLOG_IS_ON(20)) {
128 // report on how many were propogated with motions for this frame only
129 const auto propogation_type =
130 propogation_result.collectByFrame(frame_id_k);
131 std::stringstream ss;
132 ss << "Propogation result: " << propogation_type.size()
133 << " objects for frame " << frame_id_k;
134
135 size_t n_used_motion = 0;
136 for (const auto& [_, type] : propogation_type) {
137 if (type == PropogateType::Propogate) {
138 n_used_motion++;
139 }
140 }
141 ss << " of which " << n_used_motion << " were motion propogated";
142 VLOG(100) << ss.str();
143 }
144 }
145
146 VLOG(50) << "Updated object pose cache";
147 object_pose_cache_ = object_poses;
148 LOG(INFO) << "ending MotionWorldAccessor::postUpdateCallback";
149}

151void WorldMotionFormulation::dynamicPointUpdateCallback(
152 const PointUpdateContextType& context, UpdateObservationResult& result,
153 gtsam::Values& new_values, gtsam::NonlinearFactorGraph& new_factors) {
154 const auto lmk_node = context.lmk_node;
155 const auto frame_node_k_1 = context.frame_node_k_1;
156 const auto frame_node_k = context.frame_node_k;
157
158 auto dynamic_point_noise = noise_models_.dynamic_point_noise;
159 auto theta_accessor = this->accessorFromTheta();
160
161 const gtsam::Key object_point_key_k_1 =
162 lmk_node->makeDynamicKey(frame_node_k_1->frame_id);
163 const gtsam::Key object_point_key_k =
164 lmk_node->makeDynamicKey(frame_node_k->frame_id);
165
166 bool add_point_from_previous = context.is_starting_motion_frame;
167 bool does_previous_point_exist = new_values.exists(object_point_key_k_1) ||
168 theta_accessor->exists(object_point_key_k_1);
169
170 // check the case that the point exists at the previous frame
171 // this only happens when we have non-consequative frames ie 2 3 4 5 6 7 10 11
172 // and we are at frame 11!!
173 // logically this occurs becuase of the implementation of Formulation which
174 // checks for all the frames that the point has been observed in and only
175 // sets is_starting_motion_frame = true on the first pair of motions it does
176 // not account for inconsistencies in the frame order as pointed out...
177 // Jesse: honestly this should NOT happen but we handle the case anyway...
178 if (!add_point_from_previous && !does_previous_point_exist) {
179 // we could do many things including re-initing this point OR removing it
180 // from the tracked set
181 CHECK(lmk_node->seenAtFrame(frame_node_k_1->frame_id));
182 // if we think we shouldn't add the previous point but the previous point
183 // does not exist, add it!@!
184 add_point_from_previous = true;
185 }
186
187 // if first motion (i.e first time we have both k-1 and k), add both at k-1
188 // and k
189 if (add_point_from_previous) {
190 CHECK(!theta_accessor->exists(object_point_key_k_1));
191
192 new_factors.emplace_shared<PoseToPointFactor>(
193 frame_node_k_1->makePoseKey(), // pose key at previous frames
194 object_point_key_k_1, lmk_node->getMeasurement(frame_node_k_1).landmark,
195 dynamic_point_noise);
196 if (result.debug_info)
197 result.debug_info->getObjectInfo(context.getObjectId())
198 .num_dynamic_factors++;
199 result.updateAffectedObject(frame_node_k_1->frame_id,
200 context.getObjectId());
201
202 // add landmark at previous frame
203 const Landmark measured_k_1 =
204 lmk_node->getMeasurement(frame_node_k_1->frame_id).landmark;
205 Landmark lmk_world_k_1;
206 getSafeQuery(lmk_world_k_1,
207 theta_accessor->query<Landmark>(object_point_key_k_1),
208 gtsam::Point3(context.X_k_1_measured * measured_k_1));
209 new_values.insert(object_point_key_k_1, lmk_world_k_1);
210 if (result.debug_info)
211 result.debug_info->getObjectInfo(context.getObjectId())
212 .num_new_dynamic_points++;
213 }
214
215 // previous point must be added by the previous iteration
216 CHECK(new_values.exists(object_point_key_k_1) ||
217 theta_accessor->exists(object_point_key_k_1))
218 << "Key: " << this->formatter()(object_point_key_k_1)
219 << " and seen at frames "
220 << container_to_string(lmk_node->getSeenFrameIds());
221
222 const Landmark measured_k = lmk_node->getMeasurement(frame_node_k).landmark;
223
224 new_factors.emplace_shared<PoseToPointFactor>(
225 frame_node_k
226 ->makePoseKey(), // pose key at this (in the iteration) frames
227 object_point_key_k, measured_k, dynamic_point_noise);
228 if (result.debug_info)
229 result.debug_info->getObjectInfo(context.getObjectId())
230 .num_dynamic_factors++;
231
232 result.updateAffectedObject(frame_node_k->frame_id, context.getObjectId());
233
234 Landmark lmk_world_k;
235 getSafeQuery(lmk_world_k, theta_accessor->query<Landmark>(object_point_key_k),
236 gtsam::Point3(context.X_k_measured * measured_k));
237 new_values.insert(object_point_key_k, lmk_world_k);
238 if (result.debug_info)
239 result.debug_info->getObjectInfo(context.getObjectId())
240 .num_new_dynamic_points++;
241
242 const gtsam::Key object_motion_key_k =
243 frame_node_k->makeObjectMotionKey(context.getObjectId());
244
245 auto landmark_motion_noise = noise_models_.landmark_motion_noise;
246 new_factors.emplace_shared<LandmarkMotionTernaryFactor>(
247 object_point_key_k_1, object_point_key_k, object_motion_key_k,
248 landmark_motion_noise);
249 result.updateAffectedObject(frame_node_k_1->frame_id, context.getObjectId());
250 result.updateAffectedObject(frame_node_k->frame_id, context.getObjectId());
251 if (result.debug_info)
252 result.debug_info->getObjectInfo(context.getObjectId())
253 .num_motion_factors++;
254
255 // mark as now in map
256 is_dynamic_tracklet_in_map_.insert2(context.getTrackletId(), true);
257}

259void WorldMotionFormulation::objectUpdateContext(
260 const ObjectUpdateContextType& context, UpdateObservationResult& result,
261 gtsam::Values& new_values, gtsam::NonlinearFactorGraph& new_factors) {
262 auto frame_node_k = context.frame_node_k;
263 const gtsam::Key object_motion_key_k =
264 frame_node_k->makeObjectMotionKey(context.getObjectId());
265
266 auto theta_accessor = this->accessorFromTheta();
267
268 // skip if no motion pair availble
269 if (!context.has_motion_pair) {
270 return;
271 }
272
273 const auto frame_id = context.getFrameId();
274 const auto object_id = context.getObjectId();
275
276 if (!is_other_values_in_map.exists(object_motion_key_k)) {
277 // when we have an initial motion
278 Motion3 initial_motion = Motion3::Identity();
279
280 constexpr static bool init_H_with_identity = false;
281 if (!init_H_with_identity) {
282 map()->hasInitialObjectMotion(frame_id, object_id, &initial_motion);
283 LOG(INFO) << "Using motion from frontend " << initial_motion;
284 initial_motion =
285 gtsam::Pose3(gtsam::Rot3::Identity(), initial_motion.translation());
286 }
287
288 new_values.insert(object_motion_key_k, initial_motion);
289 is_other_values_in_map.insert2(object_motion_key_k, true);
290 }
291
292 if (frame_id < 2) return;
293
294 auto frame_node_k_1 = map()->getFrame(frame_id - 1u);
295 if (!frame_node_k_1) {
296 return;
297 }
298
299 if (FLAGS_use_smoothing_factor && frame_node_k_1->objectObserved(object_id)) {
300 // motion key at previous frame
301 const gtsam::Symbol object_motion_key_k_1 =
302 frame_node_k_1->makeObjectMotionKey(object_id);
303
304 auto object_smoothing_noise = noise_models_.object_smoothing_noise;
305 CHECK(object_smoothing_noise);
306 CHECK_EQ(object_smoothing_noise->dim(), 6u);
307
308 {
309 ObjectId object_label_k_1, object_label_k;
310 FrameId frame_id_k_1, frame_id_k;
311 CHECK(reconstructMotionInfo(object_motion_key_k_1, object_label_k_1,
312 frame_id_k_1));
313 CHECK(reconstructMotionInfo(object_motion_key_k, object_label_k,
314 frame_id_k));
315 CHECK_EQ(object_label_k_1, object_label_k);
316 CHECK_EQ(frame_id_k_1 + 1, frame_id_k); // assumes consequative frames
317 }
318
319 // if the motion key at k (motion from k-1 to k), and key at k-1 (motion
320 // from k-2 to k-1) exists in the map or is about to exist via new values,
321 // add the smoothing factor
322 if (is_other_values_in_map.exists(object_motion_key_k_1) &&
323 is_other_values_in_map.exists(object_motion_key_k)) {
324 new_factors.emplace_shared<gtsam::BetweenFactor<gtsam::Pose3>>(
325 object_motion_key_k_1, object_motion_key_k, gtsam::Pose3::Identity(),
326 object_smoothing_noise);
327 if (result.debug_info)
328 result.debug_info->getObjectInfo(context.getObjectId())
329 .smoothing_factor_added = true;
330 }
331 }
332}
