29void propogateObjectPoses(
30 ObjectPoseMap& object_poses,
31 const MotionEstimateMap& object_motions_k,
32 const gtsam::Point3Vector& object_centroids_k_1,
33 const gtsam::Point3Vector& object_centroids_k, //TODO: dont actually need this one!!
34 FrameId frame_id_k,
35 std::optional<GroundTruthPacketMap> gt_packet_map,
36 PropogatePoseResult* result
37) {
38
39 CHECK_EQ(object_motions_k.size(), object_centroids_k_1.size());
40 CHECK_EQ(object_centroids_k.size(), object_centroids_k_1.size());
41 const FrameId frame_id_k_1 = frame_id_k - 1;
42
43 //get centroid for object at k-1 using the gt pose if available, or the centroid if not
44 auto get_centroid = [=](
45 ObjectId object_id, const gtsam::Point3& centroid_k_1, gtsam::Pose3& pose_k_1) -> PropogateType
46 {
47 bool initalised_with_gt = false;
48 //if gt packet exists for this frame, use that as the rotation
49 if(gt_packet_map) {
50 if(gt_packet_map->exists(frame_id_k_1)) {
51 const GroundTruthInputPacket& gt_packet_k_1 = gt_packet_map->at(frame_id_k_1);
52
53 ObjectPoseGT object_pose_gt_k_1;
54 if(gt_packet_k_1.getObject(object_id, object_pose_gt_k_1)) {
55 pose_k_1 = object_pose_gt_k_1.L_world_;
56 initalised_with_gt = true;
57 }
58 }
59 }
60
61 if(!initalised_with_gt) {
62 //could not init with gt, use identity rotation and centroid
63 pose_k_1 = gtsam::Pose3(gtsam::Rot3::Identity(), centroid_k_1);
64
65 return PropogateType::InitCentroid;
66 }
67 else {
68 return PropogateType::InitGT;
69 }
70 };
71
72 size_t i = 0; //used to index the object centroid vectors
73 for(const auto&[object_id, motion] : object_motions_k) {
74 const auto centroid_k = object_centroids_k.at(i);
75 const auto centroid_k_1 = object_centroids_k_1.at(i);
76 const gtsam::Pose3 prev_H_world_curr = motion;
77 //new object - so we need to add at k-1 and k
78 if(!object_poses.exists(object_id)) {
79 gtsam::Pose3 pose_k_1;
80 auto propgate_result = get_centroid(object_id, centroid_k_1, pose_k_1);
81 if(result) result->insert22(object_id, frame_id_k_1,propgate_result );
82

109 object_poses.insert2(object_id , gtsam::FastMap<FrameId, gtsam::Pose3>{});
110 object_poses.at(object_id).insert2(frame_id_k_1, pose_k_1);
111 }
112
113 auto& per_frame_poses = object_poses.at(object_id);
114 //if we have a pose at the previous frame, simply apply motion
115 if(per_frame_poses.exists(frame_id_k_1)) {
116 const gtsam::Pose3& object_pose_k_1 = per_frame_poses.at(frame_id_k_1);
117 //assuming in world
118 gtsam::Pose3 object_pose_k = prev_H_world_curr * object_pose_k_1;
119 per_frame_poses.insert2(frame_id_k, object_pose_k);
120
121 //update result map
122 if(result) result->insert22(object_id, frame_id_k, PropogateType::Propogate);
123 }
124 else {
125 //no motion at the previous frame - if close, interpolate between last pose and this pose
126 //no motion used
127 const size_t min_diff_frames = 3;
128
129 //last frame SHOULD be the largest frame (as we use a std::map with std::less)
130 auto last_record_itr = per_frame_poses.rbegin();
131 const FrameId last_frame = last_record_itr->first;
132 const gtsam::Pose3 last_recorded_pose = last_record_itr->second;
133
134 //construct current pose using last poses rotation (I guess?)
135 gtsam::Pose3 current_pose = gtsam::Pose3(
136 last_recorded_pose.rotation(),
137 object_centroids_k.at(i)
138 );
139
140 CHECK_LT(last_frame, frame_id_k_1);
141 if(frame_id_k - last_frame < min_diff_frames) {
142 //apply interpolation
143 //need to map [last_frame:frame_id_k] -> [0,1] for the interpolation function
144 //with N values such that frame_id_k - last_frame + 1= N (to be inclusive)
145 const size_t N = frame_id_k - last_frame + 1;
146 const double divisor = (double)(frame_id_k - last_frame);
147 for(size_t j = 0; j < N; j++) {
148 double t = (double)j/divisor;
149
150 gtsam::Pose3 interpolated_pose = last_recorded_pose.slerp(t, current_pose, boost::none, boost::none);
151
152 FrameId frame = last_frame + j;
153 per_frame_poses.insert2(frame, interpolated_pose);
154
155 //update result map
156 if(result) result->insert22(object_id, frame, PropogateType::Interpolate);
157 }
158
159 }
160 else {
161 gtsam::Pose3 pose_k_1;
162 //last frame too far away - reinitalise with centroid!
163 VLOG(20) << "Frames too far away - current frame is " << frame_id_k << " previous frame is " << last_frame << " for object " << object_id;
164 auto propogate_result = get_centroid(object_id, centroid_k_1, pose_k_1);
165 object_poses.at(object_id).insert2(frame_id_k_1, pose_k_1);
166
167 if(result) result->insert22(object_id, frame_id_k_1, propogate_result);
168
169
170 gtsam::Pose3 object_pose_k = prev_H_world_curr * pose_k_1;
171 per_frame_poses.insert2(frame_id_k, object_pose_k);
172
173 //update result map
174 if(result) result->insert22(object_id, frame_id_k, PropogateType::Propogate);
175
176 }
177
178
179 }
180 i++;
181 }
