54template <>
55inline bool convert(const LandmarkStatus& status, pcl::PointXYZRGB& point) {
56 const gtsam::Point3 lmk = status.value();
57 const ObjectId object_id = status.objectId();
58 if (status.isStatic()) {
59 point =
60 pcl::PointXYZRGB(static_cast<float>(lmk(0)), static_cast<float>(lmk(1)),
61 static_cast<float>(lmk(2)), 0, 0, 0);
62 } else {
63 // has to recalculate this every time!!
64 const Color colour = Color::uniqueId(object_id);
65 point = pcl::PointXYZRGB(
66 static_cast<float>(lmk(0)), static_cast<float>(lmk(1)),
67 static_cast<float>(lmk(2)), static_cast<std::uint8_t>(colour.r),
68 static_cast<std::uint8_t>(colour.g),
69 static_cast<std::uint8_t>(colour.b));
70 }
71 return true;
72}

74template <typename Value, typename Point>
75void convert_pcl(const GenericTrackedStatusVector<Value>& status_vector,
76 pcl::PointCloud<Point>& cloud) {
77 for (const auto& value : status_vector) {
78 Point point;
79 // call the status type to the point type explicitly
80 // allows any type w
81 convert(value, point);
82 cloud.push_back(point);
83 }
84}

87void convert(const StatusLandmarkVector& status_vector,
88 pcl::PointCloud<Point>& cloud) {
89 convert_pcl(status_vector, cloud);

195template <typename PointT>
196void computeMomentOfInertial(
197 const typename pcl::PointCloud<PointT>::Ptr obj_cloud_ptr,
198 pcl::MomentOfInertiaEstimation<PointT>& bbx_extractor, int mean_k = 100,
199 float std_dev_thresh = 1.0) {
200 pcl::PointCloud<PointT> filtered_cloud;
201
202 // statistical outlier removal
203 pcl::StatisticalOutlierRemoval<PointT> outlier_remover;
204 outlier_remover.setInputCloud(obj_cloud_ptr);
205 outlier_remover.setMeanK(mean_k);
206 outlier_remover.setStddevMulThresh(std_dev_thresh);
207 outlier_remover.filter(filtered_cloud);
208
209 bbx_extractor.setInputCloud(
210 pcl::make_shared<pcl::PointCloud<PointT>>(filtered_cloud));
211 bbx_extractor.compute();
212}

223template <typename PointT>
224ObjectBBX findAABBFromCloud(
225 const typename pcl::PointCloud<PointT>::Ptr obj_cloud_ptr) {
226 pcl::PointCloud<PointT> filtered_cloud;
227
228 pcl::MomentOfInertiaEstimation<PointT> bbx_extractor;
229 computeMomentOfInertial(obj_cloud_ptr, bbx_extractor);
230
231 PointT min_point_AABB;
232 PointT max_point_AABB;
233 bbx_extractor.getAABB(min_point_AABB, max_point_AABB);
234
235 ObjectBBX aabb;
236 aabb.min_bbx_point_ = pclPointToGtsam(min_point_AABB);
237 aabb.max_bbx_point_ = pclPointToGtsam(max_point_AABB);
238
239 return aabb;
240}

252template <typename PointT>
253ObjectBBX findOBBFromCloud(
254 const typename pcl::PointCloud<PointT>::Ptr obj_cloud_ptr) {
255 // oriented aligned bounding box
256 pcl::MomentOfInertiaEstimation<PointT> bbx_extractor;
257 computeMomentOfInertial(obj_cloud_ptr, bbx_extractor);
258
259 PointT min_point_AABB;
260 PointT max_point_AABB;
261 PointT position;
262 Eigen::Matrix3f orientation;
263 bbx_extractor.getOBB(min_point_AABB, max_point_AABB, position, orientation);
264
265 ObjectBBX obb;
266 obb.min_bbx_point_ = pclPointToGtsam(min_point_AABB);
267 obb.max_bbx_point_ = pclPointToGtsam(max_point_AABB);
268 obb.bbx_position_ = pclPointToGtsam(position);
269 obb.orientation_ = gtsam::Rot3(orientation.cast<double>());
270
271 return obb;
272}
