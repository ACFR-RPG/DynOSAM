54template <>
55inline bool convert(const LandmarkStatus& status, pcl::PointXYZRGB& point) {
56 const gtsam::Point3 lmk = status.value();
57 const ObjectId object_id = status.objectId();
58 if (status.isStatic()) {
59 point =
60 pcl::PointXYZRGB(static_cast<float>(lmk(0)), static_cast<float>(lmk(1)),
61 static_cast<float>(lmk(2)), 0, 0, 0);
62 } else {
63 // has to recalculate this every time!!
64 const Color colour = Color::uniqueId(object_id);
65 point = pcl::PointXYZRGB(
66 static_cast<float>(lmk(0)), static_cast<float>(lmk(1)),
67 static_cast<float>(lmk(2)), static_cast<std::uint8_t>(colour.r),
68 static_cast<std::uint8_t>(colour.g),
69 static_cast<std::uint8_t>(colour.b));
70 }
71 return true;
72}

74template <typename Value, typename Point>
75void convert_pcl(const GenericTrackedStatusVector<Value>& status_vector,
76 pcl::PointCloud<Point>& cloud) {
77 for (const auto& value : status_vector) {
78 Point point;
79 // call the status type to the point type explicitly
80 // allows any type w
81 convert(value, point);
82 cloud.push_back(point);
83 }
84}

87void convert(const StatusLandmarkVector& status_vector,
88 pcl::PointCloud<Point>& cloud) {
89 convert_pcl(status_vector, cloud);

92template <typename PointT>
93bool convert(const gtsam::Point3& gtsam_point, PointT& pcl_point) {
94 pcl_point.x = static_cast<decltype(pcl_point.x)>(gtsam_point(0));
95 pcl_point.y = static_cast<decltype(pcl_point.y)>(gtsam_point(1));
96 pcl_point.z = static_cast<decltype(pcl_point.z)>(gtsam_point(2));
97 return true;
98}

207template <typename PointT>
208void computeMomentOfInertial(
209 const typename pcl::PointCloud<PointT>::Ptr obj_cloud_ptr,
210 pcl::MomentOfInertiaEstimation<PointT>& bbx_extractor, int mean_k = 100,
211 float std_dev_thresh = 1.0) {
212 pcl::PointCloud<PointT> filtered_cloud;
213
214 // statistical outlier removal
215 pcl::StatisticalOutlierRemoval<PointT> outlier_remover;
216 outlier_remover.setInputCloud(obj_cloud_ptr);
217 outlier_remover.setMeanK(mean_k);
218 outlier_remover.setStddevMulThresh(std_dev_thresh);
219 outlier_remover.filter(filtered_cloud);
220
221 bbx_extractor.setInputCloud(
222 pcl::make_shared<pcl::PointCloud<PointT>>(filtered_cloud));
223 bbx_extractor.compute();
224}

235template <typename PointT>
236ObjectBBX findAABBFromCloud(
237 const typename pcl::PointCloud<PointT>::Ptr obj_cloud_ptr) {
238 pcl::PointCloud<PointT> filtered_cloud;
239
240 pcl::MomentOfInertiaEstimation<PointT> bbx_extractor;
241 computeMomentOfInertial(obj_cloud_ptr, bbx_extractor);
242
243 PointT min_point_AABB;
244 PointT max_point_AABB;
245 bbx_extractor.getAABB(min_point_AABB, max_point_AABB);
246
247 ObjectBBX aabb;
248 aabb.min_bbx_point_ = pclPointToGtsam(min_point_AABB);
249 aabb.max_bbx_point_ = pclPointToGtsam(max_point_AABB);
250
251 return aabb;
252}

264template <typename PointT>
265ObjectBBX findOBBFromCloud(
266 const typename pcl::PointCloud<PointT>::Ptr obj_cloud_ptr) {
267 // oriented aligned bounding box
268 pcl::MomentOfInertiaEstimation<PointT> bbx_extractor;
269 computeMomentOfInertial(obj_cloud_ptr, bbx_extractor);
270
271 PointT min_point_AABB;
272 PointT max_point_AABB;
273 PointT position;
274 Eigen::Matrix3f orientation;
275 bbx_extractor.getOBB(min_point_AABB, max_point_AABB, position, orientation);
276
277 ObjectBBX obb;
278 obb.min_bbx_point_ = pclPointToGtsam(min_point_AABB);
279 obb.max_bbx_point_ = pclPointToGtsam(max_point_AABB);
280 obb.bbx_position_ = pclPointToGtsam(position);
281 obb.orientation_ = gtsam::Rot3(orientation.cast<double>());
282
283 return obb;
284}
