115template <typename PointT>
116void computeMomentOfInertial(
117 const typename pcl::PointCloud<PointT>::Ptr obj_cloud_ptr,
118 pcl::MomentOfInertiaEstimation<PointT>& bbx_extractor, int mean_k = 100,
119 float std_dev_thresh = 1.0) {
120 pcl::PointCloud<PointT> filtered_cloud;
121
122 // statistical outlier removal
123 pcl::StatisticalOutlierRemoval<PointT> outlier_remover;
124 outlier_remover.setInputCloud(obj_cloud_ptr);
125 outlier_remover.setMeanK(mean_k);
126 outlier_remover.setStddevMulThresh(std_dev_thresh);
127 outlier_remover.filter(filtered_cloud);
128
129 bbx_extractor.setInputCloud(
130 pcl::make_shared<pcl::PointCloud<PointT>>(filtered_cloud));
131 bbx_extractor.compute();
132}

143template <typename PointT>
144ObjectBBX findAABBFromCloud(
145 const typename pcl::PointCloud<PointT>::Ptr obj_cloud_ptr) {
146 pcl::PointCloud<PointT> filtered_cloud;
147
148 pcl::MomentOfInertiaEstimation<PointT> bbx_extractor;
149 computeMomentOfInertial(obj_cloud_ptr, bbx_extractor);
150
151 PointT min_point_AABB;
152 PointT max_point_AABB;
153 bbx_extractor.getAABB(min_point_AABB, max_point_AABB);
154
155 ObjectBBX aabb;
156 aabb.min_bbx_point_ = pclPointToGtsam(min_point_AABB);
157 aabb.max_bbx_point_ = pclPointToGtsam(max_point_AABB);
158
159 return aabb;
160}

172template <typename PointT>
173ObjectBBX findOBBFromCloud(
174 const typename pcl::PointCloud<PointT>::Ptr obj_cloud_ptr) {
175 // oriented aligned bounding box
176 pcl::MomentOfInertiaEstimation<PointT> bbx_extractor;
177 computeMomentOfInertial(obj_cloud_ptr, bbx_extractor);
178
179 PointT min_point_AABB;
180 PointT max_point_AABB;
181 PointT position;
182 Eigen::Matrix3f orientation;
183 bbx_extractor.getOBB(min_point_AABB, max_point_AABB, position, orientation);
184
185 ObjectBBX obb;
186 obb.min_bbx_point_ = pclPointToGtsam(min_point_AABB);
187 obb.max_bbx_point_ = pclPointToGtsam(max_point_AABB);
188 obb.bbx_position_ = pclPointToGtsam(position);
189 obb.orientation_ = gtsam::Rot3(orientation.cast<double>());
190
191 return obb;
192}
