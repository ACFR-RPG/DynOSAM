48// inherit to add more sensor data (eg imu)
49// can be of any InputImagePacketBase with this as the default. We use this so
50// we can create a FrontendInputPacketBase type with the InputPacketType as the
51// actual derived type and not the base class so different modules do not have
52// to do their own casting to get image data
53struct FrontendInputPacketBase {
54 DYNO_POINTER_TYPEDEFS(FrontendInputPacketBase)
55
56 ImageContainer::Ptr image_container_;
57 GroundTruthInputPacket::Optional optional_gt_;
58 ImuMeasurements::Optional imu_measurements;
59
60 FrontendInputPacketBase()
61 : image_container_(nullptr), optional_gt_(std::nullopt) {}
62
63 FrontendInputPacketBase(
64 ImageContainer::Ptr image_container,
65 GroundTruthInputPacket::Optional optional_gt = std::nullopt)
66 : image_container_(image_container), optional_gt_(optional_gt) {
67 CHECK(image_container_);
68
69 if (optional_gt) {
70 CHECK_EQ(optional_gt_->frame_id_, image_container_->frameId());
71 }
72 }
73
74 inline FrameId getFrameId() const {
75 return CHECK_NOTNULL(image_container_)->frameId();
76 }
77
78 inline Timestamp getTimestamp() const {
79 return CHECK_NOTNULL(image_container_)->timestamp();
80 }
81
82 virtual ~FrontendInputPacketBase() = default;
83};
84
85} // namespace dyno
