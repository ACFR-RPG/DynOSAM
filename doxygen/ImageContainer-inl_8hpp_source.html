31template<typename... SubsetImageTypes>
32ImageContainerSubset<SubsetImageTypes...> ImageContainerSubset<ImageTypes...>::makeSubset(const bool clone) const {
33 //the tuple of image wrappers to construct for the subset class
34 using Subset = ImageContainerSubset<SubsetImageTypes...>;
35 using WrappedImageTypesTupleSubset = typename Subset::WrappedImageTypesTuple;
36 static constexpr size_t SubN = Subset::N;
37
38 //wrappers to populate from the current set? what about copying etc
39 WrappedImageTypesTupleSubset subset_wrappers;
40 for (size_t i = 0; i < SubN; i++) {
41 //get the type at the new subset
42 internal::select_apply<SubN>(i, [&](auto I) {
43 using SubType = typename Subset::ImageTypeStruct<I>;
44 //find this image type in the current container as the index may be different
45 constexpr size_t current_index = This::Index<SubType>();
46 //with the index i of the subset wrapper (corresponding with type SubType)
47 //get the image wrapper in this container using the request (sub) type
48 auto current_wrapped_image = std::get<current_index>(image_storage_);
49
50 //if invalid but request, throw runtime error becuase whats the point of requesting an invalid image
51 if(!current_wrapped_image.exists()) {
52 throw std::runtime_error(
53 "Error constructing ImageContainerSubset - requested image type " + type_name<SubType>() + " is empty in parent container");
54 }
55
56 if(clone) {
57 std::get<I>(subset_wrappers) = current_wrapped_image.clone();
58 }
59 else {
60 std::get<I>(subset_wrappers) = current_wrapped_image;
61 }
62 });
63 }
64
65 //this is going to do a bunch of copies?
66 auto construct_subset = [&](auto&&... args) { return Subset(args...); };
67 return std::apply(construct_subset, subset_wrappers);
68}

70template<typename... ImageTypes>
71void ImageContainerSubset<ImageTypes...>::validateSetup() const {
72 cv::Size required_size;
73
74 for (size_t i = 0; i < N; i++) {
75 //get the type at the new subset
76 internal::select_apply<N>(i, [&](auto stream_index) {
77 using ImageTypeStruct = typename This::ImageTypeStruct<stream_index>;
78 //only check if exists?
79 if(This::exists<ImageTypeStruct>()) {
80 //if required size is empty, then first check
81 //set required size to be the image size and then compare against
82 //this size on future iterations
83 const bool is_first = required_size.empty();
84 if(is_first) {
85 required_size = This::get<ImageTypeStruct>().size();
86 }
87 else {
88 //compare against required size
89 const cv::Size incoming_size = This::get<ImageTypeStruct>().size();
90 if(incoming_size != required_size) {
91 throw ImageContainerConstructionException(
92 "Non-empty images were not all the same size. First image (type: " +
93 type_name<This::ImageTypeStruct<0u>>() + ") was of size " + to_string(required_size) +
94 " and other image (type: " + type_name<ImageTypeStruct>() + ") was of size " + to_string(incoming_size)
95 );
96 }
97 }
98 }
99 });
100 }
101}
