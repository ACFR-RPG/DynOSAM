122 inline void fillWithKeypoints(const Keypoints& keypoints)
123 {
124 // TODO(cfo): could be implemented using block operations.
125 for (size_t i = 0; i < keypoints.size(); ++i)
126 {
127 const Keypoint& keypoint = keypoints.at(i);
128 const int int_x = static_cast<int>(keypoint(0));
129 const int int_y = static_cast<int>(keypoint(1));
130 const size_t idx = getCellIndex(int_x, int_y, 1);
131 occupancy_.at(idx) = true;
132 feature_occupancy_.at(idx) = Keypoint(keypoint(0), keypoint(1));
133 }
134 }
