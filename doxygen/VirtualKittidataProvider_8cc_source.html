82class VirtualKittiFlowDataFolder : public GenericVirtualKittiImageLoader {
83
84public:
85 VirtualKittiFlowDataFolder(const std::string& path) : GenericVirtualKittiImageLoader(path) {}
86
87 cv::Mat getItem(size_t idx) override {
88 CHECK(idx > 0);
89 std::stringstream ss;
90 //index at frame -1 so that we get the flow from t-1 to t (where the idx is t)
91 ss << std::setfill('0') << std::setw(5) << idx-1;
92 const std::string file_path = full_path_ + "/flow_" + ss.str() + ".png";
93 // const std::string file_path = full_path_ + "/backwardFlow_" + ss.str() + ".png"; //specific to backwards flow
94 throwExceptionIfPathInvalid(file_path);
95 return vKittiPngToFlow(file_path);
96 }
97
98private:
99 //as per example code in https://europe.naverlabs.com/research/computer-vision/proxy-virtual-worlds-vkitti-2/
100 // In R, flow along x-axis normalized by image width and quantized to [0;2^16 – 1]
101 // In G, flow along x-axis normalized by image width and quantized to [0;2^16 – 1]
102 // B = 0 for invalid flow (e.g., sky pixels)
103 cv::Mat vKittiPngToFlow(const std::string& path) const {
104 // cv::Mat bgr = cv::imread(path, cv::IMREAD_ANYCOLOR | cv::IMREAD_ANYDEPTH);
105 cv::Mat bgr;
106 loadRGB(path, bgr);
107
108 CHECK(!bgr.empty());
109
110 int h = bgr.rows;
111 int w = bgr.cols;
112 int c = bgr.channels();
113
114 CHECK_EQ(bgr.type(), CV_16UC3) << "type was " << utils::cvTypeToString(bgr);
115 CHECK_EQ(c, 3);
116
117 cv::Mat bgr_channels[3];
118 cv::split(bgr, bgr_channels);
119 cv::Mat b_channel = bgr_channels[0];
120
121 cv::Mat bgr_float;
122 //float 32
123 bgr.convertTo(bgr_float, CV_32F);
124
125 //scaled to [0;2**16 – 1]
126 cv::Mat unscaled_out_flow = 2.0 / (std::pow(2, 16) - 1.0) * bgr_float - 1;
127
128 cv::Mat channels[2];
129 //now only take g and r channels
130 cv::split(unscaled_out_flow, channels);
131 // g,r == flow_y,x normalized by height,width
132 cv::Mat& flow_y = channels[1];
133 cv::Mat& flow_x = channels[2];
134 flow_x *= w -1.0;
135 flow_y *= h -1.0;
136
137 //re-order channels so x is first and then y
138 cv::Mat x_y_ordered_channels[] = {flow_x, flow_y};
139 cv::Mat out_flow;
140 cv::merge(x_y_ordered_channels, 2, out_flow);
141
142 //b == invalid flow flag == 0 for sky or other invalid flow
143 cv::Mat invalid = (b_channel == 0);
144 CHECK(!invalid.empty());
145 out_flow.setTo(0,invalid);
146
147 out_flow.convertTo(out_flow, CV_32F);
148
149 return out_flow;
150
151 }

256 bool isMoving(FrameId frame, ObjectId object_id) const {
257 //check current frame and prev frame
258 if(checkExists(frame, object_id, bbox_data_) && checkExists(frame-1, object_id, bbox_data_)) {
259 return bbox_data_.at(frame - 1).at(object_id).is_moving;

276 template<typename T>
277 bool checkExists(FrameId frame_id, ObjectId object_id, const FrameObjectIDMap<T>& map) const {
278 auto it = map.find(frame_id);
279 if(it == map.end()) return false;
280
281 const auto& inner_map = it->second;
282 auto it_inner = inner_map.find(object_id);

308 void loadBBoxMetaData(FrameObjectIDMap<BBoxMetaData>& bbox_metadata) {
309 std::ifstream fstream;
310 fstream.open(bbox_path, std::ios::in);
311 if(!fstream) {
312 throw std::runtime_error("Could not open file " + bbox_path + " when trying to load BBox metadata for Virtual Kitti dataset!");
313 }
314
315 LOG(INFO) << "Loading BBox metadta for virtual kitti datasetat path: " << bbox_path;
316
317 static const std::map<std::string, int> header_map = {
318 {"frame", 0},
319 {"cameraID", 1},
320 {"trackID", 2},
321 {"left", 3},
322 {"right", 4},
323 {"top", 5},
324 {"bottom", 6},
325 {"number_pixels", 7},
326 {"truncation_ratio", 8},
327 {"occupancy_ratio", 9},
328 {"isMoving", 10},
329 };
330
331 auto convert_string_bool = [](const std::string& string_bool) -> bool {
332 if(string_bool == "True") return true;
333 else if(string_bool == "False") return false;
334 else throw std::runtime_error("Unknown string bool " + string_bool + " when attempting to covnert to bool type");
335 };
336
337 bool is_first = true; //to access header
338 while (!fstream.eof()) {
339 std::vector<std::string> split_line;
340 if(!getLine(fstream, split_line)) continue;
341
342 CHECK_EQ(split_line.size(), header_map.size()) << "Header map and the line present in the bbox.txt file do not have the same length - " << container_to_string(split_line);
343
344 if(is_first) {
345 //check header and validate
346 for(const auto& [header_value, index] : header_map) {
347 const std::string value = split_line.at(index);
348 if(value != header_value) {
349 throw std::runtime_error("Header value mismatch - " + value + " != " + header_value);
350 }
351 }
352 is_first = false;
353 }
354 else {
355 const int frame = std::stoi(split_line.at(header_map.at("frame")));
356 const int camera_id = std::stoi(split_line.at(header_map.at("cameraID")));
357 const int track_id = getTrackID(split_line, header_map.at("trackID"));
358 const int left = std::stoi(split_line.at(header_map.at("left")));
359 const int right = std::stoi(split_line.at(header_map.at("right")));
360 const int top = std::stoi(split_line.at(header_map.at("top")));
361 const int bottom = std::stoi(split_line.at(header_map.at("bottom")));
362 // const int number_pixels = std::stoi(split_line.at(header_map.at("number_pixels")));
363 // const double truncation_ratio = std::stod(split_line.at(header_map.at("truncation_ratio")));
364 // const double occupancy_ratio = std::stod(split_line.at(header_map.at("occupancy_ratio")));
365 const bool is_moving = convert_string_bool(split_line.at(header_map.at("isMoving")));
366
367 if(camera_id != 0) {
368 continue; //hardcoded for only one camera atm
369 }
370
371 BBoxMetaData bbox_data;
372 bbox_data.frame = frame;
373 bbox_data.track_id = track_id;
374 bbox_data.is_moving = is_moving;
375
376 //convert to cv::Rect where image starts top right
377 cv::Point tl(left, top);
378 cv::Point br(right, bottom);
379 bbox_data.bbox = cv::Rect(tl, br);
380
381 addToFrameObjectMap(frame, track_id, bbox_data, bbox_metadata);
382 }
383 }

387 void loadPoseTxt(FrameObjectIDMap<ObjectPoseGT>& object_poses, const FrameObjectIDMap<BBoxMetaData>& bbox_metadata) {
388 std::ifstream fstream;
389 fstream.open(pose_path, std::ios::in);
390 if(!fstream) {
391 throw std::runtime_error("Could not open file " + pose_path + " when trying to load Object pose metadata for Virtual Kitti dataset!");
392 }
393
394 LOG(INFO) << "Loading Object pose metadta for virtual kitti datasetat path: " << pose_path;
395
396 static const std::map<std::string, int> header_map = {
397 {"frame", 0},
398 {"cameraID", 1},
399 {"trackID", 2},
400 {"alpha", 3},
401 {"width", 4},
402 {"height", 5},
403 {"length", 6},
404 {"world_space_X", 7},
405 {"world_space_Y", 8},
406 {"world_space_Z", 9},
407 {"rotation_world_space_y", 10},
408 {"rotation_world_space_x", 11},
409 {"rotation_world_space_z", 12},
410 {"camera_space_X", 13},
411 {"camera_space_Y", 14},
412 {"camera_space_Z", 15},
413 {"rotation_camera_space_y", 16},
414 {"rotation_camera_space_x", 17},
415 {"rotation_camera_space_z", 18},
416 };
417
418 bool is_first = true; //to access header
419 while (!fstream.eof()) {
420 std::vector<std::string> split_line;
421 if(!getLine(fstream, split_line)) continue;
422
423
424 CHECK_EQ(split_line.size(), header_map.size()) << "Header map and the line present in the pose.txt file do not have the same length";
425
426 if(is_first) {
427 //check header and validate
428 for(const auto& [header_value, index] : header_map) {
429 CHECK_EQ(split_line.at(index), header_value);
430 }
431 is_first = false;
432 }
433 else {
434
435 const int frame = std::stoi(split_line.at(header_map.at("frame")));
436 const int camera_id = std::stoi(split_line.at(header_map.at("cameraID")));
437 const int track_id = getTrackID(split_line, header_map.at("trackID"));
438
439 if(camera_id != 0) {
440 continue; //hardcoded for only one camera atm
441 }
442
443 // const double alpha = std::stod(split_line.at(header_map.at("alpha")));
444 const double width = std::stod(split_line.at(header_map.at("width")));
445 const double height = std::stod(split_line.at(header_map.at("height")));
446 const double length = std::stod(split_line.at(header_map.at("length")));
447 // const double world_space_X = std::stod(split_line.at(header_map.at("world_space_X")));
448 // const double world_space_Y = std::stod(split_line.at(header_map.at("world_space_Y")));
449 // const double world_space_Z = std::stod(split_line.at(header_map.at("world_space_Z")));
450 // const double rotation_world_space_y = std::stod(split_line.at(header_map.at("rotation_world_space_y")));
451 // const double rotation_world_space_x = std::stod(split_line.at(header_map.at("rotation_world_space_x")));
452 // const double rotation_world_space_z = std::stod(split_line.at(header_map.at("rotation_world_space_z")));
453 const double camera_space_X = std::stod(split_line.at(header_map.at("camera_space_X")));
454 const double camera_space_Y = std::stod(split_line.at(header_map.at("camera_space_Y")));
455 const double camera_space_Z = std::stod(split_line.at(header_map.at("camera_space_Z")));
456 const double rotation_camera_space_y = std::stod(split_line.at(header_map.at("rotation_camera_space_y")));
457 const double rotation_camera_space_x = std::stod(split_line.at(header_map.at("rotation_camera_space_x")));
458 const double rotation_camera_space_z = std::stod(split_line.at(header_map.at("rotation_camera_space_z")));
459
460 ObjectPoseGT object_pose_gt;
461 object_pose_gt.frame_id_ = frame;
462 object_pose_gt.object_id_ = track_id;
463 object_pose_gt.object_dimensions_ = gtsam::Vector3(width, height, length);
464
465 {
466 // assign r vector
467 // double y = rotation_camera_space_y + (3.1415926 / 2); // +(3.1415926/2)
468 double y = rotation_camera_space_y; // +(3.1415926/2)
469 double x = rotation_camera_space_x;
470 double z = rotation_camera_space_z;
471
472 // the angles are in radians.
473 double cy = cos(y);
474 double sy = sin(y);
475 double cx = cos(x);
476 double sx = sin(x);
477 double cz = cos(z);
478 double sz = sin(z);
479
480 double m00, m01, m02, m10, m11, m12, m20, m21, m22;
481
482 m00 = cy * cz + sy * sx * sz;
483 m01 = -cy * sz + sy * sx * cz;
484 m02 = sy * cx;
485 m10 = cx * sz;
486 m11 = cx * cz;
487 m12 = -sx;
488 m20 = -sy * cz + cy * sx * sz;
489 m21 = sy * sz + cy * sx * cz;
490 m22 = cy * cx;
491
492 gtsam::Matrix33 rot;
493 rot(0, 0) = m00;
494 rot(0, 1) = m01;
495 rot(0, 2) = m02;
496 rot(1, 0) = m10;
497 rot(1, 1) = m11;
498 rot(1, 2) = m12;
499 rot(2, 0) = m20;
500 rot(2, 1) = m21;
501 rot(2, 2) = m22;
502
503 gtsam::Vector3 translation(camera_space_X, camera_space_Y, camera_space_Z);
504
505 object_pose_gt.L_camera_ = gtsam::Pose3(gtsam::Rot3(rot), translation);
506 }
507
508 CHECK(checkExists(frame, track_id, bbox_metadata));
509
510 //check asociated bboxdata has the right info for this object
511 const BBoxMetaData& bbox_data = bbox_metadata.at(frame).at(track_id);
512 CHECK_EQ(bbox_data.frame, frame);
513 CHECK_EQ(bbox_data.track_id, track_id);
514
515 object_pose_gt.bounding_box_ = bbox_data.bbox;
516 addToFrameObjectMap(frame, track_id, object_pose_gt, object_poses);
517
518 }
519
520 }

523 void loadCameraPoses(std::vector<gtsam::Pose3>& camera_poses) {
524 std::ifstream fstream;
525 fstream.open(extrinsic_path, std::ios::in);
526 if(!fstream) {
527 throw std::runtime_error("Could not open file " + extrinsic_path + " when trying to load camera exterinsics for Virtual Kitti dataset!");
528 }
529
530 LOG(INFO) << "Camera pose data for virtual kitti datasetat path: " << extrinsic_path;
531
532 bool is_first = true;
533
534 gtsam::Pose3 initial_pose = gtsam::Pose3::Identity();
535 bool set_initial_pose = false;
536
537 while (!fstream.eof()) {
538 std::vector<std::string> split_line;
539 if(!getLine(fstream, split_line)) continue;
540
541 //frame camera_id, 3x4 camera pose (r11, r12, r13, t1....)
542 //header: frame cameraID r1,1 r1,2 r1,3 t1 r2,1 r2,2 r2,3 t2 r3,1 r3,2 r3,3 t3 0 0 0 1
543 constexpr static size_t kExtrinsicFileLength = 18u;
544 CHECK_EQ(split_line.size(), kExtrinsicFileLength);
545
546 if(is_first){
547
548 is_first = false;
549 }
550 else {
551 const int frame = std::stoi(split_line.at(0));
552 const int camera_id = std::stoi(split_line.at(1));
553
554 if(camera_id != 0) {
555 continue; //hardcoded for only one camera atm
556 }
557
558 //sanity check to ensure things are added in order and at the right index (frame is zero indexed)
559 //check only after we check the camera id
560 CHECK_EQ(frame, camera_poses.size());
561
562 const double r11 = std::stod(split_line.at(2));
563 const double r12 = std::stod(split_line.at(3));
564 const double r13 = std::stod(split_line.at(4));
565 const double t1 = std::stod(split_line.at(5));
566 const double r21 = std::stod(split_line.at(6));
567 const double r22 = std::stod(split_line.at(7));
568 const double r23 = std::stod(split_line.at(8));
569 const double t2 = std::stod(split_line.at(9));
570 const double r31 = std::stod(split_line.at(10));
571 const double r32 = std::stod(split_line.at(11));
572 const double r33 = std::stod(split_line.at(12));
573 const double t3 = std::stod(split_line.at(13));
574
575 gtsam::Matrix33 rot;
576 rot(0, 0) = r11;
577 rot(0, 1) = r12;
578 rot(0, 2) = r13;
579 rot(1, 0) = r21;
580 rot(1, 1) = r22;
581 rot(1, 2) = r23;
582 rot(2, 0) = r31;
583 rot(2, 1) = r32;
584 rot(2, 2) = r33;
585
586 //last 4 values should be [0 0 0 1] for homogenous matrix form so we just check
587 CHECK_EQ(std::stod(split_line.at(14)), 0);
588 CHECK_EQ(std::stod(split_line.at(15)), 0);
589 CHECK_EQ(std::stod(split_line.at(16)), 0);
590 CHECK_EQ(std::stod(split_line.at(17)), 1);
591
592 gtsam::Pose3 pose(gtsam::Rot3(rot), gtsam::Point3(t1, t2, t3));
593
594
595 const static gtsam::Pose3 camera_to_world(gtsam::Rot3::RzRyRx(M_PI_2, 0, M_PI_2), gtsam::traits<gtsam::Point3>::Identity());
596 //pose is in world coordinate convention so we put into camera convention (the camera_to_world.inverse())
597 //the pose provided is actually world -> camera but we want camera -> world eg T_world_camera, so we apply the inverse transform
598 pose = camera_to_world.inverse() * pose.inverse();
599
600 if(!set_initial_pose) {
601 initial_pose = pose;
602 set_initial_pose = true;
603 }
604
605 // offset initial pose so we start at "0, 0, 0"
606 pose = initial_pose.inverse() * pose;
607 camera_poses.push_back(pose);
608 }
609
610 }

614 //only include objects that are moving - we will then use the returned vector as the input
615 //to the mask loader to remove masks for stationary objects
616 std::vector<GroundTruthInputPacket> constructGTPackets(
617 const FrameObjectIDMap<ObjectPoseGT>& object_poses,
618 const std::vector<gtsam::Pose3>& camera_poses
619 ) const
620 {
621
622 std::vector<GroundTruthInputPacket> gt_packets;
623 for(size_t frame_id = 0; frame_id < camera_poses.size(); frame_id++) {
624 GroundTruthInputPacket gt_packet;
625 gt_packet.frame_id_ = frame_id;
626 gt_packet.X_world_ = camera_poses.at(frame_id);
627
628 const auto& it = object_poses.find(frame_id);
629 if(it != object_poses.end()) {
630 auto& object_id_pose_map = it->second;
631 for(auto& [object_id, object_pose_gt] : object_id_pose_map) {
632 //query for moving
633 // if(isMoving(frame_id, object_id)) {
634 // LOG(INFO) << "Added moving object " << frame_id << " " << object_id;
635 auto object_pose = object_pose_gt;
636 object_pose.L_world_ = gt_packet.X_world_ * object_pose.L_camera_;
637 gt_packet.object_poses_.push_back(object_pose);
638 // }
639 }
640 }
641
642 //set motions
643 if(frame_id > 0) {
644 const GroundTruthInputPacket& previous_gt_packet = gt_packets.at(frame_id - 1);
645 gt_packet.calculateAndSetMotions(previous_gt_packet);

845 const std::string depth_folder = path + "/" + v_depth_folder + "/" + scene + "/" + scene_type + "/frames/depth/Camera_0";
846 const std::string forward_flow_folder = path + "/" + v_forward_flow_folder + "/" + scene + "/" + scene_type + "/frames/forwardFlow/Camera_0";

849 const std::string class_semantics_folder = path + "/" + v_class_semantics_folder + "/" + scene + "/" + scene_type + "/frames/classSegmentation/Camera_0";
850 const std::string instance_segmentation_folder = path + "/" + v_instance_segmentation_folder + "/" + scene + "/" + scene_type + "/frames/instanceSegmentation/Camera_0";
851 const std::string rgb_folder = path + "/" + v_rgb_folder + "/" + scene + "/" + scene_type + "/frames/rgb/Camera_0";
852 const std::string text_gt_folder = path + "/" + v_text_gt_folder + "/" + scene + "/" + scene_type;
853
854 throwExceptionIfPathInvalid(depth_folder);
855 throwExceptionIfPathInvalid(forward_flow_folder);
856 throwExceptionIfPathInvalid(forward_scene_flow_folder);

859 throwExceptionIfPathInvalid(rgb_folder);
860 throwExceptionIfPathInvalid(text_gt_folder);

864 auto timestamp_folder = std::make_shared<VirtualKittiTimestampLoader>(gt_loader->size());
