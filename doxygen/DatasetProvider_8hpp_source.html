193 //the data idx should also correspond with the index in the DynoDataTypesTuple tuple

205 auto per_folder_data_vector = dataset_->template getDataVector<I>();
206
207 //the data idx should also correspond with the index in the DynoDataTypesTuple tuple
208 //update data
209 std::get<I>(extra_data) = per_folder_data_vector.at(frame_id);
210
211 });
212 }
213
214 //concat the default and extra data into the full tuple
215 //this is a bit annoting as we will have to copy everything over
216 //get default data
217 DynoDataTypesTuple loaded_data = std::tuple_cat(default_data, extra_data);
218
219 if(this->data_input_callback_) {
220 auto construct_input_func = [&](auto&&... args) { return this->data_input_callback_(frame_id, args...); };
221 return std::apply(construct_input_func, loaded_data);
222 }
223 else {
224 LOG(WARNING) << "Data input callback not set with DynoDataset::setCallback!";
225 return false;
226 }

257template<typename... DataTypes>
258class DynoDatasetProvider : public DynoDataset<DataTypes...>, public DataProvider {
259
260public:
261 using BaseDynoDataset = DynoDataset<DataTypes...>;
262 using This = DynoDatasetProvider<DataTypes...>;
263 DYNO_POINTER_TYPEDEFS(This)
264
265 using BaseDynoDataset::isProviderValid;
266
267 //does not accept DataProviderModule* for dataprovider
268 DynoDatasetProvider(
269 const fs::path& dataset_path)
270 : BaseDynoDataset(dataset_path), DataProvider() {}
271
272 virtual ~DynoDatasetProvider() = default;
273
274
275 //ignore if -1. How to reset to original starting frame? we only have -1 so that in main we can call setStartingFrame without doing any checks
276 void setStartingFrame(int starting_frame) {
277 if(starting_frame != -1)
278 requested_starting_frame_id_ = starting_frame;
279 setEndingFrameFromRequested();
280 }
281 //ignore if -1
282 void setEndingFrame(int ending_frame) {
283 if(ending_frame != -1)
284 requested_ending_frame_id_ = ending_frame;
285
286 LOG(WARNING) << "Updating ending frame" << requested_ending_frame_id_;
287 setEndingFrameFromRequested();
288 }
289
300 void reset() {
301 is_first_spin_ = true;
302 active_frame_id_ = 0;
303 }
304
305 //returns the effetive size of the dataset (after the ending frame has been set)
306 //only valid after setLoaders is called, but this should happen at the constructor level anyway!
307 virtual int datasetSize() const override {
308 return (int)ending_frame_id_;
309 }
310
311 virtual bool spin() override {
312 if(!isProviderValid()) {
313 LOG(ERROR) << "Cannot spin as provider is invalid";
314 return false;
315 }
316
317 if(active_frame_id_ >= ending_frame_id_) {
318 LOG_FIRST_N(INFO, 1) << "Finished dataset";
319 emitOnFinishCallbacks();
320 return false;
321 }
322
323
324 utils::TimingStatsCollector dataset_spin_timer("dataset_spin");
325 if(!BaseDynoDataset::processSingle(active_frame_id_)) {
326 LOG(ERROR) << "Processing single frame failed at frame id " << active_frame_id_;
327 return false;
328 }
329
330 active_frame_id_++;
331 return true;
332 }
333
334 size_t getActiveFrameId() const { return active_frame_id_; }
335 size_t getEndingFrameId() const { return ending_frame_id_; }
336
337protected:
338 //overrride loaders so we can add our own post set laoders behaviour
339 //in this case, dataset size is now going to be valid so we can set the starting/end frame!
340 void setLoaders(TimestampBaseLoader::Ptr timestamp_loader,
341 dyno::DataFolder<cv::Mat>::Ptr rgb_loader,
342 dyno::DataFolder<cv::Mat>::Ptr optical_flow_loader,
343 typename DataFolder<DataTypes>::Ptr... data_folders) override
344 {
345 BaseDynoDataset::setLoaders(timestamp_loader, rgb_loader, optical_flow_loader, data_folders...);
346 //now we have a valid timestamp loader set the active/ending frame ids
347 //using the requested starting/ending frames and the actual dataset size, which we get from
348 //the base
349 setEndingFrameFromRequested();
350 }
351
352private:
353 void setEndingFrameFromRequested() {
354 //call base version - this will also check that the timestamp laoder is valid!
355 const size_t dataset_size = BaseDynoDataset::getDatasetSize();
356
357 // by default start dataset at 0
358 size_t starting_frame = 0u;
359 // by default end dataset at the last frame
360 size_t ending_frame = dataset_size;
361
362 if(requested_starting_frame_id_ >= 0) {
363 starting_frame = requested_starting_frame_id_;
364 }
365
366 if(requested_ending_frame_id_ >= 0) {
367 ending_frame = requested_ending_frame_id_;
368 }
369
370 //check ending frame first
371 if(ending_frame > dataset_size) {
372 throw DynoDatasetException("Requested ending frame is greater than the size of the dataset (" + std::to_string(ending_frame) + " > " + std::to_string(dataset_size));
373 }
374
375 if(starting_frame > ending_frame) {
376 throw DynoDatasetException("Requested starting frame is invalid - either less than 0 or greater than the ending frame (starting frame= " + std::to_string(starting_frame) + ", ending frame= " + std::to_string(ending_frame));
377 }
378
379 //update varibles actually used in the spin
380 active_frame_id_ = starting_frame;
381 ending_frame_id_ = ending_frame;
382 }
383
384private:
385 size_t active_frame_id_ = 0u;
386 size_t ending_frame_id_ = 0u; //Start at 0, but this will be set upon the first call to spin()
387
388 int requested_starting_frame_id_{-1};
389 int requested_ending_frame_id_{-1};
390
391 bool is_first_spin_{true};
392
393 using BaseDynoDataset::getDatasetSize;
394
395
396
397};
