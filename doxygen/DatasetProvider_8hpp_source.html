183 bool processSingle(size_t frame_id) {
184 // TODO: change to check with isProviderValid!!
185 if (!dataset_) {
186 LOG(ERROR) << "Dataset not loaded with setLoaders()! Skipping processing";
187 return false;
188 }
189 if (frame_id >= getDatasetSize()) {
190 throw std::runtime_error("Failure when processing a single frame_id!");
191 }
192
193 const size_t loading_idx = frame_id + 1; // odd indexing... oops
194 CHECK(default_dataset_->load(loading_idx));
195 CHECK(dataset_->load(loading_idx));
196
197 DefaultDataset::DataTypeTuple default_data;
198 typename TypedGenericDataset::DataTypeTuple extra_data;
199 constexpr size_t default_data_size =
200 std::tuple_size<DefaultDataset::DataTypeTuple>::value;
201
202 // iterate over default dataset portion of the ConcatDataTypes
203 for (size_t data_idx = 0; data_idx < default_data_size; data_idx++) {
204 internal::select_apply<default_data_size>(data_idx, [&](auto I) {
205 // I should be data_idx -> access the data vector we just loaded
206 auto per_folder_data_vector = default_dataset_->getDataVector<I>();
207
208 // the data idx should also correspond with the index in the
209 // DynoDataTypesTuple tuple update data
210 std::get<I>(default_data) = per_folder_data_vector.at(frame_id);
211 });
212 }
213
214 constexpr size_t extra_data_size =
215 std::tuple_size<typename TypedGenericDataset::DataTypeTuple>::value;
216 // //iterate over extra dataset portion of the ConcatDataTypes
217 for (size_t data_idx = 0; data_idx < extra_data_size; data_idx++) {
218 internal::select_apply<extra_data_size>(data_idx, [&](auto I) {
219 // I should be data_idx -> access the data vector we just loaded
220 auto per_folder_data_vector = dataset_->template getDataVector<I>();
221
222 // the data idx should also correspond with the index in the
223 // DynoDataTypesTuple tuple update data
224 std::get<I>(extra_data) = per_folder_data_vector.at(frame_id);
225 });
226 }
227
228 // concat the default and extra data into the full tuple
229 // this is a bit annoting as we will have to copy everything over
230 // get default data
231 DynoDataTypesTuple loaded_data = std::tuple_cat(default_data, extra_data);
232
233 if (this->data_input_callback_) {
234 auto construct_input_func = [&](auto&&... args) {
235 return this->data_input_callback_(frame_id, args...);
236 };
237 return std::apply(construct_input_func, loaded_data);
238 } else {
239 LOG(WARNING)
240 << "Data input callback not set with DynoDataset::setCallback!";
241 return false;
242 }
243 }

270template <typename... DataTypes>
271class DynoDatasetProvider : public DynoDataset<DataTypes...>,
272 public DataProvider {
273 public:
274 using BaseDynoDataset = DynoDataset<DataTypes...>;
275 using This = DynoDatasetProvider<DataTypes...>;
276 DYNO_POINTER_TYPEDEFS(This)
277
278 using BaseDynoDataset::isProviderValid;
279
280 // does not accept DataProviderModule* for dataprovider
281 DynoDatasetProvider(const fs::path& dataset_path)
282 : BaseDynoDataset(dataset_path), DataProvider() {}
283
284 virtual ~DynoDatasetProvider() = default;
285
286 // ignore if -1. How to reset to original starting frame? we only have -1 so
287 // that in main we can call setStartingFrame without doing any checks
288 void setStartingFrame(int starting_frame) {
289 if (starting_frame != -1) requested_starting_frame_id_ = starting_frame;
290 setEndingFrameFromRequested();
291 }
292 // ignore if -1
293 void setEndingFrame(int ending_frame) {
294 if (ending_frame != -1) requested_ending_frame_id_ = ending_frame;
295
296 LOG(WARNING) << "Updating ending frame" << requested_ending_frame_id_;
297 setEndingFrameFromRequested();
298 }
299
311 void reset() {
312 is_first_spin_ = true;
313 active_frame_id_ = 0;
314 }
315
316 // returns the effetive size of the dataset (after the ending frame has been
317 // set) only valid after setLoaders is called, but this should happen at the
318 // constructor level anyway!
319 virtual int datasetSize() const override { return (int)ending_frame_id_; }
320
321 virtual bool spin() override {
322 if (!isProviderValid()) {
323 LOG(ERROR) << "Cannot spin as provider is invalid";
324 return false;
325 }
326
327 if (active_frame_id_ >= ending_frame_id_) {
328 LOG_FIRST_N(INFO, 1) << "Finished dataset";
329 emitOnFinishCallbacks();
330 return false;
331 }
332
333 utils::TimingStatsCollector dataset_spin_timer("dataset_spin");
334 if (!BaseDynoDataset::processSingle(active_frame_id_)) {
335 LOG(ERROR) << "Processing single frame failed at frame id "
336 << active_frame_id_;
337 return false;
338 }
339
340 active_frame_id_++;
341 return true;
342 }
343
344 size_t getActiveFrameId() const { return active_frame_id_; }
345 size_t getEndingFrameId() const { return ending_frame_id_; }
346
347 protected:
348 // overrride loaders so we can add our own post set laoders behaviour
349 // in this case, dataset size is now going to be valid so we can set the
350 // starting/end frame!
351 void setLoaders(
352 TimestampBaseLoader::Ptr timestamp_loader,
353 dyno::DataFolder<cv::Mat>::Ptr rgb_loader,
354 dyno::DataFolder<cv::Mat>::Ptr optical_flow_loader,
355 typename DataFolder<DataTypes>::Ptr... data_folders) override {
356 BaseDynoDataset::setLoaders(timestamp_loader, rgb_loader,
357 optical_flow_loader, data_folders...);
358 // now we have a valid timestamp loader set the active/ending frame ids
359 // using the requested starting/ending frames and the actual dataset size,
360 // which we get from the base
361 setEndingFrameFromRequested();
362 }
363
364 private:
365 void setEndingFrameFromRequested() {
366 // call base version - this will also check that the timestamp laoder is
367 // valid!
368 const size_t dataset_size = BaseDynoDataset::getDatasetSize();
369
370 // by default start dataset at 0
371 size_t starting_frame = 0u;
372 // by default end dataset at the last frame
373 size_t ending_frame = dataset_size;
374
375 if (requested_starting_frame_id_ >= 0) {
376 starting_frame = requested_starting_frame_id_;
377 }
378
379 if (requested_ending_frame_id_ >= 0) {
380 ending_frame = requested_ending_frame_id_;
381 }
382
383 // check ending frame first
384 if (ending_frame > dataset_size) {
385 throw DynoDatasetException(
386 "Requested ending frame is greater than the size of the dataset (" +
387 std::to_string(ending_frame) + " > " + std::to_string(dataset_size));
388 }
389
390 if (starting_frame > ending_frame) {
391 throw DynoDatasetException(
392 "Requested starting frame is invalid - either less than 0 or greater "
393 "than the ending frame (starting frame= " +
394 std::to_string(starting_frame) +
395 ", ending frame= " + std::to_string(ending_frame));
396 }
397
398 // update varibles actually used in the spin
399 active_frame_id_ = starting_frame;
400 ending_frame_id_ = ending_frame;
401 }
402
403 private:
404 size_t active_frame_id_ = 0u;
405 size_t ending_frame_id_ =
406 0u; // Start at 0, but this will be set upon the first call to spin()
407
408 int requested_starting_frame_id_{-1};
409 int requested_ending_frame_id_{-1};
410
411 bool is_first_spin_{true};
412
413 using BaseDynoDataset::getDatasetSize;
414};
