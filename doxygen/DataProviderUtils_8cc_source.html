59gtsam::Rot3 toRightHandedRotation(const gtsam::Rot3& left_handed_rotation) {
60 const gtsam::Vector3 left_rpy = left_handed_rotation.rpy();
61
62 double right_roll = left_rpy[0];
63 double right_pitch = -left_rpy[1];
64 double right_yaw = -left_rpy[2];
65
66 return gtsam::Rot3::Ypr(right_yaw, right_pitch, right_roll);
67}

69gtsam::Vector3 toRightHandedVector(
70 const gtsam::Vector3& left_handed_vector,
71 std::optional<gtsam::Rot3> left_handed_rotation) {
72 gtsam::Vector3 tmp_vector = left_handed_vector;
73 if (left_handed_rotation) {
74 const gtsam::Rot3 right_handed_rotation =
75 toRightHandedRotation(*left_handed_rotation);
76 tmp_vector = right_handed_rotation.rotate(left_handed_vector);
77 }
78
79 gtsam::Vector3 right_handed_vector;
80 right_handed_vector[0] = tmp_vector[0];
81 right_handed_vector[1] = -tmp_vector[1];
82 right_handed_vector[2] = tmp_vector[2];
83
84 // apparently just swap axis according to Sephyr...
85
86 return right_handed_vector;
87}

99void loadPathsInDirectory(
100 std::vector<std::string>& file_paths, const std::string& folder_path,
101 const std::function<bool(const std::string&)>& condition) {
102 std::function<bool(const std::string&)> impl_condition;
103 if (condition) {
104 impl_condition = condition;
105 } else {
106 // if no condition is provided, set condition to always return true; adding
107 // all the files found
108 impl_condition = [](const std::string&) -> bool { return true; };
109 }
110
111 auto files_in_directory = getAllFilesInDir(folder_path);
112 for (const std::string file_path : files_in_directory) {
113 utils::throwExceptionIfPathInvalid(file_path);
114
115 // if condition is true, add
116 if (impl_condition(file_path)) file_paths.push_back(file_path);
117 }
118}

120void removeStaticObjectFromMask(const cv::Mat& instance_mask,
121 cv::Mat& motion_mask,
122 const GroundTruthInputPacket& gt_packet) {
123 try {
124 ImageType::SemanticMask::validate(instance_mask);
125 } catch (const InvalidImageTypeException& e) {
126 throw DynosamException(
127 "Input mask to removeStaticObjectFromMask as it"
128 "did not meet the requirements of ImageType::SemanticMask. Error: " +
129 std::string(e.what()));
130 }
131
132 ObjectIds object_ids = vision_tools::getObjectLabels(instance_mask);
133
134 instance_mask.copyTo(motion_mask);
135
136 const FrameId frame_id = gt_packet.frame_id_;
137
138 // collect only moving labels
139 std::vector<int> moving_labels;
140 for (const ObjectPoseGT& object_pose_gt : gt_packet.object_poses_) {
141 // LOG_IF(WARNING, !object_pose_gt.motion_info_)
142 // << "Object Pose GT (object " << object_pose_gt.object_id_ << ", frame
143 // " << object_pose_gt.frame_id_ << " does not have motion info set!
144 // Cannot determine of object is moving or not!";
145
146 if (!object_pose_gt.motion_info_) {
147 LOG(WARNING) << "Object Pose GT (object " << object_pose_gt.object_id_
148 << ", frame " << object_pose_gt.frame_id_
149 << " does not have motion info set! Cannot determine of "
150 "object is moving or not!";
151 continue;
152 }
153
154 const auto& motion_info = *object_pose_gt.motion_info_;
155 const ObjectId& object_id = object_pose_gt.object_id_;
156 if (motion_info.is_moving_) {
157 moving_labels.push_back(object_id);
158 }
159
160 // this is just a sanity (debug) check to ensure all the labels in the image
161 // match up with the ones we have already collected in the ground truth
162 // packet
163 const auto& it = std::find(object_ids.begin(), object_ids.end(), object_id);
164 CHECK(it != object_ids.end())
165 << "Object id " << object_id
166 << " appears in gt packet but"
167 "is not in mask when loading motion mask for Virtual Kitti at frame "
168 << frame_id;
169 }
170
171 // iterate over each object and if not moving remove
172 for (int i = 0; i < motion_mask.rows; i++) {
173 for (int j = 0; j < motion_mask.cols; j++) {
174 int label = motion_mask.at<int>(i, j);
175 if (label == 0) {
176 continue;
177 }
178
179 // check if label is in moving labels. if not, make zero!
180 auto it = std::find(moving_labels.begin(), moving_labels.end(), label);
181 if (it == moving_labels.end()) {
182 motion_mask.at<int>(i, j) = 0;
183 }
184 }
185 }
186}
