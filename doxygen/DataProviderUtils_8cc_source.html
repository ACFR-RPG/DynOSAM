71void loadSemanticMask(const std::string& image_path, const cv::Size& size,
72 cv::Mat& mask) {
73 throwExceptionIfPathInvalid(image_path);
74 CHECK(!size.empty());
75
76 mask = cv::Mat(size, CV_32SC1);
77
78 std::ifstream file_mask;
79 file_mask.open(image_path.c_str());
80
81 int count = 0;
82 while (!file_mask.eof()) {
83 std::string s;
84 getline(file_mask, s);
85 if (!s.empty()) {
86 std::stringstream ss;
87 ss << s;
88 int tmp;
89 for (int i = 0; i < mask.cols; ++i) {
90 ss >> tmp;
91 if (tmp != 0) {
92 mask.at<int>(count, i) = tmp;
93 } else {
94 mask.at<int>(count, i) = 0;
95 }
96 }
97 count++;
98 }
99 }
100
101 file_mask.close();
102}

110void toRightHandedTwist(gtsam::Vector3& right_handed_linear_velocity,
111 gtsam::Vector3& right_handed_angular_velocity,
112 const gtsam::Vector3& left_handed_linear_velocity,
113 const gtsam::Vector3& left_handed_angular_velocity,
114 std::optional<gtsam::Rot3> left_handed_rotation) {
115 right_handed_linear_velocity =
116 toRightHandedVector(left_handed_linear_velocity, left_handed_rotation);
117
118 right_handed_angular_velocity[0] = left_handed_angular_velocity[0];

123gtsam::Rot3 toRightHandedRotation(const gtsam::Rot3& left_handed_rotation) {
124 const gtsam::Vector3 left_rpy = left_handed_rotation.rpy();
125
126 double right_roll = left_rpy[0];
127 double right_pitch = -left_rpy[1];
128 double right_yaw = -left_rpy[2];
129
130 return gtsam::Rot3::Ypr(right_yaw, right_pitch, right_roll);
131}

133gtsam::Vector3 toRightHandedVector(
134 const gtsam::Vector3& left_handed_vector,
135 std::optional<gtsam::Rot3> left_handed_rotation) {
136 gtsam::Vector3 tmp_vector = left_handed_vector;
137 if (left_handed_rotation) {
138 const gtsam::Rot3 right_handed_rotation =
139 toRightHandedRotation(*left_handed_rotation);
140 tmp_vector = right_handed_rotation.rotate(left_handed_vector);
141 }
142
143 gtsam::Vector3 right_handed_vector;
144 right_handed_vector[0] = tmp_vector[0];
145 right_handed_vector[1] = -tmp_vector[1];
146 right_handed_vector[2] = tmp_vector[2];
147
148 // apparently just swap axis according to Sephyr...
149
150 return right_handed_vector;
151}

163void loadPathsInDirectory(
164 std::vector<std::string>& file_paths, const std::string& folder_path,
165 const std::function<bool(const std::string&)>& condition) {
166 std::function<bool(const std::string&)> impl_condition;
167 if (condition) {
168 impl_condition = condition;
169 } else {
170 // if no condition is provided, set condition to always return true; adding
171 // all the files found
172 impl_condition = [](const std::string&) -> bool { return true; };
173 }
174
175 auto files_in_directory = getAllFilesInDir(folder_path);
176 for (const std::string file_path : files_in_directory) {
177 throwExceptionIfPathInvalid(file_path);
178
179 // if condition is true, add
180 if (impl_condition(file_path)) file_paths.push_back(file_path);
181 }
182}

184void removeStaticObjectFromMask(const cv::Mat& instance_mask,
185 cv::Mat& motion_mask,
186 const GroundTruthInputPacket& gt_packet) {
187 try {
188 ImageType::SemanticMask::validate(instance_mask);
189 } catch (const InvalidImageTypeException& e) {
190 throw DynosamException(
191 "Input mask to removeStaticObjectFromMask as it"
192 "did not meet the requirements of ImageType::SemanticMask. Error: " +
193 std::string(e.what()));
194 }
195
196 ObjectIds object_ids = vision_tools::getObjectLabels(instance_mask);
197
198 instance_mask.copyTo(motion_mask);
199
200 const FrameId frame_id = gt_packet.frame_id_;
201
202 // collect only moving labels
203 std::vector<int> moving_labels;
204 for (const ObjectPoseGT& object_pose_gt : gt_packet.object_poses_) {
205 // LOG_IF(WARNING, !object_pose_gt.motion_info_)
206 // << "Object Pose GT (object " << object_pose_gt.object_id_ << ", frame
207 // " << object_pose_gt.frame_id_ << " does not have motion info set!
208 // Cannot determine of object is moving or not!";
209
210 if (!object_pose_gt.motion_info_) {
211 LOG(WARNING) << "Object Pose GT (object " << object_pose_gt.object_id_
212 << ", frame " << object_pose_gt.frame_id_
213 << " does not have motion info set! Cannot determine of "
214 "object is moving or not!";
215 continue;
216 }
217
218 const auto& motion_info = *object_pose_gt.motion_info_;
219 const ObjectId& object_id = object_pose_gt.object_id_;
220 if (motion_info.is_moving_) {
221 moving_labels.push_back(object_id);
222 }
223
224 // this is just a sanity (debug) check to ensure all the labels in the image
225 // match up with the ones we have already collected in the ground truth
226 // packet
227 const auto& it = std::find(object_ids.begin(), object_ids.end(), object_id);
228 CHECK(it != object_ids.end())
229 << "Object id " << object_id
230 << " appears in gt packet but"
231 "is not in mask when loading motion mask for Virtual Kitti at frame "
232 << frame_id;
233 }
234
235 // iterate over each object and if not moving remove
236 for (int i = 0; i < motion_mask.rows; i++) {
237 for (int j = 0; j < motion_mask.cols; j++) {
238 int label = motion_mask.at<int>(i, j);
239 if (label == 0) {
240 continue;
241 }
242
243 // check if label is in moving labels. if not, make zero!
244 auto it = std::find(moving_labels.begin(), moving_labels.end(), label);
245 if (it == moving_labels.end()) {
246 motion_mask.at<int>(i, j) = 0;
247 }
248 }
249 }
250}

252std::vector<std::string> trimAndSplit(const std::string& input,
253 const std::string& delimiter) {
254 std::string trim_input = boost::algorithm::trim_right_copy(input);
255 std::vector<std::string> split_line;
256 boost::algorithm::split(split_line, trim_input, boost::is_any_of(delimiter));
257 return split_line;
258}

260bool getLine(std::ifstream& fstream, std::vector<std::string>& split_lines) {
261 std::string line;
262 getline(fstream, line);
263
264 split_lines.clear();
265
266 if (line.empty()) return false;
267
268 split_lines = trimAndSplit(line);
269 return true;
