69void loadSemanticMask(const std::string& image_path, const cv::Size& size, cv::Mat& mask) {
70 throwExceptionIfPathInvalid(image_path);
71 CHECK(!size.empty());
72
73 mask = cv::Mat(size, CV_32SC1);
74
75 std::ifstream file_mask;
76 file_mask.open(image_path.c_str());
77
78 int count = 0;
79 while (!file_mask.eof())
80 {
81 std::string s;
82 getline(file_mask, s);
83 if (!s.empty())
84 {
85 std::stringstream ss;
86 ss << s;
87 int tmp;
88 for (int i = 0; i < mask.cols; ++i)
89 {
90 ss >> tmp;
91 if (tmp != 0)
92 {
93 mask.at<int>(count, i) = tmp;
94 }
95 else
96 {
97 mask.at<int>(count, i) = 0;
98 }
99 }
100 count++;
101 }
102 }
103
104 file_mask.close();
105
106}

124void loadPathsInDirectory(std::vector<std::string>& file_paths, const std::string& folder_path, const std::function<bool(const std::string&)>& condition) {
125 std::function<bool(const std::string&)> impl_condition;
126 if(condition) {
127 impl_condition = condition;
128 }
129 else {
130 //if no condition is provided, set condition to always return true; adding all the files found
131 impl_condition = [](const std::string&) -> bool { return true; };
132 }
133
134 auto files_in_directory = getAllFilesInDir(folder_path);
135 for (const std::string file_path : files_in_directory) {
136 throwExceptionIfPathInvalid(file_path);
137
138 //if condition is true, add
139 if(impl_condition(file_path)) file_paths.push_back(file_path);
140 }
141}

144void removeStaticObjectFromMask(const cv::Mat& instance_mask, cv::Mat& motion_mask, const GroundTruthInputPacket& gt_packet) {
145 try {
146 ImageType::SemanticMask::validate(instance_mask);
147 }
148 catch (const InvalidImageTypeException& e) {
149 throw DynosamException("Input mask to removeStaticObjectFromMask as it"
150 "did not meet the requirements of ImageType::SemanticMask. Error: " + std::string(e.what()));
151 }
152
153 ObjectIds object_ids = vision_tools::getObjectLabels(instance_mask);
154
155 instance_mask.copyTo(motion_mask);
156
157 const FrameId frame_id = gt_packet.frame_id_;
158
159 //collect only moving labels
160 std::vector<int> moving_labels;
161 for(const ObjectPoseGT& object_pose_gt : gt_packet.object_poses_) {
162 // LOG_IF(WARNING, !object_pose_gt.motion_info_)
163 // << "Object Pose GT (object " << object_pose_gt.object_id_ << ", frame " << object_pose_gt.frame_id_ << " does not have motion info set! Cannot determine of object is moving or not!";
164
165 if(!object_pose_gt.motion_info_) {
166 LOG(WARNING) << "Object Pose GT (object " << object_pose_gt.object_id_ << ", frame " << object_pose_gt.frame_id_ << " does not have motion info set! Cannot determine of object is moving or not!";
167 continue;
168 }
169
170 const auto& motion_info = *object_pose_gt.motion_info_;
171 const ObjectId& object_id = object_pose_gt.object_id_;
172 if(motion_info.is_moving_) {
173 moving_labels.push_back(object_id);
174 }
175
176 //this is just a sanity (debug) check to ensure all the labels in the image
177 //match up with the ones we have already collected in the ground truth packet
178 const auto& it = std::find(object_ids.begin(), object_ids.end(), object_id);
179 CHECK(it != object_ids.end()) << "Object id " << object_id << " appears in gt packet but"
180 "is not in mask when loading motion mask for Virtual Kitti at frame " << frame_id;
181
182 }
183
184 //iterate over each object and if not moving remove
185 for (int i = 0; i < motion_mask.rows; i++)
186 {
187 for (int j = 0; j < motion_mask.cols; j++)
188 {
189
190 int label = motion_mask.at<int>(i, j);
191 if(label == 0) {
192 continue;
193 }
194
195 //check if label is in moving labels. if not, make zero!
196 auto it = std::find(moving_labels.begin(), moving_labels.end(), label);
197 if(it == moving_labels.end()) {
198 motion_mask.at<int>(i, j) = 0;
199 }
200 }
201 }
202
203}

206 std::vector<std::string> trimAndSplit(const std::string& input, const std::string& delimiter) {
207 std::string trim_input = boost::algorithm::trim_right_copy(input);
208 std::vector<std::string> split_line;
209 boost::algorithm::split(split_line, trim_input, boost::is_any_of(delimiter));
