63template <typename MAP>
64gtsam::Pose3 Formulation<MAP>::getInitialOrLinearizedSensorPose(
65 FrameId frame_id) const {
66 const auto accessor = this->accessorFromTheta();
67 // sensor pose from a previous/current linearisation point
68 StateQuery<gtsam::Pose3> X_k_theta = accessor->getSensorPose(frame_id);
69
70 gtsam::Pose3 X_k_initial;
71 CHECK(this->map()->hasInitialSensorPose(frame_id, &X_k_initial));
72 // take either the query value from the map (if we have a previous
73 // initalisation), or the estimate from the camera
74 gtsam::Pose3 X_k;
75 getSafeQuery(X_k, X_k_theta, X_k_initial);
76 return X_k;
77}

185template <typename MAP>
186UpdateObservationResult Formulation<MAP>::updateStaticObservations(
187 FrameId frame_id_k, gtsam::Values& new_values,
188 gtsam::NonlinearFactorGraph& new_factors,
189 const UpdateObservationParams& update_params) {
190 typename Map::Ptr map = this->map();
191 auto accessor = this->accessorFromTheta();
192
193 auto static_point_noise = CHECK_NOTNULL(noise_models_.static_point_noise);
194
195 gtsam::SharedNoiseModel static_keypoint_noise =
196 gtsam::noiseModel::Isotropic::Sigma(2u, 3);
197 static_keypoint_noise = gtsam::noiseModel::Robust::Create(
198 gtsam::noiseModel::mEstimator::Huber::Create(0.01),
199 static_keypoint_noise);
200 // keep track of the new factors added in this function
201 // these are then appended to the internal factors_ and new_factors
202 gtsam::NonlinearFactorGraph internal_new_factors;
203
204 UpdateObservationResult result(update_params);
205
206 // VIODE!!
207 double fx = 376.0;
208 double fy = 376.0;
209 double cx = 376.0;
210 double cy = 240.0;
211 double skew = 0.0; // assuming zero skew
212
213 const auto K = boost::make_shared<gtsam::Cal3_S2>(fx, fy, skew, cx, cy);
214
215 const auto frame_node_k = map->getFrame(frame_id_k);
216 CHECK_NOTNULL(frame_node_k);
217
218 // pose estimate from frontend
219 gtsam::Pose3 T_world_camera_frontend;
220 CHECK(
221 this->map()->hasInitialSensorPose(frame_id_k, &T_world_camera_frontend));
222
223 VLOG(20) << "Looping over " << frame_node_k->static_landmarks.size()
224 << " static lmks for frame " << frame_id_k;
225 for (const auto& lmk_node : frame_node_k->static_landmarks) {
226 const gtsam::Key point_key = lmk_node->makeStaticKey();
227 // check if lmk node is already in map (which should mean it is equivalently
228 // in isam)
229 if (is_other_values_in_map.exists(point_key)) {
230 const Landmark measured = lmk_node->getMeasurement(frame_id_k).landmark;
231 // const auto measured_keypoint =
232 // lmk_node->getMeasurement(frame_id_k).keypoint;
233
234 internal_new_factors
235 .emplace_shared<gtsam::PoseToPointFactor<gtsam::Pose3, Landmark>>(
236 frame_node_k->makePoseKey(), // pose key for this frame
237 point_key, measured, static_point_noise);
238 // internal_new_factors
239 // .emplace_shared<GenericProjectionFactor>(
240 // measured_keypoint,
241 // static_keypoint_noise,
242 // frame_node_k->makePoseKey(), // pose key for this frame
243 // point_key, K);
244
245 if (result.debug_info) result.debug_info->num_static_factors++;
246 result.updateAffectedObject(frame_id_k, 0);
247
248 } else {
249 // see if we have enough observations to add this lmk
250 if (lmk_node->numObservations() < params_.min_static_observations) {
251 continue;
252 }
253
254 // this condition should only run once per tracklet (ie.e the first time
255 // the tracklet has enough observations) we gather the tracklet
256 // observations and then initalise it in the new values these should then
257 // get added to the map and map_->exists() should return true for all
258 // other times
259 const auto& seen_frames = lmk_node->getSeenFrames();
260 for (const auto& seen_frame : seen_frames) {
261 auto seen_frame_id = seen_frame->getId();
262 // only iterate up to the query frame
263 if ((FrameId)seen_frame_id > frame_id_k) {
264 break;
265 }
266
267 // if we should not backtrack, only add the current frame!!!
268 const auto do_backtrack = update_params.do_backtrack;
269 if (!do_backtrack && (FrameId)seen_frame_id < frame_id_k) {
270 continue;
271 }
272
273 const auto measured_keypoint =
274 lmk_node->getMeasurement(seen_frame).keypoint;
275
276 const Landmark& measured =
277 lmk_node->getMeasurement(seen_frame).landmark;
278 internal_new_factors.emplace_shared<PoseToPointFactor>(
279 seen_frame->makePoseKey(), // pose key at previous frames
280 point_key, measured, static_point_noise);
281
282 // internal_new_factors
283 // .emplace_shared<GenericProjectionFactor>(
284 // measured_keypoint,
285 // static_keypoint_noise,
286 // seen_frame->makePoseKey(), // pose key for this frame
287 // point_key, K);
288
289 if (result.debug_info) result.debug_info->num_static_factors++;
290 result.updateAffectedObject(seen_frame_id, 0);
291 }
292
293 // pick the one in this frame
294 const Landmark& measured =
295 MeasurementTraits::point(lmk_node->getMeasurement(frame_id_k));
296 // add initial value, either from measurement or previous estimate
297 gtsam::Point3 lmk_world;
298 getSafeQuery(lmk_world,
299 accessor->getStaticLandmark(lmk_node->tracklet_id),
300 gtsam::Point3(T_world_camera_frontend * measured));
301 new_values.insert(point_key, lmk_world);
302 is_other_values_in_map.insert2(point_key, true);
303
304 if (result.debug_info) result.debug_info->num_new_static_points++;
305 result.updateAffectedObject(frame_id_k, 0);
306 }
307 }
308
309 // update internal data structures
310 theta_.insert_or_assign(new_values);
311 factors_ += internal_new_factors;
312 new_factors += internal_new_factors;
313
314 if (result.debug_info)
315 LOG(INFO) << "Num new static points: "
316 << result.debug_info->num_new_static_points
317 << " Num new static factors "
318 << result.debug_info->num_static_factors;
319 return result;
320}

322template <typename MAP>
323UpdateObservationResult Formulation<MAP>::updateDynamicObservations(
324 FrameId frame_id_k, gtsam::Values& new_values,
325 gtsam::NonlinearFactorGraph& new_factors,
326 const UpdateObservationParams& update_params) {
327 typename Map::Ptr map = this->map();
328 auto accessor = this->accessorFromTheta();
329 // collect noise models to be used
330 auto landmark_motion_noise = noise_models_.landmark_motion_noise;
331 auto dynamic_point_noise = noise_models_.dynamic_point_noise;
332 // keep track of the new factors added in this function
333 // these are then appended to the internal factors_ and new_factors
334 gtsam::NonlinearFactorGraph internal_new_factors;
335 // keep track of the new values added in this function
336 // these are then appended to the internal values_ and new_values
337 gtsam::Values internal_new_values;
338
339 UpdateObservationResult result(update_params);
342 // constexpr static size_t kMinNumberPoints = 3u;
343
344 const FrameId frame_id_k_1 = frame_id_k - 1u;
345 VLOG(20) << "Add dynamic observations between frames " << frame_id_k_1
346 << " and " << frame_id_k;
347 const auto frame_node_k = map->getFrame(frame_id_k);
348 const auto frame_node_k_1 = map->getFrame(frame_id_k_1);
349 CHECK_NOTNULL(frame_node_k_1);
350
351 // pose estimate from frontend
352 // CHECK(parent_->initial_camera_poses_.exists(frame_id_k));
353 // const gtsam::Pose3& T_world_camera_initial_k =
354 // new_values.at<gtsam::Pose3>(CameraPoseSymbol(frame_id_k));
355
356 utils::TimingStatsCollector dyn_obj_itr_timer(this->loggerPrefix() +
357 ".dynamic_object_itr");
358 for (const auto& object_node : frame_node_k->objects_seen) {
359 DebugInfo::ObjectInfo object_debug_info;
360 const ObjectId object_id = object_node->getId();
361
362 // first check that object exists in the previous frame
363 if (!frame_node_k->objectMotionExpected(object_id)) {
364 continue;
365 }
366 // possibly the longest call?
367 // landmarks on this object seen at frame k
368 auto seen_lmks_k = object_node->getLandmarksSeenAtFrame(frame_id_k);
369
370 // if we dont have at least N observations of this object in this frame AND
371 // the previous frame
372 if (seen_lmks_k.size() < params_.min_dynamic_observations ||
373 object_node->getLandmarksSeenAtFrame(frame_id_k_1).size() <
374 params_.min_dynamic_observations) {
375 continue;
376 }
377
378 utils::TimingStatsCollector dyn_point_itr_timer(this->loggerPrefix() +
379 ".dynamic_point_itr");
380 VLOG(10) << "Seen lmks at frame " << frame_id_k << " obj " << object_id
381 << ": " << seen_lmks_k.size();
382 // iterate over each lmk we have on this object
383 for (const auto& obj_lmk_node : seen_lmks_k) {
384 CHECK_EQ(obj_lmk_node->getObjectId(), object_id);
385
386 // see if we have enough observations to add this lmk
387 if (obj_lmk_node->numObservations() < params_.min_dynamic_observations) {
388 continue;
389 }
390
391 TrackletId tracklet_id = obj_lmk_node->getId();
392
393 // if does not exist, we need to go back and all the previous measurements
394 // & factors & motions
395 if (!isDynamicTrackletInMap(obj_lmk_node)) {
396 // add the points/motions from the past
397 auto seen_frames = obj_lmk_node->getSeenFrames();
398 // assert frame observations are continuous?
399
400 // start at the first frame we want to start adding points in as we know
401 // have seen them enough times start from +1, becuase the motion index
402 // is k-1 to k and there is no motion k-2 to k-1 but the first poitns we
403 // want to add are at k-1
404 // TODO: shouldnt this actually just be frame_id_k -
405 // params_.min_dynamic_observations
406 // TODO: how to handle the if starting_motion_frame < a frame actually
407 // seen in!
408 FrameId starting_motion_frame;
409 if (update_params.do_backtrack) {
410 starting_motion_frame =
411 seen_frames.template getFirstIndex<FrameId>() +
412 1u; // as we index the motion from k
413 } else {
414 // start from the requested index, this will still mean that we will
415 // add the previous frame as always add a motion between k-1 and k
416 starting_motion_frame = frame_id_k;
417
418 if (starting_motion_frame <
419 seen_frames.template getFirstIndex<FrameId>() + 1u) {
420 // if the requested starting frame is not the first frame + 1u of
421 // the actul track (ie. the second seen frame) we cannot use it yet
422 // as we have to index BACKWARDS from the starting motion frame if
423 // we used frame_id_k as the starting_motion_frame, we would end up
424 // with an iterator pointing to the end instead, we will have to get
425 // it next frame!!!
426 continue;
427 }
428 }
429
430 auto starting_motion_frame_itr =
431 seen_frames.find(starting_motion_frame);
432 CHECK(starting_motion_frame_itr != seen_frames.end())
433 << "Starting motion frame is " << starting_motion_frame
434 << " but first frame is "
435 << seen_frames.template getFirstIndex<FrameId>();
436
437 std::stringstream ss;
438 ss << "Going back to add point on object " << object_id
439 << " at frames\n";
440
441 // iterate over k-N to k (inclusive) and all all
442 utils::TimingStatsCollector dyn_point_backtrack_timer(
443 this->loggerPrefix() + ".dynamic_point_backtrack");
444 for (auto seen_frames_itr = starting_motion_frame_itr;
445 seen_frames_itr != seen_frames.end(); seen_frames_itr++) {
446 auto seen_frames_itr_prev = seen_frames_itr;
447 std::advance(seen_frames_itr_prev, -1);
448 CHECK(seen_frames_itr_prev != seen_frames.end())
449 << " For object " << object_id;
450
451 auto query_frame_node_k = *seen_frames_itr;
452 auto query_frame_node_k_1 = *seen_frames_itr_prev;
453
454 CHECK_EQ(query_frame_node_k->frame_id,
455 query_frame_node_k_1->frame_id + 1u);
456
457 // add points UP TO AND INCLUDING the current frame
458 if (query_frame_node_k->frame_id > frame_id_k) {
459 break;
460 }
461
462 // point needs to be be in k and k-1 -> we have validated the object
463 // exists in these two frames but not the points
464 CHECK(obj_lmk_node->seenAtFrame(query_frame_node_k->frame_id));
465 if (!obj_lmk_node->seenAtFrame(query_frame_node_k_1->frame_id)) {
466 LOG(WARNING) << "Tracklet " << tracklet_id << " on object "
467 << object_id << " seen at "
468 << query_frame_node_k->frame_id << " but not "
469 << query_frame_node_k_1->frame_id;
470 break;
471 } // this miay mean this this point never gets added?
472
473 ss << query_frame_node_k_1->frame_id << " "
474 << query_frame_node_k->frame_id << "\n";
475
476 const gtsam::Pose3 T_world_camera_k_1 =
477 getInitialOrLinearizedSensorPose(query_frame_node_k_1->frame_id);
478
479 gtsam::Pose3 T_world_camera_k =
480 getInitialOrLinearizedSensorPose(query_frame_node_k->frame_id);
481
482 PointUpdateContextType point_context;
483 point_context.lmk_node = obj_lmk_node;
484 point_context.frame_node_k_1 = query_frame_node_k_1;
485 point_context.frame_node_k = query_frame_node_k;
486 point_context.X_k_measured = T_world_camera_k;
487 point_context.X_k_1_measured = T_world_camera_k_1;
488
489 // this assumes we add all the points in order and have continuous
490 // frames (which we should have?)
491 if (seen_frames_itr == starting_motion_frame_itr) {
492 point_context.is_starting_motion_frame = true;
493 }
494 utils::TimingStatsCollector dyn_point_update_timer(
495 this->loggerPrefix() + ".dyn_point_update_1");
496 // the true set of values that are added from the update
497 dynamicPointUpdateCallback(point_context, result, internal_new_values,
498 internal_new_factors);
499 // // update internal theta and factors
500 // theta_.insert(local_new_values);
501 // // add to the external new_values
502 // new_values.insert(local_new_values);
503 }
504 } else {
505 // these tracklets should already be in the graph so we should only need
506 // to add the new measurements from this frame check that we have
507 // previous point for this frame
508 PointUpdateContextType point_context;
509 point_context.lmk_node = obj_lmk_node;
510 point_context.frame_node_k_1 = frame_node_k_1;
511 point_context.frame_node_k = frame_node_k;
512 point_context.X_k_1_measured =
513 getInitialOrLinearizedSensorPose(frame_node_k_1->frame_id);
514 point_context.X_k_measured =
515 getInitialOrLinearizedSensorPose(frame_node_k->frame_id);
516 point_context.is_starting_motion_frame = false;
517 utils::TimingStatsCollector dyn_point_update_timer(
518 this->loggerPrefix() + ".dyn_point_update_2");
519 // the true set of values that are added from the update
520 // gtsam::Values local_new_values;
521 dynamicPointUpdateCallback(point_context, result, internal_new_values,
522 internal_new_factors);
523 // // update internal theta and factors
524 // theta_.insert(local_new_values);
525 // // add to the external new_values
526 // new_values.insert(local_new_values);
527 }
528 }
529 }
530 // iterate over objects for which a motion was added
531 // becuase we add lots of new points every frame, we may go over the same
532 // object many times to account for backtracking over new points over this
533 // object this is a bit inefficient as we do this iteration even if no new
534 // object values are added becuuse we dont know if the affected frames are
535 // becuase of old points as well as new points
536 // utils::TimingStatsCollector dyn_obj_affected_timer(this->loggerPrefix() +
537 // ".dyn_object_affected");
538 for (const auto& [object_id, frames_affected] :
539 result.objects_affected_per_frame) {
540 VLOG(20) << "Iterating over frames for which a motion was added "
541 << container_to_string(frames_affected) << " for object "
542 << object_id;
543
544 auto object_node = map->getObject(object_id);
545
546 std::vector<FrameId> frames_affected_vector(frames_affected.begin(),
547 frames_affected.end());
548 // TODO: this is NO LONGER the case in the object centric case
549 // TODO: should always have at least two frames (prev and current) as we
550 // must add factors on this frame and the previous frame
551 // CHECK_GE(frames_affected_vector.size(), 2u);
552 for (size_t frame_idx = 0; frame_idx < frames_affected_vector.size();
553 frame_idx++) {
554 const FrameId frame_id = frames_affected_vector.at(frame_idx);
555 auto frame_node_k_impl = map->getFrame(frame_id);
556
557 ObjectUpdateContextType object_update_context;
558 // perform motion check on this frame -> if this is the first frame (of at
559 // least 2) then there should be no motion at this frame (since it is k-1
560 // of a motion pair and there is no k-2)
561 if (frame_idx == 0) {
562 object_update_context.has_motion_pair = false;
563 } else {
564 object_update_context.has_motion_pair = true;
565 }
566 object_update_context.frame_node_k = frame_node_k_impl;
567 object_update_context.object_node = object_node;
568
569 // gtsam::Values local_new_values;
570 objectUpdateContext(object_update_context, result, internal_new_values,
571 internal_new_factors);
572 // // update internal theta and factors
573 // theta_.insert(local_new_values);
574 // // add to the external new_values
575 // new_values.insert(local_new_values);
576 }
577 }
578
579 // this doesnt really work any more as debug info is meant to be per frame?
580 if (result.debug_info && VLOG_IS_ON(20)) {
581 for (const auto& [object_id, object_info] :
582 result.debug_info->getObjectInfos()) {
583 std::stringstream ss;
584 ss << "Object id debug info: " << object_id << "\n";
585 ss << object_info;
586 LOG(INFO) << ss.str();
587 }
588 }
589
590 factors_ += internal_new_factors;
591 new_factors += internal_new_factors;
592 // update internal theta and factors
593 theta_.insert(internal_new_values);
594 // add to the external new_values
595 new_values.insert(internal_new_values);
596
597 return result;
598}

600template <typename MAP>
601void Formulation<MAP>::logBackendFromMap(const BackendMetaData& backend_info) {
602 // TODO:
603 std::string logger_prefix = this->getFullyQualifiedName();
604 const std::string suffix = backend_info.logging_suffix;
605
606 // add suffix to name if required
607 if (!suffix.empty()) {
608 logger_prefix += ("_" + suffix);
609 }
610 BackendLogger::UniquePtr logger =
611 std::make_unique<BackendLogger>(logger_prefix);
612
613 typename Map::Ptr map = this->map();
614 auto accessor = this->accessorFromTheta();
615
616 CHECK(hooks().ground_truth_packets_request);
617 const auto ground_truth_packets = hooks().ground_truth_packets_request();
618
619 CHECK_NOTNULL(backend_info.backend_params);
620 const auto& backend_params = *backend_info.backend_params;
621
622 const ObjectPoseMap object_pose_map = accessor->getObjectPoses();
623
624 for (FrameId frame_k : map->getFrameIds()) {
625 // TODO: hack - only go up to frames < full batch so we actually only
626 // include the optimised alues
627 if (backend_params.use_use_full_batch_opt &&
628 backend_params.full_batch_frame - 1 == (int)frame_k) {
629 break;
630 }
631
632 std::stringstream ss;
633 ss << "Logging data from map at frame " << frame_k;
634
635 // get MotionestimateMap
636 // const MotionEstimateMap motions = map->getMotionEstimates(frame_k);
637 {
638 const MotionEstimateMap motions = accessor->getObjectMotions(frame_k);
639 auto result =
640 logger->logObjectMotion(frame_k, motions, ground_truth_packets);
641 if (result)
642 ss << " Logged " << *result << " motions from " << motions.size()
643 << " computed motions.";
644 else
645 ss << " Could not log object motions.";
646 }
647
648 StateQuery<gtsam::Pose3> X_k_query = accessor->getSensorPose(frame_k);
649
650 if (X_k_query) {
651 logger->logCameraPose(frame_k, X_k_query.get(), ground_truth_packets);
652 } else {
653 LOG(WARNING) << "Could not log camera pose estimate at frame " << frame_k;
654 }
655
656 logger->logObjectPose(frame_k, object_pose_map, ground_truth_packets);
657
658 if (map->frameExists(frame_k)) {
659 auto static_map = accessor->getStaticLandmarkEstimates(frame_k);
660 auto dynamic_map = accessor->getDynamicLandmarkEstimates(frame_k);
661
662 CHECK(X_k_query); // actually not needed for points in world!!
663 logger->logPoints(frame_k, *X_k_query, static_map);
664 logger->logPoints(frame_k, *X_k_query, dynamic_map);
665 }
666
667 LOG(INFO) << ss.str();
668 }
669
670 logger.reset();
671}
