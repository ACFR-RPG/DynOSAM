82 gtsam::Values& values,

147
148 bool addLandmark(LmkNode& lmk, const FrameNode& frame,
149 const UpdateObservationParams& update_params,
150 gtsam::Values& values, gtsam::NonlinearFactorGraph& graph,
151 gtsam::Key& point_key, UpdateObservationResult& result,
152 std::optional<Landmark>& initial) override {
153 point_key = lmk->makeStaticKey();
154 const FrameId frame_k = FrameId(frame->getId());
155
156 const auto& params = this->formulation_->params();
157
158 if (this->isPointAdded(point_key)) {
159 CHECK(lmk->added_to_opt);
160 const auto pose_key = frame->makePoseKey();
161
162 auto [measured_point_local, measurement_covariance] =
163 MeasurementTraits::pointWithCovariance(
164 lmk->getMeasurement(frame_k));
165 CHECK_NOTNULL(measurement_covariance);
166
167 this->robustifyHuber(measurement_covariance);
168
169 auto factor = boost::make_shared<
170 gtsam::PoseToPointFactor<gtsam::Pose3, Landmark>>(
171 pose_key, point_key, measured_point_local, measurement_covariance);
172 graph.add(factor);
173 result.updateAffectedObject(frame_k, 0);
174 return true;
175 } else {
176 CHECK(!lmk->added_to_opt);
177
178 if (lmk->numObservations() < params.min_static_observations) {
179 return false;
180 }
181
182 // this condition should only run once per tracklet (ie.e the first time
183 // the tracklet has enough observations) we gather the tracklet
184 // observations and then initalise it in the new values these should
185 // then get added to the map and map_->exists() should return true for
186 // all other times
187 const auto& seen_frames = lmk->getSeenFrames();
188 for (const auto& seen_frame : seen_frames) {
189 FrameId seen_frame_id = FrameId(seen_frame->getId());
190 // only iterate up to the query frame
191 if (seen_frame_id > frame_k) {
192 break;
193 }
194
195 // if we should not backtrack, only add the current frame!!!
196 const auto do_backtrack = update_params.do_backtrack;
197 if (!do_backtrack && seen_frame_id < frame_k) {
198 continue;
199 }
200
201 const gtsam::Key pose_key = seen_frame->makePoseKey();
202 auto [measured_point_local, measurement_covariance] =
203 MeasurementTraits::pointWithCovariance(
204 lmk->getMeasurement(seen_frame_id));
205 CHECK_NOTNULL(measurement_covariance);
206
207 this->robustifyHuber(measurement_covariance);
208
209 auto factor = boost::make_shared<
210 gtsam::PoseToPointFactor<gtsam::Pose3, Landmark>>(
211 pose_key, point_key, measured_point_local,
212 measurement_covariance);
213
214 graph.add(factor);
215 result.updateAffectedObject(seen_frame_id, 0);
216 }
217
218 const Landmark& measured =
219 MeasurementTraits::point(lmk->getMeasurement(frame_k));
220
221 // TODO: should use getInitialOrLinearizedSensorPose
222 gtsam::Pose3 T_W_X;
223 CHECK(this->formulation_->map()->hasInitialSensorPose(frame_k, &T_W_X));
224
225 Landmark initial_point = T_W_X * measured;
226 initial = initial_point;
227 result.updateAffectedObject(frame_k, 0);
228
229 values.insert(point_key, initial_point);
230 this->markPointAsAdded(point_key);
231 lmk->added_to_opt = true;
232 return true;
233 }
234 }

258 struct StereoProjection : public Base {
259 StereoProjection(FormulationType* formulation) : Base(formulation) {
260 std::shared_ptr<Camera> camera =
261 CHECK_NOTNULL(formulation->sensors().camera);
262 std::shared_ptr<RGBDCamera> rgbd_camera =
263 CHECK_NOTNULL(camera->safeGetRGBDCamera());
264 K_stereo_ = rgbd_camera->getFakeStereoCalib();
265 CHECK_NOTNULL(K_stereo_);
266
267 K_stereo_->print("Stero K\n");
268 K_ = camera->getGtsamCalibration();
269 }

271 bool addLandmark(LmkNode& lmk, const FrameNode& frame,
272 const UpdateObservationParams& update_params,
273 gtsam::Values& values, gtsam::NonlinearFactorGraph& graph,
274 gtsam::Key& point_key, UpdateObservationResult& result,
275 std::optional<Landmark>& initial) override {
276 point_key = lmk->makeStaticKey();
277 const FrameId frame_k = FrameId(frame->getId());
278
279 const auto& params = this->formulation_->params();
280
281 // NOTE: not handling the case a lmk becomes an outlier once already added
282 // to the opt
283 // hoping the robust cost funcion handles this!
284 if (this->isPointAdded(point_key)) {
285 CHECK(lmk->added_to_opt);
286 const auto pose_key = frame->makePoseKey();
287
288 auto stereo_measurement =
289 MeasurementTraits::stereo(lmk->getMeasurement(frame_k));
290 // FOR NOW
291 CHECK(stereo_measurement);
292 auto [measurement, model] = *stereo_measurement;
293
294 this->robustifyHuber(model);
295
296 auto factor = boost::make_shared<GenericStereoFactor>(
297 measurement, model, pose_key, point_key, K_stereo_);
298
299 graph.add(factor);
300 result.updateAffectedObject(frame_k, 0);
301 return true;
302 } else {
303 CHECK(!lmk->added_to_opt);
304
305 using GtsamCamera = Camera::CameraImpl;
306 CameraSet<GtsamCamera> camera_set;
307 gtsam::Point2Vector measurements;
308 gtsam::SharedNoiseModel model;
309
310 // triangulate stereo measurements by treating each stereocamera as a
311 // pair of monocular cameras this is vital for good triangulation!
312 const auto& seen_frames = lmk->getSeenFrames();
313 for (const auto& frame_node_i : seen_frames) {
314 FrameId frame_id_i = frame_node_i->getId();
315 // use the initial pose
316 // in the IMU case the optimised pose will be not so good until visual
317 // odom starts working... or maybe not...
318 gtsam::Pose3 X_W_i;
319 CHECK(this->formulation_->map()->hasInitialSensorPose(frame_id_i,
320 &X_W_i));
321
322 const gtsam::Pose3 leftPose = X_W_i;
323 const gtsam::Cal3_S2 monoCal = K_stereo_->calibration();
324 const GtsamCamera leftCamera_i(leftPose, monoCal);
325 const gtsam::Pose3 left_Pose_right = gtsam::Pose3(
326 gtsam::Rot3(), gtsam::Point3(K_stereo_->baseline(), 0.0, 0.0));
327 const gtsam::Pose3 rightPose = leftPose.compose(left_Pose_right);
328 const GtsamCamera rightCamera_i(rightPose, monoCal);
329
330 // gtsam::Pose3 X_W_i =
331 // this->formulation_->getInitialOrLinearizedSensorPose(frame_id_i);
332
333 // updates the model each time, just uses the last one!
334 // auto [keypoint, model] = MeasurementTraits::keypointWithCovariance(
335 // lmk->getMeasurement(frame_id_i));
336 auto stereo_measurement =
337 MeasurementTraits::stereo(lmk->getMeasurement(frame_id_i));
338 CHECK(stereo_measurement);
339 auto [zi, model] = *stereo_measurement;
340
341 camera_set.push_back(leftCamera_i);
342 measurements.push_back(Point2(zi.uL(), zi.v()));
343 if (!std::isnan(zi.uR())) { // if right point is valid
344 camera_set.push_back(rightCamera_i);
345 measurements.push_back(Point2(zi.uR(), zi.v()));
346 }
347 }
348
349 gtsam::TriangulationParameters triangulation_params;
350 // triangulation_params.useLOST = true;
351 triangulation_params.noiseModel = model;
352 auto triangulation_result = gtsam::triangulateSafe<GtsamCamera>(
353 camera_set, measurements, triangulation_params);
354
355 if (triangulation_result) {
356 const gtsam::Point3 initial_point = *triangulation_result;
357 initial = initial_point;
358
359 double reprojection_error =
360 camera_set.reprojectionError(initial_point, measurements).norm();
361
362 // if error is too large, discard point
363 if (reprojection_error > 3.0) {
364 // mark as outlier for the front-end
365 lmk->inlier = false;
366 return false;
367 }
368 // collect factors
369 gtsam::NonlinearFactorGraph stereo_factors;
370 FrameIds frames_with_good_factors;
371 for (const auto& frame_node_i : seen_frames) {
372 const auto pose_key_i = frame_node_i->makePoseKey();
373 FrameId frame_id_i = frame_node_i->getId();
374
375 auto stereo_measurement =
376 MeasurementTraits::stereo(lmk->getMeasurement(frame_id_i));
377 // FOR NOW
378 CHECK(stereo_measurement);
379 auto [measurement, model] = *stereo_measurement;
380
381 double disparity = measurement.uL() - measurement.uR();
382 if (disparity > 0.5) {
383 this->robustifyHuber(model);
384 auto factor = boost::make_shared<GenericStereoFactor>(
385 measurement, model, pose_key_i, point_key, K_stereo_);
386 stereo_factors += factor;
387 frames_with_good_factors.push_back(frame_id_i);
388 }
389 }
390
391 if (stereo_factors.size() < 2) {
392 return false;
393 }
394
395 for (const auto good_frame_id : frames_with_good_factors) {
396 result.updateAffectedObject(good_frame_id, 0);
397 }
398 graph += stereo_factors;
399
400 values.insert(point_key, initial_point);
401 this->markPointAsAdded(point_key);
402 lmk->added_to_opt = true;
403
404 return true;
405 } else {
406 // mark as outlier for the front-end
407 lmk->inlier = false;
408 return false;
409 }
410 }
411 }
412
413 Camera::CalibrationType::shared_ptr K_;
414 StereoCalibPtr K_stereo_;
415 };

417 static std::unique_ptr<Base> makeImpl(FormulationType* formulation) {
418 const MethodType& method_type =
419 CHECK_NOTNULL(formulation)->params().static_formulation;
420 switch (method_type) {
421 case MethodType::PTP:
422 VLOG(20) << "Using Point-to-Pose formulation for Visual SLAM";
423 return std::make_unique<PTP>(formulation);
424 case MethodType::GENERIC_PROJECTION:
425 VLOG(20) << "Using Generic Projection formulation for Visual SLAM";
426 return std::make_unique<GenericProjection>(formulation);
427 case MethodType::STEREO_PROJECTION:
428 VLOG(20) << "Using Stereo Projection formulation for Visual SLAM";
429 return std::make_unique<StereoProjection>(formulation);
430 default:
431 LOG(FATAL) << "Unknown method type for Static Formulation!";
432 return nullptr;
433 }
434 }

436 private:
437 std::unique_ptr<Base> impl_;
438};

470template <typename MAP>
471gtsam::Pose3 Formulation<MAP>::getInitialOrLinearizedSensorPose(
472 FrameId frame_id) const {
473 const auto accessor = this->accessorFromTheta();
474 // sensor pose from a previous/current linearisation point
475 StateQuery<gtsam::Pose3> X_k_theta = accessor->getSensorPose(frame_id);
476
477 gtsam::Pose3 X_k_initial;
478 CHECK(this->map()->hasInitialSensorPose(frame_id, &X_k_initial));
479 // take either the query value from the map (if we have a previous
480 // initalisation), or the estimate from the camera
481 gtsam::Pose3 X_k;
482 getSafeQuery(X_k, X_k_theta, X_k_initial);
483 return X_k;
484}

552template <typename MAP>
553UpdateObservationResult Formulation<MAP>::updateStaticObservations(
554 FrameId frame_id_k, gtsam::Values& new_values,
555 gtsam::NonlinearFactorGraph& new_factors,
556 const UpdateObservationParams& update_params) {
557 typename Map::Ptr map = this->map();
558 auto accessor = this->accessorFromTheta();
559
560 // keep track of the new factors added in this function
561 // these are then appended to the internal factors_ and new_factors
562 gtsam::NonlinearFactorGraph internal_new_factors;
563
564 UpdateObservationResult result(update_params);
565
566 const size_t initial_factors_size = new_factors.size();
567 const size_t initial_values_size = new_values.size();
568
569 auto frame_node_k = map->getFrame(frame_id_k);
570 CHECK_NOTNULL(frame_node_k);
571
572 internal::StaticFormulationUpdater<MAP> static_updater(this);
573
574 VLOG(20) << "Looping over " << frame_node_k->static_landmarks.size()
575 << " static lmks for frame " << frame_id_k;
576 for (auto lmk_node : frame_node_k->static_landmarks) {
577 gtsam::Key point_key;
578 std::optional<Landmark> initial_value;
579
580 bool lmk_result = static_updater.addLandmark(
581 lmk_node, frame_node_k, update_params, new_values, internal_new_factors,
582 point_key, result, initial_value);
583
584 CHECK_EQ(point_key, lmk_node->makeStaticKey());
585 }
586
587 if (result.debug_info) {
588 result.debug_info->num_static_factors =
589 internal_new_factors.size() - initial_factors_size;
590 result.debug_info->num_new_static_points =
591 new_values.size() - initial_values_size;
592 }
593
594 // update internal data structures
595 theta_.insert_or_assign(new_values);
596 factors_ += internal_new_factors;
597 new_factors += internal_new_factors;
598
599 if (result.debug_info)
600 LOG(INFO) << "Num new static points: "
601 << result.debug_info->num_new_static_points
602 << " Num new static factors "
603 << result.debug_info->num_static_factors;
604 return result;
605}

607template <typename MAP>
608UpdateObservationResult Formulation<MAP>::updateDynamicObservations(
609 FrameId frame_id_k, gtsam::Values& new_values,
610 gtsam::NonlinearFactorGraph& new_factors,
611 const UpdateObservationParams& update_params) {
612 typename Map::Ptr map = this->map();
613 auto accessor = this->accessorFromTheta();
614 // collect noise models to be used
615 auto landmark_motion_noise = noise_models_.landmark_motion_noise;
616 auto dynamic_point_noise = noise_models_.dynamic_point_noise;
617 // keep track of the new factors added in this function
618 // these are then appended to the internal factors_ and new_factors
619 gtsam::NonlinearFactorGraph internal_new_factors;
620 // keep track of the new values added in this function
621 // these are then appended to the internal values_ and new_values
622 gtsam::Values internal_new_values;
623
624 UpdateObservationResult result(update_params);
625
626 // starting slot number is size of new factors
627 // as long as the new factor slot is calculated before adding a new factor
628 const Slot starting_factor_slot = new_factors.size();
629
630 const FrameId frame_id_k_1 = frame_id_k - 1u;
631 VLOG(20) << "Add dynamic observations between frames " << frame_id_k_1
632 << " and " << frame_id_k;
633 const auto frame_node_k = map->getFrame(frame_id_k);
634
635 utils::TimingStatsCollector dyn_obj_itr_timer(this->loggerPrefix() +
636 ".dynamic_object_itr");
637 for (const auto& object_node : frame_node_k->objects_seen) {
638 DebugInfo::ObjectInfo object_debug_info;
639 const ObjectId object_id = object_node->getId();
640
641 // first check that object exists in the previous frame
642 if (!frame_node_k->objectMotionExpected(object_id)) {
643 continue;
644 }
645 // possibly the longest call?
646 // landmarks on this object seen at frame k
647 auto seen_lmks_k = object_node->getLandmarksSeenAtFrame(frame_id_k);
648
649 // if we dont have at least N observations of this object in this frame AND
650 // the previous frame
651 if (seen_lmks_k.size() < params_.min_dynamic_observations ||
652 object_node->getLandmarksSeenAtFrame(frame_id_k_1).size() <
653 params_.min_dynamic_observations) {
654 continue;
655 }
656
657 utils::TimingStatsCollector dyn_point_itr_timer(this->loggerPrefix() +
658 ".dynamic_point_itr");
659 VLOG(10) << "Seen lmks at frame " << frame_id_k << " obj " << object_id
660 << ": " << seen_lmks_k.size();
661 // iterate over each lmk we have on this object
662 for (const auto& obj_lmk_node : seen_lmks_k) {
663 CHECK_EQ(obj_lmk_node->getObjectId(), object_id);
664
665 // see if we have enough observations to add this lmk
666 if (obj_lmk_node->numObservations() < params_.min_dynamic_observations) {
667 continue;
668 }
669
670 TrackletId tracklet_id = obj_lmk_node->getId();
671
672 // if does not exist, we need to go back and all the previous measurements
673 // & factors & motions
674 if (!isDynamicTrackletInMap(obj_lmk_node)) {
675 // add the points/motions from the past
676 auto seen_frames = obj_lmk_node->getSeenFrames();
677 // assert frame observations are continuous?
678
679 // start at the first frame we want to start adding points in as we know
680 // have seen them enough times start from +1, becuase the motion index
681 // is k-1 to k and there is no motion k-2 to k-1 but the first poitns we
682 // want to add are at k-1
683 // TODO: shouldnt this actually just be frame_id_k -
684 // params_.min_dynamic_observations
685 // TODO: how to handle the if starting_motion_frame < a frame actually
686 // seen in!
687 FrameId starting_motion_frame;
688 if (update_params.do_backtrack) {
689 starting_motion_frame =
690 seen_frames.template getFirstIndex<FrameId>() +
691 1u; // as we index the motion from k
692 } else {
693 // start from the requested index, this will still mean that we will
694 // add the previous frame as always add a motion between k-1 and k
695 starting_motion_frame = frame_id_k;
696
697 if (starting_motion_frame <
698 seen_frames.template getFirstIndex<FrameId>() + 1u) {
699 // if the requested starting frame is not the first frame + 1u of
700 // the actul track (ie. the second seen frame) we cannot use it yet
701 // as we have to index BACKWARDS from the starting motion frame if
702 // we used frame_id_k as the starting_motion_frame, we would end up
703 // with an iterator pointing to the end instead, we will have to get
704 // it next frame!!!
705 continue;
706 }
707 }
708
709 auto starting_motion_frame_itr =
710 seen_frames.find(starting_motion_frame);
711 CHECK(starting_motion_frame_itr != seen_frames.end())
712 << "Starting motion frame is " << starting_motion_frame
713 << " but first frame is "
714 << seen_frames.template getFirstIndex<FrameId>();
715
716 std::stringstream ss;
717 ss << "Going back to add point on object " << object_id
718 << " at frames\n";
719
720 // iterate over k-N to k (inclusive) and all all
721 utils::TimingStatsCollector dyn_point_backtrack_timer(
722 this->loggerPrefix() + ".dynamic_point_backtrack");
723 for (auto seen_frames_itr = starting_motion_frame_itr;
724 seen_frames_itr != seen_frames.end(); seen_frames_itr++) {
725 auto seen_frames_itr_prev = seen_frames_itr;
726 std::advance(seen_frames_itr_prev, -1);
727 CHECK(seen_frames_itr_prev != seen_frames.end())
728 << " For object " << object_id;
729
730 auto query_frame_node_k = *seen_frames_itr;
731 auto query_frame_node_k_1 = *seen_frames_itr_prev;
732
733 CHECK_EQ(query_frame_node_k->frame_id,
734 query_frame_node_k_1->frame_id + 1u);
735
736 // add points UP TO AND INCLUDING the current frame
737 if (query_frame_node_k->frame_id > frame_id_k) {
738 break;
739 }
740
741 // point needs to be be in k and k-1 -> we have validated the object
742 // exists in these two frames but not the points
743 CHECK(obj_lmk_node->seenAtFrame(query_frame_node_k->frame_id));
744 if (!obj_lmk_node->seenAtFrame(query_frame_node_k_1->frame_id)) {
745 LOG(WARNING) << "Tracklet " << tracklet_id << " on object "
746 << object_id << " seen at "
747 << query_frame_node_k->frame_id << " but not "
748 << query_frame_node_k_1->frame_id;
749 break;
750 } // this miay mean this this point never gets added?
751
752 ss << query_frame_node_k_1->frame_id << " "
753 << query_frame_node_k->frame_id << "\n";
754
755 const gtsam::Pose3 T_world_camera_k_1 =
756 getInitialOrLinearizedSensorPose(query_frame_node_k_1->frame_id);
757
758 gtsam::Pose3 T_world_camera_k =
759 getInitialOrLinearizedSensorPose(query_frame_node_k->frame_id);
760
761 PointUpdateContextType point_context;
762 point_context.lmk_node = obj_lmk_node;
763 point_context.frame_node_k_1 = query_frame_node_k_1;
764 point_context.frame_node_k = query_frame_node_k;
765 point_context.X_k_measured = T_world_camera_k;
766 point_context.X_k_1_measured = T_world_camera_k_1;
767 point_context.starting_factor_slot = starting_factor_slot;
768
769 // this assumes we add all the points in order and have continuous
770 // frames (which we should have?)
771 if (seen_frames_itr == starting_motion_frame_itr) {
772 point_context.is_starting_motion_frame = true;
773 }
774 utils::TimingStatsCollector dyn_point_update_timer(
775 this->loggerPrefix() + ".dyn_point_update_1");
776 // the true set of values that are added from the update
777 dynamicPointUpdateCallback(point_context, result, internal_new_values,
778 internal_new_factors);
779 // // update internal theta and factors
780 // theta_.insert(local_new_values);
781 // // add to the external new_values
782 // new_values.insert(local_new_values);
783 }
784 } else {
785 const auto frame_node_k_1 = map->getFrame(frame_id_k_1);
786 CHECK_NOTNULL(frame_node_k_1);
787
788 // these tracklets should already be in the graph so we should only need
789 // to add the new measurements from this frame check that we have
790 // previous point for this frame
791 PointUpdateContextType point_context;
792 point_context.lmk_node = obj_lmk_node;
793 point_context.frame_node_k_1 = frame_node_k_1;
794 point_context.frame_node_k = frame_node_k;
795 point_context.X_k_1_measured =
796 getInitialOrLinearizedSensorPose(frame_node_k_1->frame_id);
797 point_context.X_k_measured =
798 getInitialOrLinearizedSensorPose(frame_node_k->frame_id);
799 point_context.starting_factor_slot = starting_factor_slot;
800 point_context.is_starting_motion_frame = false;
801 utils::TimingStatsCollector dyn_point_update_timer(
802 this->loggerPrefix() + ".dyn_point_update_2");
803 // the true set of values that are added from the update
804 // gtsam::Values local_new_values;
805 dynamicPointUpdateCallback(point_context, result, internal_new_values,
806 internal_new_factors);
807 // // update internal theta and factors
808 // theta_.insert(local_new_values);
809 // // add to the external new_values
810 // new_values.insert(local_new_values);
811 }
812 }
813 }
814 // iterate over objects for which a motion was added
815 // becuase we add lots of new points every frame, we may go over the same
816 // object many times to account for backtracking over new points over this
817 // object this is a bit inefficient as we do this iteration even if no new
818 // object values are added becuuse we dont know if the affected frames are
819 // becuase of old points as well as new points
820 // utils::TimingStatsCollector dyn_obj_affected_timer(this->loggerPrefix() +
821 // ".dyn_object_affected");
822 for (const auto& [object_id, frames_affected] :
823 result.objects_affected_per_frame) {
824 VLOG(20) << "Iterating over frames for which a motion was added "
825 << container_to_string(frames_affected) << " for object "
826 << object_id;
827
828 auto object_node = map->getObject(object_id);
829
830 std::vector<FrameId> frames_affected_vector(frames_affected.begin(),
831 frames_affected.end());
832 // TODO: this is NO LONGER the case in the object centric case
833 // TODO: should always have at least two frames (prev and current) as we
834 // must add factors on this frame and the previous frame
835 // CHECK_GE(frames_affected_vector.size(), 2u);
836 for (size_t frame_idx = 0; frame_idx < frames_affected_vector.size();
837 frame_idx++) {
838 const FrameId frame_id = frames_affected_vector.at(frame_idx);
839 auto frame_node_k_impl = map->getFrame(frame_id);
840
841 ObjectUpdateContextType object_update_context;
842 // perform motion check on this frame -> if this is the first frame (of at
843 // least 2) then there should be no motion at this frame (since it is k-1
844 // of a motion pair and there is no k-2)
845 if (frame_idx == 0) {
846 object_update_context.has_motion_pair = false;
847 } else {
848 object_update_context.has_motion_pair = true;
849 }
850 object_update_context.frame_node_k = frame_node_k_impl;
851 object_update_context.object_node = object_node;
852
853 // gtsam::Values local_new_values;
854 objectUpdateContext(object_update_context, result, internal_new_values,
855 internal_new_factors);
856 // // update internal theta and factors
857 // theta_.insert(local_new_values);
858 // // add to the external new_values
859 // new_values.insert(local_new_values);
860 }
861 }
862
863 // this doesnt really work any more as debug info is meant to be per frame?
864 if (result.debug_info && VLOG_IS_ON(20)) {
865 for (const auto& [object_id, object_info] :
866 result.debug_info->getObjectInfos()) {
867 std::stringstream ss;
868 ss << "Object id debug info: " << object_id << "\n";
869 ss << object_info;
870 LOG(INFO) << ss.str();
871 }
872 }
873
874 factors_ += internal_new_factors;
875 new_factors += internal_new_factors;
876 // update internal theta and factors
877 theta_.insert(internal_new_values);
878 // add to the external new_values
879 new_values.insert(internal_new_values);
880
881 return result;
882}
