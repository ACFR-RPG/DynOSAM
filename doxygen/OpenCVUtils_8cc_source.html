58std::string cvTypeToString(int type)
59{
60 std::string r;
61 uchar depth = type & CV_MAT_DEPTH_MASK;
62 uchar chans = 1 + (type >> CV_CN_SHIFT);
63 switch (depth)
64 {
65 case CV_8U:
66 r = "8U";
67 break;
68 case CV_8S:
69 r = "8S";
70 break;
71 case CV_16U:
72 r = "16U";
73 break;
74 case CV_16S:
75 r = "16S";
76 break;
77 case CV_32S:
78 r = "32S";
79 break;
80 case CV_32F:
81 r = "32F";
82 break;
83 case CV_64F:
84 r = "64F";
85 break;
86 default:
87 r = "User";
88 break;
89 }
90 r += "C";
91 r += (chans + '0');
92 return r;
93}

101cv::Mat concatenateImagesHorizontally(const cv::Mat& left_img, const cv::Mat& right_img)
102{
103 cv::Mat left_img_tmp = left_img.clone();
104 if (left_img_tmp.channels() == 1)
105 {
106 cv::cvtColor(left_img_tmp, left_img_tmp, cv::COLOR_GRAY2BGR);
107 }
108 cv::Mat right_img_tmp = right_img.clone();
109 if (right_img_tmp.channels() == 1)
110 {
111 cv::cvtColor(right_img_tmp, right_img_tmp, cv::COLOR_GRAY2BGR);
112 }
113
114 cv::Size left_img_size = left_img_tmp.size();
115 cv::Size right_img_size = right_img_tmp.size();
116
117 CHECK_EQ(left_img_size.height, left_img_size.height) << "Cannot concat horizontally if images are not the same "
118 "height";
119
120 cv::Mat dual_img(left_img_size.height, left_img_size.width + right_img_size.width, CV_8UC3);
121
122 cv::Mat left(dual_img, cv::Rect(0, 0, left_img_size.width, left_img_size.height));
123 left_img_tmp.copyTo(left);
124
125 cv::Mat right(dual_img, cv::Rect(left_img_size.width, 0, right_img_size.width, right_img_size.height));
126
127 right_img_tmp.copyTo(right);
128 return dual_img;
129}

148

150

152 top_img_tmp.copyTo(top);
153
154 cv::Mat bottom(dual_img, cv::Rect(0, top_img_size.height, bottom_img_size.width, bottom_img_size.height));
155
156 bottom_img_tmp.copyTo(bottom);
157 return dual_img;

161void flowToRgb(const cv::Mat& flow, cv::Mat& rgb) {
162 CHECK(flow.channels() == 2) << "Expecting flow in frame to have 2 channels";
163
164 // Visualization part
165 cv::Mat flow_parts[2];
166 cv::split(flow, flow_parts);
167
168 // Convert the algorithm's output into Polar coordinates
169 cv::Mat magnitude, angle, magn_norm;
170 cv::cartToPolar(flow_parts[0], flow_parts[1], magnitude, angle, true);
171 cv::normalize(magnitude, magn_norm, 0.0f, 1.0f, cv::NORM_MINMAX);
172 angle *= ((1.f / 360.f) * (180.f / 255.f));
173
174 // Build hsv image
175 cv::Mat _hsv[3], hsv, hsv8, bgr;
176 _hsv[0] = angle;
177 _hsv[1] = cv::Mat::ones(angle.size(), CV_32F);
178 _hsv[2] = magn_norm;
179 cv::merge(_hsv, 3, hsv);
180 hsv.convertTo(hsv8, CV_8U, 255.0);
181
182 // Display the results
183 cv::cvtColor(hsv8, rgb, cv::COLOR_HSV2BGR);
184}

187cv::Mat labelMaskToRGB(const cv::Mat& mask, int background_label, const cv::Mat& rgb) {
188 CHECK(mask.channels() == 1) << "Expecting mask input to have channels 1";
189 CHECK(rgb.channels() == 3) << "Expecting rgb input to have channels 3";
190 cv::Mat mask_viz;
191 rgb.copyTo(mask_viz);
192
193 for (int i = 0; i < mask.rows; i++)
194 {
195 for (int j = 0; j < mask.cols; j++)
196 {
197 // background is zero
198 if (mask.at<int>(i, j) != background_label)
199 {
200
201 cv::Scalar color = Color::uniqueId(mask.at<int>(i, j));
202 // rgb or bgr?
203 mask_viz.at<cv::Vec3b>(i, j)[0] = color[0];
204 mask_viz.at<cv::Vec3b>(i, j)[1] = color[1];
205 mask_viz.at<cv::Vec3b>(i, j)[2] = color[2];
206 }
207 }
208 }
209
210 return mask_viz;
211}

219void getDisparityVis(cv::InputArray src, cv::OutputArray dst, int unknown_disparity) {
220 CHECK(!src.empty() && (src.depth() == CV_16S || src.depth() == CV_32F) &&
221 (src.channels() == 1));
222 // cv::Mat srcMat = src.getMat();
223 cv::Mat srcMat = src.getMat();
224 dst.create(srcMat.rows, srcMat.cols, CV_8UC1);
225 cv::Mat& dstMat = dst.getMatRef();
226
227 // Check its extreme values.
228 double min_val;
229 double max_val;
230 cv::minMaxLoc(src, &min_val, &max_val);
231
232 // Multiply by 1.25 just to saturate a bit the extremums.
233 double scale = 2.0 * 255.0 / (max_val - min_val);
234 srcMat.convertTo(dstMat, CV_8UC1, scale / 16.0);
235 dstMat &= (srcMat != unknown_disparity);
236}

238void drawLabeledBoundingBox(cv::Mat& image, const std::string& label, const cv::Scalar& colour, const cv::Rect& bounding_box) {
239 constexpr static double kFontScale = 0.6;
240 constexpr static int kFontFace = cv::FONT_HERSHEY_SIMPLEX;
241 constexpr static int kThickness = 1;
242
243 // Top left corner.
244 const cv::Point& tlc = bounding_box.tl();
245
246 // Display the label at the top of the bounding box.
247 int base_line;
248 cv::Size label_size = cv::getTextSize(label, kFontFace, kFontScale, kThickness, &base_line);
249
250 //draw on filled black rectangle that the object label will then be drawn over to make it easier to see
251 constexpr static int pixel_buffer = 2; //pixel buffer around the rectangle
252 const cv::Point tlc_black_rectangle(tlc.x, tlc.y - label_size.height - pixel_buffer);
253 const cv::Point brc_black_rectangle(tlc.x + label_size.width + pixel_buffer, tlc.y);
254 cv::rectangle(image, tlc_black_rectangle, brc_black_rectangle, cv::Scalar(0,0,0), -1);
255 // Put the label on the black rectangle.
256 // Note: text origin starts from the bottom left corner of the text string in the image and we add a pixel buffer along y to make it look better
257 cv::putText(image, label, cv::Point(tlc.x, tlc.y - 2), kFontFace, kFontScale, cv::Scalar(255, 255, 255), kThickness);
258 //draw bounding box with line thickness kThickness
259 cv::rectangle(image, bounding_box, colour, 2);
260}

264bool compareCvMatsUpToTol(const cv::Mat& mat1, const cv::Mat& mat2, const double& tol)
265{
266 CHECK_EQ(mat1.size(), mat2.size());
267 CHECK_EQ(mat1.type(), mat2.type());
268
269 // treat two empty mat as identical as well
270 if (mat1.empty() && mat2.empty())
271 {
272 LOG(WARNING) << "CvMatCmp: asked comparison of 2 empty matrices.";
273 return true;
274 }
275
276 // Compare the two matrices!
277 cv::Mat diff = mat1 - mat2;
278 return cv::checkRange(diff, true, nullptr, -tol, tol);
279}

286cv::Mat readOpticalFlow( const std::string& path ) {
287 using namespace cv;
288
289 Mat_<Point2f> flow;
290 std::ifstream file(path.c_str(), std::ios_base::binary);
291 if ( !file.good() )
292 return CV_CXX_MOVE(flow); // no file - return empty matrix
293
294 float tag;
295 file.read((char*) &tag, sizeof(float));
296 if ( tag != FLOW_TAG_FLOAT )
297 return CV_CXX_MOVE(flow);
298
299 int width, height;
300
301 file.read((char*) &width, 4);
302 file.read((char*) &height, 4);
303
304 flow.create(height, width);
305
306 for ( int i = 0; i < flow.rows; ++i )
307 {
308 for ( int j = 0; j < flow.cols; ++j )
309 {
310 Point2f u;
311 file.read((char*) &u.x, sizeof(float));
312 file.read((char*) &u.y, sizeof(float));
313 if ( !file.good() )
314 {
315 flow.release();
316 return CV_CXX_MOVE(flow);
317 }
318
319 flow(i, j) = u;
320 }
321 }
322 file.close();
323 return CV_CXX_MOVE(flow);
324
325}

328bool writeOpticalFlow( const std::string& path, const cv::Mat& flow) {
329 using namespace cv;
330 const int nChannels = 2;
331
332 Mat input = flow;
333 if ( input.channels() != nChannels || input.depth() != CV_32F || path.length() == 0 )
334 return false;
335
336 std::ofstream file(path.c_str(), std::ofstream::binary);
337 if ( !file.good() )
338 return false;
339
340 int nRows, nCols;
341
342 nRows = (int) input.size().height;
343 nCols = (int) input.size().width;
344
345 const int headerSize = 12;
346 char header[headerSize];
347 memcpy(header, FLOW_TAG_STRING, 4);
348 // size of ints is known - has been asserted in the current function
349 memcpy(header + 4, reinterpret_cast<const char*>(&nCols), sizeof(nCols));
350 memcpy(header + 8, reinterpret_cast<const char*>(&nRows), sizeof(nRows));
351 file.write(header, headerSize);
352 if ( !file.good() )
353 return false;
354
355// if ( input.isContinuous() ) //matrix is continous - treat it as a single row
356// {
357// nCols *= nRows;
358// nRows = 1;
359// }
360
361 int row;
362 char* p;
363 for ( row = 0; row < nRows; row++ )
364 {
365 p = input.ptr<char>(row);
366 file.write(p, nCols * nChannels * sizeof(float));
367 if ( !file.good() )
368 return false;
369 }
370 file.close();
371 return true;
372}
