87double calculateIoU(const cv::Rect& a, const cv::Rect& b) {
88 cv::Rect intersection = a & b;
89 double intersection_area = intersection.area();
90
91 // Calculate union area
92 double union_area = a.area() + b.area() - intersection_area;
93
94 // Handle case where unionArea is zero (e.g., no overlap and one or both rects
95 // have zero area)
96 if (union_area == 0) {
97 return 0.0;
98 }
99
100 // Calculate IoU
101 return intersection_area / union_area;
102}

119std::string cvTypeToString(int type) {
120 std::string r;
121 uchar depth = type & CV_MAT_DEPTH_MASK;
122 uchar chans = 1 + (type >> CV_CN_SHIFT);
123 switch (depth) {
124 case CV_8U:
125 r = "8U";
126 break;
127 case CV_8S:
128 r = "8S";
129 break;
130 case CV_16U:
131 r = "16U";
132 break;
133 case CV_16S:
134 r = "16S";
135 break;
136 case CV_32S:
137 r = "32S";
138 break;
139 case CV_32F:
140 r = "32F";
141 break;
142 case CV_64F:
143 r = "64F";
144 break;
145 default:
146 r = "User";
147 break;
148 }
149 r += "C";
150 r += (chans + '0');
151 return r;
152}

158cv::Mat concatenateImagesHorizontally(const cv::Mat& left_img,
159 const cv::Mat& right_img) {
160 cv::Mat left_img_tmp = left_img.clone();
161 if (left_img_tmp.channels() == 1) {
162 cv::cvtColor(left_img_tmp, left_img_tmp, cv::COLOR_GRAY2BGR);
163 }
164 cv::Mat right_img_tmp = right_img.clone();
165 if (right_img_tmp.channels() == 1) {
166 cv::cvtColor(right_img_tmp, right_img_tmp, cv::COLOR_GRAY2BGR);
167 }
168
169 cv::Size left_img_size = left_img_tmp.size();
170 cv::Size right_img_size = right_img_tmp.size();
171
172 CHECK_EQ(left_img_size.height, left_img_size.height)
173 << "Cannot concat horizontally if images are not the same "
174 "height";
175
176 cv::Mat dual_img(left_img_size.height,
177 left_img_size.width + right_img_size.width, CV_8UC3);
178
179 cv::Mat left(dual_img,
180 cv::Rect(0, 0, left_img_size.width, left_img_size.height));
181 left_img_tmp.copyTo(left);
182
183 cv::Mat right(dual_img, cv::Rect(left_img_size.width, 0, right_img_size.width,
184 right_img_size.height));
185
186 right_img_tmp.copyTo(right);
187 return dual_img;
188}

222void flowToRgb(const cv::Mat& flow, cv::Mat& rgb) {
223 CHECK(flow.channels() == 2) << "Expecting flow in frame to have 2 channels";
224
225 // Visualization part
226 cv::Mat flow_parts[2];
227 cv::split(flow, flow_parts);
228
229 // Convert the algorithm's output into Polar coordinates
230 cv::Mat magnitude, angle, magn_norm;
231 cv::cartToPolar(flow_parts[0], flow_parts[1], magnitude, angle, true);
232 cv::normalize(magnitude, magn_norm, 0.0f, 1.0f, cv::NORM_MINMAX);
233 angle *= ((1.f / 360.f) * (180.f / 255.f));
234
235 // Build hsv image
236 cv::Mat _hsv[3], hsv, hsv8, bgr;
237 _hsv[0] = angle;
238 _hsv[1] = cv::Mat::ones(angle.size(), CV_32F);
239 _hsv[2] = magn_norm;
240 cv::merge(_hsv, 3, hsv);
241 hsv.convertTo(hsv8, CV_8U, 255.0);
242
243 // Display the results
244 cv::cvtColor(hsv8, rgb, cv::COLOR_HSV2BGR);
245}

247void labelMaskToRGB(const cv::Mat& mask, const cv::Mat& rgb_input,
248 cv::Mat& output, float alpha, int background_label) {
249 CHECK_EQ(mask.channels(), 1u) << "Expecting mask input to have channels 1";
250 CHECK_EQ(rgb_input.channels(), 3) << "Expecting rgb input to have channels 3";
251 CHECK(cvSizeEqual(mask, rgb_input));
252
253 rgb_input.copyTo(output);
254
255 FunctionalParallelOpenCVMat process(output, [&](cv::Mat& viz, int i, int j) {
256 const ObjectId* mask_ptr = mask.ptr<ObjectId>(i);
257 const ObjectId object_label = mask_ptr[j];
258 if (object_label != background_label) {
259 cv::Scalar color = Color::uniqueId(object_label).bgra();
260
261 cv::Vec3b* viz_row = viz.ptr<cv::Vec3b>(i);
262
263 // blend colour with existing values in viz_row size we have already
264 // copied the rgb input values into the output (referenced as viz)
265 for (int c = 0; c < 3; ++c) {
266 viz_row[j][c] = static_cast<uchar>(alpha * color[c] +
267 (1.0 - alpha) * viz_row[j][c]);
268 }
269 }
270 });
271 process.run();
272}

300void getDisparityVis(cv::InputArray src, cv::OutputArray dst,
301 int unknown_disparity) {
302 CHECK(!src.empty() && (src.depth() == CV_16S || src.depth() == CV_32F) &&
303 (src.channels() == 1));
304 // cv::Mat srcMat = src.getMat();
305 cv::Mat srcMat = src.getMat();
306 dst.create(srcMat.rows, srcMat.cols, CV_8UC1);
307 cv::Mat& dstMat = dst.getMatRef();
308
309 // Check its extreme values.
310 double min_val;
311 double max_val;
312 cv::minMaxLoc(src, &min_val, &max_val);
313
314 // Multiply by 1.25 just to saturate a bit the extremums.
315 double scale = 2.0 * 255.0 / (max_val - min_val);
316 srcMat.convertTo(dstMat, CV_8UC1, scale / 16.0);
317 dstMat &= (srcMat != unknown_disparity);
318}

320void drawLabeledBoundingBox(cv::Mat& image, const std::string& label,
321 const cv::Scalar& colour,
322 const cv::Rect& bounding_box,
323 const int& bb_thickness) {
324 constexpr static double kFontScale = 0.6;
325 constexpr static int kFontFace = cv::FONT_HERSHEY_SIMPLEX;
326 // text thickness
327 constexpr static int kThickness = 1;
328
329 // Top left corner.
330 const cv::Point& tlc = bounding_box.tl();
331
332 // Display the label at the top of the bounding box.
333 int base_line;
334 cv::Size label_size =
335 cv::getTextSize(label, kFontFace, kFontScale, kThickness, &base_line);
336
337 // draw on filled black rectangle that the object label will then be drawn
338 // over to make it easier to see
339 constexpr static int pixel_buffer = 2; // pixel buffer around the rectangle
340 const cv::Point tlc_black_rectangle(tlc.x,
341 tlc.y - label_size.height - pixel_buffer);
342 const cv::Point brc_black_rectangle(tlc.x + label_size.width + pixel_buffer,
343 tlc.y);
344 cv::rectangle(image, tlc_black_rectangle, brc_black_rectangle,
345 cv::Scalar(0, 0, 0), -1);
346 // Put the label on the black rectangle.
347 // Note: text origin starts from the bottom left corner of the text string in
348 // the image and we add a pixel buffer along y to make it look better
349 cv::putText(image, label, cv::Point(tlc.x, tlc.y - 2), kFontFace, kFontScale,
350 cv::Scalar(255, 255, 255), kThickness);
351 // draw bounding box with line thickness
352 cv::rectangle(image, bounding_box, colour, bb_thickness);
353}

355void drawObjectPoseAxes(cv::Mat& image, const cv::Mat& K, const cv::Mat& D,
356 const std::vector<gtsam::Pose3>& poses_c, float scale) {
357 cv::Mat K_float, D_float;
358 K.copyTo(K_float);
359 D.copyTo(D_float);
360
361 K_float.convertTo(K_float, CV_32F);
362 D_float.convertTo(D_float, CV_32F);
363
364 // Define the 3D points for the axis (origin and points along X, Y, Z)
365 static const std::vector<cv::Point3f> axis_points = {
366 {0, 0, 0}, // Origin
367 {scale, 0, 0}, // X-axis (red)
368 {0, scale, 0}, // Y-axis (green)
369 {0, 0, scale} // Z-axis (blue)
370 };
371
372 for (const gtsam::Pose3& pose : poses_c) {
373 auto [rot_matrix, tvec] = Pose2cvmats(pose);
374
375 rot_matrix.convertTo(rot_matrix, CV_32F);
376 tvec.convertTo(tvec, CV_32F);
377
378 cv::Mat rvec;
379 cv::Rodrigues(rot_matrix,
380 rvec); // Convert rotation matrix to Rodrigues vector
381
382 std::vector<cv::Point2f> image_points;
383 cv::projectPoints(axis_points, rvec, tvec, K_float, D_float, image_points);
384
385 // Draw axes on the image
386 cv::line(image, image_points[0], image_points[1], cv::Scalar(0, 0, 255),
387 2); // X-axis in red
388 cv::line(image, image_points[0], image_points[2], cv::Scalar(0, 255, 0),
389 2); // Y-axis in green
390 cv::line(image, image_points[0], image_points[3], cv::Scalar(255, 0, 0),
391 2); // Z-axis in blue
392 }
393}

439bool compareCvMatsUpToTol(const cv::Mat& mat1, const cv::Mat& mat2,
440 const double& tol) {
441 CHECK_EQ(mat1.size(), mat2.size());
442 CHECK_EQ(mat1.type(), mat2.type());
443
444 // treat two empty mat as identical as well
445 if (mat1.empty() && mat2.empty()) {
446 LOG(WARNING) << "CvMatCmp: asked comparison of 2 empty matrices.";
447 return true;
448 }
449
450 // Compare the two matrices!
451 cv::Mat diff = mat1 - mat2;
452 return cv::checkRange(diff, true, nullptr, -tol, tol);
453}

458cv::Mat readOpticalFlow(const std::string& path) {
459 using namespace cv;
460
461 Mat_<Point2f> flow;
462 std::ifstream file(path.c_str(), std::ios_base::binary);
463 if (!file.good()) return flow; // no file - return empty matrix
464
465 float tag;
466 file.read((char*)&tag, sizeof(float));
467 if (tag != FLOW_TAG_FLOAT) return flow;
468
469 int width, height;
470
471 file.read((char*)&width, 4);
472 file.read((char*)&height, 4);
473
474 flow.create(height, width);
475
476 for (int i = 0; i < flow.rows; ++i) {
477 for (int j = 0; j < flow.cols; ++j) {
478 Point2f u;
479 file.read((char*)&u.x, sizeof(float));
480 file.read((char*)&u.y, sizeof(float));
481 if (!file.good()) {
482 flow.release();
483 return flow;
484 }
485
486 flow(i, j) = u;
487 }
488 }
489 file.close();
490 return flow;
491}

493bool writeOpticalFlow(const std::string& path, const cv::Mat& flow) {
494 using namespace cv;
495 const int nChannels = 2;
496
497 Mat input = flow;
498 if (input.channels() != nChannels || input.depth() != CV_32F ||
499 path.length() == 0)
500 return false;
501
502 std::ofstream file(path.c_str(), std::ofstream::binary);
503 if (!file.good()) return false;
504
505 int nRows, nCols;
506
507 nRows = (int)input.size().height;
508 nCols = (int)input.size().width;
509
510 const int headerSize = 12;
511 char header[headerSize];
512 memcpy(header, FLOW_TAG_STRING, 4);
513 // size of ints is known - has been asserted in the current function
514 memcpy(header + 4, reinterpret_cast<const char*>(&nCols), sizeof(nCols));
515 memcpy(header + 8, reinterpret_cast<const char*>(&nRows), sizeof(nRows));
516 file.write(header, headerSize);
517 if (!file.good()) return false;
518
519 // if ( input.isContinuous() ) //matrix is continous - treat it as a single
520 // row
521 // {
522 // nCols *= nRows;
523 // nRows = 1;
524 // }
525
526 int row;
527 char* p;
528 for (row = 0; row < nRows; row++) {
529 p = input.ptr<char>(row);
530 file.write(p, nCols * nChannels * sizeof(float));
531 if (!file.good()) return false;
532 }
533 file.close();
534 return true;
535}
