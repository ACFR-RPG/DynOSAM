315 template <typename ValueType>
316 StateQuery<ValueType> query(gtsam::Key key) const {
317 boost::optional<const gtsam::Value&> value_opt = this->getValueImpl(key);
318 if (value_opt) {
319 // Check the type and throw exception if incorrect
320 // h() split in two lines to avoid internal compiler error (MSVC2017)
321 const gtsam::Value* value = value_opt.get_ptr();
322 auto h = gtsam::internal::handle<ValueType>();
323 return StateQuery<ValueType>(key, h(key, value));
324 } else {
325 return StateQuery<ValueType>::NotInMap(key);
326 }
327 }

330 mutable std::mutex mutex_;
