460 gtsam::FastMap<ObjectId, Value> collectByFrame(FrameId frame_id) const {
461 gtsam::FastMap<ObjectId, Value> object_map;
462 for (const auto& [object_id, frame_map] : *this) {
463 if (frame_map.exists(frame_id)) {
464 object_map.insert2(object_id, frame_map.at(frame_id));
465 }
466 }
467 return object_map;
468 }
469
470 private:
471 template <typename Container, typename Return>
472 static Return atImpl(Container* container, ObjectId object_id,
473 FrameId frame_id) {
474 size_t out_of_range_flag;
475 const bool result =
476 container->existsImpl(object_id, frame_id, out_of_range_flag);
477 if (result) {
478 CHECK_EQ(out_of_range_flag, 2u);
479 return container->at(object_id).at(frame_id);
480 } else {
481 std::stringstream ss;
482 ss << "Index out of range: "
483 << ((out_of_range_flag == 0) ? " object id " : " frame id")
484 << " missing. Full query - (object id " << object_id << ", frame id "
485 << frame_id << ").";
486 throw std::out_of_range(ss.str());
487 }
488 }
489
505 bool existsImpl(ObjectId object_id, FrameId frame_id,
506 size_t& out_of_range_flag) const {
507 if (!this->exists(object_id)) {
508 out_of_range_flag = 0;
509 return false;
510 }
511
512 const auto& frame_map = this->at(object_id);
513 if (!frame_map.exists(frame_id)) {
514 out_of_range_flag = 1;
515 return false;
516 } else {
517 out_of_range_flag = 2;
518 return true;
519 }
520 }
