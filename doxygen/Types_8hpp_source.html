115template <typename T>
116class Flags {
117 public:
118 using Storage = std::underlying_type_t<T>;
119
120 Flags() : bits_(0) {}
121 explicit Flags(T value) : bits_(static_cast<Storage>(value)) {}
122
123 Flags& set(T value) {
124 bits_ |= static_cast<Storage>(value);
125 return *this;
126 }
127
128 bool has(T value) const { return (bits_ & static_cast<Storage>(value)) != 0; }
129
130 bool hasAll(T value) const {
131 return (bits_ & static_cast<Storage>(value)) == static_cast<Storage>(value);
132 }
133
134 bool hasAny(T value) const {
135 return has(value); // same logic
136 }
137
138 bool operator==(const Flags& other) const { return bits_ == other.bits_; }
139
140 bool operator==(const T& value) const { return has(value); }
141
142 bool operator!=(const Flags& other) const { return !(*this == other); }
143
144 bool operator!=(const T& value) const { return !(*this == value); }
145
146 Flags& operator=(const T& value) {
147 bits_ = static_cast<Storage>(value);
148 return *this;
149 }
150
151 Storage raw() const { return bits_; }
152
153 private:
154 Storage bits_;
155};

555 bool insert22(ObjectId object_id, FrameId frame_id, const Value& value) {
556 if (!this->exists(object_id)) {
557 this->insert2(object_id, NestedBase{});

568 result &= this->insert22(object_id, frame_id, estimate_value);

570 return result;

615 ObjectIds gatherInvolvedObjects() const {

629 for (const auto& [frame_id, _] : frame_map) {
630 frame_ids.push_back(frame_id);
631 }
632 }

637 template <typename Container, typename Return>
638 static Return atImpl(Container* container, ObjectId object_id,
639 FrameId frame_id) {
640 size_t out_of_range_flag;
641 const bool result =
642 container->existsImpl(object_id, frame_id, out_of_range_flag);
643 if (result) {
644 CHECK_EQ(out_of_range_flag, 2u);
645 return container->at(object_id).at(frame_id);
646 } else {
647 std::stringstream ss;
648 ss << "Index out of range: "
649 << ((out_of_range_flag == 0) ? " object id " : " frame id")
650 << " missing. Full query - (object id " << object_id << ", frame id "
651 << frame_id << ").";
652 throw std::out_of_range(ss.str());
653 }
654 }
655
671 bool existsImpl(ObjectId object_id, FrameId frame_id,
672 size_t& out_of_range_flag) const {
673 if (!this->exists(object_id)) {
674 out_of_range_flag = 0;
675 return false;
676 }
677
678 const auto& frame_map = this->at(object_id);
679 if (!frame_map.exists(frame_id)) {
680 out_of_range_flag = 1;
681 return false;
682 } else {
683 out_of_range_flag = 2;
684 return true;
685 }
686 }
