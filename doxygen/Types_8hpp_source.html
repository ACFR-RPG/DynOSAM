114template <typename T>
115class Flags {
116 public:
117 using Storage = std::underlying_type_t<T>;
118
119 Flags() : bits_(0) {}
120 explicit Flags(T value) : bits_(static_cast<Storage>(value)) {}
121
122 Flags& set(T value) {
123 bits_ |= static_cast<Storage>(value);
124 return *this;
125 }
126
127 bool has(T value) const { return (bits_ & static_cast<Storage>(value)) != 0; }
128
129 bool hasAll(T value) const {
130 return (bits_ & static_cast<Storage>(value)) == static_cast<Storage>(value);
131 }
132
133 bool hasAny(T value) const {
134 return has(value); // same logic
135 }
136
137 bool operator==(const Flags& other) const { return bits_ == other.bits_; }
138
139 bool operator==(const T& value) const { return has(value); }
140
141 bool operator!=(const Flags& other) const { return !(*this == other); }
142
143 bool operator!=(const T& value) const { return !(*this == value); }
144
145 Flags& operator=(const T& value) {
146 bits_ = static_cast<Storage>(value);
147 return *this;
148 }
149
150 Storage raw() const { return bits_; }
151
152 private:
153 Storage bits_;
154};

549 bool insert22(ObjectId object_id, FrameId frame_id, const Value& value) {
550 if (!this->exists(object_id)) {

631 template <typename Container, typename Return>
632 static Return atImpl(Container* container, ObjectId object_id,
633 FrameId frame_id) {
634 size_t out_of_range_flag;
635 const bool result =
636 container->existsImpl(object_id, frame_id, out_of_range_flag);
637 if (result) {
638 CHECK_EQ(out_of_range_flag, 2u);
639 return container->at(object_id).at(frame_id);
640 } else {
641 std::stringstream ss;
642 ss << "Index out of range: "
643 << ((out_of_range_flag == 0) ? " object id " : " frame id")
644 << " missing. Full query - (object id " << object_id << ", frame id "
645 << frame_id << ").";
646 throw std::out_of_range(ss.str());
647 }
648 }
649
665 bool existsImpl(ObjectId object_id, FrameId frame_id,
666 size_t& out_of_range_flag) const {
667 if (!this->exists(object_id)) {
668 out_of_range_flag = 0;
669 return false;
670 }
671
672 const auto& frame_map = this->at(object_id);
673 if (!frame_map.exists(frame_id)) {
674 out_of_range_flag = 1;
675 return false;
676 } else {
677 out_of_range_flag = 2;
678 return true;
679 }
680 }
