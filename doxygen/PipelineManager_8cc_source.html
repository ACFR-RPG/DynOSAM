45DynoPipelineManager::DynoPipelineManager(
46 const DynoParams& params, DataProvider::Ptr data_loader,
47 FrontendDisplay::Ptr frontend_display, BackendDisplay::Ptr backend_display,
48 BackendModuleFactory::Ptr factory, const ExternalHooks::Ptr external_hooks)
49 : params_(params),
50 use_offline_frontend_(FLAGS_frontend_from_file),
51 data_loader_(std::move(data_loader)),
52 displayer_(&display_queue_, params.parallelRun())
53
54{
55 LOG(INFO) << "Starting DynoPipelineManager";
56
57 CHECK(data_loader_);
58 CHECK(frontend_display);
59
60 data_interface_ =
61 std::make_unique<DataInterfacePipeline>(params_.parallelRun());
62 data_loader_->registerImageContainerCallback(
63 std::bind(&dyno::DataInterfacePipeline::fillImageContainerQueue,
64 data_interface_.get(), std::placeholders::_1));
65
66 // if an external hook exists to update the time, add callback in
67 // datainterface that will be triggered when new data is added to the output
68 // queue (from the DataInterfacePipeline) this is basically used to alert ROS
69 // to a new timestamp which we then publish to /clock
70 if (external_hooks && external_hooks->update_time) {
71 LOG(INFO) << "Added pre-queue callback to register new Timestampd data "
72 "with an external module";
73 data_interface_->registerPreQueueContainerCallback(
74 [external_hooks](const ImageContainer::Ptr image_container) -> void {
75 external_hooks->update_time(image_container->timestamp());
76 });
77 }
78
79 // ground truth
80 data_loader_->registerGroundTruthPacketCallback(
81 std::bind(&dyno::DataInterfacePipeline::addGroundTruthPacket,
82 data_interface_.get(), std::placeholders::_1));
83
84 // register single and multi IMU callbacks to the data loader
85 data_loader_->registerImuSingleCallback(std::bind(
86 static_cast<void (DataInterfacePipeline::*)(const ImuMeasurement&)>(
87 &dyno::DataInterfacePipeline::fillImuQueue),
88 data_interface_.get(), std::placeholders::_1));
89
90 data_loader_->registerImuMultiCallback(std::bind(
91 static_cast<void (DataInterfacePipeline::*)(const ImuMeasurements&)>(
92 &dyno::DataInterfacePipeline::fillImuQueue),
93 data_interface_.get(), std::placeholders::_1));
94
95 // preprocessing
96 data_interface_->registerImageContainerPreprocessor(
97 std::bind(&dyno::DataProvider::imageContainerPreprocessor,
98 data_loader_.get(), std::placeholders::_1));
99
100 // push data from the data interface to the frontend module
101 data_interface_->registerOutputQueue(&frontend_input_queue_);
102
103 CameraParams camera_params;
104 if (params_.preferDataProviderCameraParams() &&
105 data_loader_->getCameraParams().has_value()) {
106 LOG(INFO) << "Using camera params from DataProvider, not the config in the "
107 "CameraParams.yaml!";
108 camera_params = *data_loader_->getCameraParams();
109 } else {
110 LOG(INFO) << "Using camera params specified in CameraParams.yaml!";
111 camera_params = params_.camera_params_;
112 }
115
116 ImuParams imu_params;
117 if (params_.preferDataProviderImuParams() &&
118 data_loader_->getImuParams().has_value()) {
119 LOG(INFO) << "Using imu params from DataProvider, not the config in the "
120 "ImuParams.yaml!";
121 imu_params = *data_loader_->getImuParams();
122 } else {
123 LOG(INFO) << "Using imu params specified in ImuParams.yaml!";
124 imu_params = params_.imu_params_;
125 }
126
127 // update the imu params that will actually get sent to the frontend
128 params_.frontend_params_.imu_params = imu_params;
129
130 loadPipelines(camera_params, frontend_display, backend_display, factory);
131
132 std::string cuda_enabled_message;
133 utils::opencvCudaAvailable(&cuda_enabled_message);
134 LOG(INFO) << cuda_enabled_message;
135
136 launchSpinners();
137}

173bool DynoPipelineManager::spin() {
174 std::function<bool()> spin_func;
175
176 if (use_offline_frontend_) {
177 // if we have an offline frontend only spiun the frontend pipeline
178 // and no need to spin the viz (TODO: right now this is only images and not
179 // the actual pipelines...)
180 spin_func = [=]() -> bool {
181 if (frontend_pipeline_->isWorking()) {
182 if (!params_.parallelRun()) {
183 frontend_pipeline_->spinOnce();
184 if (backend_pipeline_) backend_pipeline_->spinOnce();
185 }
186 return true;
187 }
188 return false;
189 };
190 } else {
191 // regular spinner....
192 spin_func = [=]() -> bool {
193 if (data_loader_->spin() || frontend_pipeline_->isWorking() ||
194 (backend_pipeline_ && backend_pipeline_->isWorking())) {
195 if (!params_.parallelRun()) {
196 frontend_pipeline_->spinOnce();
197 if (backend_pipeline_) backend_pipeline_->spinOnce();
198 }
199 // TODO: this definately takes some time...
200 spinViz(); // for now
201 // a later problem!
202 return true;
203 }
204 return false;
205 };
206 }
207
208 utils::TimingStatsCollector timer("pipeline_spin");
209 return spin_func();
210}

220void DynoPipelineManager::launchSpinners() {
221 LOG(INFO) << "Running PipelineManager with parallel_run="
222 << params_.parallelRun();
223
224 if (params_.parallelRun()) {
225 frontend_pipeline_spinner_ = std::make_unique<dyno::Spinner>(
226 std::bind(&dyno::FrontendPipeline::spin, frontend_pipeline_.get()),
227 "frontend-pipeline-spinner");
228
229 if (backend_pipeline_)
230 backend_pipeline_spinner_ = std::make_unique<dyno::Spinner>(
231 std::bind(&dyno::BackendPipeline::spin, backend_pipeline_.get()),
232 "backend-pipeline-spinner");
233 }
234
235 data_provider_spinner_ = std::make_unique<dyno::Spinner>(
236 std::bind(&dyno::DataInterfacePipeline::spin, data_interface_.get()),
237 "data-interface-spinner");
238
239 if (frontend_viz_pipeline_)
240 frontend_viz_pipeline_spinner_ = std::make_unique<dyno::Spinner>(
241 std::bind(&dyno::FrontendVizPipeline::spin,
242 frontend_viz_pipeline_.get()),
243 "frontend-display-spinner");
244
245 if (backend_viz_pipeline_)
246 backend_viz_pipeline_spinner_ = std::make_unique<dyno::Spinner>(
247 std::bind(&dyno::BackendVizPipeline::spin, backend_viz_pipeline_.get()),
248 "backend-display-spinner");

251void DynoPipelineManager::loadPipelines(const CameraParams& camera_params,
252 FrontendDisplay::Ptr frontend_display,
253 BackendDisplay::Ptr backend_display,
254 BackendModuleFactory::Ptr factory) {
255 BackendModule::Ptr backend = nullptr;
256 BackendModuleDisplay::Ptr additional_backend_display = nullptr;
257 // the registra for the frontend pipeline
258 // this is agnostic to the actual pipeline type so we can add/register
259 // a new queue to it regardless of the derived type (as long as it is at least
260 // a MIMO, which it should be as this is the lowest type of actual pipeline
261 // with any functionality)
262 typename FrontendPipeline::OutputRegistra::Ptr frontend_output_registra =
263 nullptr;
264 const auto parallel_run = params_.parallelRun();
265
266 switch (params_.frontend_type_) {
267 case FrontendType::kRGBD: {
268 LOG(INFO) << "Making RGBDInstance frontend";
269 FrontendModule::Ptr frontend = nullptr;
270
271 CameraParams mutable_camera_params = camera_params;
272 // TODO: make this conversion to RGBD param based
273 // only update camera to rgbd if the data provider has not already set
274 if (!mutable_camera_params.hasDepthParams()) {
275 LOG(INFO) << "Updating camera params: converting to Fake Stereo camera";
276 // mutable_camera_params.setDepthParams(0.07);
277 mutable_camera_params.setDepthParams(0.1);
278 }
279 Camera::Ptr camera = std::make_shared<Camera>(mutable_camera_params);
280 CHECK_NOTNULL(camera);
281
282 if (use_offline_frontend_) {
283 // TODO: bring back!
284 LOG(FATAL) << "Not reimplemented";
285 // LOG(INFO) << "Offline RGBD frontend";
286 // using OfflineFrontend =
287 // FrontendOfflinePipeline<RegularBackendModule::ModuleTraits>;
288 // const std::string file_path =
289 // getOutputFilePath(kRgbdFrontendOutputJsonFile);
290 // LOG(INFO) << "Loading RGBD frontend output packets from " <<
291 // file_path;
292
293 // OfflineFrontend::UniquePtr offline_frontend =
294 // std::make_unique<OfflineFrontend>("offline-rgbdfrontend",
295 // file_path);
296 // // make registra so we can register queues with this pipeline
297 // frontend_output_registra = offline_frontend->getOutputRegistra();
298 // // raw ptr type becuase we cannot copy the unique ptr!! This is only
299 // // becuase we need it in the lambda function which is a temporary
300 // // solution
301 // OfflineFrontend* offline_frontend_ptr = offline_frontend.get();
302 // // set get dataset size function (bit of a hack for now, and only for
303 // // the batch optimizer so it knows when to optimize!!)
304 // get_dataset_size_ = [offline_frontend_ptr]() -> FrameId {
305 // // get frame id of the final frame saved
306 // return CHECK_NOTNULL(offline_frontend_ptr)
307 // ->getFrontendOutputPackets()
308 // .rbegin()
309 // ->first;
310 // };
311 // // convert pipeline to base type
312 // frontend_pipeline_ = std::move(offline_frontend);
313 } else {
314 frontend = std::make_shared<RGBDInstanceFrontendModule>(
315 params_, camera, &display_queue_);
316 LOG(INFO) << "Made RGBDInstanceFrontendModule";
317 // need to make the derived pipeline so we can set parallel run etc
318 // the manager takes a pointer to the base MIMO so we can have different
319 // types of pipelines
320 FrontendPipeline::UniquePtr frontend_pipeline_derived =
321 std::make_unique<FrontendPipeline>(
322 "frontend-pipeline", &frontend_input_queue_, frontend);
323 // make registra so we can register queues with this pipeline
324 frontend_output_registra =
325 frontend_pipeline_derived->getOutputRegistra();
326 frontend_pipeline_derived->parallelRun(parallel_run);
327 // conver pipeline to base type
328 frontend_pipeline_ = std::move(frontend_pipeline_derived);
329
330 get_dataset_size_ = [=]() -> FrameId {
331 CHECK(data_loader_) << "Data Loader is null when accessing "
332 "get_last_frame_ in BatchOptimizerParams";
333 return data_loader_->datasetSize();
334 };
335 }
336
337 // right now depends on the get_dataset_size_ function being det before
338 // the optimzier is created!!!
339
340 if (FLAGS_use_backend) {
341 LOG(INFO) << "Construcing RGBD backend";
342 params_.backend_params_.full_batch_frame = (int)get_dataset_size_();
343
344 Sensors sensors;
345 sensors.camera = camera;
346
347 ModuleParams module_params;
348 module_params.backend_params = params_.backend_params_;
349 module_params.sensors = sensors;
350 module_params.display_queue = &display_queue_;
351
352 BackendWrapper backend_wrapper = factory->createModule(module_params);
353
354 backend = backend_wrapper.backend;
355 additional_backend_display = backend_wrapper.backend_viz;
356 CHECK(backend);
357
358 // if(frontend && backend) {
359 // backend->registerMapUpdater(std::bind(&FrontendModule::mapUpdate,
360 // frontend.get(), std::placeholders::_1)); LOG(INFO) << "Bound map
361 // update between frontend and backend";
362 // }
363
364 } else if (use_offline_frontend_) {
365 LOG(WARNING)
366 << "FLAGS_use_backend is false but use_offline_frontend "
367 "(FLAGS_frontend_from_file) us true. "
368 << " Pipeline will load data from frontend but send it nowhere!!";
369 }
370
371 } break;
372 case FrontendType::kMono: {
373 LOG(FATAL) << "MONO Not implemented!";
374 } break;
375
376 default: {
377 LOG(FATAL) << "Not implemented!";
378 } break;
379 }
380
381 CHECK_NOTNULL(frontend_pipeline_);
382 CHECK_NOTNULL(frontend_output_registra);
383 // register output queue to send the front-end output to the viz
384 frontend_output_registra->registerQueue(&frontend_viz_input_queue_);
385
386 if (backend) {
387 backend_pipeline_ = std::make_unique<BackendPipeline>(
388 "backend-pipeline", &backend_input_queue_, backend);
389 backend_pipeline_->parallelRun(parallel_run);
390 // also register connection between front and back
391 frontend_output_registra->registerQueue(&backend_input_queue_);
392
393 backend_pipeline_->registerOutputQueue(&backend_output_queue_);
394
395 if (additional_backend_display) {
396 VLOG(10) << "Connecting BackendModuleDisplay";
397 backend_pipeline_->registerOutputCallback(
398 [additional_backend_display](const auto& output) -> void {
399 additional_backend_display->spin(output);
400 });
401 }
402 }
403
404 // right now we cannot use the viz when we load from file as do not load
405 // certain data values (e.g. camera and debug info) so these will be null -
406 // the viz's try and access these causing a seg fault. Just need to add checks
407 if (!use_offline_frontend_) {
408 if (backend && backend_display) {
409 backend_viz_pipeline_ = std::make_unique<BackendVizPipeline>(
410 "backend-viz-pipeline", &backend_output_queue_, backend_display);
411 }
412 frontend_viz_pipeline_ = std::make_unique<FrontendVizPipeline>(
413 "frontend-viz-pipeline", &frontend_viz_input_queue_, frontend_display);
414 }
415}
