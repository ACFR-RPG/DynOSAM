38DynoPipelineManager::DynoPipelineManager(const DynoParams& params, DataProvider::Ptr data_loader, FrontendDisplay::Ptr frontend_display, BackendDisplay::Ptr backend_display)
39 : params_(params),
40 use_offline_frontend_(FLAGS_frontend_from_file),
41 data_loader_(std::move(data_loader)),
42 displayer_(&display_queue_, params.parallelRun())
43
44{
45 LOG(INFO) << "Starting DynoPipelineManager";
46
47 CHECK(data_loader_);
48 CHECK(frontend_display);
49
50 data_interface_ = std::make_unique<DataInterfacePipeline>(params_.parallelRun());
51 data_loader_->registerImageContainerCallback(
52 std::bind(&dyno::DataInterfacePipeline::fillImageContainerQueue, data_interface_.get(), std::placeholders::_1)
53 );
54
55 //ground truth
56 data_loader_->registerGroundTruthPacketCallback(
57 std::bind(&dyno::DataInterfacePipeline::addGroundTruthPacket, data_interface_.get(), std::placeholders::_1)
58 );
59
60 //preprocessing
61 data_interface_->registerImageContainerPreprocessor(
62 std::bind(&dyno::DataProvider::imageContainerPreprocessor, data_loader_.get(), std::placeholders::_1)
63 );
64
65 //push data from the data interface to the frontend module
66 data_interface_->registerOutputQueue(&frontend_input_queue_);
67
68 CameraParams camera_params;
69 if(params_.preferDataProviderCameraParams() && data_loader_->getCameraParams().has_value()) {
70 LOG(INFO) << "Using camera params from DataProvider, not the config in the CameraParams.yaml!";
71 camera_params = *data_loader_->getCameraParams();
72 }
73 else {
74 LOG(INFO) << "Using camera params specified in CameraParams.yaml!";
75 camera_params = params_.camera_params_;
76 }
77
78 loadPipelines(camera_params, frontend_display, backend_display);
79 launchSpinners();
80
81}

118bool DynoPipelineManager::spin() {
119
120 std::function<bool()> spin_func;
121
122 if(use_offline_frontend_) {
123 // if we have an offline frontend only spiun the frontend pipeline
124 // and no need to spin the viz (TODO: right now this is only images and not the actual pipelines...)
125 spin_func =[=]() -> bool {
126 if(frontend_pipeline_->isWorking()) {
127 if(!params_.parallelRun()) {
128 frontend_pipeline_->spinOnce();
129 if(backend_pipeline_) backend_pipeline_->spinOnce();
130 }
131 return true;
132 }
133 return false;
134 };
135 }
136 else {
137 //regular spinner....
138 spin_func =[=]() -> bool {
139 if(data_loader_->spin() || frontend_pipeline_->isWorking()) {
140 if(!params_.parallelRun()) {
141 frontend_pipeline_->spinOnce();
142 if(backend_pipeline_) backend_pipeline_->spinOnce();
143 }
144 spinViz(); //for now
145 //a later problem!
146 return true;
147 }
148 return false;
149 };
150 }
151
152 utils::TimingStatsCollector timer("pipeline_spin");
153 return spin_func();
154
155}

164void DynoPipelineManager::launchSpinners() {
165 LOG(INFO) << "Running PipelineManager with parallel_run=" << params_.parallelRun();
166
167 if(params_.parallelRun()) {
168 frontend_pipeline_spinner_ = std::make_unique<dyno::Spinner>(std::bind(&dyno::FrontendPipeline::spin, frontend_pipeline_.get()), "frontend-pipeline-spinner");
169
170 if(backend_pipeline_)
171 backend_pipeline_spinner_ = std::make_unique<dyno::Spinner>(std::bind(&dyno::BackendPipeline::spin, backend_pipeline_.get()), "backend-pipeline-spinner");
172
173 }
174
175 data_provider_spinner_ = std::make_unique<dyno::Spinner>(std::bind(&dyno::DataInterfacePipeline::spin, data_interface_.get()), "data-interface-spinner");
176
177 if(frontend_viz_pipeline_)
178 frontend_viz_pipeline_spinner_ = std::make_unique<dyno::Spinner>(std::bind(&dyno::FrontendVizPipeline::spin, frontend_viz_pipeline_.get()), "frontend-display-spinner");
179
180 if(backend_viz_pipeline_)
181 backend_viz_pipeline_spinner_ = std::make_unique<dyno::Spinner>(std::bind(&dyno::BackendVizPipeline::spin, backend_viz_pipeline_.get()), "backend-display-spinner");

185void DynoPipelineManager::loadPipelines(const CameraParams& camera_params, FrontendDisplay::Ptr frontend_display, BackendDisplay::Ptr backend_display) {
186
187 BackendModule::Ptr backend = nullptr;
188 //the registra for the frontend pipeline
189 //this is agnostic to the actual pipeline type so we can add/register
190 //a new queue to it regardless of the derived type (as long as it is at least a MIMO,
191 //which it should be as this is the lowest type of actual pipeline with any functionality)
192 typename FrontendPipeline::OutputRegistra::Ptr frontend_output_registra = nullptr;
193 const auto parallel_run = params_.parallelRun();
194
195 switch (params_.frontend_type_)
196 {
197 case FrontendType::kRGBD: {
198 LOG(INFO) << "Making RGBDInstance frontend";
199
200 using MapType = RGBDBackendModule::MapType;
201 typename MapType::Ptr map = MapType::create();
202
203 Camera::Ptr camera = std::make_shared<Camera>(camera_params);
204 CHECK_NOTNULL(camera);
205
206
207 if(use_offline_frontend_) {
208 LOG(INFO) << "Offline RGBD frontend";
209 using OfflineFrontend = FrontendOfflinePipeline<RGBDBackendModule::ModuleTraits>;
210 const std::string file_path = getOutputFilePath(kRgbdFrontendOutputJsonFile);
211 LOG(INFO) << "Loading RGBD frontend output packets from " << file_path;
212
213 OfflineFrontend::UniquePtr offline_frontend = std::make_unique<OfflineFrontend>("offline-rgbdfrontend", file_path);
214 //make registra so we can register queues with this pipeline
215 frontend_output_registra = offline_frontend->getOutputRegistra();
216 //raw ptr type becuase we cannot copy the unique ptr!! This is only becuase
217 //we need it in the lambda function which is a temporary solution
218 OfflineFrontend* offline_frontend_ptr = offline_frontend.get();
219 //set get dataset size function (bit of a hack for now, and only for the batch optimizer so it
220 //knows when to optimize!!)
221 get_dataset_size_ = [offline_frontend_ptr]() -> FrameId {
222 //get frame id of the final frame saved
223 return CHECK_NOTNULL(offline_frontend_ptr)->getFrontendOutputPackets().rbegin()->first;
224 };
225 //convert pipeline to base type
226 frontend_pipeline_ = std::move(offline_frontend);
227 }
228 else {
229 FrontendModule::Ptr frontend = std::make_shared<RGBDInstanceFrontendModule>(params_.frontend_params_, camera, &display_queue_);
230 LOG(INFO) << "Made RGBDInstanceFrontendModule";
231 //need to make the derived pipeline so we can set parallel run etc
232 //the manager takes a pointer to the base MIMO so we can have different types of pipelines
233 FrontendPipeline::UniquePtr frontend_pipeline_derived = std::make_unique<FrontendPipeline>("frontend-pipeline", &frontend_input_queue_, frontend);
234 //make registra so we can register queues with this pipeline
235 frontend_output_registra = frontend_pipeline_derived->getOutputRegistra();
236 frontend_pipeline_derived->parallelRun(parallel_run);
237 //conver pipeline to base type
238 frontend_pipeline_ = std::move(frontend_pipeline_derived);
239
240 get_dataset_size_ = [=]() -> FrameId {
241 CHECK(data_loader_) << "Data Loader is null when accessing get_last_frame_ in BatchOptimizerParams";
242 return data_loader_->datasetSize();
243 };
244 }
245
246 //right now depends on the get_dataset_size_ function being det before the optimzier is created!!!
247
248
249 if(FLAGS_use_backend) {
250 LOG(INFO) << "Construcing RGBD backend";
251
252 //TODO: make better params!!
253 auto updater_type = static_cast<RGBDBackendModule::UpdaterType>(
254 FLAGS_backend_updater_enum
255 );
256
257 params_.backend_params_.full_batch_frame = (int)get_dataset_size_();
258
259 backend = std::make_shared<RGBDBackendModule>(params_.backend_params_, map, camera, updater_type, &display_queue_);
260 }
261 else if(use_offline_frontend_) {
262 LOG(WARNING) << "FLAGS_use_backend is false but use_offline_frontend (FLAGS_frontend_from_file) us true. "
263 << " Pipeline will load data from frontend but send it nowhere!!";
264 }
265
266 } break;
267 case FrontendType::kMono: {
268 LOG(FATAL) << "MONO Not implemented!";
269 } break;
270
271 default: {
272 LOG(FATAL) << "Not implemented!";
273 } break;
274 }
275
276 CHECK_NOTNULL(frontend_pipeline_);
277 CHECK_NOTNULL(frontend_output_registra);
278 frontend_output_registra->registerQueue(&frontend_viz_input_queue_);
279
280 if(backend) {
281 backend_pipeline_ = std::make_unique<BackendPipeline>("backend-pipeline", &backend_input_queue_, backend);
282 backend_pipeline_->parallelRun(parallel_run);
283 //also register connection between front and back
284 frontend_output_registra->registerQueue(&backend_input_queue_);
285
286 backend_pipeline_->registerOutputQueue(&backend_output_queue_);
287 }
288
289 //right now we cannot use the viz when we load from file as do not load certain data values (e.g. camera and debug info)
290 //so these will be null - the viz's try and access these causing a seg fault. Just need to add checks
291 if(!use_offline_frontend_) {
292 if(backend && backend_display) {
293 backend_viz_pipeline_ = std::make_unique<BackendVizPipeline>("backend-viz-pipeline", &backend_output_queue_, backend_display);
294 }
295 frontend_viz_pipeline_ = std::make_unique<FrontendVizPipeline>("frontend-viz-pipeline", &frontend_viz_input_queue_, frontend_display);
296
297 }
298
299
300}
