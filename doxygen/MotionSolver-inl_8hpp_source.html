45template <class SampleConsensusProblem>
46bool runRansac(
47 std::shared_ptr<SampleConsensusProblem> sample_consensus_problem_ptr,
48 const double& threshold, const int& max_iterations,
49 const double& probability, const bool& do_nonlinear_optimization,
50 gtsam::Pose3& best_pose, std::vector<int>& inliers) {
51 CHECK(sample_consensus_problem_ptr);
52 inliers.clear();
53
55 opengv::sac::Ransac<SampleConsensusProblem> ransac(max_iterations, threshold,
56 probability);
57
59 ransac.sac_model_ = sample_consensus_problem_ptr;
60
62 bool success = ransac.computeModel(0);
63
64 if (success) {
65 if (ransac.iterations_ >= max_iterations && ransac.inliers_.empty()) {
66 success = false;
67 best_pose = gtsam::Pose3();
68 inliers = {};
69 } else {
70 best_pose = utils::openGvTfToGtsamPose3(ransac.model_coefficients_);
71 inliers = ransac.inliers_;
72
73 if (do_nonlinear_optimization) {
74 opengv::transformation_t optimized_pose;
75 sample_consensus_problem_ptr->optimizeModelCoefficients(
76 inliers, ransac.model_coefficients_, optimized_pose);
77 best_pose = Eigen::MatrixXd(optimized_pose);
78 }
79 }
80 } else {
81 CHECK(ransac.inliers_.empty());
82 best_pose = gtsam::Pose3();
83 inliers.clear();
84 }
85
86 return success;
87}

89template <typename CALIBRATION>
90OpticalFlowAndPoseOptimizer::Result OpticalFlowAndPoseOptimizer::optimize(
91 const Frame::Ptr frame_k_1, const Frame::Ptr frame_k,
92 const TrackletIds& tracklets, const gtsam::Pose3& initial_pose) const {
93 utils::TimingStatsCollector timer("of_motion_solver.optimize");
94
95 using Calibration = CALIBRATION;
96 using Pose3FlowProjectionFactorCalib = Pose3FlowProjectionFactor<Calibration>;
97
98 gtsam::NonlinearFactorGraph graph;
99 gtsam::Values values;
100
101 gtsam::SharedNoiseModel flow_noise =
102 gtsam::noiseModel::Isotropic::Sigma(2u, params_.flow_sigma);
103
104 gtsam::SharedNoiseModel flow_prior_noise =
105 gtsam::noiseModel::Isotropic::Sigma(2u, params_.flow_prior_sigma);
106
107 // robust noise model!
108 flow_noise = gtsam::noiseModel::Robust::Create(
109 gtsam::noiseModel::mEstimator::Huber::Create(params_.k_huber),
110 flow_noise);
111
112 const gtsam::Pose3 T_world_k_1 = frame_k_1->getPose();
113 const gtsam::Symbol pose_key('X', 0);
114
115 gtsam::Ordering ordering;
116
117 // lambda to construct gtsam symbol from a tracklet
118 auto flowSymbol = [](TrackletId tracklet) -> gtsam::Symbol {
119 return gtsam::symbol_shorthand::F(static_cast<uint64_t>(tracklet));
120 };
121 auto gtsam_calibration = boost::make_shared<Calibration>(
122 frame_k_1->getFrameCamera().calibration());
123
124 // sanity check to ensure all tracklets are on the same object
125 std::unordered_set<ObjectId> object_id;
126
127 for (TrackletId tracklet_id : tracklets) {
128 Feature::Ptr feature_k_1 = frame_k_1->at(tracklet_id);
129 Feature::Ptr feature_k = frame_k->at(tracklet_id);
130
131 object_id.insert(feature_k_1->objectId());
132 object_id.insert(feature_k->objectId());
133
134 CHECK_NOTNULL(feature_k_1);
135 CHECK_NOTNULL(feature_k);
136
137 CHECK(feature_k_1->hasDepth());
138 CHECK(feature_k->hasDepth())
139 << " with object id " << feature_k->objectId() << " and is valid "
140 << feature_k->usable() << " and age " << feature_k->age()
141 << " previous age " << feature_k_1->age() << " kp "
142 << feature_k->keypoint();
143
144 const Keypoint kp_k_1 = feature_k_1->keypoint();
145 const Depth depth_k_1 = feature_k_1->depth();
146
147 const gtsam::Point2 flow = feature_k_1->measuredFlow();
148 CHECK(gtsam::equal(kp_k_1 + flow, feature_k->keypoint()))
149 << gtsam::Point2(kp_k_1 + flow) << " " << feature_k->keypoint()
150 << " object id " << feature_k->objectId() << " flow " << flow;
151
152 gtsam::Symbol flow_symbol(flowSymbol(tracklet_id));
153 auto flow_factor = boost::make_shared<Pose3FlowProjectionFactorCalib>(
154 flow_symbol, pose_key, kp_k_1, depth_k_1, T_world_k_1,
155 *gtsam_calibration, flow_noise);
156 graph.add(flow_factor);
157
158 // add prior factor on each flow
159 graph.addPrior<gtsam::Point2>(flow_symbol, flow, flow_prior_noise);
160
161 values.insert(flow_symbol, flow);
162 ordering += flow_symbol;
163 }
164
165 // pose at frame k
166 // put pose last and use custom ordering!
167 values.insert(pose_key, initial_pose);
168 ordering += pose_key;
169
170 // check we only have one label
171 CHECK_EQ(object_id.size(), 1u);
172 Result result;
173 result.best_result.object_id = *object_id.begin();
174
175 // double error_before = graph.error(values);
176 std::vector<double> post_errors;
177 std::unordered_set<gtsam::Key> outlier_flows;
178 // graph we will mutate by removing outlier factors
179 gtsam::NonlinearFactorGraph mutable_graph = graph;
180 gtsam::Values optimised_values = values;
181 // number of all variables
182 utils::StatsCollector("of_motion_solver.joint_of_pose_num_vars_all")
183 .AddSample(optimised_values.size());
184
185 gtsam::LevenbergMarquardtParams opt_params;
186 // for speed
187 opt_params.setMaxIterations(10);
188 // this is basically a set of prior looking factors on a pose so we know we
189 // need to eliminate the pose last to avoid fill in therefore we use our own
190 // custom ordering that has the pose last
191 // opt_params.setOrdering(ordering);
192 if (VLOG_IS_ON(200))
193 opt_params.verbosity = gtsam::NonlinearOptimizerParams::Verbosity::ERROR;
194
195 {
196 utils::TimingStatsCollector timer("of_motion_solver.LM_solve");
197 optimised_values = gtsam::LevenbergMarquardtOptimizer(
198 mutable_graph, optimised_values, opt_params)
199 .optimize();
200 }
201 // double error_after = mutable_graph.error(optimised_values);
202 // post_errors.push_back(error_after);
203
204 gtsam::FactorIndices outlier_factors;
205 // if we have outliers, enter iteration loop
206 if (params_.outlier_reject) {
207 utils::TimingStatsCollector timer("of_motion_solver.outlier_reject");
208
209 outlier_factors = factor_graph_tools::determineFactorOutliers<
210 Pose3FlowProjectionFactorCalib>(mutable_graph, optimised_values);
211
212 if (outlier_factors.size() > 0) {
213 for (size_t itr = 0; itr < 4; itr++) {
214 // currently removing factors from graph makes them nullptr
215 gtsam::NonlinearFactorGraph mutable_graph_with_null = mutable_graph;
216 for (auto outlier_idx : outlier_factors) {
217 auto factor = mutable_graph_with_null.at(outlier_idx);
218 gtsam::Symbol flow_symbol = factor->keys()[0];
219 CHECK_EQ(flow_symbol.chr(), 'f');
220
221 outlier_flows.insert(flow_symbol.key());
222 mutable_graph_with_null.remove(outlier_idx);
223 }
224 // now iterate over graph and add factors that are not null to ensure
225 // all factors are ok
226 mutable_graph.resize(0);
227 for (size_t i = 0; i < mutable_graph_with_null.size(); i++) {
228 auto factor = mutable_graph_with_null.at(i);
229 if (factor) {
230 mutable_graph.add(factor);
231 }
232 }
233
234 optimised_values.update(pose_key, initial_pose);
235 // do we use values or optimised values here?
236 optimised_values = gtsam::LevenbergMarquardtOptimizer(
237 mutable_graph, optimised_values, opt_params)
238 .optimize();
239 // error_after = mutable_graph.error(optimised_values);
240 // post_errors.push_back(error_after);
241
242 outlier_factors = factor_graph_tools::determineFactorOutliers<
243 Pose3FlowProjectionFactorCalib>(mutable_graph, optimised_values);
244
245 if (outlier_factors.size() == 0) {
246 break;
247 }
248 }
249 }
250 }
251
252 // size_t initial_size = graph.size();
253 // size_t inlier_size = mutable_graph.size();
254 // error_after = mutable_graph.error(optimised_values);
255
256 // recover values
257 result.best_result.refined_pose = optimised_values.at<gtsam::Pose3>(pose_key);
258 // result.error_before = error_before;
259 // result.error_after = error_after;
260
261 // for each outlier edge, update the set of inliers
262 for (TrackletId tracklet_id : tracklets) {
263 const gtsam::Symbol flow_sym = flowSymbol(tracklet_id);
264 const gtsam::Key flow_key = flow_sym.key();
265
266 if (outlier_flows.find(flow_key) != outlier_flows.end()) {
267 // still need to update the result as this is used to actually update the
268 // outlier tracklets in the updateFrameOutliersWithResult
269 result.outliers.push_back(tracklet_id);
270 } else {
271 gtsam::Point2 refined_flow = optimised_values.at<gtsam::Point2>(flow_sym);
272 result.best_result.refined_flows.push_back(refined_flow);
273 result.inliers.push_back(tracklet_id);
274 }
275 }
276 // number of variables after outlier removal
277 utils::StatsCollector("of_motion_solver.joint_of_pose_num_vars_inliers")
278 .AddSample(result.inliers.size());
279
280 CHECK_EQ(tracklets.size(), result.inliers.size() + result.outliers.size());
281 return result;
282}

295template <typename CALIBRATION>
296Pose3SolverResult MotionOnlyRefinementOptimizer::optimize(
297 const Frame::Ptr frame_k_1, const Frame::Ptr frame_k,
298 const TrackletIds& tracklets, const ObjectId object_id,
299 const gtsam::Pose3& initial_motion, const RefinementSolver& solver) const {
300 gtsam::NonlinearFactorGraph graph;
301 gtsam::Values values;
302
303 // noise models are chosen arbitrarily :)
304 gtsam::SharedNoiseModel landmark_motion_noise =
305 gtsam::noiseModel::Isotropic::Sigma(3u, params_.landmark_motion_sigma);
306
307 // TODO: this will break if we use 3d point error
308 gtsam::SharedNoiseModel projection_noise =
309 gtsam::noiseModel::Isotropic::Sigma(2u, params_.projection_sigma);
310
311 // make robust (I mean, why not?)
312 landmark_motion_noise = gtsam::noiseModel::Robust::Create(
313 gtsam::noiseModel::mEstimator::Huber::Create(params_.k_huber),
314 landmark_motion_noise);
315 projection_noise = gtsam::noiseModel::Robust::Create(
316 gtsam::noiseModel::mEstimator::Huber::Create(params_.k_huber),
317 projection_noise);
318
319 const gtsam::Key pose_k_1_key = CameraPoseSymbol(frame_k_1->getFrameId());
320 const gtsam::Key pose_k_key = CameraPoseSymbol(frame_k->getFrameId());
321 const gtsam::Key object_motion_key =
322 ObjectMotionSymbol(object_id, frame_k->getFrameId());
323
324 values.insert(pose_k_1_key, frame_k_1->getPose());
325 values.insert(pose_k_key, frame_k->getPose());
326 values.insert(object_motion_key, initial_motion);
327
328 auto gtsam_calibration = boost::make_shared<CALIBRATION>(
329 frame_k_1->getFrameCamera().calibration());
330
331 auto pose_prior = gtsam::noiseModel::Isotropic::Sigma(6u, 0.00001);
332 graph.addPrior(pose_k_1_key, frame_k_1->getPose(), pose_prior);
333 graph.addPrior(pose_k_key, frame_k->getPose(), pose_prior);
334
335 utils::TimingStatsCollector timer("motion_solver.object_nlo_refinement");
336
337 for (TrackletId tracklet_id : tracklets) {
338 Feature::Ptr feature_k_1 = frame_k_1->at(tracklet_id);
339 Feature::Ptr feature_k = frame_k->at(tracklet_id);
340
341 CHECK_NOTNULL(feature_k_1);
342 CHECK_NOTNULL(feature_k);
343
344 if (!feature_k_1->usable() || !feature_k->usable()) {
345 continue;
346 }
347
348 CHECK(feature_k_1->hasDepth());
349 CHECK(feature_k->hasDepth());
350
351 const Keypoint kp_k_1 = feature_k_1->keypoint();
352 const Keypoint kp_k = feature_k->keypoint();
353
354 const gtsam::Point3 lmk_k_1_world =
355 frame_k_1->backProjectToWorld(tracklet_id);
356 const gtsam::Point3 lmk_k_world = frame_k->backProjectToWorld(tracklet_id);
357
358 const gtsam::Point3 lmk_k_1_local =
359 frame_k_1->backProjectToCamera(tracklet_id);
360 const gtsam::Point3 lmk_k_local = frame_k->backProjectToCamera(tracklet_id);
361
362 const gtsam::Key lmk_k_1_key =
363 DynamicLandmarkSymbol(frame_k_1->getFrameId(), tracklet_id);
364 const gtsam::Key lmk_k_key =
365 DynamicLandmarkSymbol(frame_k->getFrameId(), tracklet_id);
366
367 // add initial for points
368 values.insert(lmk_k_1_key, lmk_k_1_world);
369 values.insert(lmk_k_key, lmk_k_world);
370
371 if (solver == RefinementSolver::PointError) {
372 LOG(FATAL) << "RefinementSolver::PointError not implemented";
373 // graph.emplace_shared<PoseToPointFactor>(
374 // pose_k_1_key, //pose key at previous frames
375 // lmk_k_1_key,
376 // lmk_k_1_local,
377 // projection_noise
378 // );
379
380 // graph.emplace_shared<PoseToPointFactor>(
381 // pose_k_key, //pose key at current frames
382 // lmk_k_key,
383 // lmk_k_local,
384 // projection_noise
385 // );
386 } else if (solver == RefinementSolver::ProjectionError) {
387 graph.emplace_shared<GenericProjectionFactor>(
388 kp_k_1, projection_noise, pose_k_1_key, lmk_k_1_key,
389 gtsam_calibration, false, false);
390
391 graph.emplace_shared<GenericProjectionFactor>(
392 kp_k, projection_noise, pose_k_key, lmk_k_key, gtsam_calibration,
393 false, false);
394 }
395
396 graph.emplace_shared<LandmarkMotionTernaryFactor>(
397 lmk_k_1_key, lmk_k_key, object_motion_key, landmark_motion_noise);
398 }
399
400 double error_before = graph.error(values);
401 std::vector<double> post_errors;
402
403 gtsam::NonlinearFactorGraph mutable_graph = graph;
404 gtsam::Values optimised_values = values;
405
406 utils::StatsCollector("motion_solver.object_nlo_refinement_num_vars_all")
407 .AddSample(optimised_values.size());
408
409 gtsam::LevenbergMarquardtParams opt_params;
410 opt_params.setMaxIterations(5);
411 if (VLOG_IS_ON(200))
412 opt_params.verbosity = gtsam::NonlinearOptimizerParams::Verbosity::ERROR;
413
414 optimised_values = gtsam::LevenbergMarquardtOptimizer(
415 mutable_graph, optimised_values, opt_params)
416 .optimize();
417 double error_after = mutable_graph.error(optimised_values);
418 // post_errors.push_back(error_after);
419
420 gtsam::FactorIndices outlier_factors =
421 factor_graph_tools::determineFactorOutliers<LandmarkMotionTernaryFactor>(
422 mutable_graph, optimised_values);
423
424 std::set<TrackletId> outlier_tracks;
425 // if we have outliers, enter iteration loop
426 if (outlier_factors.size() > 0u && params_.outlier_reject) {
427 for (size_t itr = 0; itr < 4; itr++) {
428 // currently removing factors from graph makes them nullptr
429 gtsam::NonlinearFactorGraph mutable_graph_with_null = mutable_graph;
430 for (auto outlier_idx : outlier_factors) {
431 auto factor = mutable_graph_with_null.at(outlier_idx);
432 DynamicPointSymbol point_symbol = factor->keys()[0];
433 outlier_tracks.insert(point_symbol.trackletId());
434 mutable_graph_with_null.remove(outlier_idx);
435 }
436 // now iterate over graph and add factors that are not null to ensure all
437 // factors are ok
438 mutable_graph.resize(0);
439 for (size_t i = 0; i < mutable_graph_with_null.size(); i++) {
440 auto factor = mutable_graph_with_null.at(i);
441 if (factor) {
442 mutable_graph.add(factor);
443 }
444 }
445
446 values.insert(object_motion_key, initial_motion);
447 // do we use values or optimised values here?
448 optimised_values = gtsam::LevenbergMarquardtOptimizer(
449 mutable_graph, optimised_values, opt_params)
450 .optimize();
451 error_after = mutable_graph.error(optimised_values);
452 post_errors.push_back(error_after);
453
454 outlier_factors = factor_graph_tools::determineFactorOutliers<
455 LandmarkMotionTernaryFactor>(mutable_graph, optimised_values);
456
457 if (outlier_factors.size() == 0) {
458 break;
459 }
460 }
461 }
462
463 const size_t initial_size = graph.size();
464 const size_t inlier_size = mutable_graph.size();
465 error_after = mutable_graph.error(optimised_values);
466
467 Pose3SolverResult result;
468 result.best_result = optimised_values.at<gtsam::Pose3>(object_motion_key);
469 result.outliers = TrackletIds(outlier_tracks.begin(), outlier_tracks.end());
470 result.error_before = error_before;
471 result.error_after = error_after;
472
473 // naming is confusing, but we already have the outliers but we can use the
474 // same function to find the inliers
475 determineOutlierIds(result.outliers, tracklets, result.inliers);
476
477 for (auto outlier_tracklet : outlier_tracks) {
478 Feature::Ptr feature_k_1 = frame_k_1->at(outlier_tracklet);
479 Feature::Ptr feature_k = frame_k->at(outlier_tracklet);
480
481 CHECK(feature_k_1->usable());
482 CHECK(feature_k->usable());
483
484 feature_k->markOutlier();
485 feature_k_1->markOutlier();
486 }
487
488 // number of variables after outlier removal
489 utils::StatsCollector("motion_solver.object_nlo_refinement_num_vars_inliers")
490 .AddSample(result.inliers.size());
491
492 // LOG(INFO) << "Object Motion refinement - error before: "
493 // << error_before << " error after: " << error_after
494 // << " with initial size " << initial_size << " inlier size " <<
495 // inlier_size;
496
497 return result;
498}
