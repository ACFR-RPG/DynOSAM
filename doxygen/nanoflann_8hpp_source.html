78template <typename DistanceType, typename IndexType = size_t,
79 typename CountType = size_t>
80class KNNResultSet {
81 IndexType* indices;
82 DistanceType* dists;
83 CountType capacity;
84 CountType count;
85
86 public:
87 inline KNNResultSet(CountType capacity_)
88 : indices(0), dists(0), capacity(capacity_), count(0) {}
89
90 inline void init(IndexType* indices_, DistanceType* dists_) {
91 indices = indices_;
92 dists = dists_;
93 count = 0;
94 if (capacity)
95 dists[capacity - 1] = (std::numeric_limits<DistanceType>::max)();
96 }
97
98 inline CountType size() const { return count; }
99
100 inline bool full() const { return count == capacity; }
101
102 inline void addPoint(DistanceType dist, IndexType index) {
103 CountType i;
104 for (i = count; i > 0; --i) {
105#ifdef NANOFLANN_FIRST_MATCH // If defined and two points have the same
106 // distance, the one with the lowest-index will be
107 // returned first.
108 if ((dists[i - 1] > dist) ||
109 ((dist == dists[i - 1]) && (indices[i - 1] > index))) {
110#else
111 if (dists[i - 1] > dist) {
112#endif
113 if (i < capacity) {
114 dists[i] = dists[i - 1];
115 indices[i] = indices[i - 1];
116 }
117 } else
118 break;
119 }
120 if (i < capacity) {
121 dists[i] = dist;
122 indices[i] = index;
123 }
124 if (count < capacity) count++;
125 }
126
127 inline DistanceType worstDist() const { return dists[capacity - 1]; }
128};

242template <class T, class DataSource, typename _DistanceType = T>
243struct L1_Adaptor {
244 typedef T ElementType;
245 typedef _DistanceType DistanceType;
246
247 const DataSource& data_source;
248
249 L1_Adaptor(const DataSource& _data_source) : data_source(_data_source) {}
250
251 inline DistanceType operator()(const T* a, const size_t b_idx, size_t size,
252 DistanceType worst_dist = -1) const {
253 DistanceType result = DistanceType();
254 const T* last = a + size;
255 const T* lastgroup = last - 3;
256 size_t d = 0;
257
258 /* Process 4 items with each loop for efficiency. */
259 while (a < lastgroup) {
260 const DistanceType diff0 =
261 std::abs(a[0] - data_source.kdtree_get_pt(b_idx, d++));
262 const DistanceType diff1 =
263 std::abs(a[1] - data_source.kdtree_get_pt(b_idx, d++));
264 const DistanceType diff2 =
265 std::abs(a[2] - data_source.kdtree_get_pt(b_idx, d++));
266 const DistanceType diff3 =
267 std::abs(a[3] - data_source.kdtree_get_pt(b_idx, d++));
268 result += diff0 + diff1 + diff2 + diff3;
269 a += 4;
270 if ((worst_dist > 0) && (result > worst_dist)) {
271 return result;
272 }
273 }
274 /* Process last 0-3 components. Not needed for standard vector lengths. */
275 while (a < last) {
276 result += std::abs(*a++ - data_source.kdtree_get_pt(b_idx, d++));
277 }
278 return result;
279 }
280
281 template <typename U, typename V>
282 inline DistanceType accum_dist(const U a, const V b, int) const {
283 return std::abs(a - b);
284 }
285};

294struct L2_Adaptor {
295 typedef T ElementType;
296 typedef _DistanceType DistanceType;

299
300 L2_Adaptor(const DataSource& _data_source) : data_source(_data_source) {}
301
302 inline DistanceType operator()(const T* a, const size_t b_idx, size_t size,
303 DistanceType worst_dist = -1) const {
304 DistanceType result = DistanceType();
305 const T* last = a + size;
306 const T* lastgroup = last - 3;
307 size_t d = 0;
308
309 /* Process 4 items with each loop for efficiency. */
310 while (a < lastgroup) {
311 const DistanceType diff0 = a[0] - data_source.kdtree_get_pt(b_idx, d++);
312 const DistanceType diff1 = a[1] - data_source.kdtree_get_pt(b_idx, d++);
313 const DistanceType diff2 = a[2] - data_source.kdtree_get_pt(b_idx, d++);
314 const DistanceType diff3 = a[3] - data_source.kdtree_get_pt(b_idx, d++);
315 result += diff0 * diff0 + diff1 * diff1 + diff2 * diff2 + diff3 * diff3;
316 a += 4;
317 if ((worst_dist > 0) && (result > worst_dist)) {
318 return result;
319 }
320 }
321 /* Process last 0-3 components. Not needed for standard vector lengths. */
322 while (a < last) {
323 const DistanceType diff0 = *a++ - data_source.kdtree_get_pt(b_idx, d++);
324 result += diff0 * diff0;
325 }
326 return result;
327 }

330 inline DistanceType accum_dist(const U a, const V b, int) const {
331 return (a - b) * (a - b);
332 }

443const size_t WORDSIZE = 16;
444const size_t BLOCKSIZE = 8192;
445
446class PooledAllocator {
447 /* We maintain memory alignment to word boundaries by requiring that all
448 allocations be in multiples of the machine wordsize. */
449 /* Size of machine word in bytes. Must be power of 2. */
450 /* Minimum number of bytes requested at a time from the system. Must be
451 * multiple of WORDSIZE. */
452
453 size_t remaining; /* Number of bytes left in current block of storage. */
454 void* base; /* Pointer to base of current block of storage. */
455 void* loc; /* Current location in block to next allocate memory. */
456
457 void internal_init() {
458 remaining = 0;
459 base = NULL;
460 usedMemory = 0;
461 wastedMemory = 0;
462 }
463
464 public:
465 size_t usedMemory;
466 size_t wastedMemory;
467
471 PooledAllocator() { internal_init(); }
472
476 ~PooledAllocator() { free_all(); }
477
479 void free_all() {
480 while (base != NULL) {
481 void* prev =
482 *(static_cast<void**>(base)); /* Get pointer to prev block. */
483 ::free(base);
484 base = prev;
485 }
486 internal_init();
487 }
488
493 void* malloc(const size_t req_size) {
494 /* Round size up to a multiple of wordsize. The following expression
495 only works for WORDSIZE that is a power of 2, by masking last bits of
496 incremented size to zero.
497 */
498 const size_t size = (req_size + (WORDSIZE - 1)) & ~(WORDSIZE - 1);
499
500 /* Check whether a new block must be allocated. Note that the first word
501 of a block is reserved for a pointer to the previous block.
502 */
503 if (size > remaining) {
504 wastedMemory += remaining;
505
506 /* Allocate new storage. */
507 const size_t blocksize =
508 (size + sizeof(void*) + (WORDSIZE - 1) > BLOCKSIZE)
509 ? size + sizeof(void*) + (WORDSIZE - 1)
510 : BLOCKSIZE;
511
512 // use the standard C malloc to allocate memory
513 void* m = ::malloc(blocksize);
514 if (!m) {
515 fprintf(stderr, "Failed to allocate memory.\n");
516 return NULL;
517 }
518
519 /* Fill first word of new block with pointer to previous block. */
520 static_cast<void**>(m)[0] = base;
521 base = m;
522
523 size_t shift = 0;
524 // int size_t = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) &
525 // (WORDSIZE-1))) & (WORDSIZE-1);
526
527 remaining = blocksize - sizeof(void*) - shift;
528 loc = (static_cast<char*>(m) + sizeof(void*) + shift);
529 }
530 void* rloc = loc;
531 loc = static_cast<char*>(loc) + size;
532 remaining -= size;
533
534 usedMemory += size;
535
536 return rloc;
537 }

636 inline const_reference operator[](size_type i) const { return elems[i]; }

664 // direct access to data (read-only)

1064 NodePtr divideTree(const IndexType left, const IndexType right,
1065 BoundingBox& bbox) {
1066 NodePtr node = pool.allocate<Node>(); // allocate memory
1067
1068 /* If too few exemplars remain, then make this a leaf node. */
1069 if ((right - left) <= static_cast<IndexType>(m_leaf_max_size)) {
1070 node->child1 = node->child2 = NULL; /* Mark as leaf node. */
1071 node->node_type.lr.left = left;
1072 node->node_type.lr.right = right;
1073
1074 // compute bounding-box of leaf points
1075 for (int i = 0; i < (DIM > 0 ? DIM : dim); ++i) {
1076 bbox[i].low = dataset_get(vind[left], i);
1077 bbox[i].high = dataset_get(vind[left], i);
1078 }
1079 for (IndexType k = left + 1; k < right; ++k) {
1080 for (int i = 0; i < (DIM > 0 ? DIM : dim); ++i) {
1081 if (bbox[i].low > dataset_get(vind[k], i))
1082 bbox[i].low = dataset_get(vind[k], i);
1083 if (bbox[i].high < dataset_get(vind[k], i))
1084 bbox[i].high = dataset_get(vind[k], i);
1085 }
1086 }
1087 } else {
1088 IndexType idx;
1089 int cutfeat;
1090 DistanceType cutval;
1091 middleSplit_(&vind[0] + left, right - left, idx, cutfeat, cutval, bbox);
1092
1093 node->node_type.sub.divfeat = cutfeat;
1094
1095 BoundingBox left_bbox(bbox);
1096 left_bbox[cutfeat].high = cutval;
1097 node->child1 = divideTree(left, left + idx, left_bbox);
1098
1099 BoundingBox right_bbox(bbox);
1100 right_bbox[cutfeat].low = cutval;
1101 node->child2 = divideTree(left + idx, right, right_bbox);
1102
1103 node->node_type.sub.divlow = left_bbox[cutfeat].high;
1104 node->node_type.sub.divhigh = right_bbox[cutfeat].low;
1105
1106 for (int i = 0; i < (DIM > 0 ? DIM : dim); ++i) {
1107 bbox[i].low = std::min(left_bbox[i].low, right_bbox[i].low);
1108 bbox[i].high = std::max(left_bbox[i].high, right_bbox[i].high);
1109 }
1110 }

1126 void middleSplit_(IndexType* ind, IndexType count, IndexType& index,
1127 int& cutfeat, DistanceType& cutval,
1128 const BoundingBox& bbox) {
1129 const DistanceType EPS = static_cast<DistanceType>(0.00001);
1130 ElementType max_span = bbox[0].high - bbox[0].low;
1131 for (int i = 1; i < (DIM > 0 ? DIM : dim); ++i) {
1132 ElementType span = bbox[i].high - bbox[i].low;
1133 if (span > max_span) {
1134 max_span = span;
1135 }
1136 }
1137 ElementType max_spread = -1;
1138 cutfeat = 0;
1139 for (int i = 0; i < (DIM > 0 ? DIM : dim); ++i) {
1140 ElementType span = bbox[i].high - bbox[i].low;
1141 if (span > (1 - EPS) * max_span) {
1142 ElementType min_elem, max_elem;
1143 computeMinMax(ind, count, cutfeat, min_elem, max_elem);
1144 ElementType spread = max_elem - min_elem;
1145 ;
1146 if (spread > max_spread) {
1147 cutfeat = i;
1148 max_spread = spread;
1149 }
1150 }
1151 }
1152 // split in the middle
1153 DistanceType split_val = (bbox[cutfeat].low + bbox[cutfeat].high) / 2;
1154 ElementType min_elem, max_elem;
1155 computeMinMax(ind, count, cutfeat, min_elem, max_elem);
1156
1157 if (split_val < min_elem)
1158 cutval = min_elem;
1159 else if (split_val > max_elem)
1160 cutval = max_elem;
1161 else
1162 cutval = split_val;
1163
1164 IndexType lim1, lim2;
1165 planeSplit(ind, count, cutfeat, cutval, lim1, lim2);
1166
1167 if (lim1 > count / 2)
1168 index = lim1;
1169 else if (lim2 < count / 2)
1170 index = lim2;
1171 else
1172 index = count / 2;
1173 }

1184 void planeSplit(IndexType* ind, const IndexType count, int cutfeat,
1185 DistanceType& cutval, IndexType& lim1, IndexType& lim2) {
1186 /* Move vector indices for left subtree to front of list. */
1187 IndexType left = 0;
1188 IndexType right = count - 1;
1189 for (;;) {
1190 while (left <= right && dataset_get(ind[left], cutfeat) < cutval) ++left;
1191 while (right && left <= right &&
1192 dataset_get(ind[right], cutfeat) >= cutval)
1193 --right;
1194 if (left > right || !right)
1195 break; // "!right" was added to support unsigned Index types
1196 std::swap(ind[left], ind[right]);
1197 ++left;
1198 --right;
1199 }
1200 /* If either list is empty, it means that all remaining features
1201 * are identical. Split in the middle to maintain a balanced tree.
1202 */
1203 lim1 = left;
1204 right = count - 1;
1205 for (;;) {
1206 while (left <= right && dataset_get(ind[left], cutfeat) <= cutval) ++left;
1207 while (right && left <= right &&
1208 dataset_get(ind[right], cutfeat) > cutval)
1209 --right;
1210 if (left > right || !right)
1211 break; // "!right" was added to support unsigned Index types
1212 std::swap(ind[left], ind[right]);
1213 ++left;
1214 --right;
1215 }
1216 lim2 = left;
1217 }

1242 template <class RESULTSET>
1243 void searchLevel(RESULTSET& result_set, const ElementType* vec,
1244 const NodePtr node, DistanceType mindistsq,
1245 distance_vector_t& dists, const float epsError) const {
1246 /* If this is a leaf node, then do check and return. */
1247 if ((node->child1 == NULL) && (node->child2 == NULL)) {
1248 // count_leaf += (node->lr.right-node->lr.left); // Removed since was
1249 // neither used nor returned to the user.
1250 DistanceType worst_dist = result_set.worstDist();
1251 for (IndexType i = node->node_type.lr.left; i < node->node_type.lr.right;
1252 ++i) {
1253 const IndexType index = vind[i]; // reorder... : i;
1254 DistanceType dist = distance(vec, index, (DIM > 0 ? DIM : dim));
1255 if (dist < worst_dist) {
1256 result_set.addPoint(dist, vind[i]);
1257 }
1258 }
1259 return;
1260 }
1261
1262 /* Which child branch should be taken first? */
1263 int idx = node->node_type.sub.divfeat;
1264 ElementType val = vec[idx];
1265 DistanceType diff1 = val - node->node_type.sub.divlow;
1266 DistanceType diff2 = val - node->node_type.sub.divhigh;
1267
1268 NodePtr bestChild;
1269 NodePtr otherChild;
1270 DistanceType cut_dist;
1271 if ((diff1 + diff2) < 0) {
1272 bestChild = node->child1;
1273 otherChild = node->child2;
1274 cut_dist = distance.accum_dist(val, node->node_type.sub.divhigh, idx);
1275 } else {
1276 bestChild = node->child2;
1277 otherChild = node->child1;
1278 cut_dist = distance.accum_dist(val, node->node_type.sub.divlow, idx);
1279 }
1280
1281 /* Call recursively to search next level down. */
1282 searchLevel(result_set, vec, bestChild, mindistsq, dists, epsError);
1283
1284 DistanceType dst = dists[idx];
1285 mindistsq = mindistsq + cut_dist - dst;
1286 dists[idx] = cut_dist;
1287 if (mindistsq * epsError <= result_set.worstDist()) {
1288 searchLevel(result_set, vec, otherChild, mindistsq, dists, epsError);
1289 }
1290 dists[idx] = dst;
1291 }
