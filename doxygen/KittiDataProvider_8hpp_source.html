46class KittiCameraPoseFolder : public dyno::DataFolder<gtsam::Pose3> {
47 public:
48 DYNO_POINTER_TYPEDEFS(KittiCameraPoseFolder)
49
50 std::string getFolderName() const override { return "pose_gt.txt"; }
51 gtsam::Pose3 getItem(size_t idx) override { return gt_camera_poses_.at(idx); }
52
53 private:
54 void onPathInit() override {
55 std::ifstream object_pose_stream((std::string)getAbsolutePath(),
56 std::ios::in);
57
58 bool set_initial_pose = false;
59 while (!object_pose_stream.eof()) {
60 std::string s;
61 getline(object_pose_stream, s);
62 if (!s.empty()) {
63 std::stringstream ss;
64 ss << s;
65 int t;
66 ss >> t;
67 cv::Mat cv_pose = cv::Mat::eye(4, 4, CV_64F);
68
69 ss >> cv_pose.at<double>(0, 0) >> cv_pose.at<double>(0, 1) >>
70 cv_pose.at<double>(0, 2) >> cv_pose.at<double>(0, 3) >>
71 cv_pose.at<double>(1, 0) >> cv_pose.at<double>(1, 1) >>
72 cv_pose.at<double>(1, 2) >> cv_pose.at<double>(1, 3) >>
73 cv_pose.at<double>(2, 0) >> cv_pose.at<double>(2, 1) >>
74 cv_pose.at<double>(2, 2) >> cv_pose.at<double>(2, 3) >>
75 cv_pose.at<double>(3, 0) >> cv_pose.at<double>(3, 1) >>
76 cv_pose.at<double>(3, 2) >> cv_pose.at<double>(3, 3);
77
78 gtsam::Pose3 pose = utils::cvMatToGtsamPose3(cv_pose);
79
80 if (!set_initial_pose) {
81 initial_pose_ = pose;
82 set_initial_pose = true;
83 }
84 // aign with first pose
85 pose = initial_pose_.inverse() * pose;
86
87 gt_camera_poses_.push_back(pose);
88 }
89 }
90 object_pose_stream.close();
91 }
92
93 std::vector<gtsam::Pose3> gt_camera_poses_;
94 gtsam::Pose3
95 initial_pose_; // initial camera pose used to align gt to identiy!

111 std::string getFolderName() const override { return "object_pose.txt"; }
112 GroundTruthInputPacket getItem(size_t idx) override {
113 GroundTruthInputPacket& gt_packet = constructGTPacketOnly(idx);
114
115 // if frame idx > 0 calcualte motions
116 if (idx > 0) {
117 const GroundTruthInputPacket& prev_gt_packet =

130 void onPathInit() override {
131 std::ifstream object_pose_stream((std::string)getAbsolutePath(),
132 std::ios::in);
133 CHECK(object_pose_stream.good());
134
135 // hmm we dont know the number of expected frames here...
136 while (!object_pose_stream.eof()) {
137 std::string s;
138 getline(object_pose_stream, s);
139 if (!s.empty()) {
140 std::stringstream ss;
141 ss << s;
142
143 std::vector<double> object_pose_vec(10, 0);
144 ss >> object_pose_vec[0] >> object_pose_vec[1] >> object_pose_vec[2] >>
145 object_pose_vec[3] >> object_pose_vec[4] >> object_pose_vec[5] >>
146 object_pose_vec[6] >> object_pose_vec[7] >> object_pose_vec[8] >>
147 object_pose_vec[9];
148
149 ObjectPoseGT object_pose = constructObjectPoseGT(object_pose_vec);
150 object_poses_.push_back(object_pose);
151 }
152 }
153 object_pose_stream.close();
154
155 // no guarantee that the timestamp loader onPathInit() has been called first
156 // (although in reality it will becuase) it is part of the default folder
157 // structure check that it has been initalised and then use the timestamps
158 // already loaded there to construct the gt packets
159 CHECK(timestamp_loader_->isAbsolutePathSet())
160 << "Timestamp laoder should be already initalised so we can get the "
161 "timestamps from it since they are preloaded!!";
162 const size_t n_times = timestamp_loader_->size();
163 CHECK(n_times > 0u);
164 object_ids_vector_.resize(
165 n_times); // only load up to the number of timestamps given
166 for (size_t i = 0; i < object_poses_.size(); i++) {
167 const ObjectPoseGT& obect_pose_gt = object_poses_[i];
168 size_t frame_id = obect_pose_gt.frame_id_;
169 if (frame_id >= object_ids_vector_.size()) {
170 break;
171 }
172
173 object_ids_vector_[frame_id].push_back(i);
174 }
175 }

177 // connstrict gt packet only - without calcuting the motions from a previous
178 // packet. This is used to ensure we dont do recursive getIdx calls
179 GroundTruthInputPacket& constructGTPacketOnly(size_t idx) {
180 // no guarantee (by ordering) that camera_pose_loader has been inited
181 // correctly before the get item so we add the data here and not in the
182 // onPathInit
183 CHECK(camera_pose_loader_);
184 CHECK(camera_pose_loader_->isAbsolutePathSet());
185
186 const gtsam::Pose3 camera_pose = camera_pose_loader_->getItem(idx);
187 const Timestamp timestamp = timestamp_loader_->getItem(idx);
188
189 // use cahce to avoid calculating every time
190 if (gt_packets_.exists(idx)) {
191 return gt_packets_.at(idx);
192 }
193
194 // frame_id < nTimes - 1??
195 GroundTruthInputPacket gt_packet;
196 gt_packet.timestamp_ = timestamp;
197 gt_packet.frame_id_ = idx;
198 // add ground truths for this fid
199 for (size_t i = 0; i < object_ids_vector_[idx].size(); i++) {
200 ObjectPoseGT object_pose_gt = object_poses_[object_ids_vector_[idx][i]];
201 object_pose_gt.L_world_ = camera_pose * object_pose_gt.L_camera_;
202 gt_packet.object_poses_.push_back(object_pose_gt);
203 // sanity check
204 CHECK_EQ(gt_packet.object_poses_[i].frame_id_, idx);
205 }
206 gt_packet.X_world_ = camera_pose;
207 gt_packets_.insert({idx, gt_packet});
208
209 // must return the actual stored one
210 return gt_packets_.at(idx);
211 }

213 ObjectPoseGT constructObjectPoseGT(
214 const std::vector<double>& obj_pose_gt) const {
215 // FrameID ObjectID B1 B2 B3 B4 t1 t2 t3 r1
216 // Where ti are the coefficients of 3D object location **t** in camera
217 // coordinates, and r1 is the Rotation around Y-axis in camera coordinates.
218 // B1-4 is 2D bounding box of object in the image, used for visualization.
219 CHECK(obj_pose_gt.size() == 10);
220 ObjectPoseGT object_pose;
221 object_pose.frame_id_ = obj_pose_gt[0];
222 object_pose.object_id_ = obj_pose_gt[1];
223
224 double b1 = obj_pose_gt[2];
225 double b2 = obj_pose_gt[3];
226 double b3 = obj_pose_gt[4];
227 double b4 = obj_pose_gt[5];
228
229 // convert to cv::Rect where image starts top right
230 cv::Point tl(b1, b2);
231 cv::Point br(b3, b4);
232 object_pose.bounding_box_ = cv::Rect(tl, br);
233
234 cv::Mat t(3, 1, CV_64FC1);
235 t.at<double>(0) = obj_pose_gt[6];
236 t.at<double>(1) = obj_pose_gt[7];
237 t.at<double>(2) = obj_pose_gt[8];
238
239 // from Euler to Rotation Matrix
240 cv::Mat R(3, 3, CV_64FC1);
241
242 // assign r vector
243 double y = obj_pose_gt[9] + (3.1415926 / 2); // +(3.1415926/2)
244 double x = 0.0;
245 double z = 0.0;
246
247 // the angles are in radians.
248 double cy = cos(y);
249 double sy = sin(y);
250 double cx = cos(x);
251 double sx = sin(x);
252 double cz = cos(z);
253 double sz = sin(z);
254
255 double m00, m01, m02, m10, m11, m12, m20, m21, m22;
256
257 m00 = cy * cz + sy * sx * sz;
258 m01 = -cy * sz + sy * sx * cz;
259 m02 = sy * cx;
260 m10 = cx * sz;
261 m11 = cx * cz;
262 m12 = -sx;
263 m20 = -sy * cz + cy * sx * sz;
264 m21 = sy * sz + cy * sx * cz;
265 m22 = cy * cx;
266
267 R.at<double>(0, 0) = m00;
268 R.at<double>(0, 1) = m01;
269 R.at<double>(0, 2) = m02;
270 R.at<double>(1, 0) = m10;
271 R.at<double>(1, 1) = m11;
272 R.at<double>(1, 2) = m12;
273 R.at<double>(2, 0) = m20;
274 R.at<double>(2, 1) = m21;
275 R.at<double>(2, 2) = m22;
276
277 // construct 4x4 transformation matrix
278 cv::Mat Pose = cv::Mat::eye(4, 4, CV_64F);
279 Pose.at<double>(0, 0) = R.at<double>(0, 0);
280 Pose.at<double>(0, 1) = R.at<double>(0, 1);
281 Pose.at<double>(0, 2) = R.at<double>(0, 2);
282 Pose.at<double>(0, 3) = t.at<double>(0);
283 Pose.at<double>(1, 0) = R.at<double>(1, 0);
284 Pose.at<double>(1, 1) = R.at<double>(1, 1);
285 Pose.at<double>(1, 2) = R.at<double>(1, 2);
286 Pose.at<double>(1, 3) = t.at<double>(1);
287 Pose.at<double>(2, 0) = R.at<double>(2, 0);
288 Pose.at<double>(2, 1) = R.at<double>(2, 1);
289 Pose.at<double>(2, 2) = R.at<double>(2, 2);
290 Pose.at<double>(2, 3) = t.at<double>(2);
291
292 object_pose.L_camera_ = utils::cvMatToGtsamPose3(Pose);
293 return object_pose;
294 }
295
296 private:
297 TimestampFile::Ptr timestamp_loader_;
298 KittiCameraPoseFolder::Ptr camera_pose_loader_;
299 std::vector<ObjectIds> object_ids_vector_;
300 std::vector<ObjectPoseGT> object_poses_;
301 gtsam::FastMap<FrameId, GroundTruthInputPacket> gt_packets_;
302
303 gtsam::Pose3
304 initial_pose_; // initial camera pose used to align gt to identiy!
305};

482 ImageContainer::Ptr imageContainerPreprocessor(
483 ImageContainer::Ptr image_container) override {
484 if (!image_container->hasDepth()) {
485 throw std::runtime_error(
486 "Cannot preprocess ImageContainer in Kitti dataset as no depth!");
487 }
488
489 const auto& base_line = params_.base_line;
490 const auto& depth_scale_factor = params_.depth_scale_factor;
491
492 const cv::Mat& const_disparity = image_container->getDepth();
493 cv::Mat depth;
494 const_disparity.copyTo(depth);
495 for (int i = 0; i < const_disparity.rows; i++) {
496 for (int j = 0; j < const_disparity.cols; j++) {
497 if (const_disparity.at<double>(i, j) < 0) {
498 depth.at<double>(i, j) = 0;
499 } else {
500 depth.at<double>(i, j) =
501 base_line /
502 (const_disparity.at<double>(i, j) / depth_scale_factor);
503 }
504 }
505 }
506
507 cv::Mat& disparity = image_container->get<ImageType::Depth>();
508 depth.copyTo(disparity);
509 depth.convertTo(depth, CV_64F);
510
511 return image_container;
512 }

558
559 } else {
