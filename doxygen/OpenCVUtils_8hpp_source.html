140template <typename T>
141struct convert<cv::Size_<T>> {
142 static Node encode(const cv::Size_<T>& size) {
143 Node node;
144 node["width"] = size.width;
145 node["height"] = size.height;
146 return node;
147 }
148
149 static bool decode(const Node& node, cv::Size_<T>& size) {
150 if (!node.IsMap()) {
151 return false;
152 }
153
154 size.width = node["width"].as<T>();
155 size.height = node["height"].as<T>();
156 return true;
157 }
158};
