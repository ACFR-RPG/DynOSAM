48template <typename T>
49class ParallelOpenCVMat : public cv::ParallelLoopBody {
50 public:
51 using This = ParallelOpenCVMat<T>;
52
53 ParallelOpenCVMat(cv::Mat& img_ref) : img_(img_ref) {}
54
55 int dataSize() const { return img_.rows * img_.cols; }
56
57 void operator()(const cv::Range& range) const CV_OVERRIDE {
58 const T* derived = static_cast<const T*>(this);
59 CHECK_NOTNULL(derived);
60 for (int r = range.start; r < range.end; r++) {
61 int i = r / img_.cols;
62 int j = r % img_.cols;
63 derived->call(img_, i, j);
64 }
65 }
66
67 This& run() {
68 cv::parallel_for_(cv::Range(0, dataSize()), *this);
69 return *this;
70 }
71
72 ParallelOpenCVMat& operator=(const ParallelOpenCVMat&) { return *this; };
73
74 protected:
75 virtual void call(cv::Mat& img_ref, int i, int j) const = 0;
76
77 private:
78 cv::Mat& img_;
79};

81class FunctionalParallelOpenCVMat
82 : public ParallelOpenCVMat<FunctionalParallelOpenCVMat> {
83 public:
84 using Call = std::function<void(cv::Mat&, int, int)>;

86
87 FunctionalParallelOpenCVMat(cv::Mat& img_ref, const Call& call)
88 : Base(img_ref), call_(call) {}
89
90 private:
91 // allow the base class to access the derived call
92 friend Base;
93

95 call_(img_ref, i, j);
96 }

98 private:
99 mutable Call call_;
100};

279template <typename T>
280struct convert<cv::Size_<T>> {
281 static Node encode(const cv::Size_<T>& size) {
282 Node node;
283 node["width"] = size.width;
284 node["height"] = size.height;
285 return node;
286 }
287
288 static bool decode(const Node& node, cv::Size_<T>& size) {
289 if (!node.IsMap()) {
290 return false;
291 }
292
293 size.width = node["width"].as<T>();
294 size.height = node["height"].as<T>();
295 return true;
296 }
297};
