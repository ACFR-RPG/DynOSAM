42gtsam::Pose3 cvMatToGtsamPose3(const cv::Mat& H) {
43 CHECK_EQ(H.rows, 4);
44 CHECK_EQ(H.cols, 4);
45
46 cv::Mat R(3, 3, H.type());
47 cv::Mat T(3, 1, H.type());
48
49 for (int i = 0; i < 3; i++) {
50 for (int j = 0; j < 3; j++) {
51 R.at<double>(i, j) = H.at<double>(i, j);
52 }
53 }
54
55 for (int i = 0; i < 3; i++) {
56 T.at<double>(i, 0) = H.at<double>(i, 3);
57 }
58
59 return cvMatsToGtsamPose3(R, T);
60}

73gtsam::Rot3 cvMatToGtsamRot3(const cv::Mat& R) {
74 CHECK_EQ(R.rows, 3);
75 CHECK_EQ(R.cols, 3);
76 gtsam::Matrix rot_mat = gtsam::Matrix::Identity(3, 3);
77 cv::cv2eigen(R, rot_mat);

81gtsam::Point3 cvMatToGtsamPoint3(const cv::Mat& cv_t) {
82 CHECK_EQ(cv_t.rows, 3);
83 CHECK_EQ(cv_t.cols, 1);
84 gtsam::Point3 gtsam_t;
85 gtsam_t << cv_t.at<double>(0, 0), cv_t.at<double>(1, 0),
86 cv_t.at<double>(2, 0);
87 return gtsam_t;
88}

97gtsam::Pose3 poseVectorToGtsamPose3(const std::vector<double>& vector_pose) {
98 CHECK_EQ(vector_pose.size(), 16u);
99 CHECK_EQ(vector_pose[12], 0.0);
100 CHECK_EQ(vector_pose[13], 0.0);
101 CHECK_EQ(vector_pose[14], 0.0);
102 CHECK_EQ(vector_pose[15], 1.0);
103 return gtsam::Pose3(Eigen::Matrix4d(vector_pose.data()).transpose());
104}

106gtsam::Cal3_S2 Cvmat2Cal3_S2(const cv::Mat& M) {
107 CHECK_EQ(M.rows, 3); // We expect homogeneous camera matrix.
108 CHECK_GE(M.cols, 3); // We accept extra columns (which we do not use).
109 const double& fx = M.at<double>(0, 0);
110 const double& fy = M.at<double>(1, 1);
111 const double& s = M.at<double>(0, 1);
112 const double& u0 = M.at<double>(0, 2);
113 const double& v0 = M.at<double>(1, 2);
114 return gtsam::Cal3_S2(fx, fy, s, u0, v0);
115}

123std::pair<cv::Mat, cv::Mat> Pose2cvmats(const gtsam::Pose3& pose) {
124 const gtsam::Matrix3& rot = pose.rotation().matrix();
125 const gtsam::Vector3& tran = pose.translation();
126 return std::make_pair(gtsamMatrix3ToCvMat(rot), gtsamVector3ToCvMat(tran));
127}
