36gtsam::Pose3 cvMatToGtsamPose3(const cv::Mat& H)
37{
38 CHECK_EQ(H.rows, 4);
39 CHECK_EQ(H.cols, 4);
40
41 cv::Mat R(3, 3, H.type());
42 cv::Mat T(3, 1, H.type());
43
44 for (int i = 0; i < 3; i++)
45 {
46 for (int j = 0; j < 3; j++)
47 {
48 R.at<double>(i, j) = H.at<double>(i, j);
49 }
50 }
51
52 for (int i = 0; i < 3; i++)
53 {
54 T.at<double>(i, 0) = H.at<double>(i, 3);
55 }
56
57 return cvMatsToGtsamPose3(R, T);
58}

73gtsam::Rot3 cvMatToGtsamRot3(const cv::Mat& R)
74{
75 CHECK_EQ(R.rows, 3);
76 CHECK_EQ(R.cols, 3);
77 gtsam::Matrix rot_mat = gtsam::Matrix::Identity(3, 3);
78 cv::cv2eigen(R, rot_mat);
79 return gtsam::Rot3(rot_mat);

82gtsam::Point3 cvMatToGtsamPoint3(const cv::Mat& cv_t)
83{
84 CHECK_EQ(cv_t.rows, 3);
85 CHECK_EQ(cv_t.cols, 1);
86 gtsam::Point3 gtsam_t;
87 gtsam_t << cv_t.at<double>(0, 0), cv_t.at<double>(1, 0), cv_t.at<double>(2, 0);
88 return gtsam_t;
89}

100gtsam::Pose3 poseVectorToGtsamPose3(const std::vector<double>& vector_pose) {
101 CHECK_EQ(vector_pose.size(), 16u);
102 CHECK_EQ(vector_pose[12], 0.0);
103 CHECK_EQ(vector_pose[13], 0.0);
104 CHECK_EQ(vector_pose[14], 0.0);
105 CHECK_EQ(vector_pose[15], 1.0);
106 return gtsam::Pose3(Eigen::Matrix4d(vector_pose.data()).transpose());
107}

109gtsam::Cal3_S2 Cvmat2Cal3_S2(const cv::Mat& M) {
110 CHECK_EQ(M.rows, 3); // We expect homogeneous camera matrix.
111 CHECK_GE(M.cols, 3); // We accept extra columns (which we do not use).
112 const double& fx = M.at<double>(0, 0);
113 const double& fy = M.at<double>(1, 1);
114 const double& s = M.at<double>(0, 1);
115 const double& u0 = M.at<double>(0, 2);
116 const double& v0 = M.at<double>(1, 2);
117 return gtsam::Cal3_S2(fx, fy, s, u0, v0);
118}

126std::pair<cv::Mat, cv::Mat> Pose2cvmats(const gtsam::Pose3& pose) {
127 const gtsam::Matrix3& rot = pose.rotation().matrix();
128 const gtsam::Vector3& tran = pose.translation();
129 return std::make_pair(gtsamMatrix3ToCvMat(rot), gtsamVector3ToCvMat(tran));
