39StateQuery<gtsam::Pose3> WorldPoseAccessor::getObjectMotion(FrameId frame_id, ObjectId object_id) const {
40 const auto frame_node_k = map()->getFrame(frame_id);
41 CHECK_NOTNULL(frame_node_k);
42 const auto object_motion_key = frame_node_k->makeObjectMotionKey(object_id);
43
44 if(frame_id < 2) {
45 //if the current frame is 1 then skip as we never get a frame that is 0
46 //this is because the first frame (frame_id=0) is never sent to the backend
47 //as at least two frames (0 and 1) are needed to track!
48 return StateQuery<gtsam::Pose3>::NotInMap(object_motion_key);
49 }
50
51 const auto object_pose_k = this->getObjectPose(frame_id, object_id);
52 const auto object_pose_k_1 = this->getObjectPose(frame_id - 1u, object_id);
53
54 if(object_pose_k && object_pose_k_1) {
55 // ^w_{k-1}H_k = ^wL_k \: ^wL_{k-1}^{-1}
56 const gtsam::Pose3 motion = object_pose_k.get() * object_pose_k_1->inverse();
57 return StateQuery<gtsam::Pose3>{object_motion_key, motion};
58 }
59 return StateQuery<gtsam::Pose3>::NotInMap(object_motion_key);
60}
61StateQuery<gtsam::Pose3> WorldPoseAccessor::getObjectPose(FrameId frame_id, ObjectId object_id) const {
62 const auto frame_node = map()->getFrame(frame_id);
63 if(!frame_node) {
64 return StateQuery<gtsam::Pose3>::InvalidMap();
65 }
66
67 // CHECK(frame_node) << "Frame Id is null at frame " << frame_id;
68 return this->query<gtsam::Pose3>(
69 frame_node->makeObjectPoseKey(object_id)
70 );

84void WorldPoseFormulation::dynamicPointUpdateCallback(
85 const PointUpdateContextType& context, UpdateObservationResult& result,
86 gtsam::Values& new_values,
87 gtsam::NonlinearFactorGraph& new_factors) {
88 const auto lmk_node = context.lmk_node;
89 const auto frame_node_k_1 = context.frame_node_k_1;
90 const auto frame_node_k = context.frame_node_k;
91
92 auto dynamic_point_noise = noise_models_.dynamic_point_noise;
93
94 auto theta_accessor = this->accessorFromTheta();
95
96 const gtsam::Key object_point_key_k_1 = lmk_node->makeDynamicKey(frame_node_k_1->frame_id);
97 const gtsam::Key object_point_key_k = lmk_node->makeDynamicKey(frame_node_k->frame_id);
98
99 // if first motion (i.e first time we have both k-1 and k), add both at k-1 and k
100 if(context.is_starting_motion_frame) {
101 new_factors.emplace_shared<PoseToPointFactor>(
102 frame_node_k_1->makePoseKey(), //pose key at previous frames
103 object_point_key_k_1,
104 lmk_node->getMeasurement(frame_node_k_1).landmark,
105 dynamic_point_noise
106 );
107 // object_debug_info.num_dynamic_factors++;
108 result.updateAffectedObject(frame_node_k_1->frame_id, context.getObjectId());
109 if(result.debug_info) result.debug_info->getObjectInfo(context.getObjectId()).num_dynamic_factors++;
110
111 //add landmark at previous frame
112 const Landmark measured_k_1 = lmk_node->getMeasurement(frame_node_k_1->frame_id).landmark;
113 Landmark lmk_world_k_1;
114 getSafeQuery(
115 lmk_world_k_1,
116 theta_accessor->query<Landmark>(object_point_key_k_1),
117 gtsam::Point3(context.X_k_1_measured * measured_k_1)
118 );
119 new_values.insert(object_point_key_k_1, lmk_world_k_1);
120 if(result.debug_info) result.debug_info->getObjectInfo(context.getObjectId()).num_new_dynamic_points++;
121 }
122
123
124 // previous point must be added by the previous iteration
125 CHECK(new_values.exists(object_point_key_k_1) || theta_accessor->exists(object_point_key_k_1));
126
127 const Landmark measured_k = lmk_node->getMeasurement(frame_node_k).landmark;
128
129 new_factors.emplace_shared<PoseToPointFactor>(
130 frame_node_k->makePoseKey(), //pose key at this (in the iteration) frames
131 object_point_key_k,
132 measured_k,
133 dynamic_point_noise
134 );
135 if(result.debug_info) result.debug_info->getObjectInfo(context.getObjectId()).num_dynamic_factors++;
136 result.updateAffectedObject(frame_node_k->frame_id, context.getObjectId());
137
138 Landmark lmk_world_k;
139 getSafeQuery(
140 lmk_world_k,
141 theta_accessor->query<Landmark>(object_point_key_k),
142 gtsam::Point3(context.X_k_measured * measured_k)
143 );
144 new_values.insert(object_point_key_k, lmk_world_k);
145 if(result.debug_info) result.debug_info->getObjectInfo( context.getObjectId()).num_new_dynamic_points++;
146
147
148 const gtsam::Key object_pose_k_1_key = frame_node_k_1->makeObjectPoseKey(context.getObjectId());
149 const gtsam::Key object_pose_k_key = frame_node_k->makeObjectPoseKey(context.getObjectId());
150
151 auto landmark_motion_noise = noise_models_.landmark_motion_noise;
152 new_factors.emplace_shared<LandmarkMotionPoseFactor>(
153 object_point_key_k_1,
154 object_point_key_k,
155 object_pose_k_1_key,
156 object_pose_k_key,
157 landmark_motion_noise
158 );
159 result.updateAffectedObject(frame_node_k_1->frame_id, context.getObjectId());
160 result.updateAffectedObject(frame_node_k->frame_id, context.getObjectId());
161 if(result.debug_info) result.debug_info->getObjectInfo(context.getObjectId()).num_motion_factors++;
162
163 // mark as now in map
164 is_dynamic_tracklet_in_map_.insert2(context.getTrackletId(), true);
165
166}

169void WorldPoseFormulation::objectUpdateContext(
170 const ObjectUpdateContextType& context, UpdateObservationResult& result,
171 gtsam::Values& new_values, gtsam::NonlinearFactorGraph& new_factors)
172{
173 auto frame_node_k = context.frame_node_k;
174 const gtsam::Key object_pose_key_k = frame_node_k->makeObjectPoseKey(context.getObjectId());
175
176 auto theta_accessor = this->accessorFromTheta();
177
178 if(!is_other_values_in_map.exists(object_pose_key_k)) {
179 FrameId frame_id_k_1 = context.getFrameId() - 1u;
180 //try and propogate from previous initalisation
181 StateQuery<gtsam::Pose3> pose_k_1_query = theta_accessor->getObjectPose(
182 frame_id_k_1,
183 context.getObjectId()
184 );
185
186 //first frame the object is seen in
187 const FrameId first_seen_frame = context.object_node->getFirstSeenFrame();
188
189 //takes me from k-1 to k
190 Motion3 motion;
191 gtsam::Pose3 object_pose_k;
192 //we have a motion from k-1 to k and a pose at k
193 if(map()->hasInitialObjectMotion(context.getFrameId(), context.getObjectId(), &motion) && pose_k_1_query) {
194 object_pose_k = motion * pose_k_1_query.get();
195 CHECK_NE(first_seen_frame, context.getFrameId());
196
197 }
198 else {
199 //no motion or previous pose, so initalise with centroid translation and identity rotation
200 const auto[centroid_initial, result] = theta_accessor->computeObjectCentroid(context.getFrameId(), context.getObjectId());
201 CHECK(result);
202 gtsam::Pose3 initial_object_pose(gtsam::Rot3::Identity(), centroid_initial);
203
204 getSafeQuery(
205 object_pose_k,
206 theta_accessor->query<gtsam::Pose3>(object_pose_key_k),
207 initial_object_pose
208 );
209
210 // CHECK_EQ(first_seen_frame, context.getFrameId());
211 }
212
213
214 //for experiments and testing
215 if(FLAGS_init_LL_with_identity && pose_k_1_query) {
216 object_pose_k = pose_k_1_query.get();
217 }
218 if(FLAGS_use_identity_rot_L_for_init) {
219 auto tmp_pose = gtsam::Pose3(gtsam::Rot3::Identity(), object_pose_k.translation());
220 object_pose_k = tmp_pose;
221 }
222 if(FLAGS_corrupt_L_for_init) {
223 object_pose_k = utils::perturbWithNoise<gtsam::Pose3>(object_pose_k, FLAGS_corrupt_L_for_init_sigma);
224 }
225
226 LOG(INFO) << "Adding object pose " << object_pose_k << " object id " << context.getObjectId();
227
228 new_values.insert(object_pose_key_k, object_pose_k);
229 is_other_values_in_map.insert2(object_pose_key_k, true);
230 VLOG(50) << "Adding object pose key " << DynoLikeKeyFormatter(object_pose_key_k);
231 }
232
233 if(FLAGS_use_smoothing_factor) {
234 const auto frame_id = context.getFrameId();
235 const auto object_id = context.getObjectId();
236 if(frame_id < 2) return;
237
238 auto frame_node_k_2 = map()->getFrame(frame_id - 2u);
239 auto frame_node_k_1 = map()->getFrame(frame_id - 1u);
240
241 if (!frame_node_k_2 || !frame_node_k_1) { return; }
242
243 //pose key at k-2
244 const gtsam::Symbol object_pose_key_k_2 = frame_node_k_2->makeObjectPoseKey(object_id);
245 //pose key at previous frame (k-1)
246 const gtsam::Symbol object_pose_key_k_1 = frame_node_k_1->makeObjectPoseKey(object_id);
247
248 auto object_smoothing_noise = noise_models_.object_smoothing_noise;
249 CHECK(object_smoothing_noise);
250 CHECK_EQ(object_smoothing_noise->dim(), 6u);
251
252 {
253 ObjectId object_label_k_2, object_label_k_1, object_label_k;
254 FrameId frame_id_k_2, frame_id_k_1, frame_id_k;
255 CHECK(reconstructPoseInfo(object_pose_key_k_2, object_label_k_2, frame_id_k_2));
256 CHECK(reconstructPoseInfo(object_pose_key_k_1, object_label_k_1, frame_id_k_1));
257 CHECK(reconstructPoseInfo(object_pose_key_k, object_label_k, frame_id_k));
258 CHECK_EQ(object_label_k_2, object_label_k);
259 CHECK_EQ(object_label_k_1, object_label_k);
260 CHECK_EQ(frame_id_k_1 + 1, frame_id_k); //assumes consequative frames
261 CHECK_EQ(frame_id_k_2 + 1, frame_id_k_1); //assumes consequative frames
262
263 }
264
265 //if the motion key at k (motion from k-1 to k), and key at k-1 (motion from k-2 to k-1)
266 //exists in the map or is about to exist via new values, add the smoothing factor
267 if(is_other_values_in_map.exists(object_pose_key_k_1) &&
268 is_other_values_in_map.exists(object_pose_key_k) &&
269 is_other_values_in_map.exists(object_pose_key_k_2)) {
270
271 new_factors.emplace_shared<LandmarkPoseSmoothingFactor>(
272 object_pose_key_k_2,
273 object_pose_key_k_1,
274 object_pose_key_k,
275 object_smoothing_noise
276 );
277 if(result.debug_info) result.debug_info->getObjectInfo(object_id).smoothing_factor_added=true;
278 VLOG(50) << "Adding smoothing " << DynoLikeKeyFormatter(object_pose_key_k_2) << " -> " << DynoLikeKeyFormatter(object_pose_key_k);
279 }
280
281
282 }
