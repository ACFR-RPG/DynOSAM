46struct UpdateObservationResult {
47 gtsam::FastMap<ObjectId, std::set<FrameId>> objects_affected_per_frame; //per frame
48 DebugInfo::Optional debug_info{};
49
50 //TODO: debug info
51 inline UpdateObservationResult& operator+=(const UpdateObservationResult& oth) {
52 for(const auto& [key, value] : oth.objects_affected_per_frame) {
53 objects_affected_per_frame[key].insert(value.begin(), value.end());
54 }
55 return *this;
56 }
57
58 void updateAffectedObject(FrameId frame_id, ObjectId object_id) {
59 if(!objects_affected_per_frame.exists(object_id)) {
60 objects_affected_per_frame.insert2(object_id, std::set<FrameId>{});
61 }
62 objects_affected_per_frame[object_id].insert(frame_id);
63 }
64};
