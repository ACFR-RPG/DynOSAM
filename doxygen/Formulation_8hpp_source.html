77struct UpdateObservationResult {
78 gtsam::FastMap<ObjectId, std::set<FrameId>>
79 objects_affected_per_frame; // per frame
80 DebugInfo::Optional debug_info{};
81
82 // Incremental interface
83 gtsam::ISAM2UpdateParams isam_update_params;
84
85 UpdateObservationResult() {}
86
87 UpdateObservationResult(const UpdateObservationParams& update_params) {
88 if (update_params.enable_debug_info) {
89 this->debug_info = DebugInfo();
90 }
91 }
92
93 void updateAffectedObject(FrameId frame_id, ObjectId object_id) {
94 if (!objects_affected_per_frame.exists(object_id)) {
95 objects_affected_per_frame.insert2(object_id, std::set<FrameId>{});
96 }
97 objects_affected_per_frame[object_id].insert(frame_id);
98 }
99};
