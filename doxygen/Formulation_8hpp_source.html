62struct UpdateObservationResult {
63 gtsam::FastMap<ObjectId, std::set<FrameId>>
64 objects_affected_per_frame; // per frame
65 DebugInfo::Optional debug_info{};
66
67 UpdateObservationResult() {}
68
69 UpdateObservationResult(const UpdateObservationParams& update_params) {
70 if (update_params.enable_debug_info) {
71 this->debug_info = DebugInfo();
72 }
73 }
74
75 // TODO: debug info
76 inline UpdateObservationResult& operator+=(
77 const UpdateObservationResult& oth) {
78 for (const auto& [key, value] : oth.objects_affected_per_frame) {
79 objects_affected_per_frame[key].insert(value.begin(), value.end());
80 }
81 return *this;
82 }
83
84 void updateAffectedObject(FrameId frame_id, ObjectId object_id) {
85 if (!objects_affected_per_frame.exists(object_id)) {
86 objects_affected_per_frame.insert2(object_id, std::set<FrameId>{});
87 }
88 objects_affected_per_frame[object_id].insert(frame_id);
89 }
90};
