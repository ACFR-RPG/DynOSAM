126Pose3SolverResult EgoMotionSolver::geometricOutlierRejection2d2d(
127 Frame::Ptr frame_k_1, Frame::Ptr frame_k,
128 std::optional<gtsam::Rot3> R_curr_ref) {
129 // get correspondences
130 RelativePoseCorrespondences correspondences;
131 // this does not create proper bearing vectors (at leas tnot for 3d-2d pnp
132 // solve) bearing vectors are also not undistorted atm!!
133 {
134 utils::ChronoTimingStats track_dynamic_timer("mono_frame_correspondences");
135 frame_k->getCorrespondences(correspondences, *frame_k_1,
136 KeyPointType::STATIC,
137 frame_k->imageKeypointCorrespondance());
138 }
139
140 Pose3SolverResult result;
141
142 const size_t& n_matches = correspondences.size();
143
144 if (n_matches < 5u) {
145 result.status = TrackingStatus::FEW_MATCHES;
146 return result;
147 }
148
149 gtsam::Matrix K = camera_params_.getCameraMatrixEigen();
150 K = K.inverse();
151
152 TrackletIds tracklets;
153 // NOTE: currently without distortion! the correspondences should be made into
154 // bearing vector elsewhere!
155 BearingVectors ref_bearing_vectors, cur_bearing_vectors;
156 for (size_t i = 0u; i < n_matches; i++) {
157 const auto& corres = correspondences.at(i);
158 const Keypoint& ref_kp = corres.ref_;
159 const Keypoint& cur_kp = corres.cur_;
160
161 gtsam::Vector3 ref_versor = (K * gtsam::Vector3(ref_kp(0), ref_kp(1), 1.0));
162 gtsam::Vector3 cur_versor = (K * gtsam::Vector3(cur_kp(0), cur_kp(1), 1.0));
163
164 ref_versor = ref_versor.normalized();
165 cur_versor = cur_versor.normalized();
166
167 ref_bearing_vectors.push_back(ref_versor);
168 cur_bearing_vectors.push_back(cur_versor);
169
170 tracklets.push_back(corres.tracklet_id_);
171 }
172
173 RelativePoseAdaptor adapter(ref_bearing_vectors, cur_bearing_vectors);
174
175 const bool use_2point_mono = params_.ransac_use_2point_mono && R_curr_ref;
176 if (use_2point_mono) {
177 adapter.setR12((*R_curr_ref).matrix());
178 }
179
180 gtsam::Pose3 best_result;
181 std::vector<int> ransac_inliers;
182 bool success = false;
183 if (use_2point_mono) {
184 success = runRansac<RelativePoseProblemGivenRot>(
185 std::make_shared<RelativePoseProblemGivenRot>(adapter,
186 params_.ransac_randomize),
187 params_.ransac_threshold_mono, params_.ransac_iterations,
188 params_.ransac_probability, params_.optimize_2d2d_pose_from_inliers,
189 best_result, ransac_inliers);
190 } else {
191 success = runRansac<RelativePoseProblem>(
192 std::make_shared<RelativePoseProblem>(
193 adapter, RelativePoseProblem::NISTER, params_.ransac_randomize),
194 params_.ransac_threshold_mono, params_.ransac_iterations,
195 params_.ransac_probability, params_.optimize_2d2d_pose_from_inliers,
196 best_result, ransac_inliers);
197 }
198
199 if (!success) {
200 result.status = TrackingStatus::INVALID;
201 } else {
202 constructTrackletInliers(result.inliers, result.outliers, correspondences,
203 ransac_inliers, tracklets);
204 // NOTE: 2-point always returns the identity rotation, hence we have to
205 // substitute it:
206 if (use_2point_mono) {
207 CHECK(R_curr_ref->equals(best_result.rotation()));
208 }
209 result.status = TrackingStatus::VALID;
210 result.best_result = best_result;
211 }
212
213 return result;
214}

234Pose3SolverResult EgoMotionSolver::geometricOutlierRejection3d2d(
235 const AbsolutePoseCorrespondences& correspondences,
236 std::optional<gtsam::Rot3> R_curr_ref) {
237 utils::ChronoTimingStats timer("motion_solver.solve_3d2d");
238 Pose3SolverResult result;
239 const size_t& n_matches = correspondences.size();
240
241 if (n_matches < 5u) {
242 result.status = TrackingStatus::FEW_MATCHES;
243 VLOG(5) << "3D2D tracking failed as there are to few matches" << n_matches;
244 return result;
245 }
246
247 gtsam::Matrix K = camera_params_.getCameraMatrixEigen();
248 K = K.inverse();
249
250 TrackletIds tracklets, inliers, outliers;
251 // NOTE: currently without distortion! the correspondences should be made into
252 // bearing vector elsewhere!
253 BearingVectors bearing_vectors;
254 Landmarks points;
255 for (size_t i = 0u; i < n_matches; i++) {
256 const AbsolutePoseCorrespondence& corres = correspondences.at(i);
257 const Keypoint& kp = corres.cur_;
258 // make Bearing vector
259 gtsam::Vector3 versor = (K * gtsam::Vector3(kp(0), kp(1), 1.0));
260 versor = versor.normalized();
261 bearing_vectors.push_back(versor);
262
263 points.push_back(corres.ref_);
264 tracklets.push_back(corres.tracklet_id_);
265 }
266
267 VLOG(20) << "Collected " << tracklets.size() << " initial correspondances";
268
269 const double reprojection_error = params_.ransac_threshold_pnp;
270 const double avg_focal_length =
271 0.5 * static_cast<double>(camera_params_.fx() + camera_params_.fy());
272 const double threshold =
273 1.0 - std::cos(std::atan(std::sqrt(2.0) * reprojection_error /
274 avg_focal_length));
275
276 AbsolutePoseAdaptor adapter(bearing_vectors, points);
277
278 if (R_curr_ref) {
279 adapter.setR(R_curr_ref->matrix());
280 }
281
282 gtsam::Pose3 best_result;
283 std::vector<int> ransac_inliers;
284
285 bool success;
286 {
287 utils::ChronoTimingStats timer("motion_solver.solve_3d2d.ransac");
288 success = runRansac<AbsolutePoseProblem>(
289 std::make_shared<AbsolutePoseProblem>(adapter,
290 AbsolutePoseProblem::KNEIP),
291 threshold, params_.ransac_iterations, params_.ransac_probability,
292 params_.optimize_3d2d_pose_from_inliers, best_result, ransac_inliers);
293 }
294
295 constructTrackletInliers(result.inliers, result.outliers, correspondences,
296 ransac_inliers, tracklets);
297
298 if (success) {
299 if (result.inliers.size() < 5u) {
300 result.status = TrackingStatus::FEW_MATCHES;
301 } else {
302 result.status = TrackingStatus::VALID;
303 result.best_result = best_result;
304 }
305
306 } else {
307 result.status = TrackingStatus::INVALID;
308 }
309
310 return result;
311}

313void OpticalFlowAndPoseOptimizer::updateFrameOutliersWithResult(
314 const Result& result, Frame::Ptr frame_k_1, Frame::Ptr frame_k) const {
315 utils::ChronoTimingStats timer("of_motion_solver.update_frame");
316 const auto& image_container_k = frame_k->image_container_;
317 const cv::Mat& motion_mask = image_container_k.objectMotionMask();
318
319 auto camera = frame_k->camera_;
320 const auto& refined_inliers = result.inliers;
321 const auto& refined_flows = result.best_result.refined_flows;
322
323 // outliers from the result. We will update this vector with new outliers
324 auto refined_outliers = result.outliers;
325
326 for (size_t i = 0; i < refined_inliers.size(); i++) {
327 TrackletId tracklet_id = refined_inliers.at(i);
328 gtsam::Point2 refined_flow = refined_flows.at(i);
329
330 Feature::Ptr feature_k_1 = frame_k_1->at(tracklet_id);
331 Feature::Ptr feature_k = frame_k->at(tracklet_id);
332
333 CHECK_EQ(feature_k->objectId(), result.best_result.object_id);
334
335 const Keypoint kp_k_1 = feature_k_1->keypoint();
336 Keypoint refined_keypoint = kp_k_1 + refined_flow;
337
338 // check boundaries?
339 if (!camera->isKeypointContained(refined_keypoint)) {
340 refined_outliers.push_back(tracklet_id);
341 continue;
342 }
343
344 ObjectId predicted_label =
345 functional_keypoint::at<ObjectId>(refined_keypoint, motion_mask);
346 if (predicted_label != result.best_result.object_id) {
347 refined_outliers.push_back(tracklet_id);
348 // TODO: other fields of the feature does not get updated? Inconsistencies
349 // as measured flow, predicted kp etc are no longer correct!!?
350 continue;
351 }
352
353 // update current keypoint
354 feature_k->keypoint(refined_keypoint);
355 // update refined frlow
356 feature_k_1->measuredFlow(refined_flow);
357 // update refined predicted keypoint
358 feature_k_1->predictedKeypoint(refined_keypoint);
359
360 // Logic is a bit convoluted and dependant on other things
361 // if we have optical flow (from k to k+1 due to historical reaseons)
362 // then use this to set the predicted keypoint
363 // if we are tracking using the provided flow then this MUST get set
364 // otherwise tracking will fail as tracking with provided optical flow uses
365 // the predicted keypoint to get the next keypoint!
366 if (image_container_k.hasOpticalFlow()) {
367 const cv::Mat& flow_image = image_container_k.opticalFlow();
368 const int x = functional_keypoint::u(refined_keypoint);
369 const int y = functional_keypoint::v(refined_keypoint);
370 double flow_xe = static_cast<double>(flow_image.at<cv::Vec2f>(y, x)[0]);
371 double flow_ye = static_cast<double>(flow_image.at<cv::Vec2f>(y, x)[1]);
372
373 OpticalFlow new_measured_flow(flow_xe, flow_ye);
374 feature_k->measuredFlow(new_measured_flow);
375 // TODO: check predicted flow is within image
376 Keypoint predicted_kp = Feature::CalculatePredictedKeypoint(
377 refined_keypoint, new_measured_flow);
378 feature_k->predictedKeypoint(predicted_kp);
379 } else {
380 // we dont have a predicted flow but we assume we are using KLT tracking
381 // in this case we dont have a predicted keypoint.
382 // just use the refined flow as the measured flow
383 feature_k->measuredFlow(refined_flow);
384 Keypoint predicted_kp =
385 Feature::CalculatePredictedKeypoint(refined_keypoint, refined_flow);
386 feature_k->predictedKeypoint(predicted_kp);
387 }
388 }
389
390 // update tracks
391 for (const auto& outlier_tracklet : refined_outliers) {
392 Feature::Ptr feature_k_1 = frame_k_1->at(outlier_tracklet);
393 Feature::Ptr feature_k = frame_k->at(outlier_tracklet);
394
395 CHECK(feature_k_1->usable());
396 CHECK(feature_k->usable());
397
398 feature_k->markOutlier();
399 feature_k_1->markOutlier();
400 }
401
402 // refresh depth information for each frame
403 CHECK(frame_k->updateDepths());
404}

473ObjectMotionSolver::Result ObjectMotionSolver::solve(Frame::Ptr frame_k,
474 Frame::Ptr frame_k_1) {
475 ObjectIds failed_object_tracks;
476 MotionEstimateMap motion_estimates;
477
478 // if only 1 object, no point parallelising
479 if (motion_estimates.size() <= 1) {
480 for (const auto& [object_id, observations] :
481 frame_k->object_observations_) {
482 if (!solveImpl(frame_k, frame_k_1, object_id, motion_estimates)) {
483 VLOG(5) << "Could not solve motion for object " << object_id
484 << " from frame " << frame_k_1->getFrameId() << " -> "
485 << frame_k->getFrameId();
486 failed_object_tracks.push_back(object_id);
487 }
488 }
489 } else {
490 std::mutex mutex;
491 // paralleilise the process of each function call.
492 tbb::parallel_for_each(
493 frame_k->object_observations_.begin(),
494 frame_k->object_observations_.end(),
495 [&](const std::pair<ObjectId, SingleDetectionResult>& pair) {
496 const auto object_id = pair.first;
497 if (!solveImpl(frame_k, frame_k_1, object_id, motion_estimates)) {
498 VLOG(5) << "Could not solve motion for object " << object_id
499 << " from frame " << frame_k_1->getFrameId() << " -> "
500 << frame_k->getFrameId();
501
502 std::lock_guard<std::mutex> lk(mutex);
503 failed_object_tracks.push_back(object_id);
504 }
505 });
506 }
507
508 // remove objects from the object observations list
509 // does not remove the features etc but stops the object being propogated to
510 // the backend as we loop over the object observations in the constructOutput
511 // function
512 for (auto object_id : failed_object_tracks) {
513 frame_k->object_observations_.erase(object_id);
514 }
515
516 ObjectPoseMap poses;
517 ObjectMotionMap motions;
518
519 updateMotions(motions, motion_estimates, frame_k, frame_k_1);
520 updatePoses(poses, motion_estimates, frame_k, frame_k_1);
521 return {motions, poses};
522}

531void ObjectMotionSovlerF2F::updatePoses(
532 ObjectPoseMap& object_poses, const MotionEstimateMap& motion_estimates,
533 Frame::Ptr frame_k, Frame::Ptr frame_k_1) {
534 gtsam::Point3Vector object_centroids_k_1, object_centroids_k;
535
536 for (const auto& [object_id, motion_estimate] : motion_estimates) {
537 auto object_points = FeatureFilterIterator(
538 const_cast<FeatureContainer&>(frame_k_1->dynamic_features_),
539 [object_id, &frame_k](const Feature::Ptr& f) -> bool {
540 return Feature::IsUsable(f) && f->objectId() == object_id &&
541 frame_k->exists(f->trackletId()) &&
542 frame_k->isFeatureUsable(f->trackletId());
543 });
544
545 gtsam::Point3 centroid_k_1(0, 0, 0);
546 gtsam::Point3 centroid_k(0, 0, 0);
547 size_t count = 0;
548 for (const auto& feature : object_points) {
549 gtsam::Point3 lmk_k_1 =
550 frame_k_1->backProjectToCamera(feature->trackletId());
551 centroid_k_1 += lmk_k_1;
552
553 gtsam::Point3 lmk_k = frame_k->backProjectToCamera(feature->trackletId());
554 centroid_k += lmk_k;
555
556 count++;
557 }
558
559 centroid_k_1 /= count;
560 centroid_k /= count;
561
562 centroid_k_1 = frame_k_1->getPose() * centroid_k_1;
563 centroid_k = frame_k->getPose() * centroid_k;
564
565 object_centroids_k_1.push_back(centroid_k_1);
566 object_centroids_k.push_back(centroid_k);
567 }
568
569 if (FLAGS_init_object_pose_from_gt) {
570 CHECK(object_motion_params.ground_truth_packets_request)
571 << "FLAGS_init_object_pose_from_gt is true but no ground truth packets "
572 "hook is set!";
573
574 const auto ground_truth_packets =
575 object_motion_params.ground_truth_packets_request();
576 LOG_IF(WARNING, !ground_truth_packets.has_value())
577 << "FLAGS_init_object_pose_from_gt but no ground truth provided! "
578 "Object poses will be initalised using centroid!";
579
580 dyno::propogateObjectPoses(object_poses_, motion_estimates,
581 object_centroids_k_1, object_centroids_k,
582 frame_k->getFrameId(), ground_truth_packets);
583 } else {
584 dyno::propogateObjectPoses(object_poses_, motion_estimates,
585 object_centroids_k_1, object_centroids_k,
586 frame_k->getFrameId());
587 }
588
589 object_poses = object_poses_;
590}

630Motion3SolverResult ObjectMotionSovlerF2F::geometricOutlierRejection3d2d(
631 Frame::Ptr frame_k_1, Frame::Ptr frame_k, const gtsam::Pose3& T_world_k,
632 ObjectId object_id) {
633 utils::ChronoTimingStats timer("motion_solver.object_solve3d2d");
634 AbsolutePoseCorrespondences dynamic_correspondences;
635 // get the corresponding feature pairs
636 bool corr_result = frame_k->getDynamicCorrespondences(
637 dynamic_correspondences, *frame_k_1, object_id,
638 frame_k->landmarkWorldKeypointCorrespondance());
639
640 const size_t& n_matches = dynamic_correspondences.size();
641
642 TrackletIds all_tracklets;
643 std::transform(dynamic_correspondences.begin(), dynamic_correspondences.end(),
644 std::back_inserter(all_tracklets),
645 [](const AbsolutePoseCorrespondence& corres) {
646 return corres.tracklet_id_;
647 });
648 CHECK_EQ(all_tracklets.size(), n_matches);
649
650 Pose3SolverResult geometric_result =
651 EgoMotionSolver::geometricOutlierRejection3d2d(dynamic_correspondences);
652 Pose3SolverResult pose_result = geometric_result;
653
654 Motion3SolverResult motion_result;
655 motion_result.status = pose_result.status;
656
657 if (pose_result.status == TrackingStatus::VALID) {
658 TrackletIds refined_inlier_tracklets = pose_result.inliers;
659
660 {
661 CHECK_EQ(pose_result.inliers.size() + pose_result.outliers.size(),
662 n_matches);
663
664 // debug only (just checking that the inlier/outliers we get from the
665 // geometric rejection match the original one)
666 TrackletIds extracted_all_tracklets = refined_inlier_tracklets;
667 extracted_all_tracklets.insert(extracted_all_tracklets.end(),
668 pose_result.outliers.begin(),
669 pose_result.outliers.end());
670 CHECK_EQ(all_tracklets.size(), extracted_all_tracklets.size());
671 }
672
673 gtsam::Pose3 G_w = pose_result.best_result.inverse();
674 if (object_motion_params.refine_motion_with_joint_of) {
675 OpticalFlowAndPoseOptimizer flow_optimizer(
676 object_motion_params.joint_of_params);
677 // Use the original result as the input to the refine joint optical flow
678 // function the result.best_result variable is actually equivalent to
679 // ^wG^{-1} and we want to solve something in the form e(T, flow) =
680 // [u,v]_{k-1} + {k-1}_flow_k - pi(T^{-1}^wm_{k-1}) so T must take the
681 // point from k-1 in the world frame to the local frame at k-1 ^wG^{-1} =
682 //^wX_k \: {k-1}^wH_k (which takes does this) but the error term uses the
683 // inverse of T hence we must parse in the inverse of G
684 auto flow_opt_result = flow_optimizer.optimizeAndUpdate<CalibrationType>(
685 frame_k_1, frame_k, refined_inlier_tracklets,
686 pose_result.best_result);
687 // still need to take the inverse as we get the inverse of G out
688 G_w = flow_opt_result.best_result.refined_pose.inverse();
689 // inliers should be a subset of the original refined inlier tracks
690 refined_inlier_tracklets = flow_opt_result.inliers;
691
692 VLOG(10) << "Refined object " << object_id
693 << "pose with optical flow - error before: "
694 << flow_opt_result.error_before.value_or(NaN)
695 << " error_after: " << flow_opt_result.error_after.value_or(NaN);
696 }
697 // still need to take the inverse as we get the inverse of G out
698 gtsam::Pose3 H_w = T_world_k * G_w;
699
700 if (object_motion_params.refine_motion_with_3d) {
701 VLOG(10) << "Refining object motion pose with 3D refinement";
702 MotionOnlyRefinementOptimizer motion_refinement_graph(
703 object_motion_params.object_motion_refinement_params);
704 auto motion_refinement_result =
705 motion_refinement_graph.optimizeAndUpdate<CalibrationType>(
706 frame_k_1, frame_k, refined_inlier_tracklets, object_id, H_w);
707
708 // should be further subset
709 refined_inlier_tracklets = motion_refinement_result.inliers;
710 H_w = motion_refinement_result.best_result;
711 }
712
713 motion_result.status = pose_result.status;
714 motion_result.best_result = Motion3ReferenceFrame(
715 H_w, Motion3ReferenceFrame::Style::F2F, ReferenceFrame::GLOBAL,
716 frame_k_1->getFrameId(), frame_k->getFrameId());
717 motion_result.inliers = refined_inlier_tracklets;
718 determineOutlierIds(motion_result.inliers, all_tracklets,
719 motion_result.outliers);
720
721 // sanity check that we have accounted for all initial matches
722 CHECK_EQ(motion_result.inliers.size() + motion_result.outliers.size(),
723 n_matches);
724 }
725
726 // needed when running things like TartanAir S where we get vew few point on
727 // the object...
728 // if (motion_result.inliers.size() < 30) {
729 // motion_result.status = TrackingStatus::FEW_MATCHES;
730 // }
731
732 return motion_result;
733}

1227void SquareRootInfoFilterGTSAM::update(
1228 const std::vector<gtsam::Point3>& P_w_list,
1229 const std::vector<gtsam::Point2>& z_obs_list, const gtsam::Pose3& X_W_k,
1230 const int num_irls_iterations) {
1231 if (P_w_list.empty() || P_w_list.size() != z_obs_list.size()) {
1232 LOG(WARNING) << "SRIF Update skipped: Input lists empty/mismatched.";
1233 return;
1234 }
1235
1236 const size_t num_measurements = P_w_list.size();
1237 const size_t total_rows = num_measurements * 2;
1238 const size_t state_dim = 6;
1239
1240 gtsam::noiseModel::mEstimator::Huber::shared_ptr huber =
1241 gtsam::noiseModel::mEstimator::Huber::Create(0.01);
1242
1243 double huber_delta_ = 1.2;
1244
1245 LOG(INFO) << "SRIF update with " << num_measurements << " measurements";
1246
1247 // 1. Calculate Jacobians (H) and Linearized Residuals (y_lin)
1248 // These are calculated ONCE at the linearization point and are fixed
1249 // for all IRLS iterations.
1250 gtsam::Matrix H_stacked = gtsam::Matrix ::Zero(total_rows, state_dim);
1251 gtsam::Vector y_linearized = gtsam::Vector::Zero(total_rows);
1252
1253 // Pre-calculate constant values
1254 gtsam::Pose3 G_w = X_W_k.inverse() * H_linearization_point_;
1255 gtsam::Matrix6 Ad_G_w = G_w.AdjointMap();
1256 gtsam::Matrix6 CorrectionFactor = -Ad_G_w;
1257
1258 const gtsam::Matrix22 R_inv = R_noise_.inverse();
1259
1260 // whitened error
1261 gtsam::Vector initial_error = gtsam::Vector::Zero(num_measurements);
1262
1263 gtsam::PinholeCamera<gtsam::Cal3_S2> camera(G_w.inverse(), *K_gtsam_);
1264 for (size_t i = 0; i < num_measurements; ++i) {
1265 const Point3& P_w = P_w_list[i];
1266 const Point2& z_obs = z_obs_list[i];
1267 gtsam::Matrix26 J_pi; // 2x6
1268 gtsam::Point2 z_pred;
1269 try {
1270 // Project using the *linearization point*
1271 z_pred = camera.project(P_w, J_pi);
1272 } catch (const gtsam::CheiralityException& e) {
1273 LOG(WARNING) << "Warning: Point " << i << " behind camera. Skipping.";
1274 continue; // Skip this measurement
1275 }
1276
1277 // Calculate linearized residual y_lin = z - h(x_nom)
1278 gtsam::Vector2 y_i_lin = z_obs - z_pred;
1279
1280 // Calculate EKF Jacobian H_i = J_pi * CorrectionFactor
1281 Eigen::Matrix<double, 2, 6> H_ekf_i = J_pi * CorrectionFactor;
1282
1283 size_t row_idx = i * 2;
1284 H_stacked.block<2, 6>(row_idx, 0) = H_ekf_i;
1285 y_linearized.segment<2>(row_idx) = y_i_lin;
1286
1287 double error_sq = y_i_lin.transpose() * R_inv * y_i_lin;
1288 double error = std::sqrt(error_sq);
1289 initial_error(i) = error;
1290 }
1291
1292 // 2. Store the prior information
1293 gtsam::Matrix6 R_info_prior = R_info_;
1294 gtsam::Vector6 d_info_prior = d_info_;
1295
1296 gtsam::Vector re_weighted_error = gtsam::Vector::Zero(num_measurements);
1297
1298 // 3. --- Start Iteratively Reweighted Least Squares (IRLS) Loop ---
1299 // cout << "\n--- SRIF Robust Update (IRLS) Started ---" << endl;
1300 for (int iter = 0; iter < num_irls_iterations; ++iter) {
1301 // 3a. Get current state estimate (from previous iteration)
1302 gtsam::Vector6 delta_w =
1303 R_info_.triangularView<Eigen::Upper>().solve(d_info_);
1304 gtsam::Pose3 H_current_mean = H_linearization_point_.retract(delta_w);
1305 // Need to validate this:
1306 // We intentionally do not recalculate the Jacobian block (H_stacked).
1307 // to solve for the single best perturbation (delta_w) relative to the
1308 // single linearization point (W_linearization_point_) that we had at the
1309 // start of the update.
1310 gtsam::Pose3 G_w = X_W_k.inverse() * H_current_mean;
1311 PinholeCamera<Cal3_S2> camera_current(G_w.inverse(), *K_gtsam_);
1312
1313 gtsam::Vector weights = gtsam::Vector::Ones(num_measurements);
1314
1315 for (size_t i = 0; i < num_measurements; ++i) {
1316 const gtsam::Point3& P_w = P_w_list[i];
1317 const gtsam::Point2& z_obs = z_obs_list[i];
1318 gtsam::Point2 z_pred_current;
1319 try {
1320 z_pred_current = camera_current.project(P_w);
1321 } catch (const gtsam::CheiralityException&) {
1322 LOG(WARNING) << "Warning: Point " << i << " behind camera. Skipping.";
1323 weights(i) = 0.0; // Skip point
1324 continue;
1325 }
1326
1327 // Calculate non-linear residual
1328 gtsam::Vector2 y_nonlinear = z_obs - z_pred_current;
1329
1330 // //CHECK we cannot do this with gtsam's noise models (we can...)
1331 // // Calculate Mahalanobis-like distance (whitened error)
1332 double error_sq = y_nonlinear.transpose() * R_inv * y_nonlinear;
1333 double error = std::sqrt(error_sq);
1334
1335 re_weighted_error(i) = error;
1336
1337 // Calculate Huber weight w(e) = min(1, delta / |e|)
1338 weights(i) = (error <= huber_delta_) ? 1.0 : huber_delta_ / error;
1339 if (/*weights(i) < 0.99 &&*/ iter == num_irls_iterations - 1) {
1340 VLOG(20) << " [Meas " << i << "] Final Weight: " << weights(i)
1341 << " (Error: " << error << ")";
1342 }
1343 }
1344
1345 // 3c. Construct the giant matrix A_qr for this iteration
1346 gtsam::Matrix A_qr =
1347 gtsam::Matrix::Zero(total_rows + state_dim, state_dim + 1);
1348
1349 for (size_t i = 0; i < num_measurements; ++i) {
1350 double w_i = weights(i);
1351 if (w_i < 1e-6) continue; // Skip 0-weighted points
1352
1353 // R_robust = R / w_i (Note: w is |e|^-1, so R_robust = R * |e|/delta)
1354 // R_robust_inv = R_inv * w_i
1355 // We need W_i such that W_i^T * W_i = R_robust_inv
1356 // W_i = sqrt(w_i) * R_inv_sqrt
1357
1358 gtsam::Matrix22 R_robust_i = R_noise_ / w_i;
1359 gtsam::Matrix22 L_robust_i = R_robust_i.llt().matrixL();
1360 gtsam::Matrix22 R_robust_inv_sqrt = L_robust_i.inverse();
1361
1362 size_t row_idx = i * 2;
1363
1364 // Fill matrix [ W_i * H | W_i * y_lin ]
1365 A_qr.block<2, 6>(row_idx, 0) =
1366 R_robust_inv_sqrt * H_stacked.block<2, 6>(row_idx, 0);
1367 A_qr.block<2, 1>(row_idx, 6) =
1368 R_robust_inv_sqrt * y_linearized.segment<2>(row_idx);
1369 }
1370
1371 // 3d. Fill the prior part of the QR matrix
1372 // This is *always* the original prior
1373 A_qr.block<6, 6>(total_rows, 0) = R_info_prior;
1374 A_qr.block<6, 1>(total_rows, 6) = d_info_prior;
1375
1376 // 3e. Perform QR decomposition
1377 Eigen::HouseholderQR<gtsam::Matrix> qr(A_qr);
1378 gtsam::Matrix R_full = qr.matrixQR().triangularView<Eigen::Upper>();
1379
1380 // 3f. Extract the new R_info and d_info for the *next* iteration
1381 R_info_ = R_full.block<6, 6>(0, 0);
1382 d_info_ = R_full.block<6, 1>(0, 6);
1383 }
1384
1385 // (Optional) Enforce R is upper-triangular
1386 R_info_ = R_info_.triangularView<Eigen::Upper>();
1387
1388 LOG(INFO) << "--- SRIF Robust Update Complete --- initial error "
1389 << initial_error.norm()
1390 << " re-weighted error: " << re_weighted_error.norm();
1391}

1645 // R_robust = R / w_i (Note: w is |e|^-1, so R_robust = R * |e|/delta)
1646 // R_robust_inv = R_inv * w_i
1647 // We need W_i such that W_i^T * W_i = R_robust_inv
1648 // W_i = sqrt(w_i) * R_inv_sqrt
1649
1650 gtsam::Matrix33 R_robust_i = R_noise_ / w_i;
1651 gtsam::Matrix33 L_robust_i = R_robust_i.llt().matrixL();
1652 gtsam::Matrix33 R_robust_inv_sqrt = L_robust_i.inverse();
1653
1654 size_t row_idx = i * ZDim;
1655
1656 A_qr.block<ZDim, StateDim>(row_idx, 0) =
1657 R_robust_inv_sqrt * H_stacked.block<3, StateDim>(row_idx, 0);
1658 A_qr.block<ZDim, 1>(row_idx, StateDim) =
1659 R_robust_inv_sqrt * y_linearized.segment<3>(row_idx);
1660 }
1661
1662 // 3d. Fill the prior part of the QR matrix
1663 // This is *always* the original prior
1664 A_qr.block<StateDim, StateDim>(total_rows, 0) = R_info_prior;
1665 A_qr.block<StateDim, 1>(total_rows, StateDim) = d_info_prior;
1666
1667 // 3e. Perform QR decomposition
1668 Eigen::HouseholderQR<gtsam::Matrix> qr(A_qr);
1669 gtsam::Matrix R_full = qr.matrixQR().triangularView<Eigen::Upper>();
1670
1671 // 3f. Extract the new R_info and d_info for the *next* iteration
1672 R_info_ = R_full.block<StateDim, StateDim>(0, 0);
1673 d_info_ = R_full.block<StateDim, 1>(0, StateDim);
1674 }
1675
1676 // (Optional) Enforce R is upper-triangular
1677 R_info_ = R_info_.triangularView<Eigen::Upper>();
1678
1679 Result result;
1680 result.error = initial_error.norm();
1681 result.reweighted_error = re_weighted_error.norm();
1682 return result;
1683

1686 // << " re-weighted error: " << re_weighted_error.norm();
