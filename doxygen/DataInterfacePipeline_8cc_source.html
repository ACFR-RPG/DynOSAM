49FrontendInputPacketBase::ConstPtr DataInterfacePipeline::getInputPacket() {
50 if (isShutdown()) {
51 return nullptr;
52 }
53
54 utils::StatsCollector queue_size_stats("data-interface_queue_size #");
55 queue_size_stats.AddSample(packet_queue_.size());
56
57 bool queue_state;
58 ImageContainer::Ptr packet = nullptr;
59
60 if (parallel_run_) {
61 queue_state = packet_queue_.pop(packet);
62 } else {
63 queue_state = packet_queue_.popBlocking(packet);
64 }
65
66 if (!queue_state) {
67 // LOG_EVERY_N(WARNING, 10)
68 // << "Module: " << MIMO::module_name_ << " - queue is down";
69 return nullptr;
70 }
71
72 CHECK(packet);
73
74 GroundTruthInputPacket::Optional ground_truth;
75 if (ground_truth_packets_.find(packet->frameId()) !=
76 ground_truth_packets_.end()) {
77 VLOG(5) << "Gotten ground truth packet for frame id " << packet->frameId()
78 << ", timestamp=" << packet->timestamp();
79 ground_truth = ground_truth_packets_.at(packet->frameId());
80 }
81
82 const Timestamp& timestamp = packet->timestamp();
83 ImuMeasurements::Optional imu_meas;
84 imu_meas.emplace();
85 FrameAction action = getTimeSyncedImuMeasurements(timestamp, &(*imu_meas));
86 switch (action) {
87 case FrameAction::Use:
88 CHECK(imu_meas);
89 break;
90 case FrameAction::Wait:
91 case FrameAction::Drop:
92 imu_meas.reset();
93 break;
94 }
95
96 auto frontend_input =
97 std::make_shared<FrontendInputPacketBase>(packet, ground_truth);
98 frontend_input->imu_measurements = imu_meas;
99 return frontend_input;
100}

109ImuInterfaceHandler::FrameAction
110ImuInterfaceHandler::getTimeSyncedImuMeasurements(const Timestamp& timestamp,
111 ImuMeasurements* imu_meas) {
112 if (imu_buffer_.isShutdown() || imu_buffer_.size() == 0u) {
113 return FrameAction::Drop;
114 }
115
116 CHECK_NOTNULL(imu_meas);
117 CHECK_LT(timestamp_last_frame_, timestamp)
118 << "Image timestamps out of order: " << timestamp_last_frame_
119 << "[s] (last) >= " << timestamp << "[s] (curr)";
120
121 if (imu_buffer_.size() == 0) {
122 VLOG(1) << "No IMU measurements available yet, dropping this frame.";
123 return FrameAction::Drop;
124 }
125
126 if (timestamp_last_frame_ == InvalidTimestamp) {
127 // TODO(Toni): wouldn't it be better to get all IMU measurements up to
128 // this
129 // timestamp? We should add a method to the IMU buffer for that.
130 VLOG(1) << "Skipping first frame, because we do not have a concept of "
131 "a previous frame timestamp otherwise.";
132 timestamp_last_frame_ = timestamp;
133 return FrameAction::Drop;
134 }
135
136 // // Do a very coarse timestamp correction to make sure that the IMU data
137 // // is aligned enough to send packets to the front-end. This is assumed
138 // // to be very inaccurate and should not be enabled without some other
139 // // actual time alignment in the frontend
140 // if (do_coarse_imu_camera_temporal_sync_) {
141 // ImuMeasurement newest_imu;
142 // imu_data_.imu_buffer_.getNewestImuMeasurement(&newest_imu);
143 // // this is delta = imu.timestamp - frame.timestamp so that when querying,
144 // // we get query = new_frame.timestamp + delta = frame_delta +
145 // imu.timestamp imu_timestamp_correction_ = newest_imu.timestamp_ -
146 // timestamp; do_coarse_imu_camera_temporal_sync_ = false; LOG(WARNING) <<
147 // "Computed intial coarse time alignment of "
148 // << UtilsNumerical::NsecToSec(imu_timestamp_correction_)
149 // << "[s]";
150 // }
151
152 // imu_time_shift_ can be externally, asynchronously modified.
153 // Caching here prevents a nasty race condition and avoids locking
154 // const Timestamp curr_imu_time_shift = imu_time_shift_ns_;
155 // const Timestamp imu_timestamp_last_frame =
156 // timestamp_last_frame_ + imu_timestamp_correction_ +
157 // curr_imu_time_shift;
158 // const Timestamp imu_timestamp_curr_frame =
159 // timestamp + imu_timestamp_correction_ + curr_imu_time_shift;
160
161 const Timestamp imu_timestamp_last_frame = timestamp_last_frame_;
162 const Timestamp imu_timestamp_curr_frame = timestamp;
163
164 // NOTE: using interpolation on both borders instead of just the upper
165 // as before because without a measurement on the left-hand side we are
166 // missing some of the motion or overestimating depending on the
167 // last timestamp's relationship to the nearest imu timestamp.
168 // For some datasets this caused an incorrect motion estimate
169 ThreadsafeImuBuffer::QueryResult query_result =
170 imu_buffer_.getImuDataInterpolatedBorders(
171 imu_timestamp_last_frame, imu_timestamp_curr_frame,
172 &imu_meas->timestamps_, &imu_meas->acc_gyr_);
173 // logQueryResult(timestamp, query_result);
174
175 switch (query_result) {
176 case ThreadsafeImuBuffer::QueryResult::kDataAvailable:
177 break; // handle this below
178 case ThreadsafeImuBuffer::QueryResult::kDataNotYetAvailable:
179 return FrameAction::Wait;
180 case ThreadsafeImuBuffer::QueryResult::kQueueShutdown:
181 // MISO::shutdown();
182 return FrameAction::Drop;
183 case ThreadsafeImuBuffer::QueryResult::kDataNeverAvailable:
184 timestamp_last_frame_ = timestamp;
185 return FrameAction::Drop;
186 case ThreadsafeImuBuffer::QueryResult::kTooFewMeasurementsAvailable:
187 default:
188 return FrameAction::Drop;
189 }
190
191 timestamp_last_frame_ = timestamp;
192
193 // adjust the timestamps for the frontend
194 // imu_meas->timestamps_.array() -=
195 // imu_timestamp_correction_ + curr_imu_time_shift;
196 VLOG(200) << "////////////////////////////////////////// Creating packet!\n"
197 << "STAMPS IMU rows : \n"
198 << imu_meas->timestamps_.rows() << '\n'
199 << "STAMPS IMU cols : \n"
200 << imu_meas->timestamps_.cols() << '\n'
201 << "STAMPS IMU: \n"
202 << imu_meas->timestamps_ << '\n'
203 << "ACCGYR IMU rows : \n"
204 << imu_meas->acc_gyr_.rows() << '\n'
205 << "ACCGYR IMU cols : \n"
206 << imu_meas->acc_gyr_.cols() << '\n'
207 << "ACCGYR IMU: \n"
208 << imu_meas->acc_gyr_;
209 return FrameAction::Use;
210}
