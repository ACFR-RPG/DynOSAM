49FrontendInputPacketBase::ConstPtr DataInterfacePipeline::getInputPacket() {
50 if (isShutdown()) {
51 return nullptr;
52 }
53
54 utils::StatsCollector queue_size_stats("data-interface_queue_size #");
55 queue_size_stats.AddSample(packet_queue_.size());
56
57 bool queue_state;
58 ImageContainer::Ptr packet = nullptr;
59
60 if (parallel_run_) {
61 queue_state = packet_queue_.pop(packet);
62 } else {
63 queue_state = packet_queue_.popBlocking(packet);
64 }
65
66 if (!queue_state) {
67 // LOG_EVERY_N(WARNING, 10)
68 // << "Module: " << MIMO::module_name_ << " - queue is down";
69 return nullptr;
70 }
71
72 CHECK(packet);
73
74 GroundTruthInputPacket::Optional ground_truth;
75 if (ground_truth_packets_.find(packet->getFrameId()) !=
76 ground_truth_packets_.end()) {
77 VLOG(5) << "Gotten ground truth packet for frame id "
78 << packet->getFrameId();
79 ground_truth = ground_truth_packets_.at(packet->getFrameId());
80 }
81
82 return std::make_shared<FrontendInputPacketBase>(packet, ground_truth);
83}

89bool DataInterfacePipelineImu::getTimeSyncedImuMeasurements(
90 const Timestamp& timestamp, ImuMeasurements* imu_meas) {
91 CHECK_NOTNULL(imu_meas);
92 CHECK_LT(timestamp_last_sync_, timestamp)
93 << "Timestamps out of order:\n"
94 << " - Last Frame Timestamp = " << timestamp_last_sync_ << '\n'
95 << " - Current Timestamp = " << timestamp;
96
97 if (imu_buffer_.size() == 0) {
98 VLOG(1) << "No IMU measurements available yet, dropping this frame.";
99 return false;
100 }
101
102 // Extract imu measurements between consecutive frames.
103 if (dyno::fpEqual(timestamp_last_sync_, 0.0)) {
104 // TODO(Toni): wouldn't it be better to get all IMU measurements up to
105 // this
106 // timestamp? We should add a method to the IMU buffer for that.
107 VLOG(1) << "Skipping first frame, because we do not have a concept of "
108 "a previous frame timestamp otherwise.";
109 timestamp_last_sync_ = timestamp;
110 return false;
111 }
112
113 ThreadsafeImuBuffer::QueryResult query_result =
114 ThreadsafeImuBuffer::QueryResult::kDataNeverAvailable;
115 bool log_error_once = true;
116 while (!MIMO::isShutdown() &&
117 (query_result = imu_buffer_.getImuDataInterpolatedUpperBorder(
118 timestamp_last_sync_, timestamp, &imu_meas->timestamps_,
119 &imu_meas->acc_gyr_)) !=
120 ThreadsafeImuBuffer::QueryResult::kDataAvailable) {
121 VLOG(1) << "No IMU data available. Reason:\n";
122 switch (query_result) {
123 case ThreadsafeImuBuffer::QueryResult::kDataNotYetAvailable: {
124 if (log_error_once) {
125 LOG(WARNING) << "Waiting for IMU data...";
126 log_error_once = false;
127 }
128 continue;
129 }
130 case ThreadsafeImuBuffer::QueryResult::kQueueShutdown: {
131 LOG(WARNING)
132 << "IMU buffer was shutdown. Shutting down DataInterfacePipeline.";
133 MIMO::shutdown();
134 return false;
135 }
136 case ThreadsafeImuBuffer::QueryResult::kDataNeverAvailable: {
137 LOG(WARNING)
138 << "Asking for data before start of IMU stream, from timestamp: "
139 << timestamp_last_sync_ << " to timestamp: " << timestamp;
140 // Ignore frames that happened before the earliest imu data
141 timestamp_last_sync_ = timestamp;
142 return false;
143 }
144 case ThreadsafeImuBuffer::QueryResult::kTooFewMeasurementsAvailable: {
145 LOG(WARNING) << "No IMU measurements here, and IMU data stream already "
146 "passed this time region"
147 << "from timestamp: " << timestamp_last_sync_
148 << " to timestamp: " << timestamp;
149 return false;
150 }
151 case ThreadsafeImuBuffer::QueryResult::kDataAvailable: {
152 LOG(FATAL) << "We should not be inside this while loop if IMU data is "
153 "available...";
154 return false;
155 }
156 }
157 }
158 timestamp_last_sync_ = timestamp;
159
160 VLOG(10) << "////////////////////////////////////////// Creating packet!\n"
161 << "STAMPS IMU rows : \n"
162 << imu_meas->timestamps_.rows() << '\n'
163 << "STAMPS IMU cols : \n"
164 << imu_meas->timestamps_.cols() << '\n'
165 << "STAMPS IMU: \n"
166 << imu_meas->timestamps_ << '\n'
167 << "ACCGYR IMU rows : \n"
168 << imu_meas->acc_gyr_.rows() << '\n'
169 << "ACCGYR IMU cols : \n"
170 << imu_meas->acc_gyr_.cols() << '\n'
171 << "ACCGYR IMU: \n"
172 << imu_meas->acc_gyr_;
173
174 return true;
175}
