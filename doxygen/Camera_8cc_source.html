40void Camera::project(const Landmark& lmk, Keypoint* kpt) const {
41 CHECK_NOTNULL(kpt);
42 // I believe this project call in gtsam is quite inefficient as it goes
43 // through a cascade of calls... Only useful if you want gradients I guess...
44 *kpt = camera_impl_->project2(lmk);
45}

47void Camera::project(const Landmarks& lmks, Keypoints* kpts) const {
48 CHECK_NOTNULL(kpts)->clear();
49 const auto& n_lmks = lmks.size();
50 kpts->resize(n_lmks);
51 // Can be greatly optimized with matrix mult or vectorization
52 for (size_t i = 0u; i < n_lmks; i++) {
53 project(lmks[i], &(*kpts)[i]);
54 }
55}

61bool Camera::isKeypointContained(const Keypoint& kpt) const {
62 return kpt(0) >= 0.0 && kpt(0) < camera_params_.ImageWidth() &&
63 kpt(1) >= 0.0 && kpt(1) < camera_params_.ImageHeight();

66void Camera::backProject(const Keypoints& kps, const Depths& depths, Landmarks* lmks) const {
67 CHECK_NOTNULL(lmks)->clear();
68 lmks->reserve(kps.size());
69 CHECK_EQ(kps.size(), depths.size());
70 CHECK(camera_impl_);
71 for (size_t i = 0u; i < kps.size(); i++) {
72 Landmark lmk;
73 backProject(kps[i], depths[i], &lmk);
74 lmks->push_back(lmk);
75 }
76}
77void Camera::backProject(const Keypoint& kp, const Depth& depth, Landmark* lmk) const {
78 CHECK(lmk);
79 backProject(kp, depth, lmk, gtsam::Pose3::Identity());

93void Camera::backProjectFromZ(const Keypoint& kp, const double Z, Landmark* lmk, const gtsam::Pose3& X_world, gtsam::OptionalJacobian<3, 3> Dp) const {
94 CHECK(lmk);
95 const auto calibration = camera_impl_->calibration();
96 //Convert (distorted) image coordinates uv to intrinsic coordinates xy
97 Eigen::Matrix<double, 2, decltype(calibration)::dimension> Dcal; //unused
98 gtsam::Matrix22 Dcal_point;
99 gtsam::Point2 pc = calibration.calibrate(kp, Dcal, Dcal_point);
100
101 // projection equation x = f(X/Z). The f is used to normalize the image cooridinates and since we already have
102 // normalized cordinates xy, we can omit the f
103 const double X = pc(0) * Z;
104 const double Y = pc(1) * Z;
105
106 const gtsam::Point3 P_camera(X, Y, Z);
107
108 //jacobian of the transforFrom function w.r.t P_camera
109 gtsam::Matrix33 H_point;
110 *lmk = X_world.transformFrom(P_camera, boost::none, H_point);
111
112 if(Dp) {
113 //TODO: test!!!
114 //cal1 should be top row of Dcal_point which determines how the pc(0) changes with input u, v
115 const double dcal1_du = Dcal_point(0,0);
116 const double dcal1_dv = Dcal_point(0,1);
117
118 //cal1 should be top row of Dcal_point which determines how the pc(1) changes with input u, v
119 const double dcal2_du = Dcal_point(1, 0);
120 const double dcal2_dv = Dcal_point(1, 1);
121
122 gtsam::Matrix33 J;
123 //top row is how X changes with u, v and Z
124 J(0, 0) = dcal1_du * Z;
125 J(0, 1) = dcal1_dv * Z;
126 J(0, 2) = pc(0);
127
128 //how Y changes with u, v and Z
129 J(1, 0) = dcal2_du * Z;
130 J(1, 1) = dcal2_dv * Z;
131 J(1, 2) = pc(1);
132
133 //how Z changes with Z
134 J(2, 0) = 0;
135 J(2, 1) = 0;
136 J(2, 2) = 1;
137
138 //J is computed as the jacobian without the affect of X_world
139 // f = X_world * C(u, v, Z)
140 // = X_world * P_camera
141 //df/dfuvz = df/dC * dC/duvz
142 // = df/dC * J
143 //since C -> C(u, v, z) = point in R^3, df/dC is the Jacobian of X_world * P_camera -> Rotation component of X_world, or H_point
144 *Dp = H_point * J;
145 }
146
147}

155bool Camera::isLandmarkContained(const Landmark& lmk, Keypoint* keypoint) const {
156 Keypoint kp;
157
158 try {
159 project(lmk, &kp);
160 }
161 catch(const gtsam::CheiralityException&) {
162 //if CheiralityException then point is behind camera
163 return false;
164 }
165
166 const bool result = isKeypointContained(kp);
167 if(keypoint) *keypoint = kp;
168 return result;
169}
