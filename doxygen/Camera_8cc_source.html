50void Camera::project(const Landmark& lmk, Keypoint* kpt) const {
51 CHECK_NOTNULL(kpt);
52 // I believe this project call in gtsam is quite inefficient as it goes
53 // through a cascade of calls... Only useful if you want gradients I guess...
54 *kpt = camera_impl_->project2(lmk);

57void Camera::project(const Landmarks& lmks, Keypoints* kpts) const {
58 CHECK_NOTNULL(kpts)->clear();
59 const auto& n_lmks = lmks.size();
60 kpts->resize(n_lmks);
61 // Can be greatly optimized with matrix mult or vectorization
62 for (size_t i = 0u; i < n_lmks; i++) {
63 project(lmks[i], &(*kpts)[i]);
64 }
65}

76void Camera::backProject(const Keypoints& kps, const Depths& depths,
77 Landmarks* lmks) const {
78 CHECK_NOTNULL(lmks)->clear();
79 lmks->reserve(kps.size());
80 CHECK_EQ(kps.size(), depths.size());
81 CHECK(camera_impl_);
82 for (size_t i = 0u; i < kps.size(); i++) {
83 Landmark lmk;
84 backProject(kps[i], depths[i], &lmk);
85 lmks->push_back(lmk);
86 }
87}

107void Camera::backProjectFromZ(const Keypoint& kp, const double Z, Landmark* lmk,
108 const gtsam::Pose3& X_world,
109 gtsam::OptionalJacobian<3, 3> Dp) const {
110 CHECK(lmk);
111 const auto calibration = camera_impl_->calibration();
112 // Convert (distorted) image coordinates uv to intrinsic coordinates xy
113 Eigen::Matrix<double, 2, decltype(calibration)::dimension> Dcal; // unused
114 gtsam::Matrix22 Dcal_point;
115 gtsam::Point2 pc = calibration.calibrate(kp, Dcal, Dcal_point);
116
117 // projection equation x = f(X/Z). The f is used to normalize the image
118 // cooridinates and since we already have normalized cordinates xy, we can
119 // omit the f
120 const double X = pc(0) * Z;
121 const double Y = pc(1) * Z;
122
123 const gtsam::Point3 P_camera(X, Y, Z);
124
125 // jacobian of the transforFrom function w.r.t P_camera
126 gtsam::Matrix33 H_point;
127 *lmk = X_world.transformFrom(P_camera, boost::none, H_point);
128
129 if (Dp) {
130 // TODO: test!!!
131 // cal1 should be top row of Dcal_point which determines how the pc(0)
132 // changes with input u, v
133 const double dcal1_du = Dcal_point(0, 0);
134 const double dcal1_dv = Dcal_point(0, 1);
135
136 // cal1 should be top row of Dcal_point which determines how the pc(1)
137 // changes with input u, v
138 const double dcal2_du = Dcal_point(1, 0);
139 const double dcal2_dv = Dcal_point(1, 1);
140
141 gtsam::Matrix33 J;
142 // top row is how X changes with u, v and Z
143 J(0, 0) = dcal1_du * Z;
144 J(0, 1) = dcal1_dv * Z;
145 J(0, 2) = pc(0);
146
147 // how Y changes with u, v and Z
148 J(1, 0) = dcal2_du * Z;
149 J(1, 1) = dcal2_dv * Z;
150 J(1, 2) = pc(1);
151
152 // how Z changes with Z
153 J(2, 0) = 0;
154 J(2, 1) = 0;
155 J(2, 2) = 1;
156
157 // J is computed as the jacobian without the affect of X_world
158 // f = X_world * C(u, v, Z)
159 // = X_world * P_camera
160 // df/dfuvz = df/dC * dC/duvz
161 // = df/dC * J
162 // since C -> C(u, v, z) = point in R^3, df/dC is the Jacobian of X_world *
163 // P_camera -> Rotation component of X_world, or H_point
164 *Dp = H_point * J;
165 }
166}

173bool Camera::isLandmarkContained(const Landmark& lmk,
174 Keypoint* keypoint) const {
175 Keypoint kp;
176
177 try {
178 project(lmk, &kp);
179 } catch (const gtsam::CheiralityException&) {
180 // if CheiralityException then point is behind camera
181 return false;
182 }
183
184 const bool result = isKeypointContained(kp);
185 if (keypoint) *keypoint = kp;
186 return result;
187}
