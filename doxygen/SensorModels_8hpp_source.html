197template <typename T, int D = gtsam::traits<T>::dimension>
198class MeasurementWithCovariance {
199 public:
200 using This = MeasurementWithCovariance<T, D>;
202 enum { dimension = D };
204 using Covariance = Eigen::Matrix<double, D, D>;
206 using Sigmas = Eigen::Matrix<double, D, 1>;
207 MeasurementWithCovariance() = default;
208 MeasurementWithCovariance(const T& measurement,
209 const gtsam::SharedGaussian& model)
210 : measurement_(measurement), model_(model) {}
211 MeasurementWithCovariance(const T& measurement)
212 : measurement_(measurement), model_(nullptr) {}
213 MeasurementWithCovariance(const T& measurement, const Covariance& covariance)
214 : measurement_(measurement),
215 model_(gtsam::noiseModel::Gaussian::Covariance(covariance)) {}
216 MeasurementWithCovariance(const T& measurement, const Sigmas& sigmas)
217 : measurement_(measurement),
218 model_(gtsam::noiseModel::Diagonal::Sigmas(sigmas)) {}
219
220 MeasurementWithCovariance(const std::pair<T, Covariance>& pair)
221 : MeasurementWithCovariance(pair.first, pair.second) {}
222
223 const T& measurement() const { return measurement_; }
224 const gtsam::SharedGaussian& model() const { return model_; }
225 bool hasModel() const { return model_ != nullptr; }
226
227 Covariance covariance() const {
228 if (hasModel()) {
229 return model_->covariance();
230 } else {
231 return Covariance::Zero();
232 }
233 }
234
237
238 operator const T&() const { return measurement(); }
239 operator Covariance() const { return covariance(); }
240
241 friend std::ostream& operator<<(std::ostream& os,
242 const This& measurement_with_cov) {
243 os << "m: " << measurement_with_cov.measurement()
244 << ", cov: " << measurement_with_cov.covariance()
245 << ", has model: " << std::boolalpha << measurement_with_cov.hasModel();
246 return os;
247 }
248
249 inline void print(const std::string& s = "") const {
250 std::cout << s << *this << std::endl;
251 }
252
253 inline bool equals(const MeasurementWithCovariance& other,
254 double tol = 1e-8) const {
255 return gtsam::traits<T>::Equals(measurement_, other.measurement_, tol) &&
256 utils::equateGtsamSharedValues(model_, other.model_, tol);
257 }
258
259 private:
261 T measurement_;
264 gtsam::SharedGaussian model_{nullptr};
265};
