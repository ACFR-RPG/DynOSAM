206template <typename T, int D = gtsam::traits<T>::dimension>
207class MeasurementWithCovariance {
208 public:
209 using This = MeasurementWithCovariance<T, D>;
211 inline constexpr static auto dimension = D;
213 using Covariance = Eigen::Matrix<double, D, D>;
215 using Sigmas = Eigen::Matrix<double, D, 1>;
216
221 MeasurementWithCovariance() = default;
222
233 MeasurementWithCovariance(const T& measurement,
234 const gtsam::SharedGaussian& model)
235 : measurement_(measurement), model_(CHECK_NOTNULL(model)) {
236 if (static_cast<int>(model->dim()) != This::dimension) {
237 DYNO_THROW_MSG(DynosamException)
238 << "Invalid gtsam::SharedGaussian model provided to "
239 << "MeasurementWithCovariance. Type dimensions are "
240 << This::dimension << " but noise models dims are " << model->dim();
241 }
242 }
243
254 explicit MeasurementWithCovariance(const T& measurement)
255 : measurement_(measurement), model_(nullptr) {}
256
265 MeasurementWithCovariance(const T& measurement, const Covariance& covariance)
266 : measurement_(measurement),
267 model_(gtsam::noiseModel::Gaussian::Covariance(covariance)) {}
268
278 MeasurementWithCovariance(const T& measurement, const Sigmas& sigmas)
279 : measurement_(measurement),
280 model_(gtsam::noiseModel::Diagonal::Sigmas(sigmas)) {}
281
282 explicit MeasurementWithCovariance(const std::pair<T, Covariance>& pair)
283 : MeasurementWithCovariance(pair.first, pair.second) {}
284
285 const T& measurement() const { return measurement_; }
286 const gtsam::SharedGaussian& model() const { return model_; }
287 bool hasModel() const { return model_ != nullptr; }
288
289 Covariance covariance() const {
290 if (hasModel()) {
291 return model_->covariance();
292 } else {
293 return Covariance::Zero();
294 }
295 }
296
299
300 operator const T&() const { return measurement(); }
301 operator Covariance() const { return covariance(); }
302
303 friend std::ostream& operator<<(std::ostream& os,
304 const This& measurement_with_cov) {
305 os << "m: " << measurement_with_cov.measurement()
306 << ", cov: " << measurement_with_cov.covariance()
307 << ", has model: " << std::boolalpha << measurement_with_cov.hasModel();
308 return os;
309 }
310
311 inline void print(const std::string& s = "") const {
312 std::cout << s << *this << std::endl;
313 }
314
315 inline bool equals(const MeasurementWithCovariance& other,
316 double tol = 1e-8) const {
317 return gtsam::traits<T>::Equals(measurement_, other.measurement_, tol) &&
318 utils::equateGtsamSharedValues(model_, other.model_, tol);
319 }
320
321 private:
323 T measurement_;
326 gtsam::SharedGaussian model_{nullptr};
327};
