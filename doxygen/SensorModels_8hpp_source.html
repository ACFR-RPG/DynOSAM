202template <typename T, int D = gtsam::traits<T>::dimension>
203class MeasurementWithCovariance {
204 public:
205 using This = MeasurementWithCovariance<T, D>;
207 inline constexpr static auto dimension = D;
209 using Covariance = Eigen::Matrix<double, D, D>;
211 using Sigmas = Eigen::Matrix<double, D, 1>;
213 using Optional = std::optional<This>;
214
219 MeasurementWithCovariance() = default;
220
231 MeasurementWithCovariance(const T& measurement,
232 const gtsam::SharedGaussian& model)
233 : measurement_(measurement), model_(CHECK_NOTNULL(model)) {
234 if (static_cast<int>(model->dim()) != This::dimension) {
235 DYNO_THROW_MSG(DynosamException)
236 << "Invalid gtsam::SharedGaussian model provided to "
237 << "MeasurementWithCovariance. Type dimensions are "
238 << This::dimension << " but noise models dims are " << model->dim();
239 }
240 }
241
252 explicit MeasurementWithCovariance(const T& measurement)
253 : measurement_(measurement), model_(nullptr) {}
254
255 explicit MeasurementWithCovariance(const std::pair<T, Covariance>& pair)
256 : MeasurementWithCovariance(pair.first, pair.second) {}
257
267 MeasurementWithCovariance(const T& measurement, const Covariance& covariance)
268 : measurement_(measurement),
269 model_(gtsam::noiseModel::Gaussian::Covariance(covariance, false)) {}
270
281 static MeasurementWithCovariance FromSigmas(const T& measurement,
282 const Sigmas& sigmas) {
283 return MeasurementWithCovariance(
284 measurement, gtsam::noiseModel::Diagonal::Sigmas(sigmas));
285 }
286
287 const T& measurement() const { return measurement_; }
288 const gtsam::SharedGaussian& model() const { return model_; }
289 bool hasModel() const { return model_ != nullptr; }
290
291 Covariance covariance() const {
292 if (hasModel()) {
293 return model_->covariance();
294 } else {
295 return Covariance::Zero();
296 }
297 }
298
301 operator const T&() const { return measurement(); }
302 operator Covariance() const { return covariance(); }
303
314 operator std::pair<T, gtsam::SharedNoiseModel>() const {
315 return {measurement(), model()};
316 }
317
318 friend std::ostream& operator<<(std::ostream& os,
319 const This& measurement_with_cov) {
320 os << "m: " << measurement_with_cov.measurement()
321 << ", cov: " << measurement_with_cov.covariance()
322 << ", has model: " << std::boolalpha << measurement_with_cov.hasModel();
323 return os;
324 }
325
326 inline void print(const std::string& s = "") const {
327 std::cout << s << *this << std::endl;
328 }
329
330 inline bool equals(const MeasurementWithCovariance& other,
331 double tol = 1e-8) const {
332 return gtsam::traits<T>::Equals(measurement_, other.measurement_, tol) &&
333 utils::equateGtsamSharedValues(model_, other.model_, tol);
334 }
335
336 private:
338 T measurement_;
341 gtsam::SharedGaussian model_{nullptr};
342};
