48class BackendExperimentsNode : public DynoNode {
49 public:
50 BackendExperimentsNode() : DynoNode("dynosam_experiments") {
51 RCLCPP_INFO_STREAM(this->get_logger(), "Starting BackendExperimentsNode");
52
53 auto data_loader = this->createDataProvider();
54 auto params = this->getDynoParams();
55
56 CameraParams camera_params;
57 if (params.preferDataProviderCameraParams() &&
58 data_loader->getCameraParams().has_value()) {
59 LOG(INFO) << "Using camera params from DataProvider, not the config in "
60 "the CameraParams.yaml!";
61 camera_params = *data_loader->getCameraParams();
62 } else {
63 LOG(INFO) << "Using camera params specified in CameraParams.yaml!";
64 camera_params = params.camera_params_;
65 }
66
67 Camera::Ptr camera = std::make_shared<Camera>(camera_params);
68
69 using BackendModuleTraits = RGBDBackendModule::ModuleTraits;
70 using MapType = RGBDBackendModule::MapType;
71 using MeasurementType = RGBDBackendModule::MeasurementType;
72
73 typename MapType::Ptr map = MapType::create();
74
75 LOG(INFO) << "Offline RGBD frontend";
76 const std::string file_path =
77 getOutputFilePath(kRgbdFrontendOutputJsonFile);
78
79 using OfflineFrontend =
80 FrontendOfflinePipeline<RGBDBackendModule::ModuleTraits>;
81
82 OfflineFrontend::UniquePtr offline_frontend =
83 std::make_unique<OfflineFrontend>("offline-rgbdfrontend", file_path,
84 FLAGS_ending_frame);
85
86 // raw ptr type becuase we cannot copy the unique ptr!! This is only becuase
87 // we need it in the lambda function which is a temporary solution
88 OfflineFrontend* offline_frontend_ptr = offline_frontend.get();
89
90 // TODO: make better params!!
91 auto updater_type =
92 static_cast<RGBDBackendModule::UpdaterType>(FLAGS_backend_updater_enum);
93
94 params.backend_params_.full_batch_frame = offline_frontend->endingFrame();
95
96 auto backend = std::make_shared<RGBDBackendModule>(
97 params.backend_params_, map, camera, updater_type);
98
99 backend_pipeline_ = std::make_unique<BackendPipeline>(
100 "backend-pipeline", &backend_input_queue_, backend);
101 backend_pipeline_->parallelRun(params.parallelRun());
102 // also register connection between front and back
103 offline_frontend_ptr->registerOutputQueue(&backend_input_queue_);
104 // NO OUTPUT!!
105
106 // convert pipeline to base type
107 frontend_pipeline_ = std::move(offline_frontend);
108 }
109
110 // TODO: copy pasted from DynoManager class - streamline/make shutdown hook
111 ~BackendExperimentsNode() {
112 // TODO: make shutdown hook!
113 writeStatisticsSamplesToFile("statistics_samples.csv");
114 writeStatisticsModuleSummariesToFile();
115 }
116
117 bool spinOnce() override {
118 RCLCPP_INFO_STREAM_THROTTLE(this->get_logger(), *this->get_clock(), 2000,
119 getStats());
120 if (frontend_pipeline_->isWorking()) {
121 frontend_pipeline_->spinOnce();
122 backend_pipeline_->spinOnce();
123 return true;
124 }
125 return false;
126 }
127
128 private:
129 PipelineBase::UniquePtr frontend_pipeline_{nullptr};
130 BackendPipeline::UniquePtr backend_pipeline_{nullptr};
131 FrontendPipeline::OutputQueue backend_input_queue_;
132};
133
134} // namespace dyno

136int main(int argc, char* argv[]) {
137 auto non_ros_args = dyno::initRosAndLogging(argc, argv);
138
139 rclcpp::NodeOptions options;
140 options.arguments(non_ros_args);
141 options.use_intra_process_comms(true);
142
143 rclcpp::executors::SingleThreadedExecutor exec;
144 auto ros_pipeline = std::make_shared<dyno::BackendExperimentsNode>();
145
146 exec.add_node(ros_pipeline);
147 while (rclcpp::ok()) {
148 if (!ros_pipeline->spinOnce()) {
149 break;
150 }
151 exec.spin_some();
152 }
153
154 ros_pipeline.reset();
155}
